هونه‌رمه‌ند (  فواد عوسمان ئه‌حمه‌د) له‌ سلێمانی له‌ساڵی  ١٩٧٨له‌ دایك بووه‌، ده‌رچوی په‌یمانگای هونه‌ره‌ جوانه‌كانی سلێمانیه‌ به‌شی شانۆ، له‌گه‌ڵ به‌رده‌وام دان به‌كاری هونه‌ری خۆی له‌سه‌ره‌تای ساڵی ١٩٩٨ له‌ كه‌ناڵی خاك كاری كردووه‌ وه‌ك ( بێژه‌ر، مونتێر، دۆبله‌ر) پاشان له‌ كه‌ناڵی كوردسات كاری كردووه‌ تاساڵی ٢٠٠٢ له‌و ماوه‌یه‌دا له‌ سه‌رجه‌م دراماكانی هونه‌رمه‌ندی حسێن میسری به‌شداری كردوه‌،  پاشان چووه‌ته‌ ده‌ره‌وه‌ی وڵات و له‌وڵاتی به‌ریتانیا تاساڵی ٢٠١٠ نیشته‌جێبوه‌، له‌گه‌ڵ گه‌رانه‌وه‌ی بۆ كوردستان له‌ساڵی ٢٠١٠ له‌ كۆمپانیای خاوێ فیلم وه‌ك سه‌رپه‌رشتیاری دۆبلاژ كاری كردوه‌ و سه‌ر په‌رشتی زیاتر له‌ ١٨ كاری خاوێی كردوه‌،  له‌هه‌مان كاتدا به‌رپرسی به‌شی راگه‌یاندنی گروپی كۆمپانیاكانی قه‌یوانه‌و به‌رده‌وامی به‌كاره‌ هونه‌ریه‌كانی ده‌دات، له‌م دیمانه‌یه‌دا چه‌ند پرسیارێكمان ئاراسته‌ی هونه‌رمه‌ند كر و به‌م جۆره‌ وه‌ڵامی پرسیاره‌كانی داینه‌وه‌.
هه‌رگیز له‌ كاری هونه‌ری دانه‌براوم به‌ڵام له‌به‌ر ئه‌وه‌ی ئه‌و ماوه‌ زۆره‌ له‌ كوردستان نه‌بوم ئازیزانم كاره‌كانی منیان نه‌دیوه‌، من ئه‌وماوه‌یه‌ی كه‌له‌ ده‌ره‌وه‌ به‌شداری كاری نواندنم له‌ فلیما كردووه‌ و پاشانیش جگه‌ له‌ پێشكه‌ش كردنی چه‌ند شانۆگه‌ریه‌ك له‌وێ ئێمه‌ ساڵی ٢٠٠٩ شانۆی ( شه‌وێك له‌وێ )مان نمایشكرد له‌ وڵاتی به‌ریتانیا پاشان له‌ سه‌ر داوه‌تی وه‌زاره‌تی رۆشنبیری له‌ شاره‌كانی ( سلێمانی، هه‌ولێر، دهۆك، زاخۆ) نمایشمان كرد كه‌ له‌ ده‌رهێنانی خۆم بوو، بۆیه‌ ده‌كرێت بڵێم له‌وكاره‌ درامیانه‌ ئه‌و چه‌ن ساڵه‌ به‌شدار نه‌بوم چونكه‌ له‌ كوردستان نه‌بوم به‌ڵام خۆش به‌ختانه‌ ئێستا لێره‌ جێگیرم .
باوه‌ڕم به‌وه‌یه‌ كه‌ ئێمه‌ ره‌خنه‌ بگرین ده‌بێت به‌ زمانی گوڵ قسه‌ بكه‌ین جا ئه‌و ره‌خنه‌یه‌ سیاسی بێت هونه‌ری بێت یان هه‌ر شتێكی تر، بێگومان ئاستی هونه‌ری درامای كوردی ده‌توانم بڵێم باش نیه‌ و له‌گه‌ڵ پێویستی و پێشكه‌وتنه‌كانی دنیا نه‌رۆشتوه‌ هه‌ربۆیه‌ بینه‌ری راسته‌قینه‌ی درامای كوردی ده‌بینین روی كردۆته‌ به‌دیله‌كانی درامای كوردی وه‌كوو دراما دۆبلاژكراوه‌كان، بۆیه‌ حه‌زه‌كه‌م زۆر به‌راشكاوی بڵێم درامای كوردی به‌ خه‌ڵكی  چاك و كه‌سی به‌رێز و ناسراو گه‌شه‌ناكات، ده‌بێت پشت به‌ هێنانی توانای ده‌ره‌كی ببه‌ستین به‌ڵام خه‌ڵكانی خۆمان لێوه‌ی فێربن و دواتر ئه‌م كاره‌ پرۆفشناڵانه‌ بكه‌ینه‌ موڵكی خۆمان،  ده‌بێت كاری ده‌رهێنان و ده‌قی درامی و پشت كامێرا كه‌سانێك بیكه‌ن كه‌ ئیشی خۆیان بێت و تایبه‌ت بن له‌و بواره‌دا، مادام ئێمه‌ ئامانجمان باشكردنی ئاستی درامای كوردیه‌ ده‌بێت هه‌مو هه‌نگاوێكی بوێر بنێین له‌و بواره‌دا .
ئایا ده‌كرێت درامای كوردی وه‌ك درامای توركی و هه‌ندێك وڵاتی تر رۆڵی گرنگ ببینن له‌ راكێشانی گه‌شتیاراندا بۆ كوردستان وه‌ك دراما توركیه‌كان؟
به‌دڵنیایه‌وه‌ ده‌توانرێ، وابزانم دووكه‌س نابینی نكوڵی له‌وه‌ بكات سروشت و جوانی كوردستان سه‌ر سامی نه‌كردبێت و ئاسایشێكی باشیش ده‌سته‌به‌ر كراوه‌، له‌گه‌ڵ هه‌ر كه‌سێكی بیانی كه‌قسه‌ ده‌كه‌یت به‌لایه‌وه‌ سه‌یره‌ شوێنێكی ئاوا جوان له‌ دنیادا نه‌ناسراوه‌ و ریكلامی بۆ ناكرێت، من لێره‌وه‌ پێشنیاری ئه‌وه‌ش ده‌كه‌م به‌ لایه‌نه‌ به‌رپرسه‌كانی حكومه‌تی هه‌رێم به‌ چاوێكی وردتره‌وه‌ سه‌یری ئه‌م بواره‌ بكه‌نو ده‌ست گیرۆی هونه‌رمه‌ندان و كۆمپانیاكان بكه‌ن كه‌ له‌ خه‌می ناساندنی كوردستان و سروشته‌ دڵفرێنه‌كه‌ین و ده‌توانرێت كاری هاوبه‌ش بكرێت كه‌ شێوه‌ی ریكلام كردن وه‌ربگرێت و هه‌موان بتوانین له‌ گه‌ڵ به‌ره‌و پێشبردنی هونه‌ری درامای كوردی وڵات و نه‌ته‌وه‌كه‌شمان بناسێنین .
به‌بۆچونی من له‌ كوێ بۆشایی هه‌بێت له‌وێ هاتنه‌ناوه‌ دروست ده‌بێت، ئه‌گه‌ر زیاترو رونتر قسه‌كه‌م له‌وكاته‌ی درامای كوردی نیه‌ یان زۆر به‌شه‌رمن ده‌رده‌كه‌وێت بێگومان رێخۆشكه‌ره‌ بۆ هاتنه‌ناوه‌وه‌ی دراماو هونه‌ری بێگانه‌ ئه‌مه‌ له‌لایه‌ك، پاشان من بۆ خۆم سه‌رپه‌رشتیاری دۆبلاژی (خاوێ فیلم)م كه‌ ده‌توانم بڵێم بێركابه‌ره‌ له‌ شیوازی دۆبلاژ كردن چونكه‌ له‌كاركردنماندا پشتمان به‌ستوه‌ به‌ نواندنی ده‌نگی كه‌ به‌داخه‌وه‌ له‌ زۆر به‌ی كۆمپانیاكانی تر ئه‌وه‌ ره‌چاو ناكرێت و هه‌ندێكیشیان به‌ هونه‌ری دانانێن، ئه‌وكارانه‌ زوو ده‌چنه‌ ناو خه‌ڵك چونكه‌ هه‌موو خاڵه‌كانی درامای سه‌ركه‌وتوی تیایه‌ له‌ چیرۆك و نواندن و ده‌رهێنان و لۆكه‌یشن و ته‌واوی به‌شه‌كانی تر ئه‌وه‌ی كه‌ نیه‌ زمانی كوردیه‌ كاتێك ئه‌م كارانه‌ دۆبلاژ ده‌كرێت زۆر به‌زویی ده‌چێته‌ ناو خه‌ڵك و بینه‌رێكی بێشوماری ده‌بێت، ئه‌مه‌ جگه‌ له‌ نزیكی كه‌لتوری ئه‌و دراما كۆری و توركیانه‌ له‌ كۆمه‌ڵگای كوردیه‌وه‌.
له‌سه‌ر هه‌ڵگری دیمه‌ن و خه‌تی توندوتیژی له‌ درامه‌كاندا راسته‌ به‌داخه‌وه‌ تائێسته‌ زۆر به‌ی كه‌ناڵه‌كانی راگه‌یاندن ره‌چاوی كۆمه‌ڵێك حاڵه‌ت ناكه‌ن كه‌ ئه‌ركی خۆیانه‌، بۆ نموونه‌ ئه‌وه‌ ئه‌ركی كۆمپانیاكانی دۆبلاژ نیه‌ كه‌ی و چۆن دراماكه‌ داده‌به‌زێت و كه‌ی دوباره‌ ده‌كرێته‌وه‌، ده‌بێت ئه‌ودرامانه‌ی كه‌ هه‌ڵگری دیمه‌نی توندو تیژین به‌ پێی ریزبه‌ندی ته‌مه‌نه‌كان كه‌ناڵه‌كه‌ په‌خشی بكات هیچ نه‌بێت بڵێت ئه‌م درامایه‌ بۆ ته‌مه‌نی ئه‌وه‌نده‌ یان ئه‌وه‌نده‌ ده‌بێت به‌مه‌ش ئه‌ركی خۆی جێبه‌جێ ده‌كات پاشان ده‌بێت ئه‌وكه‌ناڵانه‌ كه‌ دووباره‌ی ده‌كه‌نه‌وه‌ له‌ رۆژدا نه‌بێت به‌وپێیه‌ی مناڵان ئاسانتر ده‌توانن سه‌یری بكه‌ن وئه‌و توندوتیژیه‌ كاردانه‌وه‌ی ده‌بیت له‌ سه‌ریان .
زۆرجار گله‌یی له‌ ئه‌كته‌ری كورد ده‌كرێت كه‌ وه‌ك ئه‌كته‌ری دراما رۆڵناگێڕن واته‌ له‌ درامادا ده‌ڵێی شانۆ ده‌كه‌ن ئێوه‌ ده‌ڵێن چی؟ 
راسته‌ كه‌م تا زۆر ئه‌و ره‌خنانه‌ هه‌یه‌ و ده‌شبینرێت، من ده‌ی گه‌رێنمه‌وه‌ بۆ دوو خاڵ یه‌كه‌م زۆر به‌ی هه‌ره‌ زۆری ئه‌كته‌ری دراماكان باگراوه‌ندی شانۆییان هه‌یه‌و بته‌وێت و نه‌ته‌وێت ته‌ئسیراتی خۆی ده‌بێت.
دووهام خاڵ به‌داخه‌وه‌ تائێستا ناتوانی له‌دوای ئه‌وه‌نده‌ ساڵ له‌ زانكۆیه‌ك یان په‌یمانگایه‌ك بخوێنیت كه‌ ئه‌كته‌ری دراما دروست كات و به‌ شێوه‌یه‌كی ئه‌كادیمی بخوینرێت، له‌گه‌ڵ ئه‌م راستیانه‌دا ده‌توانم بڵێم تاراده‌یه‌ك ئێسته‌ باشتره‌و ئه‌گه‌ر ده‌رهێنه‌ره‌كه‌ پرۆفیشناڵانه‌ كاربكات رێ به‌وه‌ نادات هه‌ر له‌ سۆنگه‌ی ئه‌وه‌وه‌یه‌ كه‌ ده‌ڵێم كه‌سی گونجاو له‌ شوێنی گونجاوابێت نه‌ك له‌ به‌ر ئه‌وه‌ی پاره‌ت هه‌یه‌ یان كه‌سێكی ناسراویت كاری ده‌رهێنان بكه‌یت .
زۆركه‌س پێی وایه‌ كه‌ درامای چپه‌ی رابردوو ئه‌وه‌نده‌ی ره‌نگدانه‌وه‌ی دراما توركیه‌كانه‌ نیو هێنده‌ ره‌نگدانه‌وه‌ی كۆمه‌ڵگای كوردی نیه‌ له‌و باره‌یه‌وه‌ بۆچوونی ئێوه‌ چۆنه‌؟ 
من هه‌رگیز له‌گه‌ڵ ئه‌و بۆچونه‌دا نیم درامای چپه‌ی رابردوو كه‌ من خۆم زۆر له‌ نزیكه‌وه‌ ئاگاداری هه‌نگاو به‌ هه‌نگاوی ئه‌وكاره‌م ده‌توانم بڵێم سه‌ره‌تای ئه‌وهانگاوه‌ راسته‌یه‌ كه‌ ده‌بێت هونه‌ری دراما روی تێ بكات ئه‌م قسه‌یه‌ بۆ ؟؟ به‌بۆچونی من له‌م درامایه‌دا ( چیرۆك و ده‌رهێنان و روناكی و كاری ده‌نگ و مۆنتاژ ) به‌خه‌ڵكانێك كراوه‌ كه‌ خاوه‌نكاری پێشینه‌و ئه‌كادیمین له‌ بواری خۆیاندا و هه‌ربۆیه‌ بینه‌ر له‌گه‌ڵ ساته‌كانی دراماكه‌ ده‌ژی و هه‌ست به‌ چنینی روداوو هه‌نگاوه‌كان ده‌كات، خاڵێكی تری ئه‌م كاره‌ هێنانه‌ ناوه‌وه‌ی كۆمه‌ڵێك كۆمپانیای خۆماڵی گه‌وره‌یه‌ كه‌ له‌ ناو دراماكا ریكلام بۆ پرۆژه‌كانی بكرێت و له‌ به‌رامبه‌ردا تۆی خاوه‌ن كۆمپانیای به‌رهه‌م هێن ده‌توانی ده‌ست كراوه‌ تر بیت له‌ هێنانی خیبره‌ی به‌شه‌ر و ئه‌ده‌واتی پێشكه‌وتوو بۆ ناو كاره‌كه‌ت و ده‌توانی له‌و دروشمه‌ كۆنه‌ نوێیه‌ رزگارمان بێت كه‌ ده‌ڵێت تاكه‌ كێشه‌ی هونه‌رمه‌ند ئه‌وه‌یه‌ كه‌ هونه‌ره‌كه‌ی ناتوانێت بیژێنێت بۆیه‌ په‌نا ده‌بات بۆ شتی تر، پاشان من پێم وایه‌ شتێكی نائاساییه‌ و نا ویژدانیه‌ كارێك ره‌تكه‌یته‌وه‌ یان وه‌لای نێیت له‌ به‌ر ئه‌وه‌ی سه‌ركه‌وتنه‌كه‌ی له‌ فڵان درامای توركی یان فڵان درامای كۆری یان عه‌ره‌بی ده‌چێت، بێگومان من ناڵێم ئه‌م كاره‌ بێ كه‌م و كورتی و هه‌ڵه‌یه‌ تۆ كاتێك كه‌ ئێشده‌كایت ته‌جروبه‌ ده‌كه‌یت و له‌وه‌شا بێگومه‌ن هه‌ڵه‌ ده‌كه‌یت به‌ڵام ده‌كرێت ئه‌مه‌ ببێته‌ هه‌وێنی ئیشێكی باشتر و باشتر له‌ داهاتودا چونكه‌ من پێم وایه‌ خه‌ڵكه‌كه‌مان له‌وه‌ باشتر ده‌هێنن و كه‌ی كاری جوانیان بینی بینه‌رو بیسه‌ری ده‌بن.
هه‌ربۆیه‌ ده‌توانم بڵێم ده‌یانی وه‌ك كارئه‌كته‌ری ڤیان و هێڤی و ئازاد و ئامانج و كاوه‌ و كلارا له‌ كۆمه‌ڵگاكه‌ماندا بونی هه‌یه‌و  رۆژانه‌ به‌رمان ده‌كه‌ون.
