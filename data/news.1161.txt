نزیك به‌دوو ساڵ ده‌بێت له‌به‌ره‌كانی‌ پێشه‌وه‌دا شه‌ڕی‌ دژ به‌داعش ده‌كه‌ن، به‌ڵام تائێستا موچه‌یان نیه‌‌و به‌پێی‌ پێویست پاڵپشتی‌ داراییش ناكرێن، ئه‌وان وه‌كو فه‌وجێكی‌ سه‌ربه‌خۆی‌ هێزی‌ پێشمه‌رگه‌ی‌ كوردستان خۆیان رێكخستووه‌ له‌ژێر ناوی‌؛ هێزی‌ دژه‌تیرۆری شه‌هید ئه‌كره‌م كوڤلی.
ده‌ركه‌وتنی‌ رۆڵی‌ زیاتر ئه‌و هێزه‌، مانگێك له‌مه‌وبه‌ر بوو كاتێك داعش له‌ 3ی‌ (ئایار/5)ی‌ ئه‌مساڵدا هێرشی‌ به‌رفراوانی‌ كرده‌ سه‌ر تلسقۆف، ئه‌وكات هێزی‌ دژه‌تیرۆری‌ شه‌هید ئه‌كره‌م به‌هاوكاری‌ هێزێكی‌ تایبه‌تی‌ ئه‌مریكی‌ به‌توندی‌ روبه‌روی‌ گروپه‌كه‌ بوونه‌وه‌.
له‌و شه‌ڕه‌دا به‌رونی‌ هاوكاری‌ راسته‌وخۆی‌ هێزی‌ زه‌مینی‌‌و راوێژكارانی‌ ئه‌مریكا بۆ پێشمه‌رگه‌ ده‌ركه‌وت، كاتێك پێكه‌وه‌ شه‌ڕی‌ داعشیان ده‌كرد، هه‌ر له‌و پێكدادانه‌دا، سه‌ربازێكی‌ ئه‌مریكا كوژرا.
لیوا وه‌حید كوڤلی، ئێستا سه‌رپه‌رشتی ئه‌و هێزه‌ ده‌كات‌و بۆ خه‌ندان باسی‌ له‌وه‌كرد، كه‌ ئه‌وكاته‌ی‌ داعش شه‌نگال‌و زوممارو كه‌سكێی‌ گرتووه‌ له‌ساڵی‌ 2014دا، ئه‌وان وه‌كو خزم‌و براو هاوڕێ نزیكه‌ی 50 كه‌س بون كه‌ له‌كه‌سكێ‌ به‌رگرییان كرد.
وتی‌ "ئه‌و رۆژه‌، نزیكه‌ی 40 هاوڵاتییان رزگاركرد‌و له‌ناوچه‌ی زومماریش ئه‌و پێشمه‌رگانه‌مان رزگاركرد كه‌ گه‌مارۆدرابون، له‌و شه‌ڕه‌دا یه‌ك شه‌هیدمان هه‌بوو".
كۆبونه‌وه‌ی‌ ئه‌و 50 چه‌كداره‌، وه‌كو خۆیان ده‌ڵێن، دوابه‌دوای‌ په‌یامێكی‌ مه‌سعود بارزانی‌ بووه‌، كه‌ داوای‌ له‌خه‌ڵكی كوردستان كردووه‌ هاوكاری پێشمه‌رگه‌ بكه‌ن‌و بچن بۆ به‌ره‌كانی جه‌نگ.
له‌دوای‌ ده‌ستپێكردنه‌وه‌ی‌ هێرشه‌كانی‌ داعشه‌وه‌ بۆ سه‌ر ناوچه‌ كوردستانیه‌كان، سه‌دان هاوڵاتی‌ خۆبه‌خش چه‌كیان هه‌ڵگرتووه‌‌و له‌زۆربه‌ی‌ میحوه‌ره‌كاندا، هاوشانی‌ پێشمه‌رگه‌ شه‌ڕی‌ دژ به‌داعش ده‌كه‌ن.
مولازم سه‌عد ئه‌حمه‌د، له‌هێزی دژه‌تیرۆری وه‌حید كۆڤلیه‌‌و بۆ خه‌ندان ئاماژه‌ی‌ به‌وه‌دا "له‌و رۆژه‌وه‌ی هێزه‌كان دروستبووه‌، به‌و چه‌كه‌ سه‌ره‌تاییانه‌ی پێمانه‌ نه‌مانهێشتوه‌ یه‌ك بست خاكی‌ كوردستان بگیرێت".
هێزی‌ دژه‌تیرۆری شه‌هید ئه‌كره‌م كوڤلی، دوای‌ به‌شدارییان له‌چه‌ند شه‌ڕێكدا‌و بڵاوبونه‌وه‌ی‌ هه‌واڵی‌ به‌رگرییه‌كانیان، ژماره‌یان له‌ 50 كه‌سه‌وه‌ بۆ نزیكه‌ی‌ 450 چه‌كدار زیادده‌كات.
"هێزه‌كه‌مان گه‌یشته‌ نزیكه‌ی‌ 500 كه‌س، به‌ڵام دواتر ژماره‌یه‌م كه‌مبووه‌وه‌، ئه‌وه‌ش به‌هۆی‌ ئه‌وه‌ی‌ بژێوی ژیانیان سه‌خت بوه‌ یان نه‌یانتوانیوه‌ ماڵ‌و منداڵ به‌خێو بكه‌ن، ئه‌گینا ئێستا به‌س موچه‌یه‌كی كه‌میش هه‌بێت به‌هه‌زاران كه‌س ئاماده‌یه‌ ببێته‌ پێشمه‌رگه‌ له‌و هێزه‌ی ئێمه‌، هه‌رچه‌نده‌ ئامانجی ئێمه‌ پاره‌ نیه‌ به‌ڵكو پاراستنی خاكی‌ كوردستانه‌".
مولازم سه‌عد وایوت.
یه‌كێك له‌نیگه‌رانیه‌كانی‌ ئه‌و هێزه‌ ئه‌وه‌یه‌، له‌كاتێكدا نزیك به‌دوو ساڵ ده‌بێت له‌زۆربه‌ی‌ به‌ره‌كانی‌ پێشه‌وه‌دا شه‌ڕی‌ دژ به‌داعشیان كردووه‌‌و ده‌ستكه‌وتیان به‌ده‌ستهێناوه‌، به‌ڵام هیچ هاوكارییه‌كی‌ دارایی‌ یان مووچه‌یه‌كیان نییه‌.
فه‌رمانده‌ی‌ هێزی‌ دژه‌تیرۆری شه‌هید ئه‌كره‌م كوڤلی ده‌ڵێت، ئه‌وان به‌بێ‌ هیچ موچه‌و پاڵپشتیه‌كی‌ دارایی شه‌ڕ ده‌كه‌ن، ته‌نیا ئه‌وه‌نه‌بێت "جاربه‌جار له‌لایه‌ن مه‌كته‌بی سه‌رۆك‌و ئاژانسی پاراستنه‌وه‌ پاداشت كراوین".
لیوا وه‌حید كوڤلی باسی‌ له‌وه‌كرد كه‌ دوای‌ ئه‌وه‌ی‌ هێزه‌كه‌یان گه‌یشتووه‌ته‌ 450 كه‌س، دواتر ژماره‌یه‌ك كه‌مبووه‌ته‌وه‌، ئه‌وه‌ش به‌هۆی فشاری دارایی‌و قه‌یرانی نه‌بونی موچه‌ چونكه‌ ده‌بوو ئه‌و كه‌سانه‌ كاربكه‌ن بۆ بژێوی خێزانیان، ئه‌گه‌رنا كه‌س ئاماده‌نه‌بوو وازبهێنێت.
هێزه‌كه‌ ئێستا 220 كه‌سی‌ ماوه‌ته‌وه‌، له‌خه‌ڵكی هه‌ولێر‌و شه‌نگال و دهۆك و سۆران‌و زوممار له‌گه‌ڵ چه‌ند ناوچه‌یه‌كی‌ تر پێكهاتوون، هه‌مویان وه‌كو خۆبه‌خش شه‌ڕ ده‌كه‌ن.
ره‌حت عیدۆ، ته‌مه‌نی 28 ساڵه‌‌و له‌مانگی‌ 2ی‌ ساڵی‌ رابردوو په‌یوه‌ندی‌ به‌هێزی‌ دژه‌تیرۆری‌ ئه‌كره‌م كۆڤلیه‌وه‌ كردووه‌، ئه‌و به‌خه‌ندانی‌ وت "هیچ كاتێك ئاماده‌نیم ده‌ستبه‌رداری مام واحید (وه‌حید كوڤلی‌)بم، هه‌رچه‌نده‌ ئێمه‌ موچه‌یه‌كی ره‌سمیمان نیه‌ به‌ڵام مام واحید واینه‌كردوه‌ زۆر پێمانه‌وه‌ دیاربێت، چونكه‌ جگه‌ له‌و پاداشتانه‌ ئێمه‌ بۆیه‌ك هه‌فته‌ 10 هه‌زار دینارمان پێده‌دات، ئه‌وه‌ش له‌گیرفانی خۆیه‌تی".
ئه‌و پێشمه‌رگه‌یه‌ داوا له‌وه‌زاره‌تی پێشمه‌رگه‌ ده‌كات موچه‌یان بۆ دابینبكات، چونكه‌ جگه‌ له‌وه‌ی‌ به‌شێكیان له‌خانووی‌ كرێدان، زۆرینه‌یان خاوه‌ن ماڵ‌و منداڵن.
شه‌ڕی‌ داعش، هاوكات بوو له‌گه‌ڵ ئه‌و قه‌یرانه‌ داراییه‌ی‌ له‌سه‌ره‌تای‌ ساڵی‌ 2014وه‌ هه‌رێمی‌ كورستانی‌ گرته‌وه‌، به‌هۆیه‌وه‌ موچه‌ی‌ فه‌رمانبه‌ران‌و هێزه‌كانی‌ پێشمه‌رگه‌، له‌واده‌ی‌ خۆیدا دابه‌شناكرێت.
فه‌رمانده‌ی‌ هێزی‌ دژه‌تیرۆری شه‌هید ئه‌كره‌م كوڤلی وتی‌ "ماوه‌ی دووساڵه‌ وه‌زاره‌تی‌ پێشمه‌رگه‌ به‌ڵێنی‌ داوه‌ موچه‌ بۆ هێزه‌كه‌ ببڕنه‌وه‌، به‌ڵام ئه‌وه‌ نه‌كراوه‌‌و راست ناكه‌ن، له‌ كاتێكدا ژن و كوڕی ئه‌و به‌رپرسانه‌ ئێستا به‌سه‌یاره‌ی ئاخر مۆدیڵ ده‌سورێنه‌وه‌و سه‌فه‌ری وڵاتان ده‌كه‌ن به‌ڵام موچه‌یه‌ك به‌قابیلی‌ ئێمه‌ نازانن".
وه‌حید كوڤلی‌ ده‌ڵێت "من بۆ پاره‌ ئه‌وه‌ ناكه‌م، به‌ڵام پێشمه‌رگه‌كانم پێویستیان پێیه‌تی و منداڵیان هه‌یه‌و كرێچین، به‌خوای گه‌وره‌ من خانوم فرۆشتوه‌‌و زێری خێزانم‌و سه‌یاره‌ی برا شه‌هیده‌كه‌م فرۆشتوه‌ به‌س بۆ ئه‌وه‌ی پێشمه‌رگه‌كانم برسی نه‌بن‌و چاویان له‌ده‌ستی خه‌ڵكی نه‌بێت".
ئه‌و سوپاسی‌ مه‌سعود بارزانی‌‌و مه‌سرور بارزانی‌ ده‌كات، وه‌كو خۆی‌ باسیكرد، ته‌نیا ئه‌وان هاوكارییان كردون له‌رووی‌ داراییه‌وه‌.
ئه‌ندامی‌ هێزی‌ دژه‌تیرۆری‌ شه‌هید ئه‌كره‌م كوڤلی‌، ته‌مه‌نیان له‌نێوان 20 بۆ 50 ساڵدایه‌، تائێستا وه‌كو هێزێكی ره‌سمی نه‌ناسێنراون، بۆیه‌ به‌هێزێكی‌ سه‌ربه‌خۆ خۆیان ناوده‌به‌ن.
ئه‌و هێزه‌ فه‌رمان‌و بڕیار له‌ مه‌سعود بارزانی و مه‌سرور بارزانی وه‌رده‌گرن، بڕیاریشه‌ له‌داهاتوودا ببنه‌ هێزێكی دژه‌ تیرۆر سه‌ر به‌ ئه‌نجومه‌نی ئاسایشی هه‌رێمی‌ كوردستان، بۆ ئه‌وه‌ش زانیارییان لێوه‌رگیراوه‌.
به‌وته‌ی‌ وه‌حید كوڤلی‌.
له‌شه‌ڕه‌كه‌ی‌ سه‌ره‌تای‌ مانگی‌ رابردووی‌ هێزه‌كانی‌ پێشمه‌رگه‌ی‌ كوردستان له‌گه‌ڵ داعش له‌میحوه‌ری‌ تلسقۆف، هێزی‌ دژه‌تیرۆری‌ شه‌هید ئه‌كره‌م رۆڵی‌ به‌رچاویان هه‌بوو له‌روبه‌ڕبونه‌وه‌كاندا، ئه‌وه‌ش به‌پێی‌ ئه‌و ئه‌و گرته‌ ڤیدیۆییانه‌ی‌ دوای‌ پێكدادانه‌كان‌و هه‌واڵی‌ به‌شێك له‌ئاژانسه‌كان له‌باره‌ی‌ ئه‌و هێزه‌وه‌ بلاوبونه‌وه‌.
وه‌حید كوڤلی‌ باسی‌ شه‌ڕه‌كه‌ی‌ تلسقۆف ده‌كات‌و ده‌ڵێت كاتێك داعش هێرشیكردوه‌ ئه‌و له‌لوبنان بووه‌ بۆ چاره‌سه‌ری‌ نه‌خۆشی‌، له‌گه‌ڵ گه‌یشتنی‌ هه‌واڵی‌ شه‌ڕه‌كه‌ گه‌راوه‌ته‌وه‌ بۆ كوردستان‌و كاتژمێر 4ی‌ ئێواره‌ گه‌یشتووه‌ته‌وه‌ هێزه‌كه‌ی‌.
ئه‌و وتی‌ "هێزه‌كه‌م هه‌شت كاتژمێر به‌رگرییان كردبوو، شه‌هیدێك و سێ‌ بریندارمان هه‌بوو به‌ڵام ئه‌و به‌رگریه‌ بوو وایكرد هێزه‌كه‌ی ئێمه‌ وه‌كو شێر ناو ده‌ربكات، جه‌نانی كاك مه‌سرور و كاك مه‌نسور‌و جه‌نابی سه‌رۆك سوپاسی ئه‌و نه‌به‌ردیه‌ی ئێمه‌یان كرد".
به‌شێك له‌و كه‌سانه‌ی‌ ئێستا له‌هێزی‌ دژه‌تیرۆری‌ شه‌هید ئه‌كره‌م-دان، پێشتر له‌چوارچێوه‌ی‌ هێزه‌كانی‌ پێشمه‌رگه‌دا ئه‌ركیان هه‌بووه‌ یان سالانێك خزمه‌تیان كردووه‌‌و شاره‌زاییان په‌یداكردووه‌، له‌ئێستاشدا به‌شێكیان پله‌ی‌ جیاوازیان هه‌یه‌‌و به‌وته‌ی‌ وه‌حید كۆڤلی‌ كاتێك ده‌خرێنه‌ سه‌ر ئه‌نجومه‌نی‌ ئاسایش پێداچونه‌وه‌ به‌ پله‌كانیان ده‌كرێت.
ئه‌و هێزه‌ چه‌ك‌و پێداویستیه‌ سه‌ربازیه‌كانیان، له‌ به‌رپرسی میحوه‌ره‌كان‌و ناوچه‌كانی‌ تری‌ هێزی‌ پێشمه‌رگه‌ وه‌رده‌گرن، به‌تایبه‌ت له‌كاتی‌ شه‌ڕه‌كاندا.
وه‌حید كۆڤلی‌ ده‌ڵێت تائێستا ده‌ستیان به‌سه‌ر سه‌دان پارچه‌ چه‌ك و كه‌ره‌سته‌ی سه‌ربازییدا گرتووه‌ كه‌ به‌ده‌ست داعشه‌وه‌ بوون له‌نێویاندا حه‌وت ئۆتۆمبێل، له‌لایه‌ن هێزه‌كانی‌ پێشمه‌رگه‌شه‌وه‌ شه‌ش ئۆتۆمبێلی‌ تریان پێدراوه‌، هه‌ر به‌و كه‌ره‌سته‌ سه‌ربازییانه‌شه‌و شه‌ڕ ده‌كه‌ن.
له‌دوای‌ دروستبونیه‌وه‌، هێزی‌ دژه‌تیرۆری شه‌هید ئه‌كره‌م كوڤلی به‌شداری‌ ده‌یان شه‌ڕیان كردووه‌ به‌رانبه‌ر داعش، له‌زۆرینه‌یاندا له‌به‌ره‌ی‌ پێشه‌وه‌دا بوون، له‌ماوه‌ی‌ نزیك به‌دوو ساڵی‌ رابردووش هه‌شت شه‌هید‌و 54 برینداریان هه‌بووه‌.
وه‌حید كۆڤلی‌، له‌گه‌ڵ دوو كوڕی‌ خۆیدا له‌به‌ره‌ی‌ شه‌ڕدایه‌‌و سێ‌ برای‌ شه‌هیدی‌ هه‌یه‌، ئێستاش ئاماده‌یه‌ "هه‌ردوو كوڕه‌كه‌م‌و خۆیشم بكه‌مه‌ قوربانی" خاكی‌ كوردستان.
