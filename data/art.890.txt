هه‌ر كه‌ زه‌نگ لێی ده‌دا من و هه‌نار و ناز و به‌سته‌ و لیلۆ ده‌چووینه‌ گۆڕه‌پانی قوتابخانه‌ .
زۆر جار باسی ئه‌و وانه‌یه‌مان ده‌كرد كه‌ خوێندمان و باسی هه‌ڵسو كه‌وته‌ نه‌رم و نیانه‌كانی مامۆستا (به‌ختیار) یشمان ده‌كرد كه‌ چۆن خۆی خۆشه‌ویست كردووه‌ لامان و حه‌ز ده‌كه‌ین هه‌موو ڕۆژێك له‌ جیاتی وانه‌یه‌ك ، دوو وانه‌ یان زیاترمان پێ بڵێت ... له‌ هه‌مان كاتیشدا خۆزگه‌مان ده‌خواست كه‌ گشت مامۆستاكانی تریش وا بوونایه‌...ئێمه‌ له‌گه‌ڵ ئه‌وه‌ی كه‌ به‌رنامه‌یه‌كی تایبه‌تمان دانابوو بۆ كۆششكردن و یاریكردن و دیداری یه‌كتری ، قوتابییه‌كی زیره‌ك بووین له‌ خوێندندا و له‌ خوێندنه‌وه‌یشدا چالاك بووین .
چونكه‌ ده‌مانزانی خوێندنه‌وه‌ پێویستییه‌كی ژیانه‌ بۆ هه‌موو مرۆڤێك به‌ تایبه‌تی بۆ ئێمه‌ی قوتابی .
بۆیه‌ ئاگاداری زۆر له‌و گۆڤار و په‌رتووك و نامیلكانه‌ بووین كه‌ بۆ منداڵان ده‌رده‌چوون..هه‌موومان ده‌مان كڕی و ده‌مانخوێندنه‌وه‌ ، به‌ڵام حه‌زمان ده‌كرد كه‌ هه‌موو هاوڕێكانی تریشمان بیكڕن و بیخوێننه‌وه‌ ، بۆیه‌ كه‌ پێمان ده‌وتن ده‌یان وت نازانین له‌ كوێ‌ بیانكڕین، كه‌ ئه‌مه‌ش لای من بوو به‌ كێشه‌یه‌ك و پێویستی به‌ چاره‌سه‌ر هه‌بوو .
بۆ ماوه‌یه‌ك بیرم له‌مه‌ كرده‌وه‌ ..به‌ تایبه‌تی ئه‌و كاتانه‌ی كه‌ ده‌چووم بۆ فرۆشگای قوتابخانه‌كه‌مان.
ئه‌وه‌ی به‌لامه‌وه‌ سه‌یر بوو ئه‌و پیاوه‌ی كه‌ فرۆشگای قوتابخانه‌كه‌مانی ده‌برد به‌ڕێوه‌ ،هه‌ندێك له‌و شتانه‌ی كه‌ ده‌یهێنان و ده‌یفرۆشتن زیانی به‌ قوتابیان ده‌گه‌یاند له‌ روی ته‌ندروستییه‌وه‌ و له‌ هه‌مان كاتیشدا ده‌بووه‌ هۆی پیسكردنی قوتابخانه‌كه‌ش..چه‌ند شتێكی تریش كه‌ ده‌مزانی به‌ هیچ شێوه‌یه‌ك سوودیان نییه‌.بۆیه‌ رۆژێك لای هاوڕێكانم باسمكرد ئه‌وانیش هه‌مان راو بۆچوونی منیان هه‌بوو ..وتیان چار چییه‌؟ منیش وتم بۆ نه‌چینه‌ لای كاك ئه‌حمه‌د كه‌ خاوه‌نی فرۆشگای قوتابخانه‌كه‌مانه‌ ئه‌مه‌ی لا باس بكه‌ین.لیلۆ به‌ پێكه‌نینه‌وه‌ وتی لیره‌ گیان ئه‌ی بۆ نه‌چینه‌ لای به‌ڕێوه‌به‌ر و لای ئه‌و باسی بكه‌ین.
به‌سته‌ وتی جا ئێمه‌ وا باسیشمان كرد لایان ئه‌ی له‌ جێیی ئه‌وانه‌ چی بهێنێ‌ ؟ !
منیش وتم ماوه‌یه‌كه‌ بیر ده‌كه‌مه‌وه‌ دامناوه‌ كه‌ چی له‌ جیاتی ئه‌وانه‌ بهێنێ‌ .
هه‌موو پێكه‌وه‌ وتیان ئاده‌ی له‌ جیاتی ئه‌وانه‌ چی بهێنێ‌ ئه‌ی هاوڕێ‌ قسه‌ زله‌كه‌مان .منیش پێم وتن با له‌ جیاتی ئه‌وانه‌ ئه‌و گۆڤاره‌ تایبه‌تانه‌ بهێنێ‌ كه‌ مانگانه‌ بۆ منداڵان ده‌رده‌چن له‌گه‌ڵ ئه‌و نامیلكه‌ و په‌رتوكانه‌ی كه‌ تایبه‌تن به‌ منداڵان..خۆ ئه‌گه‌ر هه‌ر نووسه‌رێكیش نامیلكه‌ و په‌رتوكی ڕێگه‌ پێدراوی هێنا بۆ قوتابخانه‌ ئه‌ها ئه‌وه‌ی كه‌ ساڵانه‌ ده‌یان هێنن و به‌ڕه‌زامه‌ندی به‌ڕێوه‌به‌رێتی په‌روه‌رده‌.
با ئه‌وه‌ هه‌ر له‌ فرۆشگاكه‌دا دایبنێن به‌ نرخێكی گونجاو بیده‌نێ‌ تا ئه‌ویش له‌وێ‌ بیفرۆشێته‌وه‌ به‌ منداڵان .
ئه‌مه‌ هه‌م قوتابخانه‌كه‌ كێشه‌ی بۆ دروست نابێ‌ و هه‌م سیمایه‌كی جوانیش ده‌به‌خشێ‌ به‌ فرۆشگاكه‌ و من دڵنیام هه‌موو قوتابیان به‌ ئاسانی ده‌توانن بیكڕن ..هه‌موومان له‌سه‌ر ئه‌مه‌ ڕێكه‌وتین ، به‌ڵام یه‌كه‌م جار چووینه‌ لای به‌ڕێوه‌به‌ر ، كه‌ بۆمان باسكرد به‌لایه‌وه‌ زۆر باش بوو ، به‌ڵێنی هاوكاری پێداین و وتی منیش مانگانه‌ هه‌موو گۆڤاره‌كان ده‌كڕم و له‌ په‌ڕتووكخانه‌ی قوتابخانه‌دا دایئه‌نێین ،ئیتر له‌ گه‌ڵمان هات بۆ لای كاك ئه‌حمه‌د ، ئه‌وانه‌ی هه‌موو پێوت ، ئه‌ویش ته‌ماشایه‌كی پڕ له‌ خۆشه‌ویستی ئێمه‌ی كرد و لای خۆیه‌وه‌ وتی منیش وه‌كو ئێوه‌ حه‌زم له‌ خوێندنه‌وه‌یه‌ له‌و كاته‌دا كه‌ ئێوه‌ له‌ پۆلدان و وانه‌ ده‌خوێنن منیش لێره‌دا ده‌خوێنمه‌وه‌ ..من ئاماده‌م له‌ به‌یانییه‌وه‌ ئه‌و شتانه‌ی كه‌ ئێوه‌ لاتان په‌سه‌ند نییه‌ لای ببه‌م و جارێكی تر نه‌یهێنم و گۆشه‌یه‌كی بچوكیش ته‌رخان بكه‌م بۆ گۆڤار و نامیلكه‌ی منداڵان تا منیش هاوكار بم له‌ دابین كردنی ئه‌و هۆكارانه‌ی كه‌ بۆ ڕۆشنبیر كردنی ئێوه‌ پێویستن.....ئێمه‌ش زۆر سوپاسی به‌ڕێوه‌به‌رمان كرد و ده‌ستخۆشیمان له‌ كاك ئه‌حمه‌دیش كرد و به‌ خۆشییه‌وه‌ گه‌ڕاینه‌وه‌ پۆل .
ئیتر له‌و ڕۆژه‌وه‌ هه‌موو قوتابییه‌كان چێژێكی زۆر له‌و خواردنه‌ به‌سووده‌ وه‌رده‌گرن كه‌ ئێمه‌ بووینه‌ هۆی په‌یداكردنی ، كه‌ ئه‌ویش خوێندنه‌وه‌ بوو......
