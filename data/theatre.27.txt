که‌زی ده‌چێت بۆ لای ئازادی میوانیان و باوکی  ،  وێنه‌که‌یان نیشان ده‌دات ، ئازاد وێنه‌که‌له‌که‌زی وه‌رده‌گرێت به‌ڵام که‌زی سه‌رنجی ده‌چێته‌سه‌ر ده‌سته‌پێچراوه‌کانی .
ئۆتۆمۆبیلێکی پۆلیس به‌ناو شه‌قامێکدا ده‌ڕوات ،گوێمان له‌ده‌نگی په‌یوه‌ندی کردنی پۆلیسه‌کانی ناو ئۆتۆمبیله‌که‌یه ‌له‌گه‌ڵ ده‌زگای سه‌ره‌کی پۆلیس .
پاسێکی پۆلیس له‌به‌رده‌‌رم ده‌رگای سه‌ره‌کی ئاپارتمانێکی گه‌وره‌دا وه‌ستاوه‌،  پاش  که‌مێك ده‌رگای ئاپارتمانه‌که ‌ده‌کرێته‌وه‌، پۆلیسێك ده‌رده‌که‌وێت به‌دوایدا شێرکۆی باوکی که‌زی چانتایه‌کی به‌ده‌سته‌وه‌یه‌و پاشان دایکی که‌زی که‌ده‌ستی که‌زی گرتوه  ‌و دوو پۆلیسی تر به‌ده‌رده‌که‌ون ، پۆلیسه‌کان سواری پاسه‌پۆلیسیه‌که‌یان ده‌که‌ن ، دوو پۆلیس له‌دواوه‌له‌گه‌ڵیان سوار ده‌بن، پۆلیسی سێهه‌میش له‌پێشه‌وه‌له ‌ته‌نیشت پۆلیسی شۆفێرداده‌نیشێت  ، پاسه‌که‌ده‌که‌وێته‌جووڵه‌.

8.... کۆمه‌ڵێك خانوو ... ده‌ره‌وه‌یه‌... رۆژه‌.
کۆمه‌ڵێك خانوو ده‌بینرێ چوارده‌وریان به‌دیوارو سیم گیراوه‌، واته‌جێگه‌یه‌کی ده‌ستبه‌سه‌ربوونه‌،‌منداڵ وگه‌وره‌و بچووك که‌به‌گشتی قژیان ره‌شه  ‌دێن وده‌چن ، له‌م کاته‌دا لیزای قژ زه‌رد ده‌رده‌که‌وێت ده‌فته‌رێکی گه‌وره‌ی به‌ده‌سته‌وه‌یه ‌و جانتایه‌كی له‌شاندایه‌، دیاره‌بۆ ناونیشان و ژماره‌ی خانوویه‌ك ده‌گه‌ڕێت .
جێگه‌ی دانیشتن ناو کابینه‌که‌یه‌، لیزا و که‌زی و دایك وباوکی  دانیشتوون خه‌ریکی گفتوگۆن ، لیزاو که‌زی به‌یه‌که‌وه‌به‌یه‌ك ده‌نگ په‌ڕه‌کاغه‌زی ده‌فته‌رێکی وێنه‌کردن ده‌بژێرن که‌لیزا بۆی هێناوه‌و به‌زمانی هۆڵندیه‌.
لیزا و که‌زی /    بیست و پێنچ ، بیست و شه‌ش بیست و حه‌وت ، بیست و هه‌شت ، بیست و نۆ سی .
شێرکۆ /   به‌وهیوایه‌ی لیزا ئه‌و هه‌واڵه‌ت جێبه‌جێکه‌ن و  به‌زووترین کات ده‌رچین له‌م که‌مپه‌و بتوانین له‌م ووڵاته‌دا بمێنینه‌وه‌.
به‌یان /   به‌ڕاستی ئێمه ‌سوپاسی تۆ و فه‌رمانده‌ی  په‌نابه‌ریی   ده‌که‌ین که‌ئاوا به‌خۆبه‌خشی هه‌وڵمان بۆ ده‌ده‌ن .
لیزا /      تۆ ڕاست ده‌که‌یت که‌زی له‌بیرم چوو له‌ده‌فته‌ری  رۆژمێره‌که‌م  بنووسم که‌ی و کام ڕۆژ بگه‌ڕێمه‌وه‌  بۆ لاتان
کابینه‌که‌ی ماڵی که‌زییه‌، کامێرا له‌ده‌ست وپه‌نچه‌ی لیزاوه ‌له‌دیمه‌نی پێشووه‌وه‌ده‌چێته‌سه‌ر ده‌ست و په‌نچه‌ی که‌زی له‌دیمه‌نی تازه‌دا .
ده‌ست وپه‌نچه‌ی که‌زی نیشان ده‌درێ خه‌ریکی وێنه‌کێشانه‌‌چه‌ند جارێك گرژی ده‌مووچاوی که‌زی نیشانده‌درێ به‌پێی دروست کردنی وێنه‌که‌ی .
کۆمه‌ڵێ کابینه‌ن، جێگه‌ی  که‌مپی په‌نابه‌ره‌کانه‌، که‌زی ده‌بینرێ له‌گه‌ڵ باوك ودایکی له‌ده‌ره‌وه‌پیاسه‌ده‌که‌ن تائه‌وکاته‌ی که‌زی به‌ره‌و لای ته‌لبه‌نی کابینه‌کان راده‌کات سه‌یری ده‌ره‌وه‌ده‌کات، دیمه‌نێکی جوانی که‌ناڵێکی ئاو ده‌بینێ کۆمه‌ڵێ قازو مراوی مه‌له‌ی تێدا ده‌که‌ن .
ده‌باشه ئه‌چمه‌وه ‌ئه‌مشه‌و هه‌موو ده‌فته‌ره‌که‌ی  لیزا ده‌که‌م به‌وێنه‌و ئیتر ته‌واو ده‌بم و به‌یانی ده‌چین بۆ ئه‌وێ .
باوك و دایکی که‌زی دانیشتوون سه‌یری ته‌له‌فیزۆن  ده‌که‌ن، له‌م کاته‌دا که‌زی به‌خۆی و ده‌فته‌ری وێنه‌که‌ی خۆی ده‌کات به‌ژووردا.
ده‌ره‌وه‌یه‌،لای دیواری به‌سیم کراوی چوارده‌وری کابینه‌کانه‌،( میشێله‌)  له‌به‌رده‌م سیمه‌که‌دا وه‌ستاوه‌و سه‌یری دیمه‌نی جوانی ده‌ره‌وه‌ده‌کات که‌پێکهاتووه‌له‌کۆمه‌ڵێ دارو که‌ناڵێکی ئاو ، هه‌مان جێگه‌یه‌که‌که‌زی وباوك ودایکی پێشتر بینیبوویان .
که‌زی به‌ره‌و لای میشێله ‌دێت و دێته‌کادره‌وه‌، وورده‌وورده‌نزیك ده‌بێته‌وه ‌له‌دیواری   سیمه‌که‌به‌شه‌رمێکه‌وه‌سه‌یری میشێله‌ده‌کات و پاشان سه‌یری ده‌ره‌وه ‌ده‌کات .
نا بێت  ئه‌بێت په‌ڕه‌کانی ده‌فته‌ره‌که‌رۆژی  یه‌ك وێنه‌ی تێدا بکێشم که‌ته‌واو بوو ئه‌توانم بڕۆم بۆلای ئه‌و قازانه‌.
ده‌فته‌ری چی ؟ من به ‌باوك و دایکیشم ووت ،    ووتیان نابێت .... ناهێڵن .
که‌زی دێته‌نزیکییه‌وه‌وه‌ك  ئه‌وه‌ی بیه‌وێت تێیبگه‌یه‌نێ ده‌ست ده‌کات به‌قسه‌کردن .
ژووره‌که‌ی لیزایه‌له‌فه‌رمانگه‌که‌، لیزا خه‌ریکی خوێندنه‌وه‌یه ‌له‌سه‌ر کۆمپیته‌ر.
لیزا به‌رده‌وامه‌له‌خوێندنه‌وه ‌تا ئه‌و کاته‌ی خه‌به‌رێکی ناخۆش ده‌خوێنێته‌وه ، ‌زۆر بێتاقه‌ت ده‌بێت و واز له‌خوێندنه‌وه‌دێنێ  هه‌ناسه‌یه‌کی قووڵ ده‌داته‌وه و له‌به‌رخۆیه‌وه‌ده‌ڵێت .
ژورێکی به‌ڕێوه‌بردنی په‌نابه‌ره‌کانه‌، ‌که‌سێکی به‌رپرسی هۆڵندی ته‌مه‌ن (47)ساڵ  که‌ناوی (یانه‌) له‌سه‌رکووسیه‌ك له‌پشتی مێزێك دانیشتووه ‌، باوك و دایکی که‌زی له‌پێشی مێزه‌که‌دا چاوه‌ڕێن  ، کۆمه‌ڵێك کاغه‌زیش له‌سه‌ر مێزه‌که‌یه‌،به‌ته‌له‌فۆن  وه‌رگێڕێكیان له‌گه‌ڵدایه‌که‌زمانی هۆله‌ندیان بۆ وه‌رده‌گێڕێته‌سه‌ر  زمانی کوردی.
به‌ڵێ ئه‌مه ‌دوا هه‌مین جارتان بوو بۆ     داوای په‌نابه‌ری و کێسه‌که‌تان به‌ته‌واوی داخراوه‌له‌لا یه‌ن دادگاوه‌و وه‌زاره‌تیش   رازیبوونی خۆیان له‌سه‌ر ده‌ربڕیوه‌.
یان /     هه‌ربۆیه ‌ده‌بێت خۆتان ئاماده‌که‌ن له‌دوو    رۆژی داهاتوودا له‌گه‌ڵ چه‌ند که‌سێکی تردا  وه‌ك ئێوه‌کێسیان داخراوه‌ده‌گه‌ڕێنرێنه‌وه‌   بۆ ووڵاتی خۆتان .
خۆتان ئاماده‌که‌ن  له‌دوو رۆژی داهاتوودا  له‌گه‌ڵ چه‌ند که‌سێکی وه‌ك خۆتان ده‌گه‌ڕێنرێنه‌وه  بۆ وڵاتی خۆتان .
باوک ودایکی که‌زی به‌ته‌واوی ڕه‌نگ و ڕوویان ده‌گۆڕرێت و زۆر بێتاقه‌ده‌بن تاده‌گاته‌ئه‌وه‌ی  شێرکۆ له‌نیوه‌ی رسته‌کانی ده‌نگی وه‌رگێڕ له‌ته‌له‌فۆنه‌که‌وه‌گوێبیستی ده‌نگێکی  خاوکراوه‌ی زۆر قه‌به‌ده‌بێ ولێی تێناگا و ره‌نگی ده‌گۆڕرێ وسه‌ری ده‌خاته‌نێو هه‌ردوو ده‌ستی و له‌وه‌ده‌چێ ببوورێته‌وه‌.
لای سیمڕێژ کراوی چوارده‌وری کابینه‌کانه‌،که‌زی و میشێله ‌له‌هه‌مان جێگه‌ی رۆژانی پێشوو وه‌ستاون و که‌زی ده‌فته‌ری وێنه‌که‌ی پێیه‌و  خه‌ریکی وێنه‌کێشانه‌، میشێله‌ش سه‌یری ده‌ست وپه‌نچه‌ی که‌زی ده‌کات که‌چۆن وێنه ده‌کێشێ .
ئه‌م وێنه‌یه‌ته‌واوکه‌م  ئممممم.. ئه‌و سێ په‌ڕه‌   سپیه‌م ده‌مێنێت .. ئنجا سێ ڕۆژی تر لیزا   دێته‌وه‌‌بۆ ئێره‌.. دوای ده‌چینه ‌ده‌ره‌وه‌و ..       ئه‌چین بۆ لای قازه‌سپیه‌کان .
به‌رده‌م کابینه‌که‌ی ماڵی که‌زیه‌، که‌زی وباوکی سه‌رگه‌رمی وێنه‌کانی که‌زین یه‌ك یه‌ك له‌ده‌فته‌ری وێنه‌که‌ی ده‌که‌نه‌وه‌و به‌دیوارو په‌نچه‌ره‌کانی کابینه‌که‌دا هه‌ڵیده‌واسن، جار جاره‌ش ئه‌وانه‌ی به‌وێدا تێده‌په‌ڕن  ده‌وه‌ستن و سه‌یری وێنه‌کان ده‌که‌ن  له‌نێویاندا میشێله‌ش به‌ده‌رده‌که‌وێ.
میشێله‌دێت بۆ لای که‌زی و پاش که‌مێك به‌رده‌م کابینه‌که‌قه‌ڕه‌باڵغیه‌كی پێوه‌ده‌بینرێ خه‌ڵکه‌که‌خه‌ریکی سه‌یرکردنی وێنه‌کانن .
له‌م کاته‌دا ئۆتۆمۆبیلێکی پۆلیس دێته‌نێو کادره‌وه‌.
ئۆتۆمۆبیله‌که‌ی لیزا ده‌بینین ده‌که‌وێته‌نێو قه‌ره‌باڵغیه‌کی ئۆتۆمۆبیله‌وه‌له‌ڕێگای خێرا، واته‌[ئۆتۆبان]، لیزا سه‌یرێکی سه‌عاته‌که‌ی ده‌ستی ده‌کات وده‌ستێك به‌تووڕه‌یی ده‌دات به‌سوکانی ئۆتۆمۆبیله‌که یدا.

به‌رده‌م کابینه‌کانه ‌کۆمه‌ڵێك خه‌ڵك وه‌ستاون ،      باوك ودایکی که‌زی له‌کابینه‌که‌یان دێنه‌ده‌ره‌وه‌جانتایان پێیه‌و پۆلیسه‌کان له‌دوایانه‌وه‌ن  به‌ره‌و ڕووی ئۆتۆمۆبیلی پۆلیس ده‌ڕۆن .
نابێت کچه‌که‌م .ئه‌بێت له‌گه‌ڵ باوك و دایکدا بێت ، ده‌بێت ئێمه‌بچین بۆ شوێنێکی  تر .
دایکی ده‌ستی که‌زی ده‌گرێ وده‌ڕۆن له‌گه‌ڵ باوکی دا سواری ئۆتۆمۆبیلی پۆلیسی ده‌بن به‌ڵام که‌زی به‌به‌رده‌وامی لا بۆ میشێله‌ده‌کاته‌وه‌، ئۆتۆمۆبیله‌که‌له‌کادر ده‌رده‌چێت .
پاشان میشێله ‌ده‌چێته‌به‌رده‌م وێنه‌یه‌ك له‌وانه‌ی که‌زی نه‌خشاندوویه‌تی که‌وێنه‌ی ئاڵای     کوردوستانه‌و له‌دیواری کابینه‌که‌ی ده‌کاته‌وه‌، له‌وکاته‌دا لیزا زۆر به‌شپرزه‌یی دێته ‌کادره‌وه‌پاشان ده‌زانێت ماڵی که‌زی ڕۆیشتوون  به‌چاوگێڕانێکی خێرا  سه‌یری وێنه‌کان ده‌کات، لیزا وێنه‌یه‌ك له‌دیواری کابینه‌کان ده‌کاته‌وه‌و پاشان چه‌ند که‌سێکی تر هه‌ریه‌که‌و وێنه‌یه‌ك له‌دیواره‌که‌ده‌که‌نه‌وه‌.
لیزا و میشێله ‌له‌ڕیزی پێشه‌وه‌ی خه‌ڵکه‌که‌وه‌ن هه‌ریه‌که‌و وێنه‌یه‌کی له‌وانه‌ی که‌زی کردوونی ، گرتوویانه‌به‌ده‌ستیانه‌وه و هه‌نگاو بۆ پێشه‌وه‌ده‌نێن .
