 هه‌ر له‌ دوره‌وه‌ سه‌ری بۆ له‌قاندم ..  زانیم به‌كوڵه‌وه‌ چاوه‌ڕوانی دیدارم ئه‌كات  , ئه‌یه‌وێـت  شتێكم نیشاندات بۆمن هه‌م تازه‌یه‌ و , هه‌م ترسناكیش ..
  ئه‌و ڕۆژه‌ له‌ هه‌مووكاته‌كانی تری  ڕوسه‌خت تر و بێشه‌رمتر  دیاربوو ..  به‌شاگه‌شكه‌یه‌كی نائارام و ترسێكی درۆینه‌وه‌ ڕایكێشامه‌ په‌ناوه‌  , وه‌ك ئه‌وه‌ی نهێنیه‌كی یه‌كجارمه‌زنی هه‌ڵگرتبێت , هه‌ر له‌ناو له‌پی خۆیدا  شاشه‌ی مۆبایله‌كه‌ی نیشاندام ..
وه‌ك دوو سوره‌ ماسی بن گۆماوێكی ڕۆشن , وه‌ك هه‌ڵقه‌گرێی  په‌لكه‌ی چنراوی زڵفزه‌ردێك , ژنو پیاوێكی ڕوت و قوت له‌یه‌ك ئاڵابون و ته‌كانیان ئه‌دا .. خانمێكی سوركه‌ڵه‌و ناسك ,  پیاوێكی زله‌و چه‌ته‌وڵی له‌باوه‌شگرتبوو   , ئاخ و ئۆفیان خه‌ریك بوو درزی ئه‌خسته‌ شاشه‌ی ناسكی مۆبایله‌كه‌وه‌  .
پێم گوت  هه‌تیو كلیپی چی و نیشتمانی چی ؟  ئه‌مانه‌ له‌دۆخی به‌دمه‌ستیدان , به‌ده‌ستی خۆیان ئابڕوی خۆیانیان بردووه‌ .. كێ هه‌یه‌ به‌م جۆره‌ سێكسكردنی خۆی وێنه‌ بگرێـت و بڵاویكاته‌وه‌ ..؟ "
 " به‌ڵێ ئه‌و ده‌توانێت  , ئه‌وه‌ زه‌ینه‌بخانه‌ .. زه‌ینه‌بخان ,  كوڕی باش گاڵته‌ی پێ ناكرێت .. ئه‌و هه‌ر به‌م كلیپانه‌ شه‌ڕی هه‌موو كه‌سێك  ئه‌كات .. به‌ڵێنی داوه‌ هه‌موو سه‌ری مانگێك وێنه‌ی ئابڕوچونی ژن و پیاوێكمان بۆ بڵاوكاته‌وه‌ " .
ماوه‌یه‌ك بوو من ئه‌مزانی زه‌ینه‌ب یاری به‌ چ ئاگرێك ئه‌كات , به‌ چ دڕنده‌ییه‌كه‌وه‌ هێرشی كرۆدته‌ سه‌ر شه‌ره‌ف و ئابڕوی خه‌ڵكی شاره‌كه‌مان .. زۆر جار ئه‌و چیرۆكه‌ ئابروبه‌رانه‌م ئه‌خوێنده‌وه‌ كه‌ نهێنیه‌كانی شه‌وی سوری  ژن و پیاوێكی شاره‌كه‌مانی  تیائه‌خسته‌ڕو  , وه‌ك چیرۆك و سیناریۆی كورته‌ فیلمی بازرگانی به‌هه‌مو جیهاندا  بڵاوی كردنه‌وه‌ .. به‌ڵام تا ئه‌و ساته‌ی وێنه‌ ڤیدۆییه‌كانیم به‌چاوی خۆم نه‌بینی , تا هه‌موو دام و ده‌زگا پۆلیسی و په‌روه‌رده‌یی وئاینیه‌كانی شار نه‌كه‌وتنه‌ ئاگاداركردنه‌وه‌ی هاووڵاتیان له‌ مه‌ترسیه‌كانی , وه‌ك ئه‌وه‌ی خۆیان ده‌یانگوت " له‌و هه‌ڕه‌شانه‌ی نیازه‌گڵاوه‌كانی زه‌ینه‌ب بۆ دۆزی ڕه‌وای گه‌له‌ سته‌م دیده‌كه‌مان دروستی كردووه‌ " ,   نه‌مزانی كه‌وتۆته‌ بڵاوكردنه‌وه‌ی ئه‌م وێنه‌ ڤیدۆییانه‌  ..!!
ڕۆژ له‌دوای ڕۆژ زه‌ینه‌ب یاخی تر ئه‌بوو , ده‌نگدانه‌وه‌ی شه‌ڕو هه‌ڕه‌شه‌كانیشی  لای ئه‌وانه‌ی لایه‌نگیریان ئه‌كرد ,  ئه‌وانه‌یش دژایه‌تیان ئه‌كرد ,  مه‌زنتر  ئه‌بوو   .. وه‌ك ئه‌و سه‌ركرده‌ سیاسیانه‌ی هێشتا ده‌مارگیری ڕۆژگاری شۆِرشیان هه‌رلا ماوه‌ ,  كه‌وتبوه‌ بڵاوكردنه‌وه‌ی لێدوانه‌ ئاگرینه‌كانی .
به‌ په‌یامنێری كه‌ناڵێكی ئاسمانی وتووبو  :-
 " به‌ سه‌دان دۆست و  و لایه‌نگرم  له‌ناوشاردا جێ هێشتووه‌ , سه‌دانی تر هه‌ر به‌پاره‌ كارم بۆ ئه‌كه‌ن , به‌س بۆ ئه‌وه‌ی سێكسكردنی كاربه‌ده‌ستانی شار و سۆزانیه‌كانم بۆ بهێنن .. هه‌ر گیز كۆڵناده‌م له‌م ڕێگای تێكۆشانه‌ی گرتومه‌ته‌ به‌ر  ,  واز ناهێنم تاكلیپه‌كه‌ی جه‌نابی پارێزگار خۆیشی بڵاونه‌كه‌مه‌وه‌ ... ئه‌وكات تواناو هێزی خۆمیان نیشان ئه‌ده‌م  " .
,  ته‌ماشاكه‌ن  , چ كاره‌ساتێكی پێیه‌ ئه‌م ژنه‌ .. خودایه‌ , كێ هه‌یه‌ له‌شه‌ڕو گێچه‌ڵی  نه‌ترسێت , به‌تایبه‌تی له‌ شاره‌كه‌ی ئێمه‌دا , كه‌ هه‌مووان " له‌ به‌ر پیسی ئیشی شه‌ویان , وه‌ك به‌راز له‌ سێبه‌ری ملی خۆیان سڵ ئه‌كه‌نه‌وه‌ " .
ئاگام لێ بو چپه‌چپی ئێمه‌ و , قیژه‌و ئاخ و ئۆفی ناومۆبایله‌كه‌ , خه‌ریك بوو ئارامی هۆڵه‌كه‌ی ئه‌شێواند , سه‌رنجی ئه‌و خوێنه‌رانه‌ی ڕائه‌كێشا كه‌ بۆ ئه‌وه‌ هاتبون ساتێكی ئارامی خوێندنه‌وه‌ به‌رنه‌ سه‌ر ... هه‌رچه‌نده‌ مه‌ترسیه‌كی گه‌وره‌یش بوو گه‌ر بیانزانیایه‌ سه‌یری چی ئه‌كه‌ین , ئه‌وه‌چیه‌ له‌م گۆشه‌ ئارامه‌ی هۆڵه‌كه‌دا هه‌ردوكمانی كۆكردۆته‌وه‌ ..  چونكه‌ ماوه‌یه‌ك بوو ده‌زگای ئاسایشی شار لای هه‌ر كه‌سێك وێنه‌و ڤیدۆی سێكسی خۆماڵیان بگرتایه‌ توشی سزایه‌كی توندیان ئه‌كرد .. به‌و بیانوه‌ی گوایه‌ ته‌نها لایه‌نگرانی زه‌ینه‌ب خه‌ریكی بڵاوكردنه‌وه‌و ده‌ستاوده‌ستكردنی  ئه‌م (شته‌ پیسانه‌ ) ئه‌بن .
له‌ ئه‌مریكای لاتینی  كاسترۆ و هۆگۆچاڤیز   ئاژاوه‌یه‌ك سازبكه‌ن  , یه‌كسه‌ر چه‌په‌كانی لای ماڵ ئێمه‌ ده‌گیرێن و  دواتر به‌كه‌فاله‌ت ئازاد ئه‌كرێن  .
له‌ وه‌ته‌ی زه‌ینه‌بخانی به‌رپه‌نجه‌ره‌كه‌یش كه‌وتۆته‌ هه‌ڕه‌شه‌كردن  ,  ترس و نیگه‌رانیه‌كی زۆر ئه‌و كه‌سانه‌ی گرتۆته‌وه‌ كه‌ مۆبایلی ( بلۆتۆز ) یان هه‌یه‌   ,  چونكه‌ زۆر هۆ هه‌یه‌ ئه‌شێـت وایان لێ بكات وه‌ك لایه‌نگیری زه‌ینه‌بخان بژمێردرێن ..
هه‌مومان  له‌یه‌ك گه‌ڕه‌كدا گه‌وره‌ بوین ( یه‌كه‌م عاشقی , یه‌كه‌م كڕیاری , یه‌كه‌م گه‌وادی  , یه‌كه‌م دۆست و یه‌كه‌م ناحه‌زی , یه‌كه‌م كه‌س زینده‌خه‌ونی ئیرۆتیكی  پێوه‌ئه‌بینی , یه‌كه‌م كه‌س ئیمزای شاربه‌ده‌ركردنی له‌سه‌ر كۆكره‌وه‌ , یه‌كه‌م كه‌سیش ئیمزاكه‌ی بۆ كرد , یه‌كه‌م كه‌س ڕاپۆرتی نهێنی له‌سه‌ر ئه‌نوسی ... هه‌روا له‌م گه‌ڕه‌كه‌ی ئێمه‌دا بوو ) .. هه‌ر كه‌ شه‌ڕه‌ نه‌گریسه‌كانیشی  ده‌ست پێكرد , ڕابرد و ژیانی هه‌مو پیاوانی شاری كرده‌ زنجیره‌ چیرۆكێكی ئابڕوبه‌ر و له‌ چاپه‌مه‌نیه‌كانی به‌ریتانیا و ئه‌مریكا بڵاویكردنه‌وه‌ .. پێش هه‌موكه‌سێكی تر ئێمه‌  كێچ كه‌وته‌ كه‌وڵمان , كه‌وتینه‌ دۆخێكی پڕترس و نیگه‌رانیه‌وه‌ , له‌ هه‌موو چركه‌یه‌كدا چاوه‌ڕوانی ئه‌وه‌ بوین پۆلێك چه‌كداری جه‌ربه‌زه‌ هه‌ڵكوتنه‌ سه‌رمان ,  قۆڵبه‌ست و چاوبه‌ست بمانبه‌ن و زیندانێكی تاكه‌كه‌سیمان پێ ئاوه‌دان كه‌نه‌وه‌ .. له‌مه‌یش خراپتر ئه‌وه‌ بوو له‌نیو شاردا به‌لایه‌نگیری زه‌ینه‌بخان , یان وه‌ك ئه‌وه‌ی (خه‌ڵك ) ده‌یانگوت ,  ( ده‌ستو پێوه‌ندی ژنه‌حیزه‌كه‌ ) ناوئه‌براین .
100%خه‌ڵكی شار به‌هۆی ئه‌و كلیپانه‌ی زه‌ینه‌بخان بڵاویان ئه‌كاته‌وه‌ توشی ترس و دڵه‌ ڕاوكێ هاتون : -
30% یان , ئه‌و ژن و پیاوانه‌ن كه‌ پێشتر په‌نێكیان داوه‌و ئێستا ترسی بڵاوبونه‌وه‌ی وێنه‌ی ئابڕوچونی خۆیانیان هه‌یه‌ .
30% تر , ئه‌و ژنانه‌ن دڵیان له‌مێرده‌كانیان پیسه‌ , یان ئه‌و پیاوانه‌ن دڵیان له‌ژنه‌كانیان پیسه‌ .. ئه‌ترسن ئه‌م كلیپانه‌ی زه‌ینه‌بخان ئه‌نجامێكی كاره‌ساتباری چه‌ندین ساڵی ژیانی خانه‌گومانیان بداته‌ ده‌سته‌وه‌ .
ئه‌م ژنه‌  به‌هاتنی سه‌ره‌تای هه‌موو مانگێك , كلپێك بڵاوئه‌كاته‌وه‌   , ئه‌وه‌ی له‌م چیرۆكه‌دا ناومان ناوه‌ كلیپ  , زه‌ینه‌ب خۆی پێی ئه‌ڵێت " به‌رهه‌مێكی ده‌زگای وێنه‌ نیشتمانیه‌كان "  .
هه‌موو زانایانی ئاینیش به‌یه‌ك ده‌نگ و بێ دوو دڵی پێی ده‌ڵێن " تۆوی فه‌ساده‌ت " .
به‌ڵام ئه‌و بڕیاری داوه‌ هه‌موو سه‌ری مانگانێك  ڤیدۆی ئابڕوچونی یه‌كێك له‌ پیاوه‌ ناوداره‌كانی شار بڵاوبكاته‌وه‌  , له‌ كۆتایی هه‌ر وێنه‌یه‌كیشدا نوسراوه‌  :-
ئیتر ئه‌م دوو نیوه‌ دێڕه‌ , ترس و دڵه‌ڕاوكێیه‌كی ئه‌وتۆی خستۆته‌ دڵی دانیشتوانی شاره‌وه‌ , سه‌دان كه‌سی ناچار كرد  بچن بۆلای پزیشكه‌ ده‌رونیه‌كان چاره‌سه‌ریان بكات .. ئه‌وه‌ی ڕۆژێك له‌ ڕۆژان , یان بۆساتێك  , له‌ گه‌ڵ دۆستێكیدا سه‌رجێ نوستنێكی كردبێت  ... له‌ترسی هه‌ڕه‌شه‌كانی زه‌ینه‌ب ئارامی لێبڕا .
یه‌كێك له‌ئه‌فسه‌ره‌ پله‌داره‌كانی  سوپایه‌ .. یان له‌و وه‌زیرانه‌یه‌ وا كه‌س نازانێـت كارو پیشه‌ی چیه‌و به‌زیاده‌ دامه‌زراوه‌ ..
له‌ناو لایه‌نگرانی زه‌ینه‌بیشدا , وه‌ك به‌شێك له‌شه‌ڕی ده‌رونی و ڕاگه‌یاندن , ده‌نگۆی ئه‌وه‌ بڵاو ئه‌كه‌نه‌وه‌  كه‌ سۆزانیه‌كه‌ی ئه‌م جاره‌ له‌ ماڵی پاشاوه‌ دێت  .
له‌م چیرۆكه‌دا پیاوێكی ترده‌ناسین به‌ناوی  ( بابه‌ تاهیری عوریانی )ه‌وه‌  , سۆفیه‌كی شاعیره‌و  له‌وپه‌ڕی گۆشه‌گیری و ژیانی زاهیدانه‌ی خۆیه‌وه‌ , به‌عه‌شقێكی خوداناسانه‌وه‌ گیرۆده‌ بووه‌ ..  كه‌ ده‌نگوباسی ئه‌م ئاژاوه‌ و هه‌ڕشانه‌ی شاره‌كه‌ی ئێمه‌ی پێ ئه‌گات , به‌ده‌م ڕاچه‌نینێكی سۆفیانه‌وه‌ نه‌فره‌تێكی سه‌خت و یه‌كجاره‌كی ئه‌نێرێت :-  بۆ هه‌رچی خاتونی چاوله‌ده‌ر ... و  پیاوی شه‌واڵپیس و بێ فه‌ڕ ... هه‌یه‌  .
هه‌ر هه‌مان ڕۆژ  ئه‌یه‌وێت ده‌رباره‌ی ئه‌م ڕوداوانه‌ نامه‌یه‌ك بۆ هاوڕێیه‌كی خۆی بنوسێـت كه‌ له‌زه‌مین و زه‌مانێكی زۆر جیاوادا ژیاوه‌ به‌ ناوی ( شه‌ره‌فخانی به‌دلیسی ) .. ده‌ڵێن گوایه‌ هێنده‌ توڕه‌ بووه‌ , سێ جار په‌ڕه‌موچه‌كه‌ی ده‌ستی شكاوه‌ , چواره‌م جار ویستویه‌تی بچێت بۆ كوفینێتێكی شاری هه‌مه‌دان و به‌ئیمه‌یل بینێرێت , كۆماری ئیسلامی ئێران ڕێگه‌یان لێگرتووه‌ , به‌بیانوی ئه‌وه‌ی شه‌ره‌فخان له‌گه‌ڵ كورده‌ جودایخوازه‌كاندا سه‌رو كاری هه‌یه‌ و .. ئه‌مه‌یش  دژی (مه‌صڵه‌حه‌تی نیزامه‌ ) .
بۆ به‌دبه‌ختی شاره‌كه‌مان , نه‌ باباتاهیر و نه‌ میرشه‌ره‌ف خانیش هیچیان پێ ناكرێت .. خاتو زه‌ینه‌بیش  وه‌ك باسكی بانان بۆی ده‌رچووه‌ و پانتایی ئه‌وروپای لێگرتووه‌ ,  ڕۆژ له‌دوای ڕۆژ هه‌ڕه‌شه‌كانی كاریگه‌رتر ئه‌بێت , ترسی ئێمه‌یش له‌ بزیسكی ئاگره‌كه‌ی زیاتر ئه‌بێت ..
" هه‌ركه‌ گه‌یشتمه‌ ئه‌و باوه‌ڕه‌ی چیرۆكه‌كان و وێنه‌ ڤیدۆییه‌كان  توانای له‌رزاندنی كورسیه‌كه‌ی ژێر پارێزگاری نیه‌ , هیچ دوودڵ نیم له‌بڵاوكردنه‌وه‌ی كلیپه‌ ئابڕوبه‌ره‌كه‌یشی  ".
ئیتر دوای ئه‌م لێدوانه‌ , ته‌واوی خه‌ڵكی  شار گوێ قوڵاخی هه‌واڵی ده‌ركه‌وتنی كلیپه‌كه‌ی پارێزگاربون  , هه‌ركه‌سه‌و له‌ خه‌یاڵی خۆیه‌وه‌ چاوڕێی ده‌ركه‌وتنی دیمه‌نێكی كاره‌ساتباره‌ , دیمه‌نێك , یان پارێزگار یان كه‌ژاڵخانی هاوسه‌ری ده‌بنه‌ نێچیره‌كه‌ی ..
به‌ تێپه‌ڕینی ڕۆژگار كاریگه‌ری هه‌ڕه‌شه‌كانی زه‌ینه‌ب زیاتر ئه‌بو  , ئاژاوه‌و سه‌نگه‌رگیری نێوان دۆست و دوژمنه‌كانیشی هه‌ر زیاتر ئه‌بو ,  له‌ ئینته‌رنێـته‌وه‌ ده‌نگو باسی ئه‌وه‌ بڵاوكرایه‌وه‌  به‌م زوانه‌ كلیپی ( شه‌واڵ میزاوی  )بڵاو ئه‌كرێـته‌وه‌ , كه‌ گرنگترین كلیپی شاره‌ , تا ئێستا كه‌س نازانێـت كه‌سه‌كانی ناو ئه‌م كلیپه‌ كێن ,  ئه‌و پاڵه‌وانه‌ ئابڕوچوانه‌ی زه‌ینه‌ب ئاگر به‌چاره‌نوسیان ئه‌كاته‌وه‌  كێ ده‌رئه‌چێت .. ئیتر ترس و دڵه‌ڕاوكێ و كاردانه‌وه‌ی چین وتوێژه‌ جیاوازه‌كانی شار , به‌  شێوازی جیاجیاو هیستریانه‌ ده‌ستی پێ كرد :-
حه‌ره‌سی موه‌لیده‌ی گه‌ڕه‌كه‌كه‌مان , له‌داخی ئه‌وه‌ی پاسه‌وانی موه‌لیده‌ی گه‌ڕه‌كێكه‌ چل ساڵی ڕه‌به‌ق زه‌ینه‌بی تیاژیاوه‌ , وازی هێنا و چوه‌وه‌ بۆماڵی خۆی .. گوتی " قیروسیا , با له‌برسا بمرم , هه‌ر له‌وه‌ باشتره‌ خزمه‌تی موه‌لیده‌ی گه‌ڕه‌كێك بكه‌م كه‌ ... " .
زانایانی ئاینی كۆبونه‌وه‌یه‌كی به‌په‌له‌یان ئه‌نجامدا , هێنده‌ دڵگه‌رم و شپرزه‌ كۆبونه‌وه‌ چاوه‌ڕێی هاتنی كاره‌بای موه‌لیده‌كه‌یان نه‌كرد تاهۆله‌كه‌یان بۆ ڕۆشنبكاته‌وه‌  , بۆیه‌ چه‌ند ڕۆژنامه‌نوسێكی بزێو و جاسوسێكی زه‌ینه‌بخانیش له‌گه‌ڵیانا ئه‌چێـته‌ ژوره‌وه‌ ..  هه‌ر هه‌مان ڕۆژ ته‌واوی ئه‌و نهێنیانه‌ی كه‌ وه‌ك (نهێنی پیشه‌ ) ی زانایانی ئاینی وایه‌ و  نابیـَت له‌خۆیان زیاتركه‌سی تر بیزانێـت دزه‌ئه‌كاته‌ ده‌ره‌وه‌ , ته‌نانه‌ت دوای سێ كاتژمێر ده‌گاته‌وه‌ ده‌ستی زه‌ینه‌ب خۆیشی .. له‌كۆبونه‌وه‌كه‌دا بڕیار ئه‌درێت .
- ته‌واوی وتاره‌كانی هه‌ینی داهاتوو , له‌سه‌ر هه‌ڕه‌شه‌كانی ئه‌و ژنه‌ ده‌ڵه‌قۆڕه‌ بێت كه‌ بۆته‌ موسیبه‌ت بۆ شاره‌كه‌مان .
- پێویسته‌ داوابكرێت له‌ موسوڵمانان , نه‌گوێ له‌لێدوانه‌كانی بگرن , نه‌ ئه‌و درۆ و ده‌له‌سانه‌ بخوێننه‌وه‌ , كه‌ به‌ناوی ( زنجیره‌ چیرۆكێكی واقعی ژیانم ) له‌ گۆڤار و ڕۆژنامه‌ بیانیه‌كاندا بڵاوی ئه‌كاته‌وه‌ .. نه‌ ته‌ماشای ئه‌و كلیپه‌ ئابڕوبه‌رانه‌یش بكه‌ن كه‌ سه‌ری مانگان ئه‌یخاته‌ پێش چاوی هه‌موو گه‌ل .. موسوڵمانی شه‌ریف و ڕه‌سه‌ن ئه‌و كه‌سه‌یه‌  گوێ له‌زه‌ینه‌ب ناگرێت , هه‌موو وته‌و نوسین و شته‌كانی تریشی فه‌رامۆش ئه‌كات .. ئه‌وه‌ی لایه‌نگیری زه‌ینه‌ب بێـت ( ئینشائه‌الله )له‌قیامه‌تدا به‌ڕو ڕه‌شی و خه‌جاڵه‌تی له‌گه‌ڵ ئه‌و دا حه‌شر ئه‌كرێت .
من نازانم چیرۆكی پۆلیسی و پڕ ڕوداوی  سه‌رئێشه‌ بنوسم , هیچ سۆزو لایه‌نگریه‌كیشم نیه‌ بۆ زه‌ینه‌ب , له‌وانه‌یه‌ له‌تۆیش بێ لایه‌نتر سه‌یری ئه‌م ڕوداوانه‌ بكه‌م .. نه‌حه‌زم له‌زیاده‌ڕۆیكردن هه‌یه‌ له‌گێڕانه‌وه‌دا , نه‌ هاوسۆزی هیچ كاره‌كته‌رێكیش ..  به‌ڵام باوه‌ڕم پێ بكه‌ , بێ هیچ هۆكارێكی ماقوڵ له‌خۆم ئه‌ترسام , ته‌نانه‌ت به‌رنامه‌و تایبه‌تمه‌ندی ئامێری مۆبایله‌كه‌یشم به‌جۆرێك نه‌بوو یارمه‌تی ئه‌م ترسه‌م بدات , وام لێبكات له‌لایه‌نگرانی زه‌ینه‌به‌وه‌ نزیك بم .. كه‌چی هه‌ر به‌ بێ ویستی خۆم گلاندمیانه‌ ناو ئه‌و گرفت و ئاژاوانه‌ی لایه‌نگرانی زه‌ینه‌ب پێ هه‌ڵئه‌سان .
گرفته‌كان هێنده‌ ته‌شه‌نه‌یان كرد , به‌هیچ لایه‌ك كۆنترۆڵ نه‌ئه‌كرا , سه‌ركردایه‌تی وڵات گه‌یشته‌ ئه‌و باوه‌ڕه‌ی ئاسایشی نه‌ته‌وه‌یمان له‌ مه‌ترسیدایه‌ , له‌م پێناوه‌یشدا پێویسته‌ ده‌زگا به‌رپرسه‌كان  هه‌موو جوڵه‌ و هه‌وڵێكیان بخه‌نه‌ گه‌ڕ , ڕێو شوێنی ئاسایشی توند بگرنه‌به‌ر , بۆ ڕێگرتن له‌و كلیپانه‌ی له‌ ناو شاره‌وه‌ ئه‌گاته‌ ده‌ستی زه‌ینه‌ب ,  ئه‌وانه‌یش له‌ڕێگای ئینته‌رنێتـه‌وه‌ زه‌ینه‌ب بڵاویان ئه‌كاته‌وه‌ .. بۆیه‌ به‌ڕێوبه‌رایه‌تی گشتی ئاسایش له‌ چوارچێوه‌ی هه‌ڵمه‌ته‌كه‌یدا  كۆمه‌ڵێك هه‌نگاوی خۆپارێزی نا  :- 
- هه‌موو ئه‌وانه‌ی گومانی ئه‌وه‌یان له‌سه‌ر بوو جاسوسی زه‌ینه‌بن و لێره‌وه‌ كلیپی ئابڕوبه‌ری بۆئه‌نێرن , هه‌رده‌ستگیركران .
- هه‌موو ئه‌و پیاوانه‌ی ڕۆژێك له‌ڕۆژان ئاڵۆشی ناوگه‌ڵیان توشی به‌زمی كردبون و چوبونه‌ سه‌رجێی ژنێكی حه‌رام , هاندران بۆئه‌وه‌ی بكه‌ونه‌ خۆیان , نه‌وه‌ك ئاگایان لێ نه‌بوبیـَت و به‌كامێرایه‌كی شاراوه‌ وێنه‌گیرابن , یان ئه‌گه‌ر كلیپی  ئه‌م سه‌ری مانگه‌ ئه‌وانبن , دواتر چیبكه‌ن و بۆكوێ سه‌ری خۆیان هه‌ڵگرن ؟
زۆری پێ نه‌چو به‌شی هه‌ره‌زۆری گه‌نجه‌كانی شار , به‌كچ وكوڕه‌وه‌ , ئه‌وه‌ی مۆبایلی بلۆتۆزی پێ بو  خرایه‌ زیندانه‌وه‌ .
به‌و بیانوه‌ی كه‌ ئه‌م ئامێره‌ فیتنه‌یه‌ و بۆته‌ سه‌رچاوه‌ی هه‌مو كێشه‌كان , تامۆبایل نه‌هاتبو كێ ئاگای له‌كێ بو .
ده‌وڵه‌ت زانی به‌ قه‌ده‌غه‌كردنی مۆبایل كاروباری وڵات به‌ڕێوه‌ ناچێت , به‌پڕكردنی زیندانه‌كان له‌خه‌ڵكانی گومانلێكراو كێشه‌كه‌ چاره‌سه‌ر نابێت .. ناچار بوون له‌گه‌ڵ زه‌ینه‌بدا دانوستاندن بكه‌ن , به‌هه‌ر نرخێك  هه‌یه‌ له‌م شه‌ڕانگێزیه‌ی په‌شیمانی بكه‌نه‌وه‌  .. بۆیه‌ بڕیاریان دا له‌هه‌فته‌ی داهاتودا , بۆ دانوستاندن له‌گه‌ڵ زه‌ینه‌بدا نێردراوێكی نهێنی بنێرنه‌ له‌نده‌ن ..
