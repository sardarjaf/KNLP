له‌وده‌مه‌یدا كه‌به‌تاق وته‌نیا به‌سه‌ر شۆسته‌كانی بیری خۆم و به‌ژێر سێبه‌ری  داری له‌بیرچوونه‌وه‌دا  پیاسه‌م ده‌كرد، له‌پڕ گوێم له‌یه‌كێك بوو ئێژێ: ئه‌م خڕه‌ی كه‌له‌گوێزه‌ ده‌چێت كێیه‌؟ وادیاره‌ لاسایی ئێمه‌ی خورما ده‌كاته‌وه‌؟  گوێز حه‌دی چییه‌ بگا به‌خورما!
سه‌رم وه‌ك فڕێنكه‌ به‌ده‌وری خۆیا سوڕایه‌وه‌و كه‌وته‌ گه‌ڕان به‌دوای قاڵه‌قاڵێكدا كه‌بۆنی لوتبه‌رزی لێده‌هات، به‌ملاو به‌ولای داره‌كه‌شیدا هێشووی زه‌رد كه‌له‌بۆڵی حه‌زێكی گڕگرتوو ده‌چوو شۆڕ ببووه‌وه‌، ته‌زێك هات به‌گیانمداو لاشه‌م وه‌ك كێل ڕووه‌و ده‌نگه‌ده‌نگه‌كه‌  ره‌پبوو، چاوم له‌نێو كاڵانه‌كه‌یدا به‌چوار ده‌ورمدا وه‌ك  لایت كه‌وته‌ خولانه‌وه‌، تابتوانێ چه‌ند زانیارییه‌كی تازه‌ به‌ده‌ست بێنێ، به‌ڵام من هیچم نه‌بینی جگه‌ له‌تارمایی پیاوێك كه‌تیژ ئاسۆی بڕی و له‌پشتی داری له‌بیرچوونه‌وه‌ ون بوو، سه‌رم كه‌مێك هێور بووه‌وه‌ گه‌ڕایه‌وه‌ نێو بازنه‌كه‌ی خۆی، وه‌لێ ئه‌و هه‌ڵه‌شه‌یه‌ به‌رده‌وام بوو له‌سه‌ر ختوكه‌دانی پێكه‌نینه‌ یه‌ك له‌دوای یه‌كه‌كانی خۆی،  له‌گه‌ڵ ئه‌ویشدا كۆمه‌ڵێك له‌ئاپوره‌ی خه‌ڵكی شه‌یدای پێكه‌نین قاقا دایانه‌ پێكه‌نین، ئه‌وكاته‌ من وامده‌زانی ئه‌و خه‌ڵكه‌ كه‌عاده‌ته‌ن خه‌ڵكی بێ كارو كه‌سابه‌تن و هه‌میشه‌ شۆسته‌ی جاده‌كانیان كردووه‌ به‌مۆڵگه‌ی دانیشتن بۆ ئه‌وه‌ی پاروییه‌ك پێكه‌نین یان دیمه‌نێكی كۆمیدیان ده‌ستكه‌وێ، وه‌ك سوڕان و كه‌وتنی ڕێبوارێك له‌توێكه‌ مۆزێك یان له‌بنێشتێكی papay یان هه‌ر شتێكی له‌م بابه‌ته‌ كه‌ پێی بنیاده‌می لێ بخلیسكێ، هه‌موو ئه‌مانه‌ له‌بشێوی و په‌ستی ئه‌و پرسیاره‌یان دوور نه‌خستمه‌وه‌ كه‌وه‌ك نیگه‌رانییه‌ك له‌ناخمدا ده‌سوڕایه‌وه‌، چونكه‌ ئه‌و قسه‌یه‌ توڕه‌و بێزاریی خستبووه‌ دڵمه‌وه‌، به‌ڵام خۆشبه‌ختانه‌ دانم به‌خۆمدا گرت وئه‌وه‌ی پیلانگێڕان پیلانیان بۆ دارشتبوو وام نه‌كرد، جارێكیتر خه‌یاڵم به‌رز فڕی و به‌ده‌وری كۆمه‌ڵێ بیروبۆچووندا كه‌وته‌ سوڕانه‌وه‌، گه‌شتمه‌ نێو باخچه‌یه‌كی پڕ له‌گوڵ وگوڵزار،چه‌ند گوڵێكم بۆنكردوو چه‌ند دانه‌یه‌كیشم به‌بێ ویستی خۆم بڕی، هه‌موو ئه‌مكاتانه‌ش وا هه‌ستم ده‌كرد له‌لایه‌ن چه‌ند كه‌سێكی نه‌ناسراوه‌وه‌ چاودێری ده‌كرێم، كه‌سانێك كه‌هه‌ر له‌سه‌نافرو قالۆنچه‌ ده‌چن،  به‌ڵام به‌نیگایه‌كی تیژو گوێیه‌كی سوكه‌وه‌، ماوه‌یه‌ك به‌و نێوه‌دا كه‌وتمه‌گه‌ڕان،  خووخده‌م له‌گه‌ڵما گه‌وته‌ گه‌ڕان به‌بێ‌ ئه‌وه‌ی ئاسۆی هیچ ئومێدێك ببینم، هه‌روا به‌بێ ئه‌وه‌ی سه‌یری ئه‌ملاو ئه‌ولا بكه‌م قاچه‌كانم بێهوده‌ تا ئه‌و په‌ڕی بینین بردمیان، له‌وێنده‌رێ له‌و ده‌شته‌ چۆڵه‌دا ئاپوره‌ی خه‌ڵكانێك كۆببوونه‌وه‌ له‌وانه‌ی كه‌كۆمه‌ڵگا توڕیهه‌ڵدابوونه‌ ده‌ره‌وه‌ی بازنه‌ی خۆی، ئه‌وانیش به‌گسك و كه‌په‌بێڵ كه‌وتبوونه‌ پاك كردنه‌وه‌و گسكدانی ئه‌وناوه‌، له‌گه‌ڵ به‌رزبوونه‌وه‌ی ته‌پوو تۆزو خۆڵباریندا سه‌دای ده‌نگه‌ده‌نگ و هاتوهاواریان به‌رزده‌بووه‌وه‌، ماوه‌یه‌كیش له‌وێ مامه‌وه‌ سه‌یری ئه‌و خه‌ڵكه‌م ده‌كرد، به‌ڵام هێشتا ئه‌و پرسیاره‌ ژاراوییه‌ ختوكه‌ی ده‌دام و گازی له‌ناخم ده‌گرت، شتێكی ئه‌وتۆ تێنه‌گه‌یشتم، له‌به‌رخۆمه‌وه‌ وتم( كێ چاودێری خه‌ڵك بكا له‌غه‌ما ده‌مرێ)  ( چیم داوه‌ له‌كاری خه‌ڵك وخه‌ڵك چییان له‌كارم داوه‌) به‌ده‌م ڕێگاوه‌ بۆ ماوه‌یه‌كی زۆر لێوه‌كانم ئه‌مه‌یان ده‌گوته‌وه‌ تاگه‌یشتمه‌ گۆڕه‌پانی (ئازادی) له‌وێ له‌نزیكی مۆته‌ فرۆشێك بنێشتێكی(سه‌همداری ئه‌سڵیم) كڕی تۆزێك له‌ولاتریشه‌وه‌ له‌سه‌ر ته‌خته‌یه‌ك كه‌بۆ ڕێبواران دانرابوو دانیشتم، وه‌ره‌قه‌یه‌كم له‌گیرفانی كراسه‌كه‌م ده‌رهێناو ئه‌م پارچه‌ هۆنراوه‌یه‌م نووسی:
پاشان قاقه‌زه‌كه‌م خسته‌ گیرفانمه‌وه‌و هیمه‌تمدا هه‌ستم، هه‌ستمكرد خۆریش باروبنه‌ی خۆی پێچاوه‌ته‌وه‌و له‌سه‌ر سه‌فه‌ری ئاوابوونه‌، وتم بۆ بیركردنه‌وه‌ هیچم بۆ نه‌ماوه‌ته‌وه‌و ئه‌م گومانه‌ش ئه‌عسابی بنیاده‌م ئه‌پرۆكێنێ، له‌پڕ هه‌ستم كرد كۆمه‌ڵێ گه‌شتیار به‌ڕاكردن به‌ره‌و لای ده‌رگای ده‌ره‌وه‌ی حه‌وشه‌ی ئیزگه‌و ته‌له‌ویزۆن ده‌چن، منیش هه‌روا به‌غاردان ته‌كیان كه‌وتم،  دوای ماوه‌یه‌ك راكردن هه‌موویانم ره‌تداو پێشیان كه‌وتم، ماوه‌یه‌ك ڕامكرد دوایی به‌رمكرده‌ دواوه‌ هیچم نه‌بینی،  جگه‌ له‌ته‌پوتۆزو فیكه‌و هاواری سه‌یاره‌ی پۆلیس كه‌به‌شوێنی مندا ده‌هاتن، له‌ترسانا خۆمم هه‌ڵدایه‌ نێو گۆماوه‌كه‌ی پشتی بینایه‌ی ئیزگه‌و ته‌لفیزۆنه‌وه‌و به‌زمانی ئینگلیزی وتم: تۆریستم  تۆریستم، واته‌ گه‌شتیارم گه‌شتیارم، گوێم  له‌یه‌كێ له‌پۆلیسه‌كان بوو هاوارده‌كاو ده‌ڵێ: كوره‌ ئاگاداربن ترۆریسته‌ ترۆریسته‌، ئه‌وكاته‌ ئیتر پۆلیسه‌كان به‌ته‌واوی سه‌نگه‌ریان گرت و به‌(بی كه‌ی سی) ده‌ستیانكرد به‌ته‌قه‌كردن له‌گۆماوه‌كه‌، به‌ڵام باش بوو پێش له‌وه‌ من توانیم خۆم ده‌ربازكه‌م و هه‌ڵبێم، به‌و ته‌ڕوبڕییه‌وه‌ گه‌ڕامه‌وه‌ بۆ مه‌یدانه‌كه‌و چوومه‌وه‌ بۆلای مۆته‌ فرۆشكه‌ كه‌ئه‌مجاره‌یان ببوه‌ به‌ (له‌بله‌بی فرۆش)  له‌وێ دانیشتم گوێم له‌ده‌نگی ته‌قه‌ بوو به‌رده‌وام به‌ده‌سڕێژ ته‌قه‌یان له‌گۆماوه‌كه‌ ده‌كرد،خه‌ڵكێكی زۆرو زه‌وندیشم ده‌بینی به‌پڕتاو به‌ره‌و ماڵه‌كانیان هه‌ڵده‌هاتن نه‌وكا گولله‌یان به‌ركه‌وێ و برینداربن، چاوم سه‌رقاڵی بینینی ئه‌و دیمه‌نه‌بوو، یه‌كێك لێم نزیك بووه‌وه‌و خورمایه‌كی خه‌سته‌وی بۆ درێژكردم و وتی: فه‌رموو، گوتم : زۆر سوپاس خورما ناخۆم نه‌خۆشی شه‌كره‌م هه‌یه‌، پێكه‌نی و گوێلێ‌ ناوكه‌ خورماو پیسی كرده‌ كۆشم، ئه‌م كاره‌ تووشی شۆكی كردم، هه‌ستم به‌ئیهانه‌كردو ویستم شتێك بڵێم، كه‌چی ئه‌و به‌مه‌شه‌وه‌ نه‌وه‌ستاو وتی: (چییه‌ جه‌نابت نمونه‌ی وا خورما ناخۆی ، تادوێنێ بۆ ده‌نكێ خورما سواڵی هه‌زار ده‌گاتان لێمان ده‌كرد، كه‌چی ئێستا بوون به‌پیاوماقوڵ وخورما ناخۆن،  پاقله‌واتان ده‌وێ، ئێوه‌ چه‌ند پیسن؟) ئه‌و رۆژه‌ وه‌ڵامم نه‌دایه‌وه‌ چونكه‌ هێشتا زمانم فێری جنێوو ئیهانه‌كردنی خه‌ڵك  نه‌ببوو، دوباره‌ له‌دڵی خۆمه‌وه‌ وتم : ئه‌ی ئه‌و كه‌سانه‌ی ده‌بیستن ونابینن، ئه‌ی ئه‌و كه‌سانه‌ی ده‌بینن ونابیستن، له‌م قسه‌یه‌ت بۆنی خۆبه‌زلزانین ده‌كه‌م، من لێره‌ جگه‌ له‌هاوڵاتییه‌كی  وه‌كو تۆ هیچیتر نیم، له‌سه‌ر غه‌م وته‌می ژیانی رۆژانه‌م ده‌ژیم، وه‌ك تۆ شه‌و و رۆژ به‌دوای پاروویه‌ك ناندا ڕاده‌كه‌م، پێم له‌لاشه‌ی تۆ هه‌ڵده‌كه‌وێ، ڕێگای ده‌ربازبوونی لێگرتووم، ئایا له‌پێناو مندا كوشتتیان یان ته‌نیا بۆ خۆشی؟ ئیتر باشه‌  بۆ تۆ من وه‌ك بوكه‌ڵه‌یه‌كی شڕ وێنا ده‌كه‌ی كه‌سێبه‌ری شه‌یتان دایپۆشی بێت،  شه‌یتانێك له‌شه‌یتانه‌كانی قه‌ده‌ر، دووای ئه‌م پرسیاره‌ی منه‌ تۆفانێك له‌پێكه‌نین هه‌ستاو له‌ئاسۆی شاردا ده‌نگی دایه‌وه‌، زۆر كه‌س تووشی ترسو تۆقین بوون، له‌وانه‌ش خۆم كه‌هاوڵاتییه‌كی پله‌ په‌نجای ژێر سفرم، دوای هه‌وڵ وكۆششێكی زۆر توانیم دوباره‌ جڵه‌وی دۆخه‌كه‌ بگرمه‌ ده‌ست وهه‌رچۆنێ بوو به‌ده‌نگێكی شه‌رمن گوتم: ئێمه‌ میلله‌تێكی بێ تواناو ده‌سه‌ڵاتین، ئموری خۆمان به‌ده‌ستی خۆمان نییه‌، نه‌له‌گه‌ڵ ئه‌وانه‌ین كه‌ ئیمانیان هێنا نه‌له‌گه‌ڵ ئه‌وانه‌ین كه‌یاخی بوون، هه‌ردوو جه‌ژنه‌كه‌مان له‌كیس چوو، ئه‌وكاته‌ی ئاوام ده‌تگوت وشه‌كانم بۆنی نێرگسیان لێده‌هات، زمانی ئه‌وانیش بۆنی نه‌فتالینی لێده‌هات، مرۆڤیش به‌دوو بچكۆڵه‌كه‌ی  زمان ودڵی مرۆڤه‌، گوێزو خورما گه‌رچی هه‌ردوكیان له‌توخمی داری هه‌میشه‌ سه‌وزن، وه‌لێ‌ ئه‌وان له‌یه‌ك كانیاو ئاو ناخۆنه‌وه‌و یه‌ك ڕه‌گیان نییه‌، كه‌شوهه‌وایان جیاوازه‌، به‌ژن و باڵاو لق وپۆیان له‌یه‌ك ناچێت، ته‌نانه‌ت مه‌لی سه‌ر چڵه‌كانیان وعاشقه‌كانیشیان هه‌مه‌ڕه‌نگه‌، ئه‌وان له‌هه‌موو شتێكدا له‌یه‌ك جیاوازن، گوێز هه‌رچه‌نده‌ توێكڵه‌كه‌ی ڕه‌قه‌ به‌ڵام دڵی نه‌رمه‌، ئه‌ما خورما هه‌رچه‌نده‌ توێكڵه‌كه‌ی نه‌رمه‌ به‌ڵام دڵی ڕه‌قه‌، دڵیش شوێنی ره‌حم و به‌زه‌ییه‌، كێش دڵی ڕه‌ق بێت بێ به‌زه‌ییه‌و په‌رێزی پیسه‌، ئه‌وه‌ش دڵی نه‌رم بێت ده‌سته‌كانی هێلانه‌ی باڵنده‌ن، وه‌لێ‌ سه‌ره‌ڕای هه‌موو ئه‌مانه‌ دارخورماو دار گوێز ده‌توانن هه‌ریه‌كه‌و له‌شوێنی خۆیان سه‌ربه‌رزو سه‌رفراز بژێن،  ئه‌گه‌ر بای خۆشه‌ویستی و موحیبه‌ت به‌سه‌ر دڵی هه‌موواندا هه‌ڵكات، دڵه‌كانیش وه‌ك ئه‌م خاكه‌ شیرینه‌ به‌ره‌حم وبه‌زه‌یی بن، هه‌روه‌ك چۆن ئه‌و خورماو گوێزی پێ به‌خشین ئه‌مانیش ئاوا خۆشه‌ویستی ودۆستایه‌تی به‌یه‌كتر ببه‌خشن.
