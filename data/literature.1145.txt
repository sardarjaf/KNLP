دوای هه‌ڵدانه‌وه‌و خوێندنه‌وه‌ی سه‌دان كتێبی مێژوویی نه‌متوانی هیچ شتێك له‌باره‌ی مێژووی ئه‌و شاره‌وه‌ بزانم، كه‌ تێدا له‌دایك بووم و ئێخه‌گیری چاره‌نووسه‌كه‌ی بووم، یان چاره‌نووسه‌كه‌ی ئێخه‌گیرم بوو، بۆیه‌ هه‌میشه‌ وه‌ك نهێنییه‌كی گه‌وره‌ی ژیانم لێم ڕوانیوه‌، به‌ڵام دوای ئه‌وه‌ی بۆ یه‌كه‌م جار ئه‌م ڕووداوه‌م بۆ هاوه‌ڵانم گێڕایه‌وه‌، ئیدی له‌وسـاوه‌ وه‌خته‌ لێم ده‌بێته‌ چیرۆكێكی ڕاسته‌قینه‌و ئێستاش وا جارێكی تر، كه‌ بۆ ئێوه‌ی ده‌گیڕمه‌وه‌ بۆخۆم لێم تێكچووه‌، نازانم ئاخۆ سوعبه‌تتان له‌گه‌ڵ ده‌كه‌م، یان هه‌ر ئه‌وه‌تا به‌ ڕاستیمه‌و ده‌مه‌وێ له‌ڕێگه‌ى ئه‌م گێڕانه‌وه‌ سه‌یره‌وه‌ زانیارییه‌كتان پێ ببه‌خشم، كه‌ سه‌رچاوه‌كه‌ی نادیاره‌و یان پێموایه‌ له‌ ئه‌نجامی ئه‌و گه‌ڕان و پشكنینه‌ زۆرو به‌رده‌وامه‌وه‌ به‌رهه‌م هاتووه‌، هه‌ربۆیه‌ مێژوونووسانى ده‌سبه‌تاڵ كه‌یفی خۆیانه‌ چۆنی ئه‌هۆننه‌وه‌و له‌ چ ڕوانگه‌یه‌كه‌وه‌ هه‌ڵسه‌نگاندنی بۆ ده‌كه‌ن، كه‌ ئه‌گه‌ری زۆر ئه‌وه‌یه‌ به‌ هه‌ڵبه‌سـت و وڕێنه‌ی چیرۆكنووسێكی شێتی بزانن، به‌ڵام دڵنیام ئه‌گه‌ر سایكۆلۆژیستێك (به‌ مه‌رجێ ده‌رچووی زانكۆ عه‌نتیكه‌كانی ئه‌م سه‌رده‌مه‌ نه‌بێ نه‌بێ) بیه‌وێ، ده‌توانێ بیكاته‌ باشترین ناونیشان بۆ لێكۆڵینه‌وه‌یه‌كی خۆی، چونكه‌ ئه‌مه‌ پێوه‌ندی به‌ دۆخی ده‌روونی و كۆمه‌ڵایه‌تی ئێسته‌ی شارێكه‌وه‌ هه‌یه‌، ئه‌و شاره‌ی كه‌ هه‌میشه‌ له‌گه‌ڵ ناوهێنانیدا پرسیارێك ئاماده‌گی هه‌یه‌، ئه‌ویش: بۆچی نه‌گبه‌تی و نه‌هامه‌تی ئێخه‌ى خه‌ڵكه‌كه‌ی، به‌ خۆمیشه‌وه‌، گرتووه‌؟ ئه‌و شاره‌ی كه‌ سه‌د ساڵیش لێی دووركه‌وییه‌وه‌، هه‌ركه‌ پێت خسته‌وه‌ سه‌ر ئه‌رزه‌كه‌ى، وا ئه‌زانی هه‌زار سـاڵه‌ لێی ده‌رنه‌چووی!!
بۆ وه‌ڵامی براده‌رانـم، كه‌ زۆرجار له‌گـه‌ڵ ئاوابوونی ڕۆژدا، پیاسه‌كانمان تا قوڵاییه‌كانی خه‌یاڵ درێـژ ده‌كرده‌وه‌، باسی ئه‌و پیاوه‌م دڵڕه‌نجاوه‌م بۆ ده‌كردن، كه‌ دوای ئه‌وه‌ی له‌ به‌رامبه‌ر خیانه‌تی ژنه‌ سه‌مه‌رقه‌ندییه‌ باڵابه‌رزو قه‌شه‌نگه‌كه‌ی چۆك دائه‌داو بڕیار ده‌دا له‌ بازرگانێكی ناوداری ئاوریشمه‌وه‌ ببێته‌ ده‌روێشێكی باباو سه‌رگه‌ردان، به‌وه‌ش سه‌رله‌به‌یانییه‌كی زووی هاوینـی ئه‌و سـاڵه‌ى كه‌ تێدا هارونه‌ ڕه‌شید ده‌سـت ده‌كا به‌ بنیاتنانی مزگه‌وته‌ گه‌وره‌كه‌ى خۆی له‌ ده‌ره‌وه‌ى شورای شـار، هه‌ندێكیش ده‌ڵێن نه‌خێر ڕاسته‌وخۆ دوای ئه‌وه‌ی ژنه‌كه‌ی له‌گه‌ڵ دۆسته‌كه‌یدا، كه‌ دیسانه‌وه‌ ئه‌ویش بازرگانێكی ناوداری ئۆزبه‌كی و هاوڕێی خۆی ده‌بێ، په‌یوه‌ندییه‌ ژێربه‌ژێره‌كه‌یان ئاشكرا ده‌بێ و ڕه‌دووی ده‌كه‌وێ، ئیدی ئه‌میش له‌ سیمای ده‌روێشێكی برینزه‌ده‌ی كۆستكه‌وتوودا له‌ به‌رزاییه‌كانی شاره‌وه‌ به‌خۆی و ماكه‌رێكی بۆزه‌وه‌ بۆی ده‌رده‌چێ و به‌ره‌و نادیار سه‌ری خۆی هه‌ڵ ده‌گرێ، به‌ڵام له‌ دوا ئاوڕدانه‌وه‌یدا له‌ شواری به‌رزی شـاری بوخارا، په‌یمانێك له‌گه‌لَ خۆی و له‌گه‌ڵ خوای خۆیدا ده‌به‌ستێ، په‌یمانێك، كه‌ دواتر ده‌بێته‌ هه‌وێنی چاره‌نووسێكی سه‌قه‌ت و هه‌میشه‌یی، په‌یمانێك، كه‌ ماكه‌ره‌بۆز ده‌ورێكی گه‌وره‌ی تیا ده‌گێڕێ، په‌یمانێك به‌وه‌ى تا به‌رگه‌ى دووریی و مه‌شه‌قه‌تی سه‌فه‌ر بگرێ، له‌ هیچ نه‌پرینگێته‌وه‌و له‌ هیچ شوێنێكیش ڕانه‌وه‌ستێ، به‌پێچه‌وانه‌شه‌وه‌ ئه‌و شوێنه‌ی تیا ته‌سلیمی قه‌ده‌ری خۆی ده‌بێ، ئاوه‌دانی بكاته‌وه‌و بیكاته‌ ئه‌و شوێنه‌ی كه‌ ئیدی خۆیشی به‌ ده‌م عیباده‌ت و نزاكانی په‌شیمانییه‌وه‌ سه‌ری تیا بنێته‌وه‌.
ئاوا و به‌م شێوه‌یه‌، ده‌روێشه‌ بوخاراییه‌ دڵشكاوه‌كه‌، كه‌وته‌ ڕێ و له‌سه‌ر په‌یمانه‌كه‌ى خۆی مایه‌وه‌، بۆیه‌ له‌ هیچ كام له‌و سه‌رزه‌مینه‌ جوان و دڵگیرانه‌ى ده‌هاتنه‌ ڕێی لای نه‌داو بۆ ساتێكیش چییه‌ نه‌حه‌سایه‌وه‌، وه‌ك ئه‌وه‌ی بیه‌وێ به‌و عه‌زابكێشانه‌، شتێك له‌ بارگرانییه‌كانی ژیانی ڕابردووی خۆی و ژیانی پڕ له‌ مه‌ینه‌تی و په‌شیمانی خۆی پاك بكاته‌وه‌، به‌ڵام هه‌میشه‌ تروسكه‌ى ئاواتێكی گه‌وره‌ له‌ ناوه‌وه‌یدا ده‌بریسكایه‌وه‌، وه‌ك ئه‌وه‌ دڵی خۆش بێ به‌ خوای خۆی و به‌و په‌یمانه‌ی پێیداوه‌.. ڕۆشت و ڕۆشت و ڕۆشـت، تا له‌ ده‌وروبه‌رى موغاره‌یلی (كوڕه‌ ڕه‌شه‌كان) ماكه‌ره‌بۆز به‌لادا هات.
شوێنێكی وشـك و برینگ و بێ ئاو، گه‌رم و تۆزاوی، به‌ زنجیره‌ ته‌پۆلكه‌یه‌كی ڕووه‌و قیبله‌ ده‌وره‌ دراو، ئیدی تێگه‌شت كه‌ ده‌بێ له‌وێدا كۆتایی به‌ په‌یمانه‌كه‌ی خۆی بێنێ، له‌گه‌ڵ ئه‌وه‌شدا هه‌ستی به‌و مه‌زڵوومییه‌ته‌ی خۆی ده‌كرد، كه‌ بۆچ له‌و هه‌موو ئاوه‌دانی و شوێنه‌ دڵگیرانه‌ی تا ئێره‌ پیا تێپه‌ڕی بوو، ماكه‌ره‌بۆز له‌ هیچ كامیاندا ڕانه‌وه‌ستاو به‌لادا نه‌هات، بۆچ له‌پڕ ئه‌م چاره‌نووسه‌ شوومه‌ ئێخه‌ى گرت و بۆچ وا خه‌ریكه‌ له‌م شوێنه‌ عاسی و دووره‌ده‌سته‌دا  كۆتایی بێت... 
ده‌روێشه‌ دڵشكاوه‌ بوخاراییه‌كه‌، ئاوا به‌و هه‌ستی پڕ له‌ دڵگرانی و داماوییه‌وه‌، یه‌كه‌م خشـتی به‌ ئاره‌قاوی ناوچاوی خۆی و له‌به‌ر تیشكی تیژی هه‌تاوی هاوینداو له‌ خۆڵـی سوێرو تووڕه‌ی ئه‌و ناوه‌ ده‌گرێته‌وه‌و به‌ ناوی یه‌زدانی به‌خشنده‌وه‌، به‌خشنده‌ی به‌خشندان، ده‌سـت هه‌ڵده‌بڕێ و به‌نییه‌تی بنیاتنانی دوا مه‌نزڵگه‌ی خۆییه‌وه‌ خشته‌كه‌ به‌ نزایه‌كی گلله‌یی ئامێزه‌وه‌، به‌ هێزێكی بێ هێزه‌وه‌، ئه‌دا به‌ ئه‌رزاو ئه‌ڵێ: باری ئیلاهی، هه‌ركه‌س دوای من خشتێكی ترى خسته‌ سه‌ر ئه‌م خشته‌و و ویستی ئه‌م شوێنه‌ی پێ ئاوه‌دان بكاته‌وه‌، خێر له‌ خۆی و له‌ ئاخیره‌تی خۆی نه‌بینێ بۆ خاتری فه‌خری كائینات و سه‌یدی مورسه‌للین... ئامین.
له‌گه‌ڵ ئامینه‌ درێژه‌كه‌ی ئه‌ودا، كه‌ ئیتر ئه‌وه‌ دواجار بوو تاڵه‌ژێیه‌كانی قوڕگی بله‌رنه‌وه‌، ئاسمانیش شریخه‌یه‌كی گه‌وره‌و ناقۆڵای لێوه‌ هات و ئیدی دوای ئه‌مه‌ گێڕانه‌وه‌كه‌ پچڕانێكی گه‌وره‌ی تێ ده‌كه‌وێ، بۆیه‌ كه‌س نازانێ كه‌ى و كێ بووه‌، كه‌ هات و خشتی دووه‌م و سێیه‌م و چواره‌م و....ی خستووه‌ته‌ سه‌ر خشته‌ بوخاراییه‌ شوومه‌كه‌و به‌و شێوه‌یه‌ش نه‌گبه‌ت دوای نه‌گبه‌ت هاتن و خشـت به‌ خشـت، نه‌فره‌ت دوای نه‌فره‌تیان نایه‌ سه‌ر ته‌مه‌ڵی ئه‌و شاره‌ی كه‌ هه‌زاران ساڵ دواتر ئێمه‌ی تێدا له‌دایك ده‌بین و ئێستایشی پێوه‌، ئه‌گه‌ر تۆ خه‌ڵكی ئه‌و شاره‌ بی، ناتوانی له‌ چاره‌نووسی ئه‌و ده‌روێشه‌ بوخاراییه‌ دڵڕه‌نجاوه‌ ده‌رباز بی و بۆ هه‌ركوێ بچی و بۆ هه‌ركوێ هه‌ڵێی، چه‌نێك بته‌وێ دوور لێی ڕاكـه‌ی، چه‌نێك بته‌وێ بۆنی هه‌وای له‌خۆت ده‌ركه‌وی غوبارو تۆزی له‌ خۆت بته‌كێنی، هه‌ر ده‌بێ دواجار گیرای نییه‌تی شووم و دوعای به‌دبه‌ختانه‌ی ئه‌و بی.
