له‌سه‌دا 36ی‌ باشترین جۆری‌ زه‌وی‌ كشتوكاڵی‌ له‌ساڵانی‌ رابردودا خراونه‌ته‌ سه‌ر وه‌زاره‌تی‌ شاره‌وانی‌، له‌كاتێكدا زیاتر له‌ 10 هه‌زار دۆنمیش ته‌لبه‌ند كراون، جگه‌ له‌و زیاده‌ڕۆییانه‌ كراونه‌ته‌ سه‌ر موڵكه‌كانی‌ هه‌ردوو وه‌زاره‌ت.
ته‌رخانكردنی‌ ئه‌و زه‌وییانه‌ به‌مه‌به‌ستی‌ به‌كارهێنانیان له‌لایه‌ن وه‌زاره‌تی‌ شاره‌وانیه‌وه‌ بۆ ماسته‌رپلانی‌ شاره‌كان، كاریگه‌ری‌ خراپی‌ له‌سه‌ر كه‌رتی‌ كشتوكاڵ دروستكردووه‌، بۆیه‌ له‌ئێستادا هه‌وڵه‌كان چڕكراونه‌ته‌وه‌ زه‌ویه‌كان بگه‌ڕێندرانه‌وه‌ وه‌زاره‌تی‌ كشتوكاڵ، ئه‌وه‌ش له‌چوارچێوه‌ی‌ لیژنه‌یه‌كدا كه‌ ئه‌نجومه‌نی‌ وه‌زیرانی‌ هه‌رێمی‌ كوردستان پێكهێناوه‌.
عه‌بدولستار مه‌جید، وه‌زیری‌ كشتوكاڵ‌و سه‌رچاوه‌ ئاوییه‌كانی‌ حكومه‌تی‌ هه‌رێم به‌خه‌ندان وت "جگه‌ له‌پاوان‌و دارستان، شه‌ش ملیۆن دۆنم زه‌وی‌ كشتوكاڵی‌ له‌هه‌رێم هه‌یه‌ به‌به‌راو‌و دێمی، له‌م ژماره‌یه‌ش ملیۆنێك‌و 659 هه‌زار‌و 191 دۆنمیان وه‌زاره‌تی‌ شاره‌وانی‌ بردویه‌تی‌ كه‌ ده‌كاته‌ سه‌دا 36ی زه‌ویه‌ كشتوكاڵیه‌كان".
پرسی‌ ئه‌و زه‌وییه‌ كشتوكاڵییانه‌، وه‌زاره‌ته‌كانی‌ كشتوكاڵ‌و شاره‌وانی‌ به‌سه‌ر دوو بۆچونی‌ جیاوازدا دابه‌شكردووه‌، له‌كاتێكدا عه‌بدلستار مه‌جید ده‌ڵێت به‌بێ‌ ئاگاداری‌ ئه‌وان زه‌وییه‌كان براون بووه‌، لایه‌نی‌ به‌رانبه‌ر ئه‌وه‌ ره‌تده‌كاته‌وه‌.
نه‌ورۆز مه‌ولود، وه‌زیری‌ شاره‌وانی‌‌و گه‌شتوگوزاری‌ حكومه‌تی‌ هه‌رێم به‌خه‌ندانی‌ وت "له‌كاتی‌ كوژاندنه‌وه‌ی‌ زه‌وی‌ كشتوكاڵی‌‌و خستنه‌ سه‌ر وه‌زاره‌تی‌ شاره‌وانی‌ به‌هۆی‌ فراوانبوونی‌ شاره‌كان، لیژنه‌یه‌ك كردویه‌تی‌ كه‌ نوێنه‌ری‌ ئێمه‌و وه‌زاره‌ته‌كانی‌ كشتوكاڵ‌و دارایی‌ تێدابووه‌".
نه‌بوونی‌ هه‌ماهه‌نگی‌ له‌رابردوودا له‌نێوان وه‌زاره‌ته‌ په‌یوه‌ندیداره‌كان بۆ دۆسیه‌ی‌ ئه‌و زه‌ویانه‌ی‌ ره‌گه‌زی‌ كشتوكاڵییان تێدایه‌، به‌گرفتی‌ ئه‌و كێشه‌یه‌ داده‌نرێت.
له‌ئێستادا پرۆتۆكۆلێك له‌ نێوان وه‌زاره‌ته‌كانی‌ شاره‌وانی‌‌و كشتوكاڵ ئیمزا كراوه‌، بۆ گه‌ڕانه‌وه‌ی‌ ئه‌و زه‌ویانه‌ی‌ ره‌گه‌زی‌ كشتوكاڵیان تێدایه‌‌و تائێستا نزیكه‌ 100 هه‌زار دۆنم وه‌رگیراوه‌ته‌وه‌، به‌شێكی‌ تریش له‌ژێر لێكۆڵینه‌وه‌‌و به‌دواداچووندا.
به‌وته‌ی‌ وه‌زیری‌ كشتوكاڵ.
عه‌بدولستار مه‌جید جه‌ختیكرده‌وه‌، كوژاندنه‌وه‌ی‌ زه‌ویه‌ كشتوكاڵیه‌كان‌و خستنه‌ سه‌ر وه‌زاره‌تی‌ شاره‌وانی‌ پێچه‌وانه‌ی‌ یاسا به‌ركاره‌كان‌و بڕیاره‌كانی‌ سه‌رۆكایه‌تی‌ هه‌رێم‌و حكومه‌ت‌و لایه‌نه‌ په‌یوه‌نداره‌كان بوه‌، زیانیشی‌ به‌هه‌رێم گه‌یاندوه‌.
دۆسیه‌ی‌ زیاده‌ڕۆیی‌ بۆ سه‌ر زه‌ویه‌ كشتوكاڵیه‌كان چه‌ند ساڵێكه‌ قسه‌‌و باسی‌ له‌باره‌وه‌ ده‌كرێت، به‌ڵام تائێستا كێشه‌كه‌ چاره‌سه‌ری‌ بنه‌ڕه‌تی‌ بۆ نه‌كراوه‌.
ئه‌نجومه‌نی‌ وه‌زاره‌نی‌ هه‌رێمی‌ كوردستان، به‌سه‌رۆكایه‌تی‌ قوباد تاڵه‌بانی‌ جێگری‌ سه‌رۆكوه‌زیران لیژنه‌یه‌كی‌ بالاَی‌ پێكهێناوه‌، به‌ئه‌ندامیه‌تی‌ هه‌موو وه‌زاره‌ت و ده‌سته‌ په‌یوه‌نداره‌كان ده‌رباره‌ی‌ چه‌ند دۆسییه‌ك له‌نێویاندا زه‌ویه‌ كشتوكاڵیه‌كان، به‌وته‌ی‌ هه‌ردوو وه‌زیری‌ كشتوكاڵ‌و شاره‌وانیش له‌ ئێستادا هه‌ماهه‌نگی له‌نێوانیاندایه‌ هه‌یه‌.
وه‌زیری‌ كشتوكاڵ رونیكرده‌وه‌، كه‌ لیژنه‌ بالاَكه‌ی‌ ئه‌نجومه‌نی‌ وه‌زیران له‌خواره‌وه‌ چه‌ندین لیژنه‌ی‌ بچووكی‌ دروست كردوه‌‌و دوو ئه‌ركیان له‌سه‌ر شانه‌، ئه‌وانیش رێگری‌ له‌ زێده‌ڕۆیی‌‌و گه‌ڕانه‌وه‌ی‌ ئه‌و زه‌وییانه‌ی‌ ره‌گه‌زی‌ كشتوكاڵییان تێداماوه‌.
جگه‌ له‌و زه‌وییانه‌ی‌ خراونه‌ته‌ سه‌ر وه‌زاره‌تی‌ شاره‌وانی‌، له‌چه‌ند ساڵی‌ رابردوو سه‌دان دۆنم زره‌وی‌ كشتوكاڵی‌ له‌لایه‌ن به‌رپرسان‌و هاوڵاتیانیشه‌وه‌، زیاده‌ڕۆییان كراوه‌ته‌سه‌ر یان ته‌لبه‌ند كراون.
عه‌بدولره‌حمان عه‌لی‌، ئه‌ندامی‌ لیژنه‌ی‌ كشتوكاڵ له‌په‌رله‌مانی‌ كوردستان بۆ خه‌ندان ئاشكرایكرد "10 بۆ 11 هه‌زار دۆنم زه‌وی‌ ته‌لبه‌ند كراون، زۆرینه‌شیان زه‌ویه‌ كشتوكاڵیه‌كانن، هه‌رچه‌نده‌ هیچ داتایه‌كی‌ فه‌رمیش له‌وباره‌یه‌وه‌ به‌رده‌ست نیه‌".
به‌پێی‌ ئاماره‌ نافه‌رمیه‌كان، 50 هه‌زار زه‌وی‌ شاره‌وانی‌ زیاده‌ڕۆییان كراوه‌ته‌سه‌ر، له‌و ژماره‌یه‌ش پشكی‌ شێر به‌ر هه‌ولێر‌و دهۆك ده‌كه‌وێت.
ئه‌وه‌ش به‌وته‌ی‌ ئه‌ندامه‌كه‌ی‌ لیژنه‌ی‌ كشتوكاڵ له‌په‌رله‌مانی‌ كوردستان.
ئه‌و پرسه‌ پێشتریش هه‌وڵدراوه‌ له‌په‌رله‌مانی‌ كوردستانه‌وه‌ تاوتوێبكرێت‌و به‌چه‌ند رێكارێك، چاره‌سه‌ری‌ گونجاوی‌ بۆ بدۆزرێته‌وه‌، به‌ڵام له‌ئێستادا‌و به‌هۆی‌ په‌ككه‌وتنی‌ دانیشتنه‌كانی‌ په‌رله‌مانه‌وه‌، نه‌توانراوه‌ چاودێری‌ كاری‌ ئه‌و لیژنه‌یه‌ی‌ ئه‌نجومه‌نی‌ وه‌زیرانیش بكرێت كه‌ كار له‌سه‌ر ئه‌و دۆسیه‌یه‌ ده‌كات.
عه‌بدولره‌حمان عه‌لی‌ ئاماژه‌ی‌ به‌وه‌دا، كه‌ لیژنه‌كه‌ی‌ ئه‌نجومه‌نی‌ وه‌زیران كاتێك پێكهێنراوه‌، ئه‌و چه‌ندینجار داوایكردووه‌ رێگری‌ له‌وه‌بكرێت كه‌ زیاتر له‌ملیۆنێك‌و 500 هه‌زار دۆنم زه‌وی‌ كشتوكاڵی‌ خراوه‌ته‌ سه‌ر وه‌زاره‌تی‌ شاره‌وانی له‌باشترین جۆری‌ زه‌وی كشتوكاڵ.
