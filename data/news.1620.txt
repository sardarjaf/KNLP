به‌وانه‌ی كه‌ داوا له‌م و له‌و ئه‌كه‌ن كه‌ رێز له‌ خواست و مافی خه‌ڵكی كوردستان بگرن، ئه‌ڵێم …
پێم وایه‌ كه‌ ئه‌مریكا و گشت وڵاتانی ئه‌وروپی باش ئه‌‌زانن كه‌ گه‌لی كورد چی ئه‌وێت و ئه‌وه‌ ئێوه‌ن كه ‌نایزانن و ئێوه‌ن كه‌ به‌ له‌ده‌ستدانی متمانه‌تان بوونه‌ته‌ هۆی زڕاندنی خه‌و و ئاواتی گه‌له‌كه‌مان.. 
زیاتر له‌ یه‌ك چاره‌كه‌ سه‌ده‌یه‌ كه‌ هه‌وڵ ئه‌ده‌ین به‌ جیهانیانی بسلمێنێن كه‌ ئێمه‌ی كوردیش هیچمان له‌ گه‌لانی تر كه‌متر نیه‌ و ئه‌توانین حوكمی خۆمان بكه‌ین .. گه‌ر بیرتان مابێت سه‌ره‌تا ناحه‌زانمان ئه‌یانوت كه‌ ( كورد ناتوانێت حوكمی خۆی بكات)… ئه‌وه‌ بوو كه‌ ئێمه‌ش وه‌ك ووڵاتانی پێشكه‌وتوو حكومڕانیمان به‌ پڕۆسه‌یكی دیموكراسیانه‌ و دڵخۆشكه‌ر ده‌ست پێ كرد، به‌ڵام داخه‌كه‌م خۆشیه‌كه‌مان زۆری نه‌خایاند و ئه‌وه‌تا تا وه‌كو ئه‌مڕۆش به‌ده‌ست داگیركه‌ره‌ ناوخۆیه‌كانه‌وه‌ ئه‌نالێنین....
به‌ درێژایی بیست و پێنج ساڵ بیانوومان بۆ خۆمان و دنیا هێنایه‌وه‌، جار وتمان ئه‌زموونمان له‌ حكومڕانیدا نیه‌، جار وتمان حكومه‌ته‌كه‌مان ساوایه‌ و هه‌ندێك جاریش بۆ ئه‌وه‌ی كه‌ حه‌یای خۆمان لای بێگانه‌ نه‌به‌ین به‌هارمان ئه‌هێنایه‌ چۆكان!!
و ئه‌مه‌ی كه‌ ئه‌مڕۆش ئه‌یبینین زاده‌ی هه‌ر هه‌مان ئه‌و سیاسه‌ت و بۆچوونانه‌ی دوێنێن.. من لێره‌یا نامه‌وێت كه‌ باس له‌ دوێنی بكه‌م .. ئه‌وه‌ی گرنگه‌ ئه‌مڕۆیه‌.. ئایا ئه‌مڕۆ توانیومانه‌ ئه‌و ئامانجه‌ بپێكین و جیاوازتر له‌ داگیركه‌رانمان حوكمی خۆمان بكه‌ین ؟ ئایا ئه‌و نان و ئازادییه‌مان به‌ده‌ست هێناوه‌‌ كه‌ له‌ ئاستی ئه‌و هه‌موو قوربانیدانه‌‌ی دوێنێمانا بووبێت؟ ئایا ئه‌توانین بڵێین كه‌ په‌رله‌مانمان هه‌یه‌؟ ئایا یاسا سه‌روه‌ره‌؟ ئایا یه‌كسانی و دادگه‌ری كۆمه‌ڵایه‌تیمان هه‌یه‌؟ ئایا و هه‌زارو یه‌ك ئایای تر؟؟ 
چۆن ئه‌كرێت به‌م سیستمه‌ی كه‌ ئه‌مڕۆ له‌ هه‌رێمه‌كه‌مانا به‌رقه‌راره‌ و كاری پێ ئه‌كرێت بڵێین دیموكراسی؟ خۆ ئه‌گه‌ر خۆشمان له‌ ڕوومان نه‌یت باسی بكه‌ین ئه‌وا هه‌موو سه‌فیرو كونسوڵخانه‌ و شانده‌ بیانیه‌كانی كه‌ دێنه‌ هه‌رێم درك به‌و راستیه‌ ئه‌كه‌ن و له‌كه‌س شاراوه‌ نیه‌ كه شه‌ڕی ئه‌مڕۆ شه‌ڕی مانه‌وه‌یه‌ له‌سه‌ر كورسی نه‌ك شه‌ڕی ده‌سته‌به‌ر كردنی نان و ئازادی و یه‌كسانی و دادپه‌روه‌ری!!
گه‌ر به‌ ناو سه‌ركرده‌كانمان له‌به‌رپرسیارێتی ڕاناكه‌ن با بفه‌رموون هه‌ر له‌ سبه‌ینێ وه‌ به‌ بێ مه‌رج دۆخی په‌رله‌مان ئاسایی بكه‌نه‌وه‌ .. گه‌ر شه‌ڕی ئه‌مڕۆ له‌ پێناو كورسیدا نیه‌ بفه‌رموون با رێگه‌ بده‌ن شێوازی هه‌ڵبژاردنی سه‌رۆكی هه‌رێم و ماوه‌ و ده‌سه‌ڵاته‌كانی به‌ یاسا و سیستم بكرێت.. گه‌ر به‌ نیازی چاك كردنی بژێوی خه‌ڵكن ده‌ با بفه‌رموون ده‌سه‌ڵاتی دارایی و ئیداری بده‌نه‌ پارێزگاكان تا بزانین بۆ نمونه‌ چه‌مچه‌ماڵێك كه‌ سه‌رچاوه‌ی خێرو بێری ئه‌م هه‌رێمه‌یه‌ هه‌ر بێ نان و كاره‌با ئه‌بێت؟ گه‌ر ئه‌یانه‌وێت یه‌كسانی بچه‌سپێنن ده‌با تاپۆ كردنی زه‌وی و خانو له‌ سه‌ر كه‌سوكاری سه‌ربه‌رزی شه‌هیدانی سلێمانیش هه‌ر وه‌ك هه‌ولێر به‌ خۆڕایی بكه‌ن.. و با مووچه‌ی پێشمه‌رگه‌ و شه‌هیدانه‌ی هه‌موو لایه‌ك وه‌ك یه‌ك لێبكه‌ن.. گه‌ر به‌ ته‌نگ بوژاندنه‌وه‌ی كشتو كاڵه‌وه‌ن ده‌با بفه‌رموون داكۆكی له‌ چڵه‌ گه‌نمێكی جوتیارێكی كورد بكه‌ن و گه‌ر به‌راستی ئه‌‌یانه‌وێت پێمان بڵێن كه‌ پاره‌ نیه‌ بۆ مووچه‌ی پارێزه‌رانی خاك و گه‌له‌كه‌مان با تێمان بگه‌یێنن كه‌ چۆن‌ و له‌ كوێوه‌‌ مووچه‌ی ئه‌و هه‌زاران كه‌سه‌ی كه‌ ئه‌‌كرێنه‌ زێره‌ڤان و ئه‌و هه‌موو حه‌شد و فه‌وج و لیوا علوجه‌ دابین ئه‌كه‌ن ؟ پێم وایه‌ باشتره‌ كه‌ پرسیارو داواكاری له‌وه‌ زیاتر نه‌كه‌م نه‌با ئه‌وه‌نده‌یتر سه‌ریان لێ بشێوێنم.. و هه‌ر ئه‌وه‌ ئه‌ڵێم كه‌ ئه‌مانه‌ ته‌نیا به‌شێكن له‌ داواكارییه‌ ڕه‌واكانی گه‌له‌كه‌مان و فه‌رموون جێبه‌جێی بكه‌ن.. و رێز له‌ خواست و مافی خه‌ڵكی كوردستان بگرن…
