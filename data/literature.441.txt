له‌هۆڵی ترانزێتی فڕۆکه‌خانه‌ی فڕانکۆرت دانیشیتبووم له‌پاش سه‌فه‌رێکی دوورو درێژ زۆر ماندوو بووم ، له‌کاتی چاوه‌ڕونیه‌کی پڕوکێنه‌ردا  که‌مێك چاوم لێك ناو سه‌رخه‌وم شکاند و جانتا بچکۆله‌که‌شم هه‌ر له‌باوه‌شما بوو، کاتێ چاوم کرده‌وه‌ له‌نێوانی نیوه‌ به‌ئاگاو نیوه‌ نوستوودا بووم ، به‌دیتنی پیاوێکی ڕیشنی که‌ ڕیشێکی چڕو پڕو ڕه‌ش و ده‌مو چاوێکی ڕه‌قه‌ڵه‌ له‌ته‌نیشتمه‌وه‌ سه‌رسام بووم، له‌ولا ئه‌م ڕیش پانه‌شه‌وه‌ چادرێکی بێ په‌نجه‌ره‌ دانیشتبوو ، چه‌ندان زه‌رفی بازاڕه‌کانی ناو فڕۆکه‌خانه‌ی  فڕانکفۆرتیان پێ بوو ،کاتێ به‌ته‌واوی له‌وسه‌خه‌وه‌ هه‌ستام زانیم خاوه‌ن ئه‌م ڕیشه‌ پانه ‌ته‌نها زه‌لامێکه‌ وجلی ئه‌فغانی له‌به‌ردایه‌ ، دارێکی بچووکی به‌قه‌باره‌ی قه‌ڵه‌مێك بچووکتری به‌ده‌سته‌وه‌ بوو زوو زووش له‌ده‌می ڕا ئه‌کرد، بۆنی ده‌میشی له‌بۆنی ده‌می یه‌کێ ئه‌چوو که‌تازه‌ له‌خه‌و هه‌ڵسابێ و مشکێکی له‌ده‌ما تۆپی بێ ، ئه‌وی له‌ناو چادره‌ بێ په‌نجه‌ره‌که‌شا بوو زانیم ئه‌بێ ژنه‌که‌ی بێت ، که‌ له‌ته‌نیشتی یه‌وه‌ زۆر بێ ده‌سه‌ڵاتانه‌ دانیشتبوو ،وه‌ سمتیشی له‌کورسیه‌که‌وه‌ ده‌رچووبوو ، چونکه‌ به‌ته‌واوی جێی ئه‌و سمته‌ زله‌ی له‌سه‌ر نه‌ئه‌بۆوه‌  ، کابرای ڕیش پانیش به‌لای چه‌پ وڕاستی خۆیدا له‌ئافره‌ته‌ بیانی یه‌کانی ئه‌وێی هه‌ڵئه‌ڕوانی وبوێری نه‌ئه‌دان(ئه‌وانه‌ی که‌له‌وقرچه‌ی گه‌رمای یولیۆیه‌دا له‌جله‌کانیان هاتبوونه‌ ده‌ر) وه‌ زۆر به‌ تامه‌زرۆییه‌وه‌ سه‌یری گۆشتی سپی (ناوه‌ڕاست)ی ئه‌کرد ،له‌به‌ر خۆشی یه‌وه‌ بۆڵه‌ی ئه‌هات گوێم لێی  بوو ئه‌یووت "ئه‌سته‌غفیره‌ڵڵا" هه‌رچه‌ند ئه‌سته‌غفیر کردن له‌سه‌یرکردنی گۆشته‌ سپی یه‌ (ناوه‌ڕاسته‌کان)  مه‌نعی نه‌ئه‌کرد،که‌زانی من سه‌یری ئه‌که‌م چاوی ئه‌گواسته‌وه‌ لایه‌کی تر، هه‌رکه‌سه‌ری به‌رزئه‌کرده‌وه‌ بۆ سه‌یرکردنی (قه‌یماغه)‌ بیانی یه‌کان من مۆڕه‌یه‌کم لێ ئه‌کرد یانی( ئیتر به‌سه‌ حاجی) ،جارو باریش که‌سه‌یری چادره‌که‌ی ته‌نیشتی ئه‌کردو سه‌یری گۆشتی سپی (ناوه‌ڕاست)ی ئه‌کرد له‌دڵی خۆیدا ئه‌یووت " سه‌یر ئه‌و شانس ونه‌وچه‌وانه‌ " ، بینیم چادره‌که‌ به‌ری کرد به‌لایه‌وه‌و شتێکی چرپاند به‌گوێیدا ، سه‌رم سوڕما که‌به‌ده‌نگێکی به‌رزو به‌زمانی عه‌ره‌بی وبه‌شێوه‌ زمانی میسری پێی ووت:
جارێکی تریش ڕووی له‌پیاوه‌که‌ وه‌رگێڕاو دیسانه‌وه‌ چرپاندی یه‌وه‌، نه‌خێر ئیشه‌که‌ زۆر په‌له‌یه‌، بینیم که‌زۆر بێ ئارامانه‌ تێ ی خوڕی و پێی ووت:
من ئاوده‌س‌ته‌که‌م به‌ده‌ست ئیشاره‌ داو ئه‌ویش به‌ژنه‌که‌ی ووت، ژنه‌که‌شی هه‌ستا و زۆر به‌په‌له‌ ڕۆیشت.
سوپاسی کردم ولێی پرسیم:
- پێم ووت : ئه‌ی باشه‌ خۆ ئه‌و هه‌ر له‌سه‌ر زه‌وی داده‌نیشت وله‌ژیانیا له‌سه‌ر کورسی دانه‌نیشتوه‌ ،ئه‌بینم تۆ له‌سه‌ر کورسی دانیشتووی .
- ئێ باشه‌ خۆ پێغه‌مبه‌ر له‌ژیانیا سواری فڕۆکه‌ی لۆفتانزا نه‌بوه‌ ،ته‌نها سواری ئه‌سپ یان حووشتر ئه‌بوو، خۆزگه‌ تۆش(له‌په‌یڕه‌وی ئه‌ودا) لێره‌ ده‌رچیت وله‌سه‌ر شه‌قامه‌که‌ی به‌رده‌م فڕۆکه‌خانه‌ یه‌که‌م حووشتر که‌بینیت ، بیگریت و یه‌کسه‌ر بۆ کۆبنهاگن ، ئه‌ترسم پێم بڵێیته‌وه‌ هه‌ر شته‌و پێداویست خۆی .
_ كه‌واته‌ سیواك سوننه‌ته‌ و فڵچه‌و ده‌رمانی دان سوننه‌ت نین ، ئه‌ی فڕۆکه‌ سوننه‌ته‌ یان نا ؟ ئه‌ی ئه‌بێ مۆبایڵ سوننه‌ت بێ ، که‌مێ له‌وه‌ پێش بینیمیت به‌مۆبایڵ قسه‌ت ئه‌کرد .
پێ ئه‌چو ته‌نگه‌ نه‌فه‌س بوو بێ له‌و بێنه‌ وبه‌رده‌یه‌دا ،زۆریش ترسام که‌له‌هه‌ر کاتێکدا بێت بته‌قێته‌وه‌، به‌ڵام دڵخۆشی خۆم به‌وه‌ ئه‌دایه‌وه‌ که‌ئێمه‌ له‌ساڵۆنی ترانزێت داین پاش ئه‌و هه‌موو پشکنینانه‌ ، جگه‌ له‌وه‌ی ئه‌فسه‌رێکی ئه‌ڵمانی لێمانه‌وه‌ زۆر نزیك بوو جارو بار سه‌رێکم بۆ ڕائه‌وه‌شان ومه‌رحابام لێ ئه‌کرد بۆ ئه‌وه‌ی چاوی لێم بێ ئه‌گه‌ر شتێكی خراپ لێم ڕوو بدات !!
ئه‌ی جه‌نابت وادیاره‌ بؤ گه‌شت وگوزار ئه‌چن بۆ دانیمارك له‌گه‌ڵ مه‌دامدا ( که‌ ئه‌و کاته‌ له‌ئاوده‌‌ست گه‌ڕابۆوه‌ وهیچیشم لێ دیار نه‌بوو که‌ئیسراحه‌تی کردوه‌ یان نا چونکه‌ ده‌مو چاوی دیار نه‌بوو).
- باشه‌ کاتێ له‌نه‌خۆشخانه‌یه‌کیت له‌دانیمارك وله‌کاتی ڕاهێناندا ئه‌گه‌ر داوات لێ کرا که‌ ئافره‌تان بپشکنی یان که‌به‌شه‌و خه‌فه‌ر ئه‌بیت وله‌گه‌ڵ سسته‌ره‌کاندا به‌ته‌نها له‌ژوورێکدا ئه‌بن ، چۆن هه‌ڵسو که‌وت ئه‌که‌ی له‌و کاته‌دا ؟
- باشه‌ بۆ دانیمارکت هه‌ڵبژارد ، باشتر نه‌بوو دکتۆراکه‌ت له‌ووڵاتی خۆت ته‌واو بکردایه‌ ، وابزانم باشتر بوو بۆ خۆشت و بۆ مه‌دامیش .
- تۆ له‌وه‌ ئه‌چێ کابرایه‌کی عیلمانی بی و قسه‌کردن بێ سوودبێ‌ له‌گه‌ڵ تۆ، پڕوپاگه‌ندی زایۆنیست وئه‌مریکا مێشکی بۆگه‌ن کردوویت .
- قه‌ینا ، هێنده‌ی مێشکه‌ بۆگه‌نه‌‌که‌م له‌گه‌ڵم بکه‌ ، ئه‌چی بۆ خوێندن یان جیهاد ، ئه‌ی پاشان تۆ له‌سه‌ر حسابی کێ ئه‌خوێنیت ؟
- من به‌زه‌ماله‌ له‌سه‌ر حسابی حکومه‌تی دانیمارکی ئه‌چم ،وه‌کو پێشم ووتی من له‌ئه‌ساسه‌وه‌ بۆ جیهاد ئه‌چم بۆ ئه‌وێ .
-داوا له‌ میلله‌تی دانیمارکی کافر ئه‌که‌م بۆ هاتنه‌ ناو دینی ئیسلامه‌وه‌ ،چونکه‌ دانیمارك ماڵی شه‌ڕه‌ هه‌تا ئه‌بن به‌ ئیسلام ئه‌وسا ئه‌به‌ێته‌ ماڵی ئاشتی .
- یانی بۆ باج کۆکردنه‌وه‌ ئه‌چی بۆ دانیمارك ؟ !!
ئه‌ی باشه‌ وا ووتمان باجت کۆکرده‌وه‌ ئه‌یده‌یته‌ کێ ؟
- یانی ئه‌وان پاره‌ی زه‌ماله‌که‌ت  بۆ ئه‌ده‌ن و میوانداری خۆشت و مه‌دامیش ئه‌که‌ن ، ئاوا پاداشتی چاکه‌یان ئه‌ده‌یته‌وه‌ ؟
- بێ گوومان موجاهیدی زۆری تیایه‌ له‌م دنیایه‌دا و هه‌موو دنیا له‌گه‌ڵمه‌ وهه‌ریه‌کێکمان به‌هه‌زاره‌ ، نابینی دنیا چۆن له‌ برا شه‌هیده‌کانمان ئه‌ترسێت  ئه‌وانه‌ی که‌له‌ڕیزا ڕاوه‌ستاون چاوه‌ڕیی شه‌هید بوونن ،به‌ ئیزنی خوا کۆتایی هه‌موو کافره‌کانی ڕۆژئاوا به‌ده‌سه‌تی خۆمان ئه‌بێت وه‌ لیوای ئیسلام له‌هه‌موو جێ یه‌کدا بڵاو ئه‌که‌ینه‌وه‌، ڕۆژئاوا ژیانیان خۆش ئه‌وێ و له‌مردن ئه‌ترسن ، به‌ڵام ئێمه‌ به‌خێرهاتنی شه‌هید بوون ئه‌که‌ین و له‌مردن له‌ڕێی خوادا ناسڵه‌مینه‌وه‌ ،هیچ سوپایه‌کش سوپای ئیمانداره‌کان که‌له‌خوا ئه‌ترسن ناشکێنێت  ،کاتێ هه‌موو دنیا ئه‌بنه‌ ئیسلام ئه‌وکاته‌ ئه‌بێته‌ کۆتایی جیهاد وه‌ کۆتایی ئه‌وه‌ی ناویان ناوه‌ " تیرۆر" .
- باشه‌ تۆ کۆمه‌ڵگه‌لێ ئایه‌تی ترت بیر چوو که‌ئه‌ڵێت : دوعا خوازی ڕێی خودا به‌ به‌ حیکمه‌ت ویامۆژگاری چاك ، ڕق له‌دیندا نیه ،ئێوه‌ دینی خۆتان و من دینی خۆم ،ئه‌گه‌ر دڵت پڕ بوو له‌ڕق وکینه‌ ده‌ورو به‌ره‌که‌شت هه‌مان شت پڕ ئه‌بن ، تۆ ئه‌وانه‌ی خۆشت ئه‌وێن ناهێنیته‌  سه‌ر ڕێی ڕاست به‌ڵام خودا خۆی ئه‌وه‌ی خۆشی بوێت ئه‌یخات ، تاکۆتایی ئه‌م ئایاتانه‌ .
