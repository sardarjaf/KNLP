له‌ دووه‌م ساڵڕۆژى داگيركردنى شارى مووسڵ له‌لايه‌ن چه‌كدارانى داعشه‌وه‌، راوێژكارى ئه‌نجوومه‌نى ئاسايشى هه‌رێمى كوردستان له‌ تويتێكدا رايگه‌ياند "بۆ زاڵبوون به‌سه‌ر داعشدا پێشمه‌رگه‌ پێويستى به‌ چه‌كى زياتره‌".
له‌ 9ى حوزه‌يران/ يۆنيۆى 2014دا چه‌كدارانى داعش له‌ سوورياوه‌ په‌ڕينه‌وه‌ بۆ عێراق و له‌ ژێر ناوى شه‌ڕى تێكشكاندنى سنووره‌كانى سايكس پيكۆدا گه‌يشتنه‌ ده‌روازه‌كانى مووسڵ و بۆ رۆژى دواتر توانيان مووسڵ داگير بكه‌ن و به‌ره‌و ناوچه‌كانى ديكه‌ى سوننه‌ نشينى عێراق په‌لبهاوێژن.
سه‌باره‌ت به‌ ده‌ركه‌وتن و گه‌شه‌ و بڵاوبوونه‌وه‌ى ئه‌و گرووپه‌ چه‌كدارييه‌، راوێژكارى ئه‌نجوومه‌نى ئاسايشى هه‌رێمى كوردستان له‌ لاپه‌ڕه‌ى خۆى له‌ تۆڕى كۆمه‌ڵايه‌تى تويته‌ر به‌ زمانى ئينگليزى نووسيويه‌تى "داعش گرووپێك بوو له‌ عێراق سه‌ريان هه‌ڵدا، له‌ سووريا گه‌شه‌ى كرد و پاشان ته‌شه‌نه‌يان سه‌ند و گه‌يشتنه‌ رۆژهه‌ڵاتى ناوه‌ڕاست و باكوورى ئه‌فريقيا و به‌وه‌ش بوونه‌ هه‌ڕه‌شه‌ بۆ سه‌ر جيهان".
سه‌باره‌ت به‌ هێزى پێشمه‌رگه‌ و پێويستييه‌كانى پێشمه‌رگه‌ بۆ به‌ره‌نگاربوونه‌وه‌ى مه‌ترسييه‌كانى داعش و كۆتايهێنان به‌و مه‌ترسييانه‌ى داعش له‌سه‌ر ناوچه‌كه‌ و جيهان درووستيان كردووه‌، مه‌سروور بارزانى له‌ تويتێكى ديكه‌دا نووسيويه‌تى "پێشمه‌رگه‌كانى ئێمه‌ له‌ ئاستى به‌رگريكردن له‌ سه‌نگه‌ره‌كانيان چه‌كيان پێيه‌، به‌ڵام له‌ ئاستى زاڵبوونى يه‌كجاره‌كى به‌سه‌ر دوژمندا پێويستيان به‌ چه‌كى زياتره"‌.
ئێستا لەرێی ئینتەر نێتەوە سزای سەرپێچی هاتوچۆی هاوڵاتیان بڵاو دەكرێتەوەو، هاوڵاتیان لە نرخی سزاكان ناڕازین
  تەنیا رێگە بۆ چارەسەركردنی كێشەكانی عێراق فیدرالیزمە هەقە ئەمەپێش پێكهێنانی حكومەتی داهاتوو بەرنامەی هەموو لایەك بێت
  نێچیرڤان بارزانی بۆ گوڵان: لەمڕۆ بەدوواوە ئامانجی پارتی ئەوەیە چۆن بتوانێت زیاتر خزمەتی خەڵكی كوردستان بكات
  عەدنان كەریم لە دیمانەیەكی تایبەتی گۆڤاری گوڵاندا: داوا لە پەرلەمانی كوردستان دەكەم لە دەركردنی یاسایەك بۆ پاراستنی فەرهەنگی كوردی پەلە بكات
  سەرۆك كۆماری عێراق بۆ گوڵان: هاوپەیمانی پارتی و یەكێتی هاوپەیمانییەكی موقەدەسە و لەسەرووی هەموو شتێكەوەیە
  عەدنان موفتی ((سەرۆكی پێشووی پەرلەمانی كوردستان)) بۆ گوڵان: لەدەستووری هەرێمدا سیستمی حكومڕانی پەرلەمانیەو بەپێی یاساش دەبێت راپرسی لەسەر دەستوور بكرێت
  مەهاباد قەرەداغی بەرپرسی پەیمانگەی یەكێتی ئافرەتانی كوردستان بۆ پرسە كۆمەڵایەتییەكان: پێویستە ئافرەت كاتێك بەشداری فیعلی سیاسەت دەكات، خۆی بۆ ئەو پۆستە تەرخان بكات
