وته‌بێژی‌ سه‌ركردایه‌تی‌ ئۆپه‌راسیۆنی‌ هاوبه‌ش، رایده‌گه‌یه‌نێت پلانی‌ ئازادكردنی‌ نه‌ینه‌وا له‌ده‌ست داعش به‌منزیكانه‌ كۆتایی دێت‌و له‌كاتی‌ خۆیدا ئه‌گه‌ر پێویست بكات ناوی‌ ئه‌و هێزانه‌ بڵاوده‌كه‌نه‌وه‌ كه‌ به‌شداریی‌ له ‌رزگاركردنی‌ نه‌ینه‌وا ده‌كه‌ن.
عه‌مید یه‌حیا زوبه‌یدی كه‌ له‌هه‌مانكاتدا وته‌بێژی سه‌ركردایه‌تی‌ ئۆپه‌راسیۆنی‌ هاوبه‌ش‌و وه‌زاره‌تی‌ به‌رگریی عیراقیشه‌، له‌م چاوپێكه‌وتنه‌یدا له‌گه‌ڵ خه‌ندان كه‌ له‌ڕێگه‌ی‌ ته‌له‌فۆنه‌وه‌ له‌سه‌ر كۆنتڕۆڵكردنه‌وه‌ی پارێزگای نه‌ینه‌وا له‌گه‌ڵیدا ئه‌نجامدراوه‌، باسله‌وه‌شده‌كات به‌پێی‌ ئه‌و پلانه‌ی‌ داڕێژراوه‌ بۆ رزگاركردنی‌ نه‌ینه‌وا ته‌نها هێزه‌ عیراقییه‌كان به‌شداری‌ ده‌كه‌ن‌و هیچ هێزێكی‌ دیكه‌ی‌ ده‌ره‌وه‌ی‌ عیراق به‌شداریی‌ ئۆپه‌راسیۆنه‌كه‌ ناكات.
سه‌ركردایه‌تی ئۆپه‌راسیۆنی هاوبه‌ش ساڵی 2012 دامه‌زراوه‌و باره‌گای سه‌ره‌كی له ‌شاری به‌غدایه‌و لقی له‌ هه‌ولێر هه‌یه‌و له‌لایه‌ن حكومه‌تی عیراق سه‌ركردایه‌تی ده‌كرێت، جگه‌ له ‌وه‌زاره‌تی به‌رگری عیراقی‌و هێزه‌ سه‌ربازییه‌كانی عیراق، سوپای ئه‌مریكاو هه‌رێمی كوردستانیش ئه‌ندامن له‌ سه‌ركردایه‌تی ئۆپه‌راسیۆنی هاوبه‌ش.
ئامانج له‌ دروستكردنی سه‌ركردایه‌تی ئۆپه‌راسیۆنی هاوبه‌ش له‌ ساڵی 2012، بۆ ئیداره‌دان‌و رێگریكردن بوو له‌ رووبه‌ڕووبوونه‌وه‌ی سه‌ربازیی له‌نێوان سوپای عیراق له‌گه‌ڵ هێزه‌كانی پێشمه‌رگه‌، له‌ناوچه‌ دابڕاوه‌كانی هه‌رێمی كوردستان.
دوای هاتنی داعش بۆ عیراق‌و ده‌ستبه‌سه‌رداگرتنی موسڵ له‌ حوزه‌یرانی 2014، ئێستا سه‌ركردایه‌تی ئۆپه‌راسیۆنی هاوبه‌ش ئه‌ركی رووبه‌ڕووبوونه‌وه‌ی داعش‌و هاوبه‌شی له‌گه‌ڵ پێشمه‌رگه‌ بۆ رزگاركردنی نه‌ینه‌واو ناوچه‌كانی دیكه‌ی له‌ ئه‌ستۆدایه‌ له‌ڕێگه‌ی داڕشتنی پلانی سه‌ربازیی هاوبه‌ش به‌هاوكاری ئه‌مریكییه‌كان.
خه‌ندان له ‌سه‌رچاوه‌یه‌كی‌ ئاگادار له‌ هێزی‌ هاوپه‌یمانانه‌وه‌ زانیویه‌تی‌، وه‌زاره‌تی‌ به‌رگری‌ عیراق و هاوپه‌یمانان سه‌رقاڵی‌ پلاندانن بۆ شه‌ڕی‌ رزگاركردنی‌ موسڵ‌و دواتر پلانه‌كه‌ راده‌ستی‌ هه‌رێم ده‌كرێت بۆ ئه‌وه‌ی‌ بۆچوونی‌ خۆیانی‌ له‌سه‌ر بڵێن.
سه‌ره‌ڕای پیاهه‌ڵدان‌و به‌باش وه‌سفكردنی هێزی‌ پێشمه‌رگه‌ له‌شه‌ڕی‌ دژ به‌ داعش‌و جه‌ختكردنه‌وه‌ له‌سه‌رئه‌وه‌ی‌ كه‌ پێشمه‌رگه‌ به‌شێكه‌ له ‌سیستمی به‌رگری عیراق، هاوكات به‌بێ ئاماژه‌دان به‌ڕۆڵی‌ سه‌ركردایه‌تیی‌ كورد‌و پێشمه‌رگه‌ له‌داڕشتنی‌ پلانی ئازادكردنی نه‌ینه‌واو ئه‌گه‌ری‌ هه‌بوونی‌ تێبینیی‌ له‌سه‌ر پلانه‌كه‌، یه‌حیا زوبه‌یدی، وتی ‌"پێشمه‌رگه‌ وه‌ك به‌شێك له‌ مه‌نزومه‌ی‌ به‌رگری‌ عیراق له ‌رزگاركردنی‌ موسڵ رۆڵیان ده‌بێت‌و له‌كاتی‌ خۆیدا ئاگاداریان ده‌كه‌ینه‌وه‌ له ‌ناوه‌ڕۆكی‌ پلانه‌كه‌".
له‌ئێستادا حكومه‌تی عیراق به‌مه‌به‌ستی ئازادكردنی موسڵ هێزێكی زۆری له ‌قه‌زای مه‌خمور جێگیركردووه‌و به‌ره‌به‌یانی 24-3-2016 ئه‌و هێزه‌ی‌ سوپای‌ عیراق به‌ پاڵپشتی‌ فڕۆكه‌ جه‌نگییه‌كانی‌ هاوپه‌یمانان هێرشی بۆسه‌ر رێكخراوی‌ داعش ده‌ستپێكرد، ئه‌وه‌ش له‌چوارچێوه‌ی هه‌وڵه‌كان بۆ گرتنه‌وه‌ی موسڵ.
وته‌بێژی‌ سه‌ركردایه‌تی‌ ئۆپه‌راسیۆنی‌ هاوبه‌ش، شه‌ڕه‌كه‌ی مه‌خمور وه‌ك قۆناغی‌ سه‌ره‌تایی‌ پلانی ئازادكردنی نه‌ینه‌وا ناوده‌بات، به‌ڵام ئه‌وه‌شی خسته‌ڕوو كه‌ پێده‌چێت پلانه‌كه‌ هه‌ندێك شتی دیكه‌شی بخرێته‌سه‌ر.
"دوای‌ ئه‌وه‌ی‌ ناوچه‌كانی‌ سه‌ر به‌ نه‌ینه‌وا له‌داعش پاكده‌كرێنه‌وه‌و هه‌موو رێگاكانی‌ گه‌یشتنی‌ هاوكاریش بۆ داعش له ‌موسڵ ده‌بڕدرێت، ئه‌وكات رزگاركردنی‌ موسڵ ده‌ستپێده‌كات" عه‌مید یه‌حیا واده‌ڵێت.
دوای هاتنی داعش بۆ عیراق، چه‌ندین هێزی چه‌كداری جیاواز دروستبوون، دیارترینیان حه‌شدی شه‌عبییه‌ كه‌ هێزی چه‌كداری شیعه‌ن‌و به‌فتوای عه‌لی سیستانی دامه‌زران‌و دواتر له‌لایه‌ن حكومه‌تی عیراقه‌وه‌ به‌ یاساییكران، هاوكات حه‌شدی نیشتمانیش هێزێكی سوننه‌ی چه‌كدارن كه‌ ئه‌سیل نوجه‌یفی پارێزگاری پێشووی نه‌ینه‌وا سه‌ركردایه‌تییان ده‌كات.
جگه‌له‌وانه‌ش چه‌ندین هێزی دیكه‌ی عه‌شایه‌ری‌و كه‌مایه‌تییه‌كانی عیراق خۆیان چه‌كداركردووه‌‌و ته‌نانه‌ت به‌شێك له‌سیاسییه‌كانیش گرووپی چه‌كداری تایبه‌ت به‌خۆیان دروستكردووه‌.
وته‌بێژی‌ سه‌ركردایه‌تی‌ ئۆپه‌راسیۆنی‌ هاوبه‌ش جه‌خت له‌سه‌ر به‌شداریی‌ حه‌شدی‌ شه‌عبی‌ ده‌كات له‌ئازادكردنی نه‌ینه‌وا به‌بێ باسكردن له‌ به‌شداریكردنی حه‌شدی نیشتمانی ‌و ده‌ڵێت "حه‌شدی‌ شه‌عبی‌ رۆڵی‌ گه‌وره‌ی‌ هه‌بووه‌ له‌شه‌ڕی‌ داعش هاوشانی‌ پێشمه‌رگه‌ كه‌ رۆڵی‌ گه‌وره‌یان هه‌بووه‌".
له‌باره‌ی‌ به‌شداریكردنی هێزه‌كانی حه‌شدی نیشتمانی، یه‌حیا زوبه‌یدی ته‌ئكیدی ده‌كرده‌وه‌ كه‌ ته‌نها ئه‌و هێزانه‌ی‌ سه‌ر به‌حكومه‌تی‌ عیراقن به‌شداریی‌ ده‌كه‌ن وه‌ك ئاماژه‌یه‌ك بۆئه‌وه‌ی كه‌ حه‌شدی‌ نیشتمانی سه‌ربه‌ حكومه‌ت نییه‌، وتیشی "به‌شداریكردنی‌ هه‌ر هێزێك له‌ رزگاركردنی‌ نه‌ینه‌وا به‌بڕیاری‌ حه‌یده‌ر عه‌بادی‌ سه‌رۆك وه‌زیرانی‌ عیراق ده‌بێت".
ئه‌وه‌ش له‌كاتێكدایه‌ پێشتر ئه‌سیل نوجێفی، سه‌رپه‌رشتیارو كه‌سی یه‌كه‌می چه‌كدارانی حه‌شدی نیشتمانی له‌ چاوپێكه‌وتنێكیدا له‌گه‌ڵ خه‌ندان دژایه‌تی بۆ به‌شداریكردنی حه‌شدی شه‌عبی له‌پڕۆسه‌ی ئازادكردنی موسڵ دووپاتكرده‌وه‌ باسیله‌وه‌كرد ئه‌گه‌ر حه‌شدی شه‌عبی به‌شداری ئه‌و پڕۆسه‌یه‌ بكات، خه‌ڵكی موسڵ هاوكاریی ئازادكردنی شاره‌كه‌یان ناكه‌ن.
ئه‌سیل نوجێفی، وتی: "ئێمه‌ تاكه‌ هێزی لۆكاڵین كه‌ هه‌موولایه‌ك پێویستی پێمانه‌ بۆ ئۆپه‌راسیۆنی رزگاركردنی موسڵ‌و توانامان هه‌یه‌ بۆ جێگیربوونه‌وه‌ له‌ناو موسڵ‌و هێزو لایه‌نگرمان له‌ناو شاره‌كه‌ هه‌یه‌".
دوای‌ رزگاركردنی‌ قه‌زای شنگال‌و به‌شێكی زۆری ناوچه‌كانی‌ ده‌وروبه‌ری‌، له‌ئێستادا جگه‌ له‌ پێشمه‌رگه‌ چه‌ند هێزێكی دیكه‌ش له‌و ناوچانه‌ بوونیان هه‌یه‌، له‌باره‌ی به‌شداریكردنی ئه‌و هێزانه‌ له‌ ئازادكردنی نه‌ینه‌وا، وته‌بێژه‌كه‌ی ئۆپه‌راسیۆنی هاوبه‌ش ئاماژه‌ی‌ به‌وه‌كرد هه‌واڵگریی‌ عیراق سه‌رقاڵی‌ ئاماده‌كردنی‌ راپۆرتێكن له‌سه‌ر هه‌بوونی‌ هه‌موو هێزه‌كان له‌ناوچه‌كاندا.
"ئه‌وان راپۆرتی‌ خۆیان ئاڕاسته‌ی‌ حه‌یده‌ر عه‌بادی‌ سه‌رۆك وه‌زیران‌و فه‌رمانده‌ی‌ گشتی‌ هێزه‌ چه‌كداره‌كان ده‌كه‌ن، بڕیاری‌ به‌شداریكردنی‌ هه‌ر هێزێكی‌ دیكه‌ جگه‌ له‌هێزه‌ عیراقییه‌كان، ده‌بێت به‌بڕیاری‌ عه‌بادی‌ بێت" یه‌حیا زوبه‌یدی وای وت.
هێزه‌كانی هاوپه‌یمانان به‌شدارن له‌ داڕشتنی پلانی ئازادكردنی نه‌ینه‌واو وه‌ك هێزی‌ ئاسمانیش به‌شداریی‌ كارای پڕۆسه‌كه‌ ده‌كه‌ن، هاوكات سوپای‌ توركیا له‌ئێستادا له ‌ناوچه‌ی به‌عشیقه‌ سه‌ربازگه‌ی كردووەته‌وه‌ بۆ مه‌شقپێكردنی‌ هێزه‌كانی حه‌شدی نیشتمانی‌و له ‌18ی‌ ئه‌م مانگه‌شدا سوپای توركیا بۆ یه‌كه‌مجار به‌شداری‌ شه‌ڕی‌ دژ به‌ داعشی‌ كرد له‌گه‌ڵ ئه‌و هێزانه‌.
له‌باره‌ی به‌شداریكردنی هێزی پیاده‌ی بیانی له‌ئازادكردنی نه‌ینه‌وا، عه‌مید یه‌حیا ده‌ڵێت "ته‌نها هێزی‌ پیاده‌ی‌ عیراق به‌شدارییده‌كه‌ن له‌ رزگاركردنی‌ نه‌ینه‌وا‌و هیچ هێزێكی‌ دیكه‌ی‌ پیاده‌ی‌ بیانی‌ به‌شداری‌ ناكه‌ن، به‌ سوپای‌ توركیاشه‌وه‌، ته‌نانه‌ت پێویستمان به‌هیچ هێزێكی‌ ده‌ره‌كیی‌ نییه‌".
وته‌بێژه‌كه‌ی سه‌ركردایه‌تی‌ ئۆپه‌راسیۆنی‌ هاوبه‌ش له‌باره‌ی‌ به‌شداریكردنی‌ سوپای‌ كۆماری‌ ئیسلامی‌ ئێران، باسیله‌وه‌كرد هه‌رچه‌نده‌ ئه‌و وڵاته‌ هاوكاری‌ عیراقی‌ كردووه‌ له‌سه‌ره‌تای‌ شه‌ڕی‌ دژ به‌ داعش، به‌ڵام یه‌كه‌ سه‌ربازییه‌كانی‌ به‌شداری‌ شه‌ڕیان نه‌كردووه‌‌و وتی "ئه‌وه‌ هیچ راست نییه‌ كه‌ سوپای‌ كۆماری‌ ئیسلامی‌ ئێران به‌شداری‌ شه‌ڕیان دژ به‌ داعش كردبێت له‌عیراق، به‌شداری‌ رزگاركردنی‌ نه‌ینه‌واش ناكه‌ن".
