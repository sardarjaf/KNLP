      هه‌وو ئه‌و خه‌یاڵانه‌ی كه‌ به‌ناو مێشكمدا دێن و ده‌چن ختووكه‌ی ئه‌وه‌م به‌ دڵ دا دێنن كه‌ ئه‌وانیش چیتر له‌ نێو مێشكدا ڕه‌وگیر نه‌كه‌م و بۆ كه‌مێكیش بێ‌ بیانخه‌مه‌وه‌ نێو بازاڕی چیرۆكه‌وه‌، ره‌نگه‌ تۆی خوێنه‌ر ئه‌وه‌ت به‌خه‌یاڵ دا بێ‌ كه‌ چیرۆكه‌كه‌ كلك و گوێی له‌ كوێن و بارسته‌و بارته‌قای چ ڕووداوێك له‌خۆده‌گرن.
من نازانم ئه‌و نامه‌یه‌ به‌ تۆ ده‌گات یان نا، ده‌مه‌وێ‌ تۆ بكه‌مه‌ پاڵه‌وان، ئاخر جه‌نابتان ببوورن جگه‌ له‌ ئێوه‌ كێ‌ ی تر له‌و وڵاته‌ پاڵه‌وانه‌؟!!
((هه‌ر ئه‌وه‌نده‌ خۆم ده‌بینمه‌وه‌ له‌سه‌ر بێِشكه‌یه‌كی جوانی سه‌رنج راكێشی بۆن خۆشی پڕ خه‌ناوكه‌و مۆری و هێنده‌ی بڵێی مێخه‌ك و شتی گرانبه‌های به‌سه‌ر سه‌ر و چاومدا شۆڕكرابوونه‌وه‌.
ئه‌و ڕۆژه‌ ئه‌وان نه‌یانده‌زانی من جیا له‌وه‌ی مێژوو نیشانیداوه‌، ره‌قه‌ڵه‌یه‌كی كه‌لله‌ پووتی تازه‌ خولقاوی بێ‌ به‌هره‌ بم، ده‌مزانی ئه‌وانه‌ چ جه‌ژنێك ده‌گێڕن، دایكم دێته‌ ژووره‌وه‌ ئۆقره‌ ناگرێ‌ تا خۆی له‌ باوه‌شم نه‌گرێ‌، كچه‌ جوانه‌كێلكه‌ره‌كان كه‌ به‌ كڕنووش بردن و چه‌مانه‌وه‌وه‌ دێنه‌ به‌رده‌م دایكم و، به‌ ڕیز ده‌وه‌ستن نایه‌ڵن دایكم ماندوو و نیگه‌ران بێ‌.))
ئه‌و هه‌فته‌یه‌ سه‌رانسه‌ری وڵات ئاهه‌نگی خۆشی گێڕاوه‌، به‌ سه‌دان ژن كچه‌كانی خۆیان پێشكه‌شی دیوه‌خانی كۆشكی خاوه‌ن شكۆ كردووه‌، سه‌دان مه‌لاو شێخ مه‌ولوودیان خوێندۆته‌وه‌، زۆر نووسه‌ری لات و پات شیعریان بۆت نووسی، ئه‌و رۆژه‌ قه‌ڵه‌می چه‌ند نووسه‌ر له‌ ده‌ستی تۆ نرا، بڕوانامه‌ی به‌رز به‌رزی فه‌خریت بۆ نووسرا، چه‌ند دكتۆر و مامۆستای زانكۆ‌و پرۆفیسۆری ئه‌و وڵاته‌ هه‌بوون كۆنفرانسی زانستییان له‌سه‌ر ده‌وری جه‌نابتان له‌ داهاتووی كۆمه‌ڵگه‌ به‌ست، هه‌ر له‌و ماوه‌یه‌، چه‌ندین نامه‌ی زانستی ماسته‌ر و دكتۆرا پێشكه‌شی تۆ كران، چه‌ند لێكۆڵینه‌وه‌ی زانستانه‌ له‌سه‌ر گرنگی خولقانتان و زه‌رووره‌تی بوونی ئێوه‌ له‌ وڵات نووسران.
پێده‌چێ‌ هه‌ر به‌و بۆنه‌یه‌ چه‌ندین كه‌سی لات و هه‌ژار و بێ‌ ده‌ره‌تان پاروێك نان‌و ئاوێكی فێنك و سارد شه‌ربه‌تێكی دڵڕفێنیان نۆش كردبێت، تۆ ئێستا بیر له‌وه‌ ده‌كه‌یته‌وه‌.
منی چیرۆكنووسیش ئه‌و رۆژه‌ له‌به‌رده‌م شه‌قامی سه‌ره‌كی شاردا به‌ چاوی خۆم بینیم، 10551 به‌ندی ئازاد بوون، چه‌ندین پڕۆژه‌ به‌ گوژمه‌كانی سه‌د ملیۆن و حه‌فسه‌د و حه‌فتا هه‌زار و نۆسه‌د و په‌نجاو پێنج دینارو نیو كرانه‌وه‌، ئه‌و مانگه‌ هه‌موو فه‌رمانبه‌رانی ده‌وڵه‌ت یه‌كی سه‌ت و بیست و یه‌كیان هاته‌ سه‌ر مووچه‌كه‌ و یه‌كی سوپاسنامه‌یه‌كیان به‌ دیاری باوكت وه‌رگرت، چه‌ند هه‌ژاری گوێ‌ به‌تۆز و چه‌ند شوانكاره‌و خاوه‌ن مه‌ڕو ماڵات، ئاژه‌ڵ و مه‌ڕو ماڵاتیان كرده‌ قوربانی و خێره‌كانیان به‌شیه‌وه‌و، خوێنه‌كانیان له‌ بن دیوارێك له‌ دیواره‌كانی كۆشكتان ڕشته‌وه‌، ئه‌و هه‌فته‌یه‌ ده‌تگوت ئێره‌ وڵاتێكی دیكه‌و جه‌نابتان فریشته‌یه‌كی سه‌ر هه‌ساره‌یه‌كی ترن، كانیاوه‌كان وه‌ك بڵێی بروسك لێی دابن و ته‌قینه‌وه‌، وڵات هه‌مووی سه‌راپا ببوو به‌ پارچه‌یه‌ك لافیته‌ی جوان، هه‌موو سه‌ره‌ كۆڵانێك به‌ ڕووی لافیته‌ ڕه‌نگاو ڕه‌نگ و تابلۆ جوانه‌كان پێده‌كه‌نین.
سه‌ربازه‌كان له‌ دوورترین قوڕنه‌ی وڵات و له‌ سه‌ر تاته‌ به‌ردێكی  بچووكدا سه‌ریان له‌ كڕنووش هه‌ڵنه‌گرت، ئاخر ته‌قه‌ی خۆشی هێنده‌ به‌ ئاسمانی وڵاتدا چوو گوێی ئاسمانی كه‌ڕ و زمانی له‌گۆ خستبوو.
ئێستا باوكت دێته‌ ژوور، یه‌ك پێ‌ هه‌موو ئه‌و پڕو پیره‌ژن و كچه‌ جوانكیله‌كانی كۆشك له‌ ده‌ورت یه‌ك ڕاست دێنه‌ سه‌ر پێ‌، پێده‌چێ‌ باوكت یه‌كه‌م نێرینه‌ بێ‌ له‌م هه‌فته‌یه‌ ڕووی تۆ ببینێ‌، ئه‌و پیاوه‌ ده‌یه‌وێ‌ بۆ یه‌كه‌م جار چاو له‌ چاوت ببڕێ‌، سه‌ری خۆی نزیك ده‌كاته‌وه‌، تۆ ده‌گریی.
له‌ په‌رده‌كانی زه‌ین و مێشكمه‌وه‌، چاوم لێیه‌، خۆت له‌یه‌ك ده‌پچڕیه‌وه‌، چاوی ئه‌و نزیك بۆته‌وه‌ له‌ چاوت، ده‌گریی، خۆت ڕاده‌پچكوڕی، ده‌سته‌ پووچه‌ بچووكه‌كه‌ت، بۆ لای خۆت به‌رز ده‌كه‌یه‌وه‌، له‌ چاوی ئه‌و ده‌چه‌قێ‌، به‌ری چاوی سوور هه‌ڵده‌گه‌ڕێ‌، تووڕه‌ ده‌بێ‌، پێده‌چێ‌ تۆ كه‌یفت به‌ جانگی ئه‌و نه‌یێ‌، خاوه‌ن شكۆ تووڕه‌ ده‌بێ‌ ڕاست ده‌بێته‌وه‌، په‌رده‌ی ژووره‌كه‌ی بۆ هه‌ڵده‌ده‌نه‌وه‌، ڕقی خۆی ده‌خواته‌وه‌، ده‌چێته‌ ده‌ره‌وه‌.
دوای بیست به‌هار له‌ چاوه‌ڕوانی منداڵ، ئه‌وه‌ی به‌ خه‌یاڵ دا دێ‌ تۆو هه‌موو منداڵانی هاو ته‌مه‌نت فووی نه‌مانیان لێ‌ بكات، ئه‌م خه‌یاڵه‌ لای دایكت ده‌دركێنێ‌، دایكت وه‌ك بڵێی بزانی باوكت بۆنی نه‌بوونی ئه‌و په‌یوه‌ستییه‌ به‌زه‌ییه‌ عاده‌تییه‌ی نێوان باوك و كوڕ بكات، تێده‌گات كه‌ له‌وانه‌یه‌ مه‌رگی ڕۆڵه‌ تاقانه‌كه‌ی ببینێ‌، هه‌موو كاره‌كه‌رو فه‌رمانبه‌رانی كۆشك خڕ ده‌كاته‌وه‌، هه‌موویان له‌بن لچی دامه‌ن و پێڵاوی پادشا ده‌نووزێنه‌وه‌، ئه‌و كاته‌ی ئه‌وه‌مانزانی من یه‌كسه‌ر ده‌ستم بۆ قه‌ڵه‌مه‌كه‌م برد ئه‌م نامه‌یه‌م بۆت نووسی:
     ئێستا ئه‌و نامه‌یه‌ ده‌نووسم نازانم ناوی چیت لێنراوه‌، ده‌زانم تۆ گه‌وره‌یت، نه‌ وه‌كو باوكتی نه‌ش وه‌كو دایكت، ده‌شێ‌ تۆ نه‌ ببیته‌ (ئه‌و نووسه‌ره‌ گوێ‌ به‌تۆزو نه‌ئه‌و مه‌لا و شێخ و مزێور ومه‌ولوود خوێن و ئه‌و هونه‌رمه‌ند و رۆژنامه‌ نووس و زوڕناژه‌ن و چاوه‌ش و ئه‌و سه‌ربازی یاوه‌ری ته‌قه‌ی خۆشی) یانه‌ی بۆ له‌ دایك بوونی تۆ دڵی باوكتیان خۆش كرد و مامه‌حه‌مه‌ییان كرد، له‌وانه‌یه‌ پێش من بۆت بگێڕنه‌وه‌.
هه‌ناسه‌كانی رۆژگار بخۆیته‌وه‌ یان نا هه‌ر ده‌بێ‌ كانیاوی رقی باوه‌ت كۆر كه‌یته‌وه‌، تۆیه‌ك نیگاكانت به‌رووی ئه‌ودا بسه‌نگرێنه‌وه‌، مه‌ودای تاریكییه‌كان ده‌سڕیته‌وه‌، ئه‌رێ‌ له‌ باوه‌شی كۆشكێكدا نه‌بی، هه‌ر بێنازه‌كه‌ی ئێمه‌ی، كێ‌ هه‌یه‌ بۆت له‌بن لچ ‌و لێوی ئه‌ودا سه‌ربنه‌وێنێ‌؟؟!
هێشتا نامه‌كه‌م ته‌واو نه‌كردبوو، به‌زه‌یی پڕ رق‌و قینی باوكت جوڵا، تۆی گه‌وره‌ كرد، له‌ باوه‌شی دیواندا هه‌ڵكوڕمای، له‌ دونیای فیستیڤاڵه‌ گه‌وره‌كاندا ناوت ده‌هات، له‌سه‌ر مێزی فه‌رمانگه‌كاندا وێنه‌یه‌كی چه‌ند ده‌وری تۆ داده‌نرا، ته‌نانه‌ت له‌ نهینی ترین ژواندا بێ‌ تۆ نائاخڤین، وێنه‌ی تۆ به‌شداره‌ له‌زه‌ین‌و بیر‌و بڕیاردان، زه‌بری تۆ بۆته‌ پێویستی، بۆ به‌ره‌كه‌ت جنێوبارانمان ده‌كه‌ی، بۆ خۆشییت رۆحمان ده‌گوشی، تۆی گوڵ ‌و زێڕ‌و، پێویست وه‌كو ئاو‌و هه‌واو هه‌ناسه‌‌و خوێن.
به‌ منداڵی له‌ په‌نجه‌ره‌ی هه‌ناوی چیرۆكمه‌وه‌ هاتیته‌ ژووره‌وه‌، به‌ گه‌وره‌یی له‌ نێو هه‌تاوی رقدا گه‌شه‌ت كرد، باڵات نه‌رمی چیرۆكی به‌زاند‌و له‌نێو شاخی قیندا ته‌مه‌نی باوكت بۆ خۆته‌‌و سه‌ره‌داوی هه‌موو ئاواته‌كانمان له‌ به‌رژه‌وه‌ندی تۆ گرێدراوه‌.
له‌وێنه‌ی زه‌ینمدا دێیته‌ قسه‌، لێم راده‌مێنیت، باوه‌شم پێدا ده‌كه‌یت، من به‌ بێ‌ ده‌نگی سه‌رم وه‌رده‌سوڕێنم، ده‌توانم وه‌كو باوكت تووڕه‌ بم، به‌ڵام زات ناكه‌م، ئێمه‌ له‌به‌رامبه‌ر هه‌موو پێشهاته‌كاندا زات ناكه‌ین، شه‌رم دامان ده‌گرێ‌.
رۆحم زه‌لیل ده‌كه‌ی، بۆم ده‌گری، پێم پێده‌كه‌نی، گاڵته‌م پێده‌كه‌ی، له‌به‌رامبه‌رت دا دۆش داماوم.
تۆ به‌بێ‌ ده‌نگی له‌ خه‌یاڵی چیرۆكم ده‌چیته‌ ده‌ره‌وه‌‌و منیش كۆتایی چیرۆكه‌كه‌م لێره‌وه‌ به‌ خوێنه‌ر سپارد.
