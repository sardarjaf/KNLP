نازانم له‌كه‌یه‌وه‌ سنووری ئه‌و ڕۆژانه‌ ده‌ستنیشانكه‌م، كه‌وه‌ك باڵنده‌یه‌ك، تازه‌ ئاشنای ئاسمانه‌كان بووه‌ و نازانێت سه‌ره‌تا و كۆتایی چ مانایه‌كیان هه‌یه‌... ئه‌و ڕۆژێك یه‌كه‌مین گه‌شتی ده‌ستپێكرد، ته‌واوی ئاسمانه‌كان تووڕه‌یی خۆیان ده‌نارده‌ خواره‌وه‌... ئه‌و ڕۆژانه‌ باڵنده‌كه‌، بێكه‌س، بێئامانج، بێنیشتیمان، له‌به‌ر ڕه‌هێڵه‌ تووڕه‌كاندا ده‌هات و ده‌چوو... باڵنده‌ی به‌دبه‌خت له‌و كاته‌دا یه‌كه‌مین پرسیاری ئه‌وه‌بوو:"به‌ڕاست ژیانی ئه‌ودیو قه‌فه‌ز هه‌ر ئه‌مه‌یه‌"؟ ئه‌و نه‌یده‌زانی باڵنده‌گه‌لێكی دی هه‌ن لانه‌یان هه‌یه‌ و عه‌شق ده‌كه‌ن، خودا ده‌ناسن و سه‌فه‌رده‌كه‌ن، نه‌یده‌زانی شه‌ڕده‌كه‌ن و جه‌ژنده‌كه‌ن... ئه‌و ئه‌وه‌شی نه‌ده‌زانی ئه‌و جێگایه‌ی پیاسه‌ی تێدا ده‌كرد ناوی ئاسمان بوو، ئه‌و ناوی ئه‌ستێره‌ و مانگ و به‌فر و باران و هه‌ساره‌كانیشی نه‌ده‌زانی... ئه‌و كه‌كاتێك بۆیه‌كه‌مجار هه‌ڵفڕی.. یان باشتره‌ بڵێم ئازادبوو، ئێواره‌یه‌كی زستان بوو، من له‌بیرم نییه‌ چ خه‌یاڵێك وایكرد بابم به‌و ئێواره‌ سارد و باراناوییه‌ ئه‌و كه‌ناریه‌ تازه‌پێگه‌یوه‌ ئازادبكات، به‌ڵام ده‌مزانی ئه‌و كاره‌ی له‌خۆڕانییه‌، هه‌ڵبه‌ت هه‌ر له‌خۆوه‌ پێم باش بوو، به‌ڵام ئێواره‌یه‌كی هێندساردبوو گریانم بۆی ده‌هات.
ئاخر باوباپیری ئه‌و باڵنده‌یه‌ی كه‌ بابم ئازادی كرد له‌نێو قه‌فه‌زێكدا ژیابوون، كه‌بابم ساڵانێكی پێشوتر به‌تایبه‌ت له‌ئێرانه‌وه‌ بۆیان هێنابوو... بابم ته‌واوی باڵنده‌كانی تری له‌نێو قه‌فه‌زه‌ كۆن و ده‌ستكرده‌كانی ناوخۆی شاردا به‌خێوده‌كرد، به‌ڵام له‌جوانی و قه‌شه‌نگی ئه‌و جووته‌ كه‌نارییه‌ی كه‌به‌ڕێكه‌وت له‌كابرایه‌كی گوندنشینی كریبوو، هه‌ر له‌خۆوه‌ نه‌یتوانیبوو وه‌ك باڵنده‌كانی دیكه‌ حیسابیان بۆبكات، بۆیه‌ ژیانی بۆ سازكردبوون له‌و قه‌فه‌زه‌ نموونه‌ییه‌دا... بابم بۆیده‌گێڕامه‌وه‌ كه‌ گوایه‌ باڵنده‌كانی دیكه‌ له‌تاو ئێره‌ییدا تووشی نه‌خۆشی هاتبوون كه‌كاتێك بینیبوویان دوو كه‌ناری نوێ‌ له‌وه‌ها قه‌فه‌زێكی قه‌شه‌نگدابژین، ئه‌وانیش به‌و هه‌موو ته‌مه‌نه‌وه‌ پاره‌یان بۆ به‌رهه‌م هێناوه‌... ئه‌و دوو كه‌ناریه‌ قه‌شه‌نگه‌ی كه‌ بابم له‌كابرا گوندییه‌كه‌ی كڕیبوو داپیر و باپیری ئه‌و كه‌ناریه‌بوون كه‌ بابم له‌ ئێواره‌ سارده‌كه‌دا ئازادی كرد.
چیرۆكی ئه‌و كه‌ناریه‌ی وایكرد بابم ئازادی كات، من نایزانم، یان باشتره‌ بڵێم ئه‌و هۆكاره‌ی بووه‌ هۆی ئازادبوونی باڵنده‌كه‌ نازانم، به‌ڵام به‌بڕوای خۆم وایده‌بینم بابیشم نازانێت بۆ له‌قه‌فه‌ز ده‌ریهێنا، هه‌ڵبه‌ت هه‌ركات ده‌كه‌وتمه‌ باسی چیرۆكی ئازادبوونی باڵنده‌كه‌، یه‌كسه‌ر بابم بێده‌نگ ده‌بوو، وه‌ك ئه‌وه‌ی هه‌ستبكات ئازادبوونی ئه‌و باڵنده‌یه‌ په‌یوه‌ندی به‌ خودی باڵنده‌كه‌وه‌هه‌یه‌، یان ئه‌وه‌ی وه‌ك جێبه‌جێكردنی په‌یامێكی ئاسمانی وه‌هابوو بۆی... به‌ڵام به‌هه‌رحاڵ ئێستاش تێنه‌گه‌یشتم له‌بابم بۆ ئه‌و كه‌ناریه‌ی له‌وه‌ها ڕۆژێكدا به‌ردا، هه‌ڵبه‌ت هۆكاری به‌ربوونی ئه‌و باڵنده‌یه‌ له‌سه‌ره‌تاوه‌ بۆم گرنگ نه‌بوو، تاكاتێك كه‌ بابم كۆچی دوایی كرد، ئاخر مردنی بابم چیرۆكێكی خۆی هه‌یه‌ و كه‌م تازۆر په‌یوه‌ندی به‌ ئازادبوونی ئه‌و باڵنده‌وه‌ هه‌یه‌.
من له‌گه‌ڵ یه‌كه‌م چركه‌دا كه‌ كه‌ناریه‌كه‌ هه‌ڵفڕی ڕووه‌و ئاسمان، ئاگام لێبوو، سه‌یرمده‌كرد تا ونبوو، ده‌مبینی نه‌یده‌توانی به‌هاوسه‌نگی بفڕێت، له‌به‌ر ڕێژنه‌ی باران و پاڵه‌په‌ستۆی بایه‌كاندا ئه‌ملاولای ده‌كرد، ده‌هاته‌وه‌ خوار و به‌ هێزێكه‌وه‌ به‌ره‌و ئاسمان ده‌ڕۆیشته‌وه‌، ئه‌وه‌ یه‌كه‌م ساتبوو كه‌ هه‌ستمكرد خێرایی دڵم زیادی ده‌كرد، هه‌رله‌خۆوه‌ ده‌ترسام ئه‌و كه‌ناریه‌ كه‌ زۆركات له‌گه‌ڵیدا ده‌كه‌وتمه‌ گفتوگۆ و باسی خۆمم بۆده‌كرد ئاینده‌یم باش نه‌ده‌بینی، نه‌مده‌زانی بۆ دڵته‌نگ بووم بۆی، بیرمده‌كرده‌وه‌ ڕه‌نگه‌ هی ئه‌وه‌بێت كه‌ زۆر دواوم له‌گه‌ڵیدا، یاخود هی ئه‌وه‌ی كه‌ من كچ بووم... ئاخر دایكم زۆركات ده‌یگووت"ئافره‌تان دڵیان ناسكتره‌ له‌ پیاوان"، هه‌رواش بوو، چونكه‌ ئه‌و ڕۆژه‌ی كه‌ناریه‌كه‌ به‌ره‌و ئاسمان فڕی من بۆی گریام، كه‌چی براكه‌م پێده‌كه‌نی، ئه‌و به‌وه‌ ده‌كه‌نی كه‌ كه‌نارییه‌كه‌ نه‌یده‌توانی له‌به‌ر باو باراندا به‌هاوسه‌نگی بفڕێت، ئای چه‌نده‌ توڕه‌بووم لێی، چه‌نده‌ی تر غه‌مم بۆ خوارد، ئه‌و ئێواره‌یه‌ی كه‌ناریه‌كه‌ ونبوو، من تا ته‌واو تاریكی شه‌وداهات له‌ هه‌یوانه‌كه‌مانه‌وه‌ ده‌مڕوانییه‌ ئاسمان، تاكو جارێكی دیكه‌ بیبنمه‌وه‌، به‌ڵام شه‌وداهات و به‌خه‌می ئه‌وه‌وه‌ نووستم.
به‌یانی به‌رله‌وه‌ی هێزگرم له‌خه‌و، هه‌ستمده‌كرد خه‌ونده‌بینم كه‌ كه‌ناریه‌كه‌ گه‌ڕاوه‌ته‌وه‌ و ئه‌وه‌تا ده‌خوێنێت، بۆیه‌ به‌خێرایی هه‌ستامه‌وه‌ و ڕووه‌و حه‌وشه‌كه‌مان ده‌رپه‌ڕیم، كه‌بینیم كه‌ناریه‌كه‌ گه‌ڕاوه‌ته‌وه‌ و له‌سه‌ر دار هه‌ناره‌كه‌ وه‌ستاوه‌... ئه‌ی خوایه‌گیان له‌و خۆشییه‌، یه‌كسه‌ر ده‌نگێكی كچانه‌م لێبه‌رزبوویه‌وه‌ و هه‌ردوو ده‌ستم هێنا به‌ده‌مووچاومدا، له‌وكاته‌دا ده‌مویست به‌ كه‌ناریه‌كه‌ بڵێم ئه‌م شه‌و تۆ له‌كوێ بوویت؟ نازانی چه‌نده‌ له‌بیری تۆدابووم؟ ئه‌ی نازانی دونیایه‌ك خه‌ونی جۆراوجۆم پێته‌وه‌ دی؟ ده‌مویست بڵێم ئێستاش بڕواناكه‌م ئه‌وه‌ تۆبی به‌سه‌ر ئه‌و داره‌وه‌؟ ئاخر وامدانابوو تۆ تازه‌ مردووی... ده‌مویست ئاله‌و هه‌ڵیت و پڵیتانه‌ی پێبڵێم، بابم به‌ جلی خه‌وه‌كانییه‌وه‌ هات و بینی باڵنده‌كه‌ گه‌ڕاوه‌ته‌وه‌ و تێڕاماوه‌، ژیان بۆئه‌وه‌نده‌ سه‌خته‌... به‌ڵام به‌رله‌وه‌ی بابم ئه‌و شتانه‌ له‌چاوی كه‌ناریه‌كه‌دا به‌دیبكات، یه‌كسه‌ر كه‌مێك ڕۆیشته‌ پێشه‌وه‌ و یه‌ك دوو ورده‌ به‌ردی تێگرد تا هه‌ڵفڕی و بۆ دووه‌مجار له‌پێشچاوم ونبووه‌وه‌!!
ته‌ماشاكه‌ تائێستاش له‌خۆم ناگه‌م بۆ چیرۆكی ئه‌و كه‌نارییه‌ به‌دبه‌خته‌ بۆ هه‌موو هه‌ڵچون و داچوونه‌كانی ژیانم ده‌هێنمه‌وه‌... زۆركات ئه‌م پرسیاره‌م له‌خۆم كردووه‌، له‌كاتی خۆشی و ناخۆشی و توڕه‌بوون و هێمنی و گریان و خه‌نیندا، هه‌مووكات یه‌كسه‌ر كه‌وتومه‌ته‌وه‌ بیری ئه‌و كه‌ناریه‌ی كه‌وا خه‌ریكم ئاشناتده‌كه‌م به‌ژیانی، ئێستاش واده‌زانم به‌رله‌وه‌ی باسی ئێستای خۆمت بۆبكه‌م پێویسته‌ بتناسێنم به‌ ژیانی باڵنده‌كه‌، ئاخر تاهه‌نووكه‌ش كه‌ئه‌م پیتانه‌ ده‌نێم به‌یه‌كه‌وه‌ واده‌زانم ژیانی ئه‌و كه‌نارییه‌، ژیانی منه‌ و نموونه‌یه‌كه‌ كه‌خودا ته‌نها به‌منی نیشانداوه‌... من له‌بیرمه‌ كه‌منداڵ بووم زۆركات خاڵم باسی ئینسانه‌ پاكه‌كانی بۆده‌كردین، ئه‌و باسی ئه‌و ڕۆحانه‌ی ده‌كرد كه‌خودا خۆشی ده‌وێن و به‌رده‌وام ئاگای لێیانه‌، به‌رده‌وام ده‌یبردینه‌وه‌ بۆ ئه‌و بابه‌ته‌ گه‌وره‌ و قورسانه‌... من له‌بیرمه‌ به‌خۆمم ده‌وت ئاخر خاڵم زۆر سه‌یره‌، ئه‌ی ئه‌و چۆن ده‌زانێت ئێمه‌ به‌م ته‌مه‌نه‌ منداڵه‌وه‌ له‌و بابه‌ته‌ گرانانه‌ ده‌گه‌ین، به‌ڵام دواتر هه‌رخۆم وه‌ڵامی خۆمم ده‌دایه‌وه‌ و ده‌موت"به‌ڵێ‌ خاڵم بێئه‌ندازه‌ زرنگه‌، چونكه‌ ئه‌و گه‌ر نه‌یزانیبا من و براكانم له‌وه‌ها بابه‌تێك ناگه‌ین، باسی نه‌ده‌كرد.... ئێستا به‌خۆم ده‌ڵێم ڕه‌نگه‌ من له‌و ئینسانانه‌بم كه‌ خاڵم باسی ده‌كرد، ده‌ڵێم ڕه‌نگه‌ خودا منی خۆشبووێت، ڕه‌نگه‌ من خاوه‌نی ڕۆحێكی پاكبم، بۆیه‌ چیرۆكێكی وه‌ك ژیانی ئه‌و كه‌ناریه‌ی پێناساندم، چیرۆكێك كه‌ته‌نها من و كه‌نارییه‌كه‌ ده‌یزانین، ته‌نها من و ئه‌و له‌واتا قوڵه‌كانی گه‌یشتین، ئاخر ڕه‌نگه‌ تۆش ئاشنای ئه‌و چیرۆكه‌ بیت، سه‌رسام بیت له‌وه‌ی ژیانی باڵنده‌یه‌ك بۆ ده‌بێت له‌هه‌موو حاڵه‌ته‌ دانسقه‌كانی ژیانمدا وه‌ك نموونه‌یه‌ك بگونجێت، بۆده‌بێت وه‌ك وه‌ڵامی ته‌واوی پرسیاره‌كان گێڕانه‌وه‌ی چیرۆكی كه‌ناریه‌كه‌ كۆتا هه‌وڵدانبووه‌ و هه‌ر به‌ڕاستیش كۆتا ڕێگه‌چاره‌بووه‌.
ته‌ماشاكه‌، من له‌و كاته‌وه‌ی ده‌توانم ژیانی ئه‌و باڵنده‌یه‌ بگێڕمه‌وه‌، هه‌ستێكی سه‌یرم لادروست بووه‌، هه‌ستی ئه‌وه‌ی گونجاندنی چیرۆكێكی ساده‌ بۆ ته‌واوی چیرۆكه‌كانی ژیانم... هه‌ستی ئه‌وه‌ی كه‌م ئینسان هه‌یه‌ له‌و به‌راوردانه‌ی من بگات... من خۆم ده‌ناسم، مێشكێكی گه‌لێك وردم هه‌یه‌، هه‌میشه‌ وه‌ك كچێكی شه‌رمن له‌په‌راوێزی ڕووداوه‌كانه‌وه‌ زۆرترین كاریگه‌ریم هه‌بووه‌ له‌سه‌ر ژیان، هه‌ڵبه‌ت مه‌به‌ستم ئه‌و ژیانه‌ لاوه‌كیانه‌یه‌ كه‌به‌ته‌نها خودا و هه‌ندێ‌ ئاده‌می دیكه‌ تێده‌گه‌ن لێی، سه‌یركه‌ من كه‌سێكی ساده‌نیم، ده‌شزانم تۆش به‌هه‌مان شێوه‌ی، به‌ڵام كه‌لله‌ڕه‌قییه‌كه‌ت و ئه‌دگاره‌ منداڵانه‌كه‌ت تێگه‌یشتنی تۆیان تێكداوه‌، بۆیه‌ زرنگه‌به‌ و له‌م ئاماژه‌ كۆتاییانه‌ی نه‌ستم بگه‌، بڕوابكه‌ ئاگام له‌حاڵته‌، ده‌زانم شه‌وانه‌ به‌دزی ماڵه‌وه‌تان ده‌گریه‌ی، ئه‌وه‌ش ده‌زانم كه‌ هه‌ندێجار له‌خۆپه‌رستیدا ده‌ته‌وێت بگریه‌یت و له‌خۆشتی بشاریه‌وه‌... به‌ڵام واهه‌سته‌كه‌م تاهه‌نووكه‌ش ده‌رفه‌تێكی باش هه‌یه‌ بۆ ئاسایی بوونه‌وه‌ی نێوانمان، هه‌ڵبه‌ت ئه‌مانه‌ی كه‌ده‌ینووسم له‌ده‌روونێكی ڕ‌و‌وخاو و شكستخواردووه‌وه‌ نیه‌، به‌ڵكو گوزارشتی ڕوخساری هه‌ندێ‌ ڕوی شته‌كانه‌ كه‌ دڵنیام تۆی خۆپه‌رست قه‌ت نه‌تده‌توانی ڕۆژێك له‌ ڕۆژان بیان نووسیت... چوزانم...باسی بكه‌یت، یان هه‌ر نه‌بێت خه‌یاڵی پێوه‌بكه‌یت، له‌گه‌ڵ هه‌موو ئه‌مانه‌شدا به‌بێ‌ ئه‌وه‌ی شه‌رم بكه‌م، به‌بێ‌ ئه‌وه‌ی هه‌ستی غروری دامگرێ‌ ده‌ڵێم گێڕانه‌وه‌ی چیرۆكی كه‌ناریه‌كه‌ ڕووی ڕاسته‌قینه‌ی منه‌ سه‌باره‌ت به‌نێوانمان.
له‌دوای ئه‌وه‌ی بۆ دووه‌مجار كه‌ناریه‌كه‌ هه‌ڵفڕیه‌وه‌ و ونبوو، شتێكی سه‌یر به‌سه‌ر بابمداهات، بابم به‌سه‌ختی نه‌خۆشكه‌وت، له‌ماوه‌ی یه‌ك دوو ڕۆژدا ڕه‌نگی مردووی لێنیشتبوو، هه‌رده‌تووت به‌ته‌واوی واده‌ی مه‌رگی نزیك بۆته‌وه‌... من وه‌ك كچێكی منداڵ به‌ترسێكه‌وه‌ به‌خۆمم ده‌موت"ئاهی ئه‌و كه‌ناریه‌ بابمی والێكرد... ئاخر من ده‌مزانی ئه‌و باڵنده‌ به‌دبه‌خته‌ سه‌رمای بوو، ده‌مزانی بۆیه‌ گه‌ڕابوویه‌وه‌ تا بێته‌وه‌ نێو قه‌فه‌زه‌كه‌ی جارانی، ده‌مزانی ئه‌و تووڕه‌بوو له‌ژیان، تووڕه‌بوو له‌ئازادی، ده‌یویست بێته‌وه‌ لانه‌ كۆنه‌كه‌ی.... به‌ڵام بابم ده‌ریكرد، ئه‌و دڵڕه‌قانه‌ ده‌ریكرد.. من بیركردنه‌وه‌یه‌كی وه‌هام لێكده‌دایه‌وه‌ و دایكم و براكانم دڵته‌نگ ده‌بوون، من نه‌مده‌وێرا باسی وه‌ها شتێك بكه‌م، چونكه‌ وامده‌بینی بابم ئه‌و باڵنده‌یه‌ی له‌به‌رئه‌وه‌ ئازادكرد كه‌ خودا خێری بۆ بنووسێت، به‌ڵام نازانم بۆ وامده‌زانی بابم تاوانباره‌؟!
من هه‌ر له‌وكاته‌دا جارێكی تر كه‌وتمه‌وه‌ بیری ئه‌و قسانه‌ی خاڵم له‌باره‌ی دین و ژیانه‌وه‌ بۆیده‌كردین، ده‌یوت هه‌ركه‌سێك به‌بێ‌ هۆ زیان به‌ گیانداره‌كان، بگه‌یه‌نێت، تاوانباره‌ و خودا لێیخۆشنابێت، ده‌یوت خودا مسقاڵه‌ زه‌رره‌ی لێوننابێت و هه‌ر سزامان ده‌دات، بۆیه‌ منداڵینه‌ ئاگاتان له‌خۆتان بێت، ئازاری گوێدرێژ و سه‌گ و پشیله‌كانی شار نه‌ده‌ن"... ده‌ڵێم ڕه‌نگه‌ ته‌واوی ئه‌و شتانه‌ی له‌ناخمدا لێكم ده‌دایه‌وه‌ له‌سه‌ر بابم هۆكاره‌كه‌ی له‌قسه‌ دینییه‌كانی خاڵمه‌وه‌ سه‌رچاوه‌ی گرتبێت، ره‌نگه‌ ته‌واوی ئه‌وكاتانه‌ی شته‌كانم ده‌هێنایه‌وه‌ سه‌ر یه‌ك رێگا، هه‌ڵه‌بووبم و ته‌نها رێكه‌وت بووبێت، ره‌نگه‌ نه‌خۆشكه‌وتنی بابم هیچ په‌یوه‌ندییه‌كی به‌ئازادكردنی ئه‌و باڵنده‌یه‌وه‌ نه‌بێت، یان ئه‌سڵه‌ن رووداوه‌كانی ئه‌و ماوه‌یه‌، دۆخێك بووبێت كه‌بتوانم ئێستامی تێدا رونكه‌مه‌وه‌...
هه‌رچی بوو نازانم كام لای شته‌كان حه‌قیقه‌ت بوون، به‌ڵام له‌وه‌ دڵنیام كه‌ ده‌بێت تۆی دڵڕه‌ق ئاشنابیت به‌م چیرۆكه‌وردیلانه‌، تالێره‌وه‌ ته‌ماشای رووداوه‌ هه‌نووكه‌ییه‌كان بكه‌یت، ئاخر من ده‌زانم تۆی خۆپه‌رست، تۆیه‌ك كه‌ پشتتكردۆته‌ راستییه‌كان و به‌وێنه‌ی شاعیرێكی عه‌به‌سی خۆتت به‌ندكردووه‌ و له‌هیچ شتێك ناپرسیت، ته‌نها و ته‌نها خه‌ریكی خواردنه‌وه‌ی كهولیت و سیگار كێشانی، تۆیه‌ك ده‌سته‌وه‌ستان له‌ئاست ته‌واوی رابردووت و ته‌سلیمی ئه‌و واقیعه‌ تڕۆهاته‌ بوویت، به‌ڕاستی ئه‌سته‌مه‌ چاوه‌ڕوانی شتێكت لێبكرێت، بۆیه‌ ئه‌م نامه‌یه‌ ده‌نووسم تا لێره‌وه‌ هێدی هێدی بتناسێنم به‌ڕاستییه‌كان... ڕه‌نگه‌ تا ئه‌م چركه‌یه‌ش به‌هه‌مان خۆپه‌رستی جارانه‌وه‌ بێیته‌وه‌هۆش و بپرسیت"ئاخۆ كێ‌ ئه‌م هه‌واڵانه‌ی منی پێدابێت"؟ یان ئه‌وه‌ی شه‌رم داتگرێت و هێنده‌ی تر وه‌ك كه‌سێكی خه‌جاڵه‌تبار توڕه‌بت له‌خۆت و له‌ من و ته‌نانه‌ت له‌ خوشكه‌كه‌شت، چونكه‌ ئه‌و به‌ته‌واوی هه‌موو ژیانی تۆی بۆ باسكردم، هه‌ڵبه‌ت ئه‌و ژیانه‌ی ده‌ڵێم كه‌له‌ دوای من گوزه‌راندوته‌، به‌ڵام عاقڵبه‌ و هێوربه‌، چونكه‌ واخه‌ریكه‌ هه‌نگاوه‌ له‌رزۆكه‌كانت هه‌ڵتده‌گرن به‌ره‌و ژیانێك كه‌ ته‌نها خۆپه‌رستیت خولقاندوویه‌تی، ئه‌گینا شته‌كان به‌وشێوه‌یه‌نین... خوشكه‌كه‌ت ده‌یوت دانت به‌هه‌موو شته‌كاندا ناوه‌، ده‌یوت په‌شیمانی و ئێستا شه‌ڕ به‌دار و به‌رد ده‌فرۆشێ‌... ده‌یوت هه‌ندێ شه‌و به‌ زرمه‌ی ئه‌و له‌خه‌و ڕاده‌چڵه‌كین، سه‌یرده‌كه‌ین به‌ بۆكس هاتووه‌ به‌دیواری ژووره‌كه‌یدا و ده‌گریه‌ت... به‌ڕاستی مایه‌ی پێكه‌نینه‌؟!
تۆ له‌م ماوه‌یه‌دا به‌و چه‌شنه‌ گۆڕابیت، مایه‌ی گاڵته‌جاڕییه‌ كه‌ بوویت به‌ پاڵه‌وانی دیوار و ئاسن و شووشه‌... ته‌ماشاكه‌ من به‌هیچ شێوه‌یه‌ك سه‌رزه‌نشتت ناكه‌م بۆئه‌وه‌ی كه‌بڵێم بۆواتكرد؟ یان بۆواتنه‌كرد؟ به‌ڵام ده‌مه‌وێت له‌م كاته‌ هه‌ستیاره‌ی ته‌مه‌نتدا هاوكاریت بكه‌م، وه‌ك خوشكه‌كه‌ت وتی "فریایكه‌وه‌"، هه‌رچه‌نده‌ حاڵی منیش له‌تۆ باشترنه‌بووه‌، به‌ڵام من ته‌نها دڵم به‌وه‌ خۆشه‌ كه‌ ئه‌گه‌ر ته‌واوی شته‌كان تێكبچن و بژاكێن به‌یه‌كدا هه‌ر ڕۆژێك ده‌بێت حه‌قیقه‌ت جیابكرێته‌وه‌.
باڵنده‌كه‌ كه‌ بۆ دووه‌مجار ونبوویه‌وه‌، و دواتر بابم به‌ خه‌ستی نه‌خۆشكه‌وت، ئیدی نه‌گه‌ڕایه‌وه‌ تا ئه‌و ڕۆژه‌ی كه‌بابم دواین فرمێسكی بۆ ڕشت و بۆ دواینجار ماڵئاوایی لێكردین... بابم مرد، به‌ڕاستی ئه‌م وشه‌یه‌ له‌ناخه‌وه‌ ده‌یگریاندم، به‌خۆمم ده‌وت"ئیتر خۆ بابم تازه‌ نابینمه‌وه‌"، ده‌موت"یانی بابم به‌ته‌واوی مرد"،"یانی ئیدی تا كۆتایی دونیا نایبینینه‌وه‌؟" ئای له‌و ڕسته‌ بێتامانه‌ی له‌و ماوه‌یه‌دا وێرانیانكردبووم، من وه‌ك كچێكی منداڵ له‌نێوان دووڕییانێكی ته‌مومژاویدابووم... هه‌ندێجار له‌ناخه‌وه‌ جنێوم به‌و باڵنده‌یه‌ ده‌دا... ده‌موت"ده‌ك تۆش بمری واده‌م بوویته‌ مایه‌ی مردن بۆ بابم"، كه‌بیرم له‌ قسه‌كانی خاڵم ده‌كرده‌وه‌، حه‌زمده‌كرد، ئه‌و باڵنده‌یه‌ بگرم و هه‌نجن هه‌نجنی بكه‌م، به‌ڵام له‌لایه‌كی تره‌وه‌ وه‌ك ئه‌وه‌ی ده‌نگێك له‌ ناخه‌وه‌ پێمبڵێت شته‌كان وه‌ها نین كه‌تۆ بیریان لێده‌كه‌یته‌وه‌، به‌ڵكو ته‌واوی ئه‌و ڕوداوانه‌ ڕێكه‌وتێكی ناخۆش بووه‌، ڕێكه‌وتێكی ناخۆش... بۆیه‌ منیش كه‌مێك ڕێگه‌م به‌و ده‌نگه‌ی ناخم ده‌دا و بیرمده‌كرده‌وه‌ و دواتر هه‌ر خۆم دڵته‌نگ ده‌بووم بۆ باڵنده‌كه‌... سه‌یركه‌ جیاوازی من و تۆ تائه‌م چركه‌یه‌ش  له‌وه‌دایه‌ كه‌تۆ هه‌رگیز نه‌تتوانیوه‌ گوێبگری بۆ ده‌وروبه‌ر، هه‌رگیز ئاماده‌یت نه‌بووه‌ دان به‌ڕاستییه‌كاندا بنێیت، به‌ڵام من به‌رده‌وام گوێم بۆ ئازاره‌كانی مێشوله‌یه‌كیش گرتووه‌، به‌رده‌وام وابیرمكردۆته‌وه‌ ڕه‌نگه‌ حه‌قیقه‌ت له‌ جێگایه‌كی هێند ساده‌دابێت كه‌ ئێمه‌ی لوتبه‌رز نه‌توانین بیبینین، نه‌توانین سڵاوێكی لێیكه‌ین و هه‌ر هیچ نه‌بێت یه‌ك دوو قسه‌ی له‌گه‌ڵدا دامه‌زرێنین.

كه‌نارییه‌كه‌ و بابم مردن، به‌ڵام من و تۆ نه‌مردووین، كه‌نارییه‌كه‌ و بابم له‌زمانی یه‌كتر زۆرباش نه‌ده‌گه‌یشتن، من و تۆش وه‌ك ئه‌وان، من و تۆ كه‌مێك لێكنه‌گه‌یشتن هه‌یه‌ له‌نێوانمان، ئه‌وانیش هه‌روابوو، كه‌نارییه‌كه‌ و بابم مردن و نه‌یانتوانی له‌یه‌ك بگه‌ن، من و تۆ زیندوین و پێویسته‌ له‌یه‌كبگه‌ین، ئه‌وان كه‌سانێك هه‌بوو كه‌ وڕێنه‌یان به‌ده‌مه‌وه‌ ده‌كردن، من و تۆش كه‌سانێكی له‌و جۆره‌ ده‌ناسین، له‌نێوان كه‌نارییه‌كه‌ و بابمدا كه‌سێك هه‌بوو له‌ هه‌ردووكیان باش تێده‌گه‌یشت، له‌نێوان من و تۆدا كه‌سێكی له‌وجۆره‌ نییه‌ و ده‌بێت خۆمان ئه‌و كه‌سه‌بین... ئه‌وان له‌پاش دووركه‌وتنه‌وه‌ له‌یه‌ك شه‌ڕێكی ساردیان هه‌بوو، من و تۆش هه‌رواین، به‌ڵام ئه‌وه‌تا من ده‌مه‌وێت نه‌یهێڵم... كه‌نارییه‌كه‌ و بابم یه‌كیان ده‌ناسی، خۆشه‌ویستیان بۆیه‌ك هه‌بوو، به‌ڵام بابم ئازادكردنی ئه‌وی به‌باش ده‌زانی و كه‌نارییه‌كه‌ به‌پێچه‌وانه‌وه‌، من و تۆش یه‌ك ده‌ناسین و یه‌كمان خۆشده‌وێت و هه‌ردووكمان جیابوونه‌وه‌مان پێخۆشنه‌بوو، به‌ڵام فریوی كه‌لله‌ڕه‌قی و خۆپه‌رستی یه‌كترمان خوارد، من و تۆ هه‌ردووكمان ده‌مانه‌وێت بگه‌ینه‌وه‌ به‌یه‌ك و تائه‌م چركه‌ساته‌ش كارێكمان بۆنه‌كردووه‌؟ تۆ خه‌ریكی مه‌ی خواردنه‌وه‌ و سیگاركێشان و تووڕه‌بوون و ئه‌م شتانه‌یت، من چاوه‌ڕوانی كارێكم كه‌واده‌زانم تۆ ده‌یكه‌یت، واده‌زانم به‌ردێك ده‌خه‌یته‌ سه‌ربه‌رد؟ به‌ڵام ئه‌وا ئه‌و چاوه‌ڕوانییه‌ ده‌شكێنم و هه‌نگاو ده‌نێم به‌وێنه‌ی كه‌نارییه‌كه‌ و ده‌مه‌وێت بگه‌ڕێمه‌وه‌ قه‌فه‌زه‌كه‌ی جاران، بگه‌ڕێمه‌وه‌ قه‌فه‌زی عه‌شق، با له‌ده‌ره‌وه‌ی ئه‌و قه‌فه‌زه‌ش بم، خۆ ئه‌و قه‌فه‌زه‌ تێكه‌ڵ به‌رۆحم بووه‌، ئه‌ی كه‌نارییه‌كه‌ت له‌بیرنییه‌ كه‌ ده‌یویست بگه‌ڕێته‌وه‌ قه‌فه‌زه‌كه‌ی جارانی، ئاخر ئه‌ویش قه‌فه‌زه‌كه‌ی له‌ناخیدابوو، نه‌وه‌ك ئه‌و قه‌فه‌زه‌ی ئێمه‌ ئاشنای بوین.. ده‌مه‌وێت له‌وێدا بمرم، له‌وێدا بیر له‌م دونیا پڕوپوچه‌ بكه‌مه‌وه‌، ده‌مه‌وێت بگه‌ڕێمه‌وه‌ لای تووڕه‌بوونه‌كانت و ده‌ستێكی هێمنانه‌یان به‌سه‌ردابهێنم، بڕوابكه‌ تووڕه‌بوونی تۆ چێژی خۆی هه‌یه‌، تێده‌گه‌ی چێژی خۆی هه‌یه‌... من هه‌موو ئه‌و باره‌ ده‌روونیانه‌ی خۆمم لێكهه‌ڵوه‌شاندووه‌ و پێم ناوه‌ به‌هه‌موو ئه‌و كه‌لله‌ڕه‌قیانه‌دا كه‌تۆ فێرتكردم، هه‌موو ئه‌و تیشكه‌ شه‌یتانیانه‌م له‌خۆمدا كوژاندۆته‌وه‌، تا لێگه‌ڕێن وه‌ك دادوه‌رێك نێوان خۆم و تۆ كه‌شف بكه‌م، لێگه‌ڕێن ده‌مار و خانه‌ی ئه‌و نهێنیانه‌ شیكه‌مه‌وه‌ كه‌ ئه‌وه‌نده‌ به‌چڕی ئاڵۆزكابوون به‌مل یه‌كدا هه‌مووكات بیركردنه‌وه‌ لێیان ماندووی ده‌كردم و ته‌نها ده‌موت لێگه‌ڕێ‌... ئێستا شته‌كان به‌جۆرێك ده‌ركه‌وتوون، یان باشتره‌ بڵێم كاردانه‌وه‌ی ڕووداوه‌كان خه‌ریكه‌ یه‌ك ئه‌نجامیان ده‌بێت، یان به‌ئه‌نجام ده‌گات.
له‌بیرتبێت كه‌نارییه‌كه‌ یه‌ك دووجار گه‌ڕایه‌وه‌ و بابم دڵڕه‌قانه‌ و به‌كۆمه‌ڵێك هۆكار كه‌ته‌نهاخۆی به‌حه‌قیقه‌تی ده‌زانین ده‌ریكرد و ده‌یوت بڕۆ ئازادبه‌... بڕۆ ژیان ببینه‌... بڕۆ ئاسمانه‌ دوور و بێكۆتاییه‌كان ته‌ماشاكه‌... بڕۆ له‌كیشوه‌ره‌كانی تر ڕابمێنه‌... به‌ڵام كه‌نارییه‌كه‌ له‌گه‌ڵ یه‌كه‌مین چركه‌ی ئازادبوونیدا، تێگه‌یشت له‌پاش ئه‌و ژیانه‌ی نێو قه‌فه‌ز، ژیانێكی تر وجودی نییه‌، تێگه‌یشت ئه‌وێ‌ هه‌موو شتێك بوو بۆ ئه‌و، نیشتیمان، ئازادی، بینینی ئه‌و هه‌موو باڵنده‌ی تر كه‌ ده‌كڕران و ده‌فرۆشران، شه‌قامه‌ دووره‌كانی بوو، ده‌نگی جۆراوجۆری باڵنده‌كانی تر بازار و شاره‌ گه‌وره‌كانی بوو، ئێوارانیش كاتێك بابم به‌خۆی و قه‌فه‌زه‌كه‌یه‌ و به‌ته‌نها ده‌یهێنایه‌ باخچه‌ بچووكه‌ی ماڵه‌كه‌مان و دار هه‌نار و گوڵه‌ جۆراوجۆره‌كانی ده‌بینی ئه‌وێ‌ بۆئه‌و وڵات و گه‌شته‌ دوره‌كانی بوو، ئێواران كه‌به‌ غه‌مگیینییه‌وه‌ بیری له‌ ژیانی ئه‌ودیو قه‌فه‌زه‌كه‌ ده‌كرده‌وه‌ بابم بێزارده‌بوو، خۆیشی دڵته‌نگ و داماو ده‌هات و یه‌كسه‌ر له‌بیری ده‌كرد و به‌خێرایی ته‌ماشایه‌كی من و براكانمی ده‌كرد و ده‌یڕوانی به‌ چیمه‌نی باخچه‌كه‌دا و هه‌ستیده‌كرد ئه‌مه‌ هه‌موو ژیانه‌بۆی، ئیدی ئه‌و دیوی قه‌فه‌ز به‌ئه‌و چی!
ئاخر ته‌واوی ئه‌وكاتانه‌ی له‌نێو ئه‌و قه‌فه‌زه‌دابوو، ئه‌و قه‌فه‌زه‌ له‌ناخیدا سه‌وزببوو، له‌ڕۆحیدا هه‌موو بیرۆكه‌كانی ئازادی تێكشكاندبوو، یان باشتره‌ بڵێم قه‌فه‌ز ببووه‌ پارچه‌ك له‌ ڕۆحی كه‌نارییه‌كه‌ و ڕۆحی ببووه‌ ته‌واوی قه‌فه‌زه‌كه‌!!
من نامه‌وێت بڵێم هه‌ڵچوونه‌كانی من و تۆ به‌م نامه‌یه‌ شیده‌كرێته‌وه‌، یاخیبوون و توڕه‌بوونه‌كانی من و تۆ له‌وه‌ها نووسینێكدا داده‌مركێته‌وه‌، به‌ڵام ده‌توانم بڵێم من و تۆ ئه‌وه‌هابوون، من و تۆ ئه‌مه‌ هه‌ڵه‌كه‌مان بوو، ده‌بێت له‌ڕوداوی كه‌نارییكه‌ و بابمه‌وه‌ تێبگه‌ین من و تۆ له‌قه‌فه‌زی یه‌كدا نه‌بێت ده‌مرین، له‌قه‌فه‌زی عه‌شقدا نه‌بێت هیچ نین... ئاخر دواجار من و تۆ وه‌ك كه‌نارییه‌كه‌ دیلی قه‌فه‌زێكین، دیلی قه‌فه‌زی عه‌شقین و به‌بێ‌ ئه‌و سنووره‌ی سه‌رما ده‌مانكوژێت!
دووكه‌ڵ و زریانه‌كانی ژیان له‌ناومانده‌بات!
بڕوابكه‌ ئێستا تێده‌گه‌م كه‌ناری داماو بۆ له‌ئازادبوونی نێو قه‌فه‌ز ناڕازیبوو، ئێستا تێده‌گه‌م ژیانی پاش باوه‌ڕ و فیكره‌ به‌هێزه‌كان هیچ نییه‌ جگه‌ له‌ گه‌ڕانه‌وه‌ بۆ ئه‌زموونه‌ ساده‌كان، ئه‌گینا له‌م ماوه‌یه‌دا كه‌ له‌دووره‌وه‌ سه‌یری ڕابردووی خۆمانم ده‌كرد ئه‌و هه‌سته‌م لادروست ده‌بوو، كه‌ زۆر كه‌س هه‌بوون ئاواتیان بووه‌ وه‌ك من و تۆ بژین، وه‌ك من و تۆ بڕواننه‌ ژیان، به‌ڵام به‌دبه‌ختی نێوانمان له‌و كاتانه‌وه‌ دروست بوو وامانزانی من و تۆ ڕێگرین له‌به‌رده‌م فڕینی یه‌كتردا، له‌پاش تێكشكانی ئه‌و هه‌موو ڕێگرانه‌ی نێوانمان، وامانزانی خۆمان ڕێگرین له‌ئاست یه‌كدا، بڕوابكه‌ ئه‌و هه‌ستانه‌ هیچ نه‌بوون جگه‌ له‌ تفی شه‌یتان كه‌ده‌یكرده‌ ڕێگاكه‌مان، هیچ نه‌بوو جگه‌ له‌و نه‌هامه‌تییه‌ی كه‌به‌ته‌نها خۆی مابوویه‌وه‌ بۆ من و تۆ.
ئه‌ی ئه‌و ئینسانه‌ی له‌ ژورێكدایت و خه‌جاڵه‌تییه‌كانت ده‌ژمێری، مه‌گه‌ر له‌م وشانه‌ تێناگه‌ی؟ ئه‌ی پیاوێك یه‌كپارچه‌ له‌ خۆپه‌رستی و شه‌رم، مه‌گه‌ر ڕووداوی كه‌نارییه‌كه‌ و بابمت بۆ ڕوننه‌كرایه‌وه‌؟ تۆ له‌و كونه‌ ڕێویه‌دا و ژیان به‌م دڵڕه‌قی خۆیه‌وه‌، قه‌د یه‌كناگرنه‌وه‌ به‌م چه‌شنه‌، تۆ كه‌ دونیات بووه‌ به‌ مه‌ستبوون و وازت له‌ هه‌ڵمژینی هه‌وای سپێده‌ هێناوه‌ به‌م شێوه‌یه‌ ناگه‌یته‌وه‌ به‌كه‌ناری خۆشبه‌ختییه‌ وێرانه‌كان، تۆ به‌م هه‌موو تاركیانه‌ته‌وه‌ و به‌م ڕووناكییانه‌ی ئه‌هریمه‌نه‌وه‌ ده‌ته‌وێت بجه‌نگی له‌گه‌ڵ كوێی ژیاندا؟ ده‌ته‌وێت له‌كام جێگادا ئاوڕبه‌یته‌وه‌ و له‌كام كه‌شتیدا بڕوانیت به‌ ده‌ریادا، كه‌ ده‌زانی هێشتا له‌ده‌ربه‌نده‌كه‌ گیرت كردووه‌؟ هێشتا یه‌كه‌م هه‌نگاوت نه‌ناوه‌؟ تۆ ده‌ته‌وێت به‌و میوزیكه‌ بیانیانه‌ و به‌و پێكه‌ تیژانه‌ی ده‌ستت له‌ بگه‌یته‌ حه‌قیقه‌تی بوونی خودا؟ یان ده‌ترسم وابزانی تۆ ئه‌و فه‌یله‌سوفه‌ی كه‌خاوه‌ن ڕێبازی بێده‌نگیت؟ ئاخر ئه‌ی به‌دبه‌خت له‌م نامه‌یه‌وه‌ بگه‌ڕێوه‌ بۆ زبری ڕووداوه‌كان تاده‌ستت له‌شتێك گیربێت، له‌م وڕێنانه‌ی منه‌وه‌ سه‌یری نیگاكانی ژیان بكه‌، تا عه‌شقی بێمانایی ژیان بیت.
