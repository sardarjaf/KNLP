له‌ به‌رده‌م په‌نجه‌ره‌كه‌وه‌ دیمه‌نی ده‌ره‌وه‌ دیاره‌، ئاگردانه‌كه‌ ژووره‌وه‌ی گه‌رمكردۆته‌وه‌، چۆله‌كه‌یه‌ك له‌ بنمیچی په‌نجه‌ره‌كه‌دا خۆی گرمۆڵه‌كردووه‌، وایه‌ری كاره‌باكان دوو په‌نجه‌ به‌فر له‌سه‌ریان كه‌ڵه‌ك بووه‌، ئێره‌ خانوویه‌كی شێوه‌ كوخیه‌، ماڵی ئه‌و پیره‌مێرده‌یه‌، كه‌ ده‌كرێ‌ به‌ پاڵه‌وانی ئه‌م چیرۆكه‌ ناوی ببه‌ین، ته‌واوی كه‌ره‌سته‌كان دارن، جگه‌ له‌و قه‌وانه‌ نه‌بێت، كه‌ نزیك به‌ ئاگردانه‌كه‌ دایناوه‌، به‌رده‌م په‌نجه‌ره‌كه‌ دوو كورسی و مێزێكی دار گوێزی لێیه‌، كه‌ گۆڕه‌پانێكی شه‌تڕه‌نج له‌سه‌ری دانراوه‌، ئه‌مه‌ش له‌ ئێستاوه‌ ده‌بێت به‌ شوێنی ئه‌و چیرۆكه‌ی، كه‌ ده‌یگێڕینه‌وه‌، ڕاسته‌ من نووسه‌ری ئه‌م چیرۆكه‌م، به‌ڵام من نایگێڕمه‌وه‌، به‌ڵكو ئه‌وانه‌ی پارێزگاری له‌مه‌لیك ده‌كه‌ن باشتر ئاگاداری ورده‌كارییه‌كانن، ئه‌وان ده‌زانن ئه‌و جه‌نگه‌ چۆنه‌؟ و كێ براوه‌یه‌؟ من ته‌نها ئه‌وانم خوڵقاندووه‌ و ڕێگه‌م داون قسه‌ بكه‌ن، ده‌نا مردنی ئه‌و مه‌لیكه‌ی ئه‌وان شه‌ڕی له‌ سه‌ر ده‌كه‌ن، هیچ له‌ چاره‌نووسی من ناگۆڕێت،
من پیره‌مێرده‌كه‌م، كه‌ به‌رانبه‌ر گه‌نجێك دانیشتووم، له‌ یاری شه‌ڕكردندا به‌رگری له‌ مه‌لیكه‌كه‌م ده‌كه‌م، ئه‌و مه‌لیكه‌ی كه‌ چاره‌نووسی بردنه‌وه‌ و دۆڕاندنی منی به‌ ده‌سته‌، ئه‌و مه‌لیكه‌ی كه‌ جه‌سته‌ی له‌ ته‌خته‌یه‌ و ڕوحله‌به‌ره‌كان به‌رگری لێ ده‌كه‌ن، من تا دوا هه‌ناسه‌ به‌رگری لێ ده‌كه‌م، ئاماده‌م ته‌واوی سه‌ربازه‌كان و شه‌ڕكه‌ره‌كانی دیكه‌ی له‌ پێناو بده‌م به‌ كوشت، هیچ پێویستم به‌ قه‌ڵا و ئه‌سپه‌كان نیه‌، بۆ به‌رگری كردن نه‌بێت له‌ مه‌لیكه‌كه‌م، چ كارێكم به‌ وه‌زیر و فیله‌كان نیه‌، گه‌ر نه‌بنه‌ هه‌ڕه‌شه‌ بۆ مه‌رگی مه‌لیكی به‌رانبه‌ر، ئێستا كه‌ من به‌ دیار ئه‌م ته‌خته‌ی شه‌تڕه‌نجه‌وه‌ دانیشتووم، هه‌موو هه‌وڵێكم بۆ مانه‌وه‌ی مه‌لیكه‌، تا دوا خوله‌كیش به‌رگری لێده‌كه‌م،
من شای گۆڕه‌پانه‌كه‌م، له‌ سه‌ر ئه‌م زه‌ویه‌ ڕه‌ش وسپیه‌ له‌ من گه‌وره‌تر كه‌سی تیا نیه‌، هه‌موو شتێك بۆ منه‌، ماندووبوون و شه‌هیدبوونی هه‌موو شه‌ڕكه‌ره‌كان به‌ خاتری سه‌لامه‌تی منه‌، تاجه‌كان بۆ سه‌ری منن، به‌ڵام بردنه‌وه‌كان هه‌موومانین، من و وه‌زیر و سه‌ربازێك وه‌ك یه‌ك ده‌یبه‌ینه‌وه‌، وه‌ك یه‌كیش ده‌یدۆڕێنین، سه‌ركه‌وتنی من سه‌ركه‌وتنی هه‌مووانه‌، له‌ بردنه‌وه‌دا من وسه‌ربازێك وه‌ك یه‌ك شادین، له‌ دۆڕاندنیشدا هه‌مان خه‌ممان هه‌یه‌.
من ئێستا له‌ ده‌ره‌وه‌ی گۆڕه‌پانه‌كه‌م، هێنده‌ نابێت ئه‌سپی به‌رانبه‌ر منی كوشت، نازانم بردنه‌وه‌ چی له‌ چاره‌نووسی من ده‌گۆڕێت، بردنه‌وه‌ مانه‌وه‌ی مه‌لیكه‌كه‌مان و كوژرانی ئێمه‌یه‌؟، ئه‌م قسانه‌ی ئه‌م مه‌لیكه‌ ده‌یكات، هیچی له‌ قسه‌ی سه‌ركرده‌یه‌كی درۆزن كه‌متر نیه‌، كه‌ ده‌یه‌وێت  ئینتیمای بردنه‌وه‌ لای شه‌ڕكه‌ره‌كانی دروست بكات و دواجار خۆی له‌ باوه‌شی سۆزانیه‌كدا خۆشاو بنۆشێت، ئێمه‌ بچوكترین به‌شداربووی جه‌نگین و یه‌كه‌مین قوربانی به‌رگریكردنین، هه‌میشه‌ ئه‌وانه‌ی سه‌ره‌وه‌ ئێمه‌یان بۆ به‌ كوشتدان ده‌وێت، ئه‌و پیره‌مێرده‌ من و هه‌موو هاوڕێكانم ده‌كات به‌ قوربانی چركه‌یه‌كی خۆی، دواتر ئه‌وه‌ی ئه‌و شه‌ڕی له‌سه‌ر ده‌كات ته‌نها نه‌كوژرانی مه‌لیكه‌، چ شتێی دیكه‌ لای ئه‌و به‌هایه‌كی نیه‌، ئه‌و شه‌تڕه‌نج ته‌نها له‌ پێناوی كات به‌سه‌ربردن ده‌كات، ده‌نا شه‌ڕی شه‌تڕه‌نج هه‌روا به‌ ئاسانی ناكرێت، هه‌روا بێ مه‌ته‌رسی نیه‌، ئه‌ویش حه‌زه‌ر و ترس و دوو دڵی ده‌وێت، ئێمه‌ ده‌زانین هونه‌ری شه‌ڕكردن له‌ چیدایه‌، ته‌نها شه‌ڕكه‌ر تێده‌گات ته‌كنیكی شه‌ڕكردن چۆنه‌، هه‌موو نهێنیه‌ك لای ئێمه‌یه‌، ئێمه‌ ده‌زانین به‌ره‌كانی جه‌نگ چی تیابووه‌ و چی تیاكراوه‌، ده‌زانین كام لا براوه‌یه‌، حیكایه‌تی شه‌ڕه‌كه‌ی لا ئێمه‌یه‌ و ده‌یگێڕینه‌وه‌، سه‌ركرده‌كان ته‌نها نه‌خشه‌ی نێوان قاچی ئافره‌ت شاره‌زان، ئه‌و پیره‌مێرده‌ زۆر به‌ سانای له‌ واتای جه‌نگ گه‌یشتووه‌، شه‌ڕی شه‌تڕه‌نج هێنده‌ی شه‌ڕی عه‌شق قورسه‌، پێناچێت ئه‌و خوایه‌ی من(پیره‌مێرده‌كه‌) هیچ له‌ عه‌شق بزانێت...
ئێوه‌ی سه‌رباز كێشه‌یه‌كتان هه‌یه‌، هه‌مووتان ده‌تانه‌وێ‌ ببن به‌ شا، ببن به‌ سه‌ركرده‌و سه‌رۆك، به‌ڵام هه‌رگیز نه‌تانپرسیوه‌ كه‌ ئێوه‌ سه‌رۆكبن، كێ؟ له‌ به‌رده‌متان پۆلیس بێت، من ده‌زانم شه‌ڕ جیاوازه‌، ده‌زانم دۆڕٍاند له‌ هه‌رشتێك قورستره‌، به‌ڵام دۆڕاندن له‌ شه‌تڕه‌نجدا جیاوازی هه‌یه‌، له‌ شه‌تڕه‌جدا ده‌توانی هه‌موو جه‌یشه‌كه‌ت بده‌ی به‌ كوشت له‌ پێناو مانه‌وه‌ی مه‌لیكدا، ئاساییه‌ گه‌ر بشدۆڕێیت و مه‌لیكه‌كه‌ت گه‌مارۆبدرێت، دۆڕاندن له‌ شه‌تڕه‌نجدا كاره‌سات نیه‌، بۆ سبه‌ی ده‌توانی یارییه‌كی دیكه‌ بكه‌یته‌وه‌، به‌ڵام شه‌ڕكردن له‌ ناو خوێندا به‌ هه‌موو كه‌سێك ناكرێت، ته‌نها خاڵیش كه‌ شه‌تڕه‌نج له‌ عه‌شق جیاده‌كاته‌وه‌ دۆڕاندنه‌، ده‌نا له‌ هه‌موو شتێكدا هاوبه‌شن، ئێوه‌ی سه‌رباز هێشتا له‌ هونه‌ری سیاسه‌ت نه‌گه‌یشتوون، وا ده‌زانن خه‌وتن له‌ باخه‌ڵی سۆزانیه‌كدا سیاسه‌ت نیه‌، ئه‌وه‌ی كه‌ هیچ خه‌یاڵێكیش په‌ی پێنابات، بۆ پیاوی سیاسی ئاسانه‌، ئه‌وان هه‌میشه‌ بوتڵی شه‌رابه‌كه‌یان له‌ سه‌ر سنگی به‌ڕماڵی ئێواره‌ خاڵی ده‌كه‌نه‌وه‌، نیشتیمان هه‌رگیز   ئه‌وپیرۆزیه‌ نیه‌، كه‌ ئێمه‌ به‌ قیبله‌یه‌ سه‌ركرده‌كانی تێده‌گه‌ین، ڕه‌نگه‌ به‌ هۆی ته‌مه‌نمه‌وه‌ كه‌مێك له‌ بێمانای سیاسه‌ت گه‌یشتبم، به‌ڵام تۆ ڕاست ده‌كه‌ی هیچی وا له‌ عه‌شق نازانم، ڕه‌نگه‌ ته‌مه‌نی من منداڵ بێت بۆ دۆزینه‌وه‌ی هونه‌ری عه‌شق،...
ئێستا من له‌ ده‌ره‌وه‌ی گۆڕه‌پانه‌كه‌م، له‌ سه‌ر مێزه‌كه‌وه‌ ته‌ماشای ده‌ره‌وه‌ ده‌كه‌م، به‌فره‌كه‌ له‌ كه‌مبوونه‌وه‌دایه‌، ئێستا كاریگه‌ریم له‌ یارییه‌كه‌دا نه‌ماوه‌، ئه‌وه‌نده‌ نه‌بێت كوژرانی من زه‌نگی نزیكبوونه‌وه‌ی كوژرانی مه‌لیكه‌، من یه‌كێك بووم  له‌ پێنج سه‌ركرده‌ی شه‌ڕه‌كه‌، زۆربه‌مان كوژراوین، ئێستا مه‌لیكه‌كه‌ ته‌نها وه‌زیر و قه‌ڵایه‌ك له‌ ده‌وری ماون، پێده‌چێت به‌ر له‌ كردنه‌وه‌ی به‌فركه‌ كۆتایی به‌ ژیانی بێت،
من دۆڕاندنم، له‌ ژیانی هه‌موو كه‌سێكدا هه‌م، هه‌موو كه‌سێك به‌ من ئاشنایه‌، له‌گه‌ڵ هه‌موو بردنه‌وه‌یه‌كدا كه‌سێك ده‌دۆڕێت، ئه‌گه‌ر كه‌سی دۆڕاو نه‌بێت، بردنه‌وه‌ كوا بوونی هه‌یه‌، براوه‌ له‌ كێی بردۆته‌وه‌؟، منیش جیاوازم، لای هه‌ركه‌سه‌ و به‌ شێوه‌یه‌ك ده‌رده‌كه‌وم، ئه‌وانه‌ی له‌ شه‌تڕه‌نج و یارییه‌كانی تردا ده‌دۆڕێن، كه‌مترین سزای منیان وه‌رگرتووه‌، ئه‌وانه‌ی نیشتیمان ده‌دۆڕێن، له‌ هه‌موو كه‌س بارقورسترن، چونكه‌ تا مردن نمیشی خه‌فه‌تباری ده‌كه‌ن، به‌ڵام عه‌شق ته‌نها یارییه‌ كه‌ بردنه‌وه‌ی تیانیه‌ و هه‌موویان ده‌دۆڕێن، كوژرانی ئه‌و مه‌لیكه‌ش دۆڕاندنی كابرایه‌كی ڕیش سپییه‌، كه‌ ئه‌مڕۆ منی هێناوه‌ته‌ قسه‌، چ شێتیه‌كه‌ به‌ دۆڕاندنێكی وه‌ها مرۆڤ خه‌مبار بێت، من سه‌ركرده‌ ده‌ناسم، هه‌موو گه‌له‌كه‌ی له‌گه‌ڵیا دۆڕاوه‌ و به‌خه‌یاڵیدا نه‌هاتووه‌، سه‌ركرده‌ ده‌ناسم دوای ڕوخانی داگیركه‌ری وڵاته‌كه‌ی، ده‌ركه‌ووتووه‌ سیخوڕبووه‌ و هه‌ر خۆی به‌ براوه‌ زانیوه‌، هه‌موو دۆڕاوه‌كان دێنه‌وه‌ لای من، من هه‌موویان ده‌ناسم، كه‌س نیه‌ به‌ دۆڕاندن خه‌فه‌تبار بێت، ته‌نها نمایشكردنه‌،
من چیرۆكنووسه‌كه‌م، دواتر ناوم ده‌چێته‌ سه‌ر ئه‌م چیرۆكه‌ی وا ده‌یخوێنیته‌وه‌، كوژرانی ئه‌و مه‌لیكه‌ هیچ په‌یوه‌ندی به‌ منه‌وه‌ نییه‌، به‌ڵكو قه‌ده‌ر وای كردووه‌ ئه‌و مه‌لیكه‌ ماتبكرێت و پیره‌مێرده‌كه‌ بدۆڕێت، گرنگیش نیه‌ با بدۆڕێت، له‌ هه‌موو دوو مرۆڤێكدا ده‌بێت یه‌كێكیان بدۆڕێت، له‌ جه‌نگی خودا و شه‌یتانیشدا یه‌كێكیان هه‌ر ده‌دۆڕێن.
