     ئێستاكه‌ وه‌كو جاران حه‌وسه‌ڵه‌ی كارم نه‌ماوه‌، به‌ڵكو كوڕه‌ تاقانه‌كه‌مم كردۆته‌ جێنشینم، هه‌رچه‌نده‌ هه‌ندێجار له‌سه‌ر وه‌سییه‌تی ئه‌و كه‌سانه‌ی كه‌ به‌ر له‌ مردنیان داوایان كردووه‌ من گۆڕیان بۆ هه‌ڵبكه‌نم، ده‌چمه‌ گۆڕستان و به‌ هه‌ناسه‌ سوارییه‌كی زۆره‌وه‌ كار ده‌كه‌م، ماوه‌ی چه‌ند ساڵێكه‌ له‌بری ئه‌وه‌ی له‌ شه‌وێكی دره‌نگه‌وه‌ وه‌ئاگابم و دوای نوێژی به‌یانی بچمه‌ سه‌ركار، به‌وپه‌ڕی دڵنیاییه‌وه‌ چاوه‌ڕێی ئه‌و كاته‌ ده‌كه‌م، وه‌كو هه‌موو ئه‌وانه‌ی به‌ ده‌سته‌كانی خۆم به‌ خاكم سپاردوون، منیش به‌رنه‌ گۆڕستان و له‌و گۆڕه‌م بنێن، كه‌ هه‌ر له‌ ته‌مه‌نی بیست و دوو ساڵیمه‌وه‌ بۆ خۆمم ئاماده‌كردووه‌.
به‌ڕه‌حمه‌ت بێت وه‌ستا كه‌ریم كه‌ پیاوێكی نورانیی و دونیا دیده‌بوو، هه‌میشه‌ پێیده‌گوتم: كوڕك ڕاسته‌ هه‌مووان مردنێك قه‌رزدارین، به‌ڵام كێ ده‌ڵێت گۆڕمان به‌ نسیب ده‌بێت، هه‌زاران كه‌س له‌ جه‌نگه‌كاندا ده‌بنه‌ قوربانی و كه‌س نازانێت چییان به‌سه‌ر دێت، چه‌ندین كه‌سیش مردنیان له‌ ناو ئاوه‌كاندایه‌ و بۆ هه‌میشه‌ نادۆزرێنه‌وه‌.
كاركردنم له‌ نێو گۆڕستاندا ئه‌گه‌رچی بژێوی ژیانمی له‌سه‌ر بوو، به‌ڵام هه‌میشه‌ به‌ دیوێكی تردا لێمده‌ڕوانی، به‌جۆرێك هه‌ستم ده‌كرد عه‌شقێكی جاویدانم هه‌یه‌ بۆی و له‌ قوڵایمه‌وه‌ دێته‌ ده‌رێ‌، چونكه‌ هه‌ر له‌ مناڵیمه‌وه‌ خۆشه‌ویستیم بۆ گۆڕستان هه‌بوو، بیرمه‌ بیرمه‌ له‌گه‌ڵ مناڵانی گه‌ڕه‌ك ده‌چووینه‌ سه‌ر دارئه‌رخه‌وانه‌كانی نێو حه‌وشه‌ی مه‌زاری شێخ و به‌ دڵی خۆمان گوڵه‌ ئه‌رخه‌وانمان ده‌خوارد، هه‌ر كاتێكیش له‌ دووره‌وه‌ زێوانه‌كه‌ ده‌رده‌كه‌وت ڕامان ده‌كرد، ئه‌ویش له‌ دوومان ده‌هات و به‌ به‌رد و دار و جنێو تا ده‌روازه‌ی گۆڕستانه‌كه‌ ڕاوی ده‌ناین.
ئامانجی من له‌ ڕۆیشتم بۆ گۆڕستان جیاوازبوو له‌ هه‌موو مناڵه‌كانیتر، ئه‌وان ته‌نیا ده‌هاتن دار و دره‌خته‌كانی گۆڕستان بشكێننه‌وه‌، یان نووسراوی سه‌ركێلی گۆڕه‌كان بشێوێنن، هه‌ركاتێكیش له‌ دووره‌وه‌ ته‌رمێكیان ببینیایه‌ به‌ هه‌موو تینیان هه‌ڵده‌هاتن، به‌ڵام من ده‌مویست له‌ نێو ئه‌و بێده‌نگییه‌ی ئه‌وێدا هه‌ناسه‌یه‌كی قوڵ هه‌ڵبمژم و به‌رامه‌ی ڕه‌یحانه‌ و گوڵه‌ به‌یبوون و گوڵه‌كانیتر ته‌واو مه‌ستم بكه‌ن، ده‌مویست له‌ ژێر سێبه‌ری دارئه‌رخه‌وان و دارزه‌یتوون سنه‌وبه‌ره‌كانی ئه‌وێدا سه‌رخه‌وێك بشكێنم و تێیدا خه‌ون به‌ ژیانه‌وه‌ ببینم، ده‌مویست بیر له‌و ئافره‌ته‌ بكه‌مه‌وه‌ كه‌ به‌رله‌ناشتنی توڕه‌كه‌یه‌ك قژه‌ سه‌ریان نایه‌ ناو گۆڕه‌كه‌ی، یان له‌ ئێسك و پروسكی ئه‌و مناڵانه‌ ورد ببمه‌وه‌ كه‌ به‌ ده‌می سه‌گه‌ دڕه‌كانی شه‌و گۆڕه‌كانیان هه‌ڵده‌درایه‌وه‌، ئه‌مانه‌ و چه‌ندین مه‌به‌ستی تر هۆكاربوون بۆ ئه‌وه‌ی نه‌توانم له‌ گۆڕهه‌ڵكه‌ندن زیاتر هیچ كارێكیتر هه‌ڵبژێرم.
بۆیه‌ هه‌ر كه‌ گه‌یشتمه‌ یانزه‌ ساڵ، ڕۆژێك چوومه‌ لای وه‌ستا كه‌ریم و دوای تۆزێك خۆسه‌رقاڵكردن به‌ دامێنی كراسه‌كه‌مه‌وه‌، به‌ ترسێكه‌وه‌ گوتم: وه‌ستا حه‌ز ده‌كه‌م لای تۆ كار بكه‌م، ئه‌ویش به‌ بزه‌یه‌كه‌وه‌ لێم وردبووه‌وه‌ و گوتی: ڕۆڵه‌ گیان كاره‌كه‌ی من بۆ تۆ سه‌خته‌، هه‌وڵبده‌ لای دووكانی هه‌لاجه‌كه‌ی نزیك ماڵی خۆتان كار بكه‌ی.
ئه‌و ئێواره‌یه‌ به‌بێتاقه‌تییه‌وه‌ چوومه‌وه‌ بۆ ماڵه‌وه‌، به‌ڵام هه‌ر كۆڵم نه‌دا و دوای ماوه‌یه‌ك هه‌ر چۆنێك بێت لای وه‌ستا كه‌ریم جێگه‌ی خۆمم كرده‌وه‌، له‌ سه‌ره‌تادا باوكم كه‌ وه‌ستای ده‌باغی بوو، پێی ناخۆش بوو له‌وێ‌ كاربكه‌م، ئه‌و حه‌زی ده‌كرد له‌گه‌ڵ خۆی بڕۆمه‌ كار و نه‌چمه‌ گۆڕستان، به‌ڵام دوای چه‌ندین جار توڕه‌بوون و ته‌نانه‌ت لێدانیش بێزاربوو له‌ ڕێگریكردنم.
دوای دوو ساڵ وه‌ستا كه‌ریم هه‌موو كارێكی پێده‌سپاردم، چونكه‌ دڵنیابوو له‌وه‌ی ده‌توانم زۆر به‌ ڕێكوپێكی بۆ ئه‌نجامبده‌م، جار جاره‌ له‌ ده‌ست و بازوومی ده‌ڕوانی و له‌ ڕوخساریدا سه‌رسوڕمانم ده‌خوێنده‌وه‌، له‌ ڕاستیدا ئه‌و پێی وابوو كه‌س ناتوانێت هێنده‌ی ئه‌و گۆڕیچه‌ی ڕێك و پته‌و دروست بكات، به‌ڵام كه‌ ده‌یبینی به‌و ورد بینییه‌وه‌ كار له‌سه‌ر گۆڕیچه‌كان ده‌كه‌م و هه‌وڵده‌ده‌م به‌ جوانترین شێوه‌ بیاننه‌خشێنم، ده‌ستخۆشی لێده‌كردم و ماندوونه‌بوونی بۆ ده‌خواستم.
من كاری گۆڕهه‌ڵكه‌ندنم به‌ هونه‌رێكی قورس ده‌زانی، له‌ كاركردنم له‌ نێو هه‌ر گۆڕێكدا بیرم له‌وه‌ ده‌كرده‌وه‌، تۆزێكیتر به‌ هه‌موو جوانی و ماناكانییه‌وه‌ به‌ هه‌موو خه‌ون و ئومێده‌ جۆراوجۆره‌كانییه‌وه‌ ده‌بێته‌ نیشته‌جێیه‌كی هه‌میشه‌یی ئێره‌.
به‌یانیان زوو ده‌چوومه‌ بۆ كاركردن، تاكو به‌ ته‌واوی دیمه‌نه‌كانی سپێده‌ی گۆڕستان ببینم، بۆنی نسێی گۆڕه‌كان لوتم پڕ بكه‌ن، بڕوانمه‌ كرانه‌وه‌ی ئه‌و گوڵانه‌ی له‌ شه‌رمیان بۆ مانگه‌ شه‌و ده‌مووچاوی خۆیان داده‌پۆشن، به‌یانیانی گۆڕستان چێژ و ئارامییه‌كی وه‌های پێده‌به‌خشیم له‌ هیچ شوێنێكیتر هه‌ستم پێنه‌ده‌كرد.
هه‌رگیز ئه‌و به‌یانییه‌م بیرناچێته‌وه‌ كه‌ دوو عاشقی به‌له‌نگاز، له‌ ژێر چه‌رداخی گۆڕێكدا به‌ ده‌م لوشكه‌ لوشكی گریانه‌وه‌ وه‌نه‌وز ده‌یبردنه‌وه‌، هه‌رچه‌ندم كرد و كۆشام نه‌متوانی لێیان بچمه‌ پێشه‌وه‌، چونكه‌ هه‌ر له‌ دووره‌وه‌ به‌ گه‌رمی عه‌شق بارگاوییان كردم و به‌ جۆرێك خرۆشام گریانم بۆیان ده‌هات، هه‌ر ئه‌و به‌یانییه‌ش بوو كه‌ كوڕه‌ گه‌وره‌كه‌ی وه‌ستا كه‌ریم به‌ شپرزه‌ییه‌كی زۆره‌ هاته‌ لام و داوای لێكردم بڕۆم و گۆڕێك بۆ باوكی هه‌ڵبكه‌نم، له‌و كاته‌دا هه‌مان ئه‌و ڕۆژانه‌م هاته‌وه‌ بیر كه‌ وه‌ستا كه‌ریم پێی ده‌گوتم: كوڕم هه‌مووان مردنێك قه‌رزدارین به‌ڵام كێ ده‌ڵێت گۆڕمان به‌ نسیب ده‌بێت؟
      له‌گه‌ڵ ئه‌و هه‌موو هه‌ستكردنه‌م به‌ جوانی له‌ ناو گۆڕستاندا ئه‌وه‌ی به‌رده‌وام توڕه‌ی ده‌كردم، ئه‌و كۆمه‌ڵه‌ پیاوه‌ بوون كه‌ ئێوارانی هه‌ینی ده‌هاتنه‌ گۆڕپانی ده‌روازه‌ی گۆڕستانه‌كه‌ و سه‌گیان به‌ شه‌ر ده‌دا، هێنده‌ی نه‌ده‌برد له‌ نێو خۆیاندا ده‌بووه‌ ده‌مه‌قاڵێیان و به‌شه‌ڕ ده‌هاتن، ئه‌وه‌شی ئارامی ده‌كردمه‌وه‌ ئه‌و پیره‌مێرده‌ بوو كه‌ ڕۆژانه‌ ده‌هاته‌ سه‌ر گۆڕی هاوسه‌ره‌كه‌ی و لێی ده‌ڕوانی، هه‌ندێجار له‌به‌رخۆیه‌وه‌ ورته‌ ورتی ده‌كرد، وه‌ك ئه‌وه‌ی قسه‌ی بۆ بكات و ئه‌ویش وه‌ڵامی بداته‌وه‌.
كوڕێكی گه‌نجی خوێن گه‌رمیش هه‌بوو، هه‌موو عه‌سرانێك به‌ خۆی و هارمۆنیكاكه‌یه‌وه‌ ده‌هاته‌ گۆڕستان و تا ماندوو ده‌بوو ئاوازێكی وه‌های ده‌ژه‌ند ئاو ده‌زایه‌ چاوه‌كانی و ته‌زوویه‌كی شادیبه‌خشی به‌سه‌ر گۆڕكاندا ده‌پرژاند، ڕۆژێكیان چوومه‌ لای و پرسیم: تۆ بۆچی دێیته‌ گۆڕستان؟ له‌ وه‌ڵامدا گوتی: دێم و ده‌مه‌وێت مۆسیقا بكه‌مه‌ ڕۆحێكی ئه‌به‌دی بۆ مردووه‌كان، چونكه‌ له‌ گۆڕستاندا مۆسیقا قوڵتر ڕۆده‌چێته‌ تاریكاییه‌كانی ناخه‌وه‌، له‌دوای هه‌ر هاتنێكمدا بۆ ئێره‌ ڕۆحی خۆم و مردووه‌كان نۆژه‌ن ده‌بنه‌وه‌، به‌جۆرێك واده‌زانم ئێستا له‌ دایكبووین، هه‌رگیز لێره‌ هه‌ست به‌ مردن ناكه‌م، چونكه‌ ده‌بینم له‌ ئه‌نجامی هاتنم بۆ ئێره‌ فێری شتی زیاتر ده‌بم له‌ باره‌ی ژیانه‌وه‌، من گۆڕستان وه‌كو كتێبێكی گه‌وره‌ ده‌بینم، هه‌ر گۆڕێكیش لاپه‌ڕه‌یه‌كه‌ له‌و كتێبه‌، هه‌ر لاپه‌ڕه‌یه‌كیش دونیایه‌ك سه‌رگوزشته‌ و چیرۆك و شیعری له‌ خۆگرتووه‌، تۆ ته‌ماشاكه‌ ئه‌و هه‌موو عاشقه‌، ئه‌و هه‌موو پاڵه‌وانه‌، ئه‌و هه‌موو بلیمه‌ته‌، چۆن پاڵكه‌وتوون،....
     له‌ ڕاستیدا من له‌ ته‌مه‌نمدا عاشق به‌ هیچ ئافره‌تێك نه‌بووم، ته‌نانه‌ت به‌ هاوسه‌ره‌كه‌شم،كه‌ كوڕێكی تاقانه‌م لێی هه‌یه‌، هه‌موو جارێك دایكم پێی ده‌گوتم: كوڕم به‌ دڵی خۆت خێزان هه‌ڵبژێره‌، ئێمه‌ هیچ كارمان به‌سه‌رته‌وه‌ نییه‌.
به‌ڵام من قه‌ت هه‌ستم به‌ عه‌شقم نه‌كردووه‌ بۆ ئافره‌ت، هه‌میشه‌ ئافره‌ت لای من ئه‌و ده‌رگایه‌ بووه‌ كه‌ حه‌زم به‌ كردنه‌وه‌ی نه‌كردووه‌، ئه‌گه‌رچی له‌وه‌ دڵنیابووم ئه‌و دیو ئه‌و ده‌رگایه‌ پڕیه‌تی له‌ نهێنی، جیاواز له‌ هه‌موو پیاوێكیتر به‌رده‌وام دووره‌ په‌رێز و كه‌مدووبووم له‌گه‌ڵ ئافره‌تدا، ته‌نانه‌ت له‌ دوا ساته‌كانی ژنهێنانیشدا هه‌ر دوو دڵبووم.
زۆرجار كه‌ له‌ باڵكۆنی ماڵه‌كه‌ماندا به‌ده‌م كێشانی سیگار و ده‌رهێنانی خۆڵی چه‌ندین ساڵه‌ی گۆڕستان له‌ نێو نینۆكه‌كانمدا، خه‌یاڵ ده‌مباته‌وه‌، گومانده‌كه‌م له‌وه‌ی كه‌ ڕه‌نگه‌ ئه‌م تاقه‌ كوڕه‌م زاده‌ی عه‌شقی من و گۆڕستان بێت نه‌ك ژنه‌كه‌م، چونكه‌ ده‌بینم ئه‌ویش هاوشێوه‌ی خۆم عه‌شقێكی ئێكجار جوانی هه‌یه‌ بۆ گۆڕستان و گۆڕهه‌ڵكه‌ندن، هه‌ر بۆیه‌ هه‌میشه‌ به‌ ده‌نگێكی ماندووه‌وه‌ بانگی ده‌م و پێیده‌لێم: كوڕم من ده‌مرم، ئه‌وه‌ی هێشتا زووه‌ بۆ مردنی تۆیت، ده‌خوازم بتوانی هه‌ست به‌ ئارامی گۆڕستان و ده‌رك به‌ جوانییه‌كانی بكه‌یت، كوڕم هه‌ركات هه‌ستتكرد چێژ له‌ ماندووبوونه‌كانی گۆڕستان وه‌رناگریت، واز له‌ كاره‌كه‌ت بهێنه‌و خۆت به‌ كارێكی تره‌وه‌ سه‌رقاڵ بكه‌، چونكه‌ مانه‌وه‌ له‌سه‌ر كارێك كه‌ چێژی تێدا نه‌بێت مردنه‌...مردن...
     ئێستاش هه‌مان ئاره‌زووی جارانم هه‌یه‌ بۆ گۆڕهه‌ڵكه‌ندن، به‌ڵام به‌ده‌ست خۆم نییه‌!
ته‌مه‌ن له‌گه‌ڵ هه‌ڵكشانیدا، توانا و هێزی جارانی لێبڕیووم، ئه‌وه‌ی كه‌مێك دڵخۆشم ده‌كات كوڕه‌كه‌مه‌، من ئێستا ماندوو، ماندووم.
