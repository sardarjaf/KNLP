له‌دوای شه‌وێكی زۆر درێژ به‌ئاگا هاته‌وه‌، هه‌رچه‌نده‌ هێشتا تاریكی وه‌ك په‌رده‌ییه‌كی ره‌ش به‌سه‌رشاره‌وه‌ داپۆشرابوو، كوچه‌و كۆڵانه‌كانی شاردبووه‌وه‌، بۆیه‌كه‌مجاربوو له‌مێژووی ئه‌و شاره‌ هه‌مووكاتژمێره‌كان له‌یه‌ك كاتدا له‌ده‌رخستنی كات به‌درۆ كه‌وتبوونه‌وه‌، هه‌ركه‌سێ له‌و كاته‌دا به‌خه‌به‌رهاتبێ یه‌كسه‌ر بێ سێ ودوو ئه‌و پرسیاره‌ی له‌خۆی ده‌كرد، كات زووه‌ یان كاتژمێره‌كه‌م په‌كی كه‌وتووه‌ ناتوانێت له‌گه‌ڵ ره‌وتی كاتدا ته‌ریب بێت، به‌ڵام دوای هه‌ندێ لێكدانه‌وه‌، داخۆ ده‌بێت چه‌ند كاتژمێر خه‌وتبێتم؟ بۆچی خۆر هه‌ڵنایه‌ت؟ هه‌ندێ چاویان بڕیبووه‌ ئاسمان وهه‌ندێكیش چاویان له‌سه‌ر میلی سه‌عاته‌كان دانابوو، هیچ كه‌س نه‌یده‌زانی چ روویداوه‌؟ له‌ تاریكێكی ئه‌نگوسته‌ چاودا هه‌نگاوی كه‌سێكی نابینای بۆپێشه‌وه‌ ده‌هاویشت، بۆخۆیشی نه‌یده‌زانی بۆ كوێ ده‌ڕوات، به‌ڵام گرنگ له‌وه‌یه‌ له‌چوارچێوه‌ی ماڵێكه‌وه‌ ده‌ربچێته‌ ده‌ره‌وه‌، هه‌رله‌ده‌رگای حه‌وشه‌ هاته‌ ده‌ره‌وه‌، روواڵه‌تی هه‌موو خانوه‌كانی به‌یه‌ك شێوه‌ بینی، ته‌نانه‌ت سیمای كه‌سه‌كانیش له‌دووره‌وه‌ هه‌رله‌یك ده‌چوون، كاتێك به‌كۆڵانه‌ ته‌نگه‌به‌ره‌كاندا تێده‌په‌ڕی گوێی له‌ گریان وشین و زایه‌ڵه‌ی مناڵ وژنان بوو، تاك وته‌را كه‌سی له‌سه‌ر شه‌قامه‌كان ده‌بینی، دوكان وكۆگاكان كڵۆمدرابوون، هه‌ستێكی نامۆ له‌نێو گه‌نجینه‌ی بیركردنه‌وه‌ی سه‌رتاپێی هێنابووه‌ له‌رزین، راچه‌ڵه‌كا له‌وه‌ی كه‌ لێیده‌ترسا رووبدات.
تاریكی به‌ته‌نها ئاسمان و زه‌وی نه‌گرتبووه‌ به‌ڵكو نیگای كه‌سه‌كانیشی ده‌مامكدابوو، به‌ڵام ئه‌وه‌ی به‌لاوه‌ زۆر سه‌یر بوو وێڕای ئه‌و تاریكیه‌ هێشتا ده‌یتوانی ته‌ماشای گرژی ولۆچی نێو چاوانی هه‌موو كه‌سێك بكات، ترس ونیگه‌رانی بیركردنه‌وه‌یان ببینێ، ئه‌گه‌ر به‌چاویش ئه‌وه‌ی نه‌بینایه‌ له‌له‌ره‌و ته‌وژمی ده‌نگی خه‌ڵك هه‌ستی پێده‌كرد.
بێ هیچ مه‌به‌ست ونیازێك له‌پێش سه‌مونخانه‌یه‌ك كه‌چه‌ند كه‌سێكی لێ‌ بوو بۆماوه‌ییه‌ك به‌بێده‌نگی راوه‌ستا ، باس هه‌رباسی ئه‌و تاریكییه‌ بوو كه‌چه‌ندین سه‌عات به‌سه‌ر واده‌ی دیاریكراوی تێپه‌ڕببوو به‌ڵام هه‌ر نه‌یده‌ویست بڕه‌وێته‌وه‌و رووناكیان بۆ بگه‌ڕێنێته‌وه‌، گه‌نجێك باسی ئه‌وه‌یكرد هه‌ساره‌ییه‌كی گه‌وره‌ پێش تیشكی رووناكی خۆری گرتووه‌و ناهێڵێت رووناكی بگاته‌ سه‌ر زه‌وی، یه‌كێكی تر باسی له‌وه‌ كرد تارمایی ئه‌م تاریكیه‌ جگه‌ له‌گوناهه‌كانی مرۆڤ هیچی تر نیه‌، به‌ ئه‌سپایی لێیان دوور كه‌وته‌وه‌، له‌هاتن ورۆیشتنیدا هیچ په‌رۆشیان پێوه‌ دیار نه‌بوو، چونكه‌ ئه‌و به‌بێده‌نگی هات وبه‌بێده‌نگیش به‌جێیانی هێشت، هه‌ندێ سه‌مونی كڕی وبه‌ره‌و ماڵه‌وه‌ گه‌ڕایه‌وه‌، ئه‌و كاته‌ی سه‌مونه‌كه‌ی هه‌ڵگرت بیری له‌وه‌ ده‌كرده‌وه‌ له‌تاریكی و رووناكیدا سكی برسی نانی ده‌وێت، خه‌ریكی ئاماده‌كردنی ژه‌مه‌ خۆراكێك بوو، چونكه‌ له‌پاش مردنی دایكی به‌ته‌نیا له‌و خانوه‌ ده‌ژیا، ئه‌و كاته‌ بیری له‌وه‌ ده‌كرده‌وه‌ باشبوو دایكی له‌ژیاندا نه‌مابوو ئه‌گه‌ربۆ ئه‌م كاته‌ بمایه‌ مێشكی سه‌ری ده‌برد، له‌باره‌ی ئه‌و تاریكیه‌ كه‌هیچ كه‌س سه‌رچاوه‌كه‌ی نه‌ده‌زانی، ئه‌و كاته‌ی خه‌ریكبوو یه‌كه‌م پاروو قوتبدات ئه‌وه‌ی به‌بیردا هات، ئه‌مه‌ ژه‌مه‌ خۆراكی نیوه‌ڕۆیه‌ یان به‌یانی... كات به‌گوێره‌ی سه‌عاته‌ هه‌ڵواسراوه‌كه‌ی ژووره‌وه‌ یه‌كی پاش نیوه‌ڕۆ بوو، به‌پێوه‌ری تاریكیش نیوه‌ی شه‌و بوو، خۆی به‌خوێندنه‌وه‌ی ئه‌و كتێبه‌وه‌ خه‌ریك كرد كه‌زۆر كه‌س بڕوای وابوو هه‌موو شتێكی تێدایه‌ به‌ڵام ئه‌و بڕوای وابوو چیرۆكی هه‌موو كه‌سێكی تیایه‌، هه‌موو كه‌س ده‌توانێت تارمایی سێبه‌ری خۆی تیایه‌ بدۆزێته‌وه‌ كه‌له‌هه‌موو كاتێكی بیركردنه‌وه‌و نیگه‌رانیدا به‌شێوه‌ییه‌كی به‌رده‌وام ده‌گه‌ڕایه‌وه‌ سه‌ر ئه‌و كتێبه‌ بۆ ئه‌وه‌ ده‌گه‌ڕا به‌ڵكو چیرۆكی خۆی تیا بدۆزێته‌وه‌، چه‌ندین ساڵ له‌هه‌وڵێكی بێهووده‌ بوو، هیچ كات نه‌یتوانی بوو به‌مه‌رامی خۆی بگات، چیرۆكی خۆی له‌ناو كتێبه‌كه‌ بدۆزێته‌وه‌، له‌باوه‌شی ئه‌و تاریكیه‌ی رۆژدا بۆ چه‌ندین سه‌عات خه‌ریكی خوێندنه‌وه‌ی كتێبه‌كه‌ بوو زۆر چیرۆكی ناوازه‌ی تیا خوێنده‌وه‌، چیرۆكی ئه‌و پیاوه‌ی له‌قوڵایی ده‌ریادا زیندانیكرا، چیرۆكی ئه‌وپیاوه‌ی ئاینده‌ی ده‌خوێنده‌وه‌و رێگری له‌پێش روودانی داده‌نا، سه‌رگوزشته‌ی ئه‌و خه‌ڵكه‌ی خۆشترین خواردنیان ده‌خوارد كه‌چی تامه‌زرۆی هه‌ندێ خۆراك بوون كه‌له‌برسێیتیشدا هیچ كه‌س تامه‌زرۆی ناكات، چیرۆكی ئه‌و برایانه‌ی سته‌م له‌براكه‌یان ده‌كه‌ن، چه‌ندجارێك ده‌یویست ته‌ماشای سه‌عاته‌ هه‌ڵواسراوه‌كه‌ی سه‌ر دیواره‌كه‌ بكات به‌ڵام زوو په‌شیمان ده‌بووه‌وه‌ چونكه‌ بڕوای وابوو مادام تاریكی شاری نگرۆ كردووه‌ ئه‌و ساته‌؛ كات هیچ گرنگی نامێنێت، كاتێ كتێبه‌كه‌ی به‌ده‌سته‌وه‌ بوو هه‌ستی ماندوو بوونێكی تاقه‌ پروكێنه‌ر به‌شێوه‌ی ئازارێكی ناوه‌خت جه‌سته‌ی داگیركرد، له‌ساتێك له‌نێوان ئاگایی وخه‌وتندا وا هه‌ستیده‌كرد ئه‌م دێرانه‌ی له‌كتێبه‌كه‌ خوێنده‌وه‌" ئه‌و كه‌سه‌ی به‌ته‌نیا له‌تاریكیدا بیر له‌چیرۆكی خۆیده‌كاته‌وه‌، چیرۆكی هه‌موو خه‌ڵك فه‌رامۆشده‌كات، هه‌ر به‌ته‌نها ده‌ژێت وده‌مرێت....." خه‌وێكی ئێكجار قورس پێڵوی چاوی داخست و وشه‌كانی تری بۆ نه‌خوێندرایه‌وه‌، له‌خه‌ونیشیدا كه‌وتبووه‌ سۆراغی وشه‌ ته‌واوكه‌ره‌كان به‌ڵام هیچی لێ ده‌ستنه‌كه‌وت.
سبه‌ی كاتێ به‌خه‌به‌ر هاته‌وه‌ خۆر سه‌رله‌نوێ رووناكی په‌خشكردبوو، دارو دیواری به‌تیشكی زێڕینی ره‌نگڕێژكردبوو، ته‌نها شتێك له‌و به‌یانیه‌ له‌قوڵایی بیركردنه‌وه‌ی شه‌پۆلیده‌دا خوێندنه‌وه‌ی ته‌واوی وشه‌كانی ناو كتێبه‌كه‌ بوو، به‌ڵام له‌به‌ر ئه‌وه‌ی كاتی ده‌وامه‌كه‌ی نزیكببووه‌وه‌ بێ ئه‌وه‌ی هیچ شتێكی تر بكات به‌خێرایی خۆی گۆڕی و به‌ره‌و فه‌رمانگه‌كه‌ی به‌ڕێكه‌وت ، هێشتا حه‌په‌ساندنی رۆژی پێشوو به‌رینه‌دابوو هه‌ر سه‌یری ئاسمان وخۆر و تیشكی رووناكی ده‌كرد، هه‌ندێجار له‌وه‌ به‌گومانبوو نه‌یده‌توانی متمانه‌ به‌وه‌ بكات ئه‌وه‌ی دوێنێ روویداوه‌ به‌شێك بێت له‌ژیان به‌ڵكو هه‌ر به‌ته‌نها خه‌ونێكی تره‌ له‌و خه‌ونانه‌ی له‌شه‌وه‌ درێژه‌كان ده‌بینرێ.
له‌ناو فه‌رمانگه‌كه‌یدا له‌سه‌ر زمانی هاوڕێكانی ئه‌وه‌ی بۆ ده‌ركه‌وت  تاریكی دوێنێ  راستییه‌كی حاشا هه‌ڵنه‌گربوو، قسه‌و باسی هاوڕێكانی له‌باره‌ی تاریكی دوێنێوه‌ ئه‌ویان سه‌رسام كردبوو، نیگه‌رانی وترسیش به‌رووی هیچ كامێكیانه‌وه‌ نه‌ڕه‌ویبووه‌وه‌، به‌ڵام هه‌موو ئه‌و باسانه‌ی كه‌له‌سه‌ر تاریكیه‌كه‌ ده‌كراو ئه‌و ترسانه‌ی له‌لای هاوڕێكانی هه‌بوو نه‌بووه‌ مایه‌ی ئه‌وه‌ی چیرۆكه‌كه‌ی خۆی له‌بیر به‌رنه‌وه‌، پاش ئه‌وه‌ی له‌كاره‌كه‌ی گه‌ڕایه‌وه‌ ماڵه‌وه‌ بێ ئه‌وه‌ی پێڵاوه‌كه‌شی له‌پێ دابكه‌نێت یه‌كسه‌ر رووی له‌كتێبه‌كانی كردو ئه‌و كتێبه‌ی هه‌ڵگرت كه‌دوێنێ ده‌یخوێنده‌وه‌، به‌ڵام رووداوێكی سه‌یرتری له‌رووداوی تاریكیه‌كه‌ی دوێنێ بینی كه‌به‌جارێ حه‌په‌سا، به‌روخسارێكی په‌شۆكاوه‌وه‌ جارێكی تر سه‌یری به‌رگی كتێبه‌كه‌ی كرد نه‌وه‌ك به‌هه‌ڵه‌ ئه‌و كتێبه‌ی هه‌ڵگرتبێت، لاپه‌ڕه‌كانی هه‌ڵدایه‌وه‌ بێ سوود بوو، به‌شپرزه‌یی وبه‌راكردن له‌ماڵه‌ هاته‌ ده‌ره‌وه‌، دوای ئه‌و رۆژه‌ش ئه‌وه‌ی به‌لاوه‌ زۆر سه‌یر بوو نه‌یده‌زانی چ هێزێك پاڵنه‌ری بوو كه‌ ئه‌وه‌نده‌ به‌خێرایی ئه‌ویان گه‌یانده‌ كتێبخانه‌كه‌ی ناو بازاڕ له‌وێ داوای ئه‌و كتێبه‌ی كرد، خاوه‌ن كتێبخانه‌كه‌ كتێبێكی داوه‌ ده‌ستی، هه‌موو لاپه‌ڕه‌كانی هه‌ڵداوه‌، له‌گه‌ڵ هه‌ر لاپه‌ڕه‌یه‌كدا زیاتر ترسی لێده‌نیشت، به‌خاوه‌ن كتێبخانه‌كه‌ی راگه‌یاند، ئه‌ویش وێرای ئه‌وه‌ی به‌لایه‌وه‌ رووداوێكی سه‌یر بوو، نوسخه‌یه‌كی تری له‌و كتێبه‌ی بۆ هێنا، لاپه‌ڕه‌كانی ئه‌وه‌شی هه‌ڵدایه‌وه‌ هه‌مان رووداوی بینی، ئه‌و ساته‌ تووشی سه‌راسیمه‌ بوو، خاوه‌ن كتێبخانه‌كه‌ش هاوشێوه‌ی ئه‌و په‌شۆكا به‌ڕاده‌ییه‌ك حه‌په‌سابوو، كه‌ به‌ئه‌سته‌م توانی له‌سه‌ر كورسییه‌كه‌ دابنشێت، ئه‌و ئێواره‌یه‌ چه‌ندین كتێبخانه‌ی تر گه‌ڕا، به‌ڵام له‌ ئه‌نجامدا زانی ئه‌و كتێبه‌ی ئه‌و به‌دوایدا ده‌گه‌ڕێ، كه‌ به‌بڕوای خۆی چیرۆكی هه‌موو كه‌سێكی تیایه‌ له‌هه‌ر جێگاییه‌ك ئه‌و كتێبه‌ی بینی سه‌یری كرد لاپه‌ڕه‌كانی سپی بۆته‌وه‌و ته‌نها وشه‌ییه‌كی تیا نه‌ماوه‌.
به‌نائومێدی له‌تاریكی ئێواره‌ وه‌ختێكی دره‌نگدا گه‌ڕایه‌وه‌ وبیری هه‌ر له‌لای ئه‌وه‌بوو ئیتر چیرۆكی هه‌موو كه‌سێك له‌تاریكی وون بوو.
