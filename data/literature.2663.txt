هێشتا ده‌كرێ‌ بڵێم كه‌ به‌یانی دانه‌هاتووه‌، خه‌ریكم سه‌یری بارینی به‌فر ده‌كه‌م وه‌ك ئه‌وه‌ی پارچه‌ په‌ڕۆ بن به‌سه‌رماندا ببارن، ئه‌م كڵۆ به‌فرانه‌ چه‌ند ئاواتیانه‌ له‌ رێگادا به‌یه‌ك بگه‌ن‌و ژوان ببه‌ستن، بۆنی یه‌كتر ده‌كه‌ن، شان له‌ شانی یه‌كتر ده‌خشێنن، لێك زیز ده‌بنه‌وه‌، با ده‌یانهێنێ‌‌و ده‌یانبا، به‌ڵام تا نه‌گه‌نه‌ سه‌ر زه‌وی ئاوێزانی یه‌كتر نابن.
نازانم بۆ ئه‌و ته‌زووه‌ خۆشه‌م به‌ له‌شدا ده‌هات، پێده‌چوو تازه‌ به‌فر بوو له‌م ساڵدا یان خۆشم نازانم، له‌شم هه‌مووی ختووكه‌ ده‌درێ‌، هه‌ستم دنه‌م ده‌دات، گیانم ده‌كه‌وێته‌ خرپه‌و جۆش و ته‌زووی خۆشی ته‌واوی گیانم داده‌گرێ‌.
ئه‌و چاره‌گه‌ی دوایی شه‌وێ‌ خه‌وم لێ‌ هه‌ڕما، له‌خۆشی به‌فر، له‌ سه‌رمای تازه‌و له‌ دیمه‌نی به‌ر تاریكایی.
هێند جوان بوو نه‌مده‌زانی كه‌ به‌یانیش دابێت دیمه‌نی به‌فر ته‌واو سه‌رسامم ده‌كات، سپێدێ‌ كه‌ سیپاڵی سپیایه‌تی ڕۆژ ختووكه‌ی داو شه‌و به‌ غه‌مگینی باری تێكنا، هه‌ستامه‌وه‌ ته‌ماشای دیمه‌نی به‌فر بزواندمی و به‌قه‌د بستێك، به‌فر له‌سه‌ر لق و كه‌دی داره‌كان هه‌ڵستابوو.
ته‌واو دیمه‌نێكی سه‌رنج ڕاكێش بوو، شه‌پڵارۆخه‌كه‌م(1) ده‌ستدایێ‌ و به‌ر هه‌یوان و ده‌رگای ماڵه‌وه‌م ماشت، كۆره‌ ڕێی پێش ماڵان كه‌ به‌فری كه‌وتبووه‌ سه‌ر به‌ ئاسته‌م ده‌ناسرایه‌وه‌، نازانم چی وای لێكرد بڕۆم پێ‌ به‌ پێ‌ حه‌زم ده‌كرد به‌فره‌كه‌ بشكێنم هه‌نگاوه‌كانم زۆر له‌ بن یه‌ك و بۆڕ ده‌هاویشتن تا هه‌رچی زوو رۆحی به‌فر بپه‌ستمه‌وه‌، چه‌ند گوناحه‌ ئه‌نفال‌و قركردنی به‌فر، كه‌ پێی له‌سه‌ریان داده‌گری، رۆحیان ده‌فیچێته‌وه‌، هه‌ناویان ده‌ته‌قێته‌وه‌‌و به‌ هێواشی زینده‌ به‌ عارد ده‌كرێن.
دونیا كش و مات، جگه‌ له‌ جریوه‌ جریوه‌ی چۆله‌كه‌ و دمك دمكانه‌كانیان‌و قوقوی كه‌ڵه‌شێر، هیچی ترم نه‌ده‌بیست، ناو به‌ ناویش بۆڕه‌ی ڕه‌شه‌وڵاغێك و زه‌ڕه‌ی كه‌رێك هۆشی ده‌بزواندمه‌وه‌.
كه‌ پێم داده‌نا وه‌ك بڵێی نه‌رماییه‌ك بپه‌ستمه‌وه‌، پێ‌ له‌ رۆحێكی شل‌و خاو بنێم، ئاوێكی كه‌م له‌ بنه‌وه‌ی به‌فر ده‌نگی لێوه‌ ده‌هات، هه‌تا گه‌یشتمه‌ ده‌رگاكه‌، به‌فر وا چه‌سپاوی كردبوو ئه‌گه‌ر خۆی نه‌كرابایه‌وه‌ زه‌حمه‌ت بوو له‌ ڕێسمه‌ ببردرێ‌.
بیرم كرده‌وه‌ ده‌رگای مزگه‌وت هه‌میشه‌ كراوه‌یه‌ ده‌نا ئه‌و پیره‌مێرده‌ قور وه‌سه‌ره‌، به‌خۆیی و كه‌م ئه‌ندامییه‌كه‌ی خۆی نه‌ده‌لوا ئاوا له‌ناو به‌فردا بله‌نگێته‌وه‌.
ئاوڕێك ده‌ده‌مه‌ دواوه‌ شوێنه‌كه‌م به‌رزه‌ ته‌واوی خانووی ئه‌و ماڵانه‌ هۆده‌یه‌ك یان دوانیان ڕه‌شایی و ده‌وری كولانكه‌ی سۆبه‌ی دارانیان دووكه‌ڵی لێهه‌ڵده‌ستێ‌.
له‌ به‌رامبه‌ر ده‌رگاكه‌ی ده‌وه‌ستم،  زه‌رداییه‌ك به‌قه‌د دوو بست و نیو ره‌نگی به‌فره‌كه‌ی گۆڕیووه‌.
گومانی ئه‌وه‌ ده‌به‌م میزی ئه‌و بێت، ده‌بێ‌ به‌م به‌یانییه‌ رشتبێتی.
بێ‌ ئه‌وه‌ی ده‌نگ و نووزه‌ی ببیستم له‌خۆوه‌ نازانم بۆ ئه‌وه‌ دێته‌ خه‌یاڵم به‌ نه‌زمێكی غه‌مگینه‌وه‌ سه‌رم نزیك ده‌رگاكه‌ی ده‌كه‌مه‌وه‌‌و دێمه‌ گۆ: مام پیرۆ خۆت ئاماده‌ بكه‌ من عیزرائیلم، هاتووم رۆحت بكێشم، سێ‌ جاران ده‌ڵێمه‌وه‌.
هیچ وه‌ڵامێكم نابیستم، ره‌نگبێ‌ بترسێ‌ نه‌وێری نوقه‌ بكا.
چه‌ند خوله‌كێك ڕاده‌مێنم، ناهه‌قی نییه‌ به‌م به‌یانییه‌ زووه‌و به‌م سه‌رماو سۆڵه‌وه‌، ئاخر مه‌رگیش ده‌رگاو دووڕیانێكه‌ ده‌بێ‌ پێیدا تێپه‌ڕین.
من ڕاده‌چڵه‌كێم به‌ده‌ست خۆم نییه‌ ده‌ترسێم، خۆ به‌فری سه‌ر دارقه‌یسییه‌كه‌ چه‌ند به‌ بێ‌ ده‌نگی‌و به‌بێ‌ شه‌رمی ته‌ماشام ده‌كا، ترس له‌ سپی‌و پاكیی به‌فر، له‌ جریو جریووی شه‌ڕی چۆله‌كه‌كان ته‌واوی گیانم ده‌كه‌وێته‌ له‌رزین، له‌ شوێنی خۆم ده‌گه‌ڕێمه‌وه‌.
ده‌چمه‌وه‌ پێش ماڵ.
دایكم له‌به‌ر ده‌رگاوه‌یه‌، دان ده‌داته‌ مریشكه‌كان، سه‌رسام ده‌بێ‌ كه‌ من ده‌بینێ‌، شه‌پڵارۆخه‌كه‌ له‌ بنمیچی ده‌رگاوه‌ داندراوه‌، پێش ئه‌وه‌ی دایكم قسه‌ بكا ده‌ڵێم دایه‌ ده‌چم به‌فری سه‌ربان ده‌ماڵم.
جارێك به‌فرماڵه‌كه‌ به‌سه‌ر به‌فره‌كه‌ دادێنم هه‌ست ده‌كه‌م ده‌ستم ته‌واو سڕ بوون خۆم ڕێك دێنم، ددانه‌كانم له‌یه‌ك گیرو له‌پی ده‌ستم له‌یه‌ك ده‌خشێنم، دادم نادا ئاوڕ بۆ دواوه‌م ده‌ده‌مه‌وه‌، لووله‌ی بۆڕییه‌ سۆبه‌كه‌ ئاگری لێ‌ هه‌ڵده‌ستێ‌ ده‌ستم له‌به‌ر تینی گه‌رماكه‌ داده‌نێم، تومه‌ز له‌گۆ چوون ئازاریان زیاتر ده‌بێ‌، دێمه‌وه‌ ژوور له‌تاو ئازارم فرمێسكم له‌چاو دێنه‌ خوار، خودایه‌ رقی گه‌رمی به‌فر بۆ ئاوا سارده‌، ره‌نگی سپی به‌فر بۆ وا رۆح ره‌شه‌؟!
ده‌چمه‌وه‌ بن نوێنه‌كه‌م خۆم ڕێك دێنم دوای ماوه‌یه‌ك دایكم نان تێده‌كات، بانگم ده‌كا .. من پێده‌چێ‌ له‌ ژووری مام پیرۆ و له‌ هه‌یوانی مزگه‌وت به‌م به‌یانییه‌ ترسابم ، ترسه‌كه‌ تا دێ‌ پتر ده‌بێ‌ وه‌خته‌ بڵێم دایه‌ هێندێك ڕه‌ژوو و ترس ده‌ركردن.
نانه‌كه‌م بۆ ناخورێ‌، هێڵێنجم دێنێ‌، دایكم خۆم به‌ قوربان ده‌كا، زۆرم له‌گه‌ڵ ده‌كا بێ‌ سووده‌، من جگه‌ له‌ ترسه‌كه‌، نیگه‌رانم.
خۆشم نازانم نیگه‌رانییه‌كه‌م له‌ چیدایه‌، له‌ خۆوه‌ بێتاقه‌تی ڕووی تێ‌ كردووم، دڵم لێ‌ ده‌دا ناتوانم هۆكاری ترس و دڕكاندنه‌كه‌شی بدركێنم.
له‌ مزگه‌وته‌وه‌ بانگ ڕاده‌هێشترێ‌: هه‌ڕوونه‌ سه‌ر قه‌بران مسوڵمانێك مردوه‌ ڕه‌حمه‌تتان لێ‌ بی.
من گومانم بۆ كاكیله‌ ده‌چێ‌، ئه‌و بریندارێك بوو برینه‌كه‌ی فره‌ سه‌خت بوو، خۆم ده‌پێچمه‌وه‌، به‌ كوته‌كه‌كه‌ پێش خۆم و ڕێگاكه‌م ده‌كوتمه‌وه‌، به‌فر و سایه‌قه‌یه‌كی گه‌لێك سارده‌، ده‌گه‌مه‌ یه‌ك دوو كه‌س ئه‌وانیش بۆ سه‌ر قه‌بران ده‌چن.
من یه‌كه‌م جاره‌ به‌بێ‌ زانینی كه‌سی مردوو ملم بۆ سه‌ر قه‌بران ناوه‌، ئه‌وان دێنه‌ پێش لێیان ده‌پرسم، موچڕكم پێدا دێ‌ له‌رز ده‌مگرێ‌، ده‌ترسم هۆكاری مردنه‌كه‌ی من بم، لێیان ده‌پرسم به‌چی مردووه‌، ئه‌وان ده‌ڵێن ئه‌م به‌یانییه‌ كه‌ سه‌ربانی مزگه‌وتیان ماشیوه‌، ته‌ماشایان كردووه‌ له‌ش ساردو سڕ له‌سه‌ر جێگایه‌كه‌ی بۆ تا هه‌تایه‌ دڵی وه‌ستاوه‌، دڵم ده‌خێورێته‌وه‌، قه‌ره‌می ده‌ست‌و پێیانم ده‌شكێ‌، له‌ شوێنی خۆم له‌ناو به‌فره‌كه‌ له‌سه‌ر چیچكان داده‌نیشم، ئه‌وان به‌ره‌و قه‌برستان ده‌چن، منیش لێیان راده‌مێنم، له‌ نێوان من‌و ئه‌واندا دونیایه‌ك خه‌یاڵ هه‌یه‌، ئه‌وان یه‌ك یه‌ك ‌و دوو دوو ده‌رۆن، له‌نێوان من‌و قه‌برستاندا له‌پێش چاو‌و خه‌یاڵمدا دونیایه‌كی سپی، رۆحی مام پیرۆ سوور سوور دێت‌و ده‌چێ‌، سوورێك تامی مردن‌و ره‌نگی مردنی له‌ به‌فرێكی سپیدا له‌خۆدا هه‌ڵگرتووه‌.
به‌فرێك بۆنی سپی مردنێكی سپی له‌ هه‌ناوی زه‌وییه‌كی سوور‌و ره‌شدا هێناوه‌.
هه‌موو ئه‌و شه‌پلارۆخ‌و به‌فرماڵانه‌ی به‌ زه‌ویماندا هێنان، مردنێكی سپی وه‌ك به‌فریان ختووكه‌دا.
