ئه‌وه‌ نزیكه‌ی مانگێكه‌ له‌ كاره‌كه‌م ده‌ریان كردوومه‌ ، ئاخر من له‌هیچ كارێكی ژیانمدا ئه‌وه‌م قبووڵ نه‌بووه‌ ، كه‌ چه‌شنی ڕۆبۆتێك ، یان هه‌روه‌ك كۆیله‌یه‌ك سه‌یربكرێم و مامه‌ڵه‌م له‌ كه‌ڵدا بكرێت ، ته‌نانه‌ت ئه‌وه‌شم قبوڵ نییه‌ كه‌ بۆچوون و هه‌سته‌كانم له‌ ژێر هه‌ست و بۆجوون كه‌لێكی تردا دا ته‌پێنن و خۆیان له‌سه‌رم فه‌رزبكه‌ن ،وه‌لێ ده‌ركردنم له‌و كاره‌ په‌یوه‌ندی به‌مانه‌وه‌ نه‌بووه‌ كه‌باسم لێیانه‌وه‌ كرد ، به‌ڵكو په‌یوه‌ندی به‌وه‌وه‌ هه‌یه‌ كه‌ وا ناهه‌قی ده‌رحه‌ق به‌ دۆستێكمه‌وه‌ كرا ، مرۆ ده‌بێت له‌ ته‌واوی كایه‌كانی ژیانیدا هه‌ڵوێستی هه‌بێت ، مرۆی بێ هه‌ڵوێست ویژدانی دا ته‌پیوه‌ و ناخێكی مردۆخی هه‌یه‌ ، ئه‌مه‌ درێژ ده‌بێته‌وه‌ بۆ پرۆسه‌ی نووسینیش .
تۆ سه‌یركه‌ كاتێك نووسه‌رێك خاوه‌ن هه‌ڵوێست نییه‌ ، نووسینه‌كانی هه‌میشه‌ جێگای بێبایه‌خن ، ده‌شێت وه‌ها سه‌یری زۆربه‌ی نووسه‌رانی كوردیی بكه‌ین ، نووسه‌ری بێ هه‌ڵوێست ، نووسه‌رێكی مردووه‌ ، له‌ ڕوانگه‌ی ویژدانییه‌وه‌ ، هه‌ربۆیه‌ كاتێك ئه‌م ته‌رزه‌ نووسه‌رانه‌  به‌ناو ساڵه‌وه‌ ده‌جن>>ته‌مه‌نیان هه‌ڵده‌كشێت<<خاوه‌ن دیدگای تایبه‌ت به‌ خۆیان نین ، به‌واتایه‌كی كورتتر : ئه‌م نووسه‌رانه‌ نابنه‌ خاوه‌ن جیهانبینی تایبه‌ت به‌ خۆیان ، نووسه‌ر ده‌بێت یاخی ببێت ، ئاخر یاخیبوون مه‌رج نییه‌ هه‌رله‌لای كه‌سی چه‌وساوه‌وه‌ بێت به‌رامبه‌ر به‌ كه‌سی زۆر دار ، ده‌شێت له‌لای كه‌سێك دروست ببێت كه‌وه‌ك بینه‌ر(ته‌ماشاكه‌ر)سه‌یری ئه‌و چه‌وسانه‌وه‌یه‌ ده‌كات ، ئه‌وه‌ جا له‌ئه‌ركی مرۆدۆستی و نووسه‌ربوون ، له‌لایه‌نی ئیمانیشه‌وه‌ دواجار هه‌ر وه‌هاش ده‌كه‌وێته‌وه‌ ، ئه‌وه‌ته‌ له‌ ئاینی ئیسلام و له‌ ئینجیلیشدا ئاماژه‌ به‌وه‌ كراوه‌ ، كه‌ چیت بۆخۆت پێ خۆشه‌ ، ده‌بێت وه‌هاش پێت خۆش بێت بۆ برای دینیت ، وه‌ به‌ پێچه‌وانه‌شه‌وه‌ ،پێویسته‌ لێره‌دا ئه‌وه‌ش بزانرێت ، كه‌وا چه‌مكی هه‌ڵوێست له‌ چه‌مكی یاخیبوون جیا بكرێته‌وه‌ ، چه‌مكی یاخیبوونیش له‌ چه‌مكی رقلێبوونه‌وه‌جیا بكرێنه‌وه‌ .
نووسه‌ری جواننووس و خودان جاویدانیبوون پێویستی به‌ هه‌را و زه‌نای ده‌روێشه‌كانی ده‌ورووبه‌ری نییه‌ ، به‌ڵكو له‌ و په‌ڕی گۆشه‌گیرییه‌وه‌ ئه‌وه‌ی پێی شیاوی نووسینه‌ ، ده‌ینووسێت .
نووسه‌ر به‌یه‌ك دوو پروپاگه‌نده‌ی ڕۆژنامه‌گه‌ریی ، یان دوو نووسه‌ری هاورێی به‌ته‌مه‌نی له‌دیمانه‌یه‌كدا ئاماژه‌ به‌ناوه‌كه‌ی بكه‌ن نابێته‌ نووسه‌ری جاویدانی ، یاخود به‌ ته‌كلیف كردنی هاورێیه‌تی و ده‌عوه‌تكردنی منه‌تباری به‌نووسینێك له‌باره‌ی ده‌قێكییه‌وه‌ نابێته‌ ده‌نگێكی ڕاسته‌ قینه‌ ، ئاخر نووسه‌ری جواننووس گۆشه‌گیری و بێده‌نگی پێ شیاوتره‌ ،له‌ هه‌راو زه‌نای وه‌ها .
من نامه‌وێت هیچ كه‌سێك ببێته‌ زووڕنا ژه‌نم ، خۆشم نازانم زووڕنا بژه‌نم ، ته‌نانه‌ت سه‌ربه‌هیچ حیزب و لایه‌ن و گروپ و فكره‌یه‌ك نیم ، من سه‌ر به‌ فكری خۆمم ، سه‌ربه‌ لایه‌نی وشه‌م ، هه‌ر هێزی وشه‌ شه‌ به‌رده‌وامیم پێده‌دات و وزه‌ی ئه‌م سه‌فه‌ره‌ زیاد ده‌كات ، هه‌ربۆیه‌ ئه‌گه‌ر ئه‌ده‌ب له‌ ڕۆحیه‌تی ده‌ربراوی كۆمه‌ڵگه‌ له‌ دایك بووبێت ، ئه‌وا به‌ دڵنیاییه‌وه‌ وشه‌ش هی ئاده‌میزاده‌ .ئیشی من ئه‌وه‌یه‌ هه‌میشه‌ وشه‌م زیاتر خۆشتر بوێت ، هه‌ڵوێستم هه‌بێت ته‌نانه‌ت له‌ نێوان وشه‌ی گونجاو و وشه‌ی نه‌گونجاو، له‌ پرۆسه‌ی نووسیندا وشه‌ وزه‌ی به‌رده‌وامی زمانی نووسینه‌ ، ته‌واوی نووسینه‌كانی ئه‌ده‌بیاتیش به‌رله‌وه‌ی لێپرسینه‌وه‌ بێت ، ئه‌خلاقه‌ ، من كارم به‌و نووسه‌رانه‌وه‌ نییه‌ ، كه‌ باسی بێ ئه‌خلاقی مرۆیی و به‌ڕه‌ڵایی پرۆسه‌ی سێكسی ده‌كه‌ن و دواجار ئه‌خلاق و سێكسییان پێ ناشیرینه‌ ، ئه‌وانه‌لای من جوانن ، قه‌د گه‌ره‌كم نییه‌ له‌ روانگه‌وه‌ خوێنه‌ر ڕاوبكه‌م ، خوێنه‌ر لای من ئه‌و به‌شه‌ ته‌واو نه‌كراوه‌ی ناو نووسینه‌ كه‌ خۆم پێم  ته‌واو نه‌بووه‌ ، خوێنه‌ر زۆر له‌وه‌ گه‌وره‌تر له‌ رێگه‌ی  نووسینه‌كانمه‌وه‌ كاری بازركانییان پێوه‌بكه‌م ، یاخود چه‌واشه‌یان بكه‌م ، كارم به‌وه‌نییه‌ كی نووسینه‌كانم ده‌ خوێنێته‌وه‌ ، وه‌لێ هه‌میشه‌ حورمه‌تم بۆ ئه‌و خوێنه‌ره‌ هه‌یه‌ كه‌ ده‌مخوێنێته‌وه‌ ، ئه‌و نووسه‌رانه‌ی كه‌وا به‌م نووسینه‌مه‌وه‌ په‌ست بوونه‌و له‌ ناخه‌ بێ هه‌ڵوێسته‌كانیانه‌وه‌ ئه‌ستووربوونه‌ ، پێم باشه‌ لێره‌ وه‌ ده‌ست له‌ خوێندنه‌وه‌ی ئه‌م نووسینه‌ م هه‌ڵبگرێت ، من بۆئه‌وه‌ نانووسم هیچ كه‌سێك په‌ست بكه‌م ، له‌بیری هیچ كه‌سێكی دیاری كراویشدا نییمه‌ ، نووسین له‌ دیدی منه‌وه‌ بۆ تاكه‌ كه‌سێك نییه‌ ، به‌ڵكو له‌ تاكه‌ كه‌سێكه‌وه‌یه‌ بۆ ته‌واوی  چین و توێژه‌كانی كۆمه‌ڵگه‌.
نووسن سه‌رئێشه‌یه‌ ، ئه‌مه‌م له‌نووسینێكی تریشدا تركاندووه‌ ، من ئێستا تووشی هه‌مان سه‌ر ئێشه‌ییم بوومه‌ته‌وه‌ و بێ ئه‌ندازه‌ سه‌رم دێشێت ، به‌درێژایی رۆژی رابردوو هه‌رله‌ماڵه‌وه‌ بوومه‌ و خوێندوومه‌ته‌وه‌ ، به‌م نیوه‌ شه‌وه‌ وا هه‌ست ده‌كه‌م كه‌لله‌ سه‌رم هه‌موو دنیای لێباركراوه‌ و به‌و هه‌موو بارگرانیه‌شه‌وه‌ هه‌رله‌ سه‌ر ملمه‌،  ناو سه‌رم پڕبووه‌ له‌ مس ، یه‌كێك له‌ خووه‌ قۆڕه‌كانی من ئه‌وه‌یه‌ كه‌ ده‌خوێنمه‌وه‌ له‌ ڕۆژێكدا ده‌مه‌وێت له‌ زۆربه‌ی بواره‌كاندا بخوێنمه‌وه‌ ،واته‌ حه‌زده‌كه‌م له‌و باغه‌ پڕگوڵه‌دا چه‌پكه‌ گوڵه‌كه‌م له‌ ته‌واوی ڕه‌نگه‌كانی بێت ، هه‌ربۆیه‌ ڕۆژی رابردوو ڕۆمانێك وكتێبه‌ شیعرێك و لێكۆڵینه‌یه‌یه‌كی ئه‌ده‌بیم خوێندۆته‌وه‌ ، هه‌نووكه‌ بێز له‌ خوێندنه‌وه‌ش ده‌كه‌م ، ئاخر هه‌ر دوێنێ‌ وه‌ها به‌نێو شه‌پۆلی خه‌یاڵه‌كاندا ڕۆچووم نه‌مده‌زانی شه‌وه‌ ،  یان ڕۆژه‌ ، له‌كات و ساتیش هه‌ر مه‌پرسه‌ ، ڕه‌نگه‌ جوانیش نه‌بێت ، كه‌وا نهێنی خۆم بتركێنم ، وه‌لێ ئه‌وه‌ی پێم شیاوی باس بێت ، ده‌یتركێنم ، من هه‌نووكه‌ تووشی نه‌ خۆشییه‌كی سه‌یر و سه‌خت بوومه‌ و شته‌كانی ده‌وورو به‌رم و ڕابردووم له‌بیر ده‌چێته‌وه‌ ، هه‌زارانجار له‌بیرم  چووه‌ من كێمه‌ ؟ نازانم ده‌مه‌وێت چی بكه‌م ؟ هه‌ربۆیه‌ كه‌ بیر له‌ خۆم ده‌كه‌مه‌وه‌ شه‌رم له‌ بوونی خۆم ده‌كه‌م و تف به‌ناوه‌كه‌م داده‌كه‌م ، ئاخر زۆرجار له‌بیر ده‌چێت كه‌ ده‌ژیم ، فكرناكه‌م مرۆڤم ، تێنوومه‌ و له‌یاد ده‌كه‌م ئاو بخۆمه‌وه‌ ، ته‌نانه‌ت چه‌ند ڕۆژێك له‌مه‌و به‌ر زۆرمیزم ده‌هات و له‌بیری ئه‌وه‌ نه‌بووم میز بكه‌م ، دواجار بنی  زگم هێشا و ناجار هه‌ستام چوومه‌ ئه‌ده‌بخانه‌ ، له‌بڕی ئه‌وه‌ی قایشی كابۆكه‌م بكه‌مه‌وه‌ و زنجیریشی بۆ خواره‌وه‌ بترازێنم ، بێئاگا له‌وه‌ی كه‌وا سێ قۆپچه‌ی قه‌میسه‌كه‌م كردۆته‌وه‌ و خه‌ریكی قۆپچه‌ی چواره‌مم.
نزیكه‌ی هه‌فته‌یه‌كه‌ خه‌ریكی نووسینی شیعرێكم ، له‌ ته‌واوی ئه‌و شه‌وانه‌دا له‌به‌ر به‌دگومانی و ڕاڕاییه‌ هه‌میشه‌ییه‌كه‌م به‌سه‌ری دا ده‌چمه‌وه‌ و ده‌ستكاری ده‌كه‌م ، ئه‌مشه‌و هه‌روه‌ك شه‌وه‌كانی تر به‌دگومانییه‌كه‌م درێژه‌ی هه‌بوو ، له‌ده‌رئه‌نجامی ئه‌و به‌دگومانی و ڕاڕاییه‌مه‌وه‌ هاتمه‌ سه‌ر قه‌ناعه‌تكردن به‌ یه‌كه‌مین ده‌ستنووسی ده‌قه‌كه‌م ، بیركردنه‌وه‌یه‌ك به‌ ئه‌ندازه‌یه‌كی قووڵ بردمی و له‌بیری ئه‌و شه‌ش شه‌وه‌ گیری كردم ،كه‌ من ئه‌م شه‌ش شه‌وه‌ له‌ و په‌ڕی به‌ تاڵبوونه‌وه‌ خولاومه‌ته‌وه‌ ، هه‌ستم به‌به‌تاڵییه‌كی قووڵی رۆحی كرد ، بێئۆقره‌ ده‌ستنووسه‌كه‌م خوێنده‌وه‌ ، دێر به‌ دێر پیاسه‌م له‌ گه‌ڵ وشه‌كانیدا كرد ، یه‌ك یه‌ك وشه‌كانیم دواند ، نازانم خه‌تای كاممان بوو ، قه‌ناعه‌تمان به‌یه‌كتری نه‌ما و من و وشه‌كانی ناوی له‌یه‌كتری زوویربووین و هیچ شتێك به‌یه‌كتری نه‌به‌ستاینه‌وه‌ ،ئه‌م ده‌ستنووسه‌شم دڕاند ، هه‌روه‌ك شه‌ش ده‌ستنووسه‌كه‌ی تر ، لێره‌دا نامه‌وێت كه‌س به‌هه‌ڵه‌ له‌م نووسینه‌م تێبگات ، چۆن مه‌به‌ستم گڤه‌ گڤی نێو نووسین نییه‌ ، ته‌نانه‌ت من جگه‌ له‌وه‌ی كه‌ گوایه‌ هه‌مه‌ ، دواجار هیچ شتێكی تر نیمه‌ ، من ئه‌زموون نانووسمه‌وه‌ ، هیچ ئه‌زموونێكم نییه‌ شیاوی هه‌ڵوه‌سته‌ سه‌ركردن بێت ، ئه‌م دانپیانانه‌م بۆئه‌وه‌ نییه‌ ، كه‌ هیچ كه‌سێك ده‌ست نه‌كات به‌ نووسین ، وه‌لێ من قسه‌ی ناخم ده‌كه‌م و له‌ گه‌ل ده‌رووننی خۆمدا ده‌دوێم ، ئه‌وه‌ ڕاستییه‌كه‌ و ناتوانم له‌ ده‌ست ڕاستییه‌كانیش ڕابكه‌م ، هه‌تا ئه‌م نووسینه‌ ، نووسین بۆمن جگه‌ له‌ سه‌ر ئێشه‌ و خووه‌كی بێ كات و جه‌م هیچیتر نه‌بووه‌ .
>>ئه‌گه‌ر سه‌روه‌ت و سامانی هه‌ر وه‌ڵاتێك به‌یه‌كسانی به‌ شێوه‌یه‌كی  دادپه‌روه‌رانه‌ به‌سه‌ر تاكه‌كاندا دابه‌شبكرێت ، ئه‌وا ته‌واوی كه‌رته‌كانی ناو ئه‌و وڵاته‌ به‌ شێوه‌یه‌كی ڕێك و ئه‌كتیڤ به‌ره‌و پێش ده‌چن<< .
نازانم بۆ ئه‌وه‌م نووسی ، ڕه‌نگه‌ ئه‌وه‌م به‌وه‌ په‌یوه‌ست بێت ، كه‌من نزیكه‌ی له‌ ( ده‌ ) هه‌فته‌نامه‌ و گۆڤار وه‌كو هاوكار كارده‌كه‌م و ناشتوانم بژیم ، نووسه‌ری تریشمان هه‌یه‌ ، ته‌نیا له‌یه‌ك بڵاو كراوه‌یه‌ و مانگانه‌ له‌ سێ وتار تێنا په‌ڕێت ، له‌ بڵاو كردنه‌وه‌ ، ئه‌وا له‌و په‌ڕی خۆشنودییه‌كانی ژیان ده‌ژیت و ژیانێكی شاهانه‌ی هه‌یه‌ ، له‌وه‌ش گه‌ڕێ‌ چه‌ند كتێبی بۆ چاپ ده‌كرێ ، نه‌خۆشییه‌كه‌ی من ئه‌وه‌یه‌ كه‌من ناتوانم بژیم و وازیش له‌و خووه‌ پیسه‌ ناهێنم ، ئاخر ئه‌وه‌ م پێ خۆشگوزه‌رانی ژیان نییه‌ ، كه‌ خۆم به‌ هه‌موو شتێكه‌وه‌ هه‌ڵبواسم ، ته‌نانه‌ت هه‌نووكه‌ له‌م خۆ هه‌ڵواسینه‌ش به‌ربوومه‌ته‌وه‌ و زۆر له‌ دواوه‌ی ژیانم ، ژوورێكم هه‌یه‌ له‌و په‌ڕی په‌راوێزی ژیان ، خۆم و كۆمه‌ڵێك كتێب ، خۆم و خه‌یاڵگه‌ڵێك ، خۆم و چه‌ند نووسینێكی ته‌واو نه‌كراو ، من خه‌م بۆ خۆشگوزه‌رانی ژیان ناخۆم ، وه‌لێ خه‌می من ئه‌وه‌یه‌ كه‌ چاره‌نووسی دوو كه‌سی ترم به‌ده‌ستی خۆم ڕه‌ش و ناخۆش كردووه‌ ، كه‌ دوو خۆشه‌ویسترین كه‌سی دنیامن ، كه‌ هاو سه‌ره‌ به‌رێزه‌كه‌م و كوره‌ خنجیلانه‌كه‌من ، به‌درێژایی ته‌مه‌نی پێكه‌وه‌ بوونمان هه‌میشه‌ فێڵم لێیان كردووه‌ ، هه‌زاران جار درۆی ڕه‌شم له‌ ڕۆژگاری ڕوونییان كردووه‌ ، گوایه‌ پاره‌م نییه‌ و به‌دزیشه‌وه‌ كتێبم كڕیوه‌ ، ئاخر ئه‌مه‌ یه‌كێكه‌ له‌ كاره‌ دكتاتۆرییه‌كانی من ، ڕه‌نگه‌ بچێته‌ خانه‌ی شێتیشه‌وه‌، هه‌موو لێكدانه‌وه‌یه‌ك له‌باره‌مه‌وه‌ دروسته‌ ، هه‌زاران جار تڤم به‌ پاره‌ داكردووه‌ ، كه‌ بۆچی چاوی نییه‌ و به‌ نابینایی دروستییان كردووه‌ ، ئاخر ئه‌گه‌ر یه‌ك چاویشی هه‌بوایه‌ به‌س بوو ، نه‌ ده‌چووه‌ گیرفانی هه‌موو كه‌سێكه‌وه‌ ،  من پاره‌م ته‌نیا له‌به‌ر ئه‌وه‌ خۆش ده‌وێت وه‌سیله‌یه‌كه‌ بۆ ژیان ، ئاخر له‌وه‌ عه‌یبتر چی ؟ له‌به‌ر بێ پاره‌ی هه‌زاران كه‌ره‌ت كتێبم دزیوه‌ ، له‌دڵی خۆمدا دزینی كتێبم حه‌ڵاڵ كردووه‌ ، نازانم ئه‌وه‌ چییه‌ لێم ناگه‌ڕێ و به‌رده‌وامیم پێده‌دات ، هه‌میشه‌ شه‌ڕه‌ له‌لام ، شه‌ڕمه‌ له‌ گه‌ڵ خۆمدا ، هه‌میشه‌ له‌ سه‌ر كتێب له‌ گه‌ڵ خێزانه‌كه‌م شه‌ڕمانه‌ ، كه‌ ئه‌وه‌ درۆیه‌كی تره‌ ، گوایه‌ :((كتێب باشترین هاوڕێی مرۆڤه‌ ))، ئاخر زۆر جار كتێب ده‌بێته‌ خراپترین دوژمنی مرۆڤیش ، وه‌ك چۆن عه‌شق دڕنده‌ ده‌كاته‌ مرۆڤ و مرۆڤیش ده‌كاته‌ دڕنده‌ .
هه‌میشه‌ ماڵه‌كه‌مان پڕه‌ له‌ شه‌ڕ ، وشه‌كان له‌ ناو خۆیان به‌شه‌ڕ دێن ، ده‌ستنووسه‌كان به‌رۆكی كتێبه‌كان ده‌گرن ، پێنووسه‌كان نووكی یه‌كتری ده‌شكێنن ، كتێبه‌كان به‌رگی یه‌كتری لێده‌كه‌نه‌وه‌ ، یاده‌وه‌رییه‌كان ئاوی به‌ لووعه‌ی ماڵ ده‌كه‌نه‌وه‌ و خۆیان له‌ خه‌یاڵه‌كان بسمیل ده‌كه‌ن ، ئاوێنه‌ هه‌میشه‌ له‌ گه‌ڵ وێنه‌كانی ناویدا عه‌داوه‌ته‌ ،  تاریكی ده‌ستی به‌ خوێنی ئه‌جنده‌ كان سوور و سپی بووه‌ ، هێنده‌ شه‌ڕ له‌ ژووره‌كه‌م كراوه‌ خۆشم له‌به‌رچاو كه‌وتووه‌ ، ته‌نانه‌ت زۆرجار شه‌ڕه‌ و واهه‌ستناكه‌م ئاژاوه‌یه‌ك هه‌یه‌ ، شه‌ڕێكی شێت ژووره‌كه‌می ماره‌ كردووه‌ و تا مردن ئه‌م جه‌نگه‌ ناوه‌ستێنێت ، ئه‌مه‌ كتێب بوو ئه‌م جه‌نگه‌ی به‌رپا كرد ، هه‌ر كتێبیش ده‌بێت ئه‌م جه‌نگه‌ بوه‌ستێنێت ، تازه‌ تاقه‌تم  له‌شه‌ڕچوو ، شه‌ڕیش هیلاكبووله‌ شه‌ڕكردنم ، ئه‌مێستا خۆزگه‌ی ئه‌وه‌ ده‌خوازم خه‌ونێكی دوور و رێژ و قووڵ قووڵ بمباته‌وه‌ ، ته‌نیا ئه‌و كاته‌ له‌و خه‌وه‌ به‌ئاگا بێم ، كه‌ خاره‌ ئیزڕائیل به‌خۆیی و ده‌فته‌رێكی مه‌زنی ئه‌فسانه‌ی و پێنووسێك له‌سه‌ر سه‌رمه‌ ، چاوه‌ڕێیه‌ شیعره‌كه‌م ته‌واو بكات و بیبات له‌ و گۆڤاره‌ بڵاو بكاته‌وه‌ كه‌ لاپه‌ڕه‌كانی به‌ قه‌د ئاسمانه‌ .
هه‌زاران كه‌ره‌ت به‌ هه‌ورینم گوتووه‌ ، هه‌موو ئه‌و شتانه‌ی ده‌چنه‌وه‌ سه‌ر ڕه‌چه‌ڵه‌كی كتێب بیان سوتێنێت .
له‌ په‌راوێزترین گۆشه‌ی ئه‌م دنیایه‌دا ، جێگایه‌كی لابه‌لام بۆ ڕابخه‌ ، ده‌مه‌وێت له‌ پێناوی مه‌رگ بنووم .
