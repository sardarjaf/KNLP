سیناریۆ" له‌گه‌ڵ ده‌ستپێكردنی مۆسیقا( كوڕێك كه‌ ته‌مه‌نی نزیكه‌ی 32 ساڵ ده‌بێت له‌سه‌ر جۆلانێك دێت و ده‌چێت، بوكێكی شڕی به‌ده‌سته‌وه‌یه‌، بوكه‌كه‌ی ده‌ستی  هی خوشكه‌ بچووكه‌كه‌یه‌تی كه‌ ئه‌نفالكراوه‌ و دوای 26 ساڵ هه‌ڵیگرتووه‌).
ده‌نگی كوڕه‌كه‌ به‌تۆماركراوی نمایش ده‌كرێت و له‌گه‌ڵ بوكه‌كه‌ ده‌ست به‌ نواندن ده‌كات( هه‌موو شه‌وێك ئه‌بێت به‌ده‌م گریانه‌وه‌ سه‌ر بنێمه‌وه‌، كاتێكیش كه‌ ده‌گریم نازنم بۆخۆم ده‌گریم یاخود بۆ ئێوه‌، یان بۆ تۆ، تۆ به‌ڵێنی ئه‌وه‌ت به‌مندا كه‌ هاوسه‌رگیری له‌گه‌ڵ هیچ كه‌سێكی دنیا نه‌كه‌ی، باوكه‌ دڵ ڕه‌قه‌كه‌ت تۆی به‌مننه‌دا له‌به‌ر ئه‌وه‌ی بێ‌ كه‌سوكارم و كه‌سێكی هه‌ژاربووم، ئێستا نازانم تۆ هاوسه‌رگیریت كردووه‌، یاخود چاوه‌ڕوانی من ده‌كه‌ی، نازانم له‌چی قوژبنێكی ئه‌م وڵاته‌دای، خۆشم ده‌وێی، دڵنیابه‌ ڕۆژێك دێت هه‌ر یه‌ك ده‌بینینه‌وه‌ و له‌باوه‌شت ده‌گرم).
كوڕ: ئه‌مساڵ 26 ساڵ به‌سه‌ر كاره‌ساتی ئه‌نفال تێده‌په‌ڕێَت من هه‌ر چاوه‌ڕوانی ئێوه‌م، ئێوه‌ش ڕۆژێك بیر له‌من ناكه‌نه‌وه‌، بگه‌ڕێنه‌وه‌، تۆش( به‌ بوكه‌كه‌ی ده‌ستی ده‌ڵێت) هه‌میشه‌ ده‌مت سورده‌كه‌یت و ته‌ماشام ده‌كه‌یت( ده‌ست ده‌كات به‌گریان) بۆ وه‌ڵام ناده‌یه‌ته‌وه‌، بۆ هیچ قسه‌یه‌كم له‌گه‌ڵ ناكه‌یت، ئه‌ی تۆ پێت نه‌وتم كاكه‌ من كه‌ گه‌وره‌ بم ژن بۆ تۆ ده‌كه‌م،.. بۆ.
بۆ  وه‌ڵامناده‌یته‌وه‌.
( كوڕه‌كه‌ هه‌ڵده‌ستێت و بوكه‌كه‌ی ده‌ستی فڕیده‌دات و ته‌ماشای دیواری سه‌ر شانۆكه‌ ده‌كات كه‌ داسێك هه‌ڵواسراوه‌، به‌ده‌نگێكی به‌رز هاوار ده‌كات) من چیتر نامه‌وێت به‌ته‌نها بژیم، نامه‌وێت به‌ته‌نها بژیم،باوكه‌ گیان من كاتێك ئه‌م (داسه‌) ده‌بینم تۆم به‌یاد دێته‌وه‌، تۆ به‌م داسه‌ی ده‌ستت ژیانمانت دابین ده‌كرد، به‌ڵام ئێستا داسه‌كه‌ت بێ‌ ناز كه‌وتووه‌ و كه‌سێك نیه‌ به‌كاری بهێنێت بۆ ژیان.
ده‌نگ له‌پشت په‌رده‌وه‌ دروست ده‌بێت كه‌ باوكی كوڕه‌كه‌یه‌( كوڕه‌كه‌م  هه‌وڵبده‌ كه‌سێكی نیشتمان په‌روه‌ر به‌، خاك و ئاوی ئه‌م نیشتمانه‌ت خۆش بویت، شوێن كه‌وته‌ی راسیته‌كان به‌، ئامانجت به‌رژه‌وه‌ندی گشتی بێت،  شه‌ڕ له‌گه‌ڵ ژیان بكه‌ و كۆڵ مه‌ده‌، كۆڵدانت مایه‌ی شكستخواردنه‌، كوره‌كه‌م هه‌میشه‌ بیر له‌و كه‌سانه‌ بكه‌ره‌وه‌ كه‌ ژیانیان به‌نه‌هامه‌تی ئه‌به‌نه‌ سه‌ر، ئه‌و پاره‌ووه‌ نانه‌ی كه‌ هه‌ییته‌ بیكه‌ به‌دوو له‌ته‌وه‌ و به‌شێكی بۆخۆت و به‌شه‌كه‌ی تریش بۆ ئه‌و كه‌سانه‌ی كه‌ هیچ ژه‌مه‌ خواردنێكیان نیه‌).
كوڕ: باوكه‌ گیان من ئامۆژگاریه‌كانتم وه‌رگرتووه‌، به‌ڵام من هه‌ر چاوه‌ڕوانی ئێوه‌م، ئێوه‌ش نه‌گه‌ڕانه‌وه‌، دایكه‌ ، باوكه‌، ئه‌سما گیان خوشكه‌ بچووكه‌كه‌م، ئێوه‌ بنون و منیش لێره‌ ده‌نوم و ڕۆژێك دێت ده‌گه‌م به‌ئێوه‌( یه‌ك تف له‌خۆی ده‌كات) ده‌ڵێت:هه‌ی گه‌مژه‌ تاكه‌ی به‌قسه‌ی ئه‌م هیچ و پوچانه‌ ده‌خه‌ڵه‌تێیت، ئه‌وان پێیتیان گوت ئیسك و پروسه‌كی كه‌سوكاره‌كانتان بۆ دێنینه‌وه‌ نه‌یان هێینایه‌وه‌، ئه‌وان گوتیان چاره‌سه‌ری خه‌مه‌كانتان ده‌كه‌ین( پێده‌كه‌نی ) ئائا به‌باشترین شێواز بۆیان كردین( توڕه‌ ده‌بێت) من ئیسك و پروسكم ناوێت، من ده‌مه‌وێت ئێوه‌ی تاوانبار كه‌ده‌ستان له‌خوێنی خێزانه‌كه‌مدا بووه‌ له‌سیداره‌ بدرێن.
( كوڕه‌كه‌ پشت ده‌كات له‌بینه‌ران و ده‌ست ده‌كات به‌میزكردن، له‌وكاته‌دا ده‌نگی زه‌نگی دادگا دێت و ته‌ماشایه‌كی ناوه‌ڕاستی شانۆكه‌ ده‌كات ) من زۆر دڵخۆشم به‌م ڕۆژه‌( توڕه‌ ده‌بێت) بۆچی ڕێگانیشانده‌ربوون بۆ ئه‌وه‌ی كه‌سوكاره‌كه‌م زینده‌به‌چاڵ بكه‌ن.
( كوڕ ڕووی ده‌كاته‌ جێگایه‌كی شانۆكه‌) جه‌نابی دادوه‌ر ئه‌م تاوانبارانه‌ ده‌یان كه‌سیان زینده‌ به‌چاڵ كردووه‌، منیان به‌تاك و ته‌نیا به‌جێهیشتووه‌، ئێره‌ ئه‌و شوێنه‌یه‌ كه‌له‌گه‌ڵ خێزانه‌كه‌م ده‌ژیام، هه‌ر بۆیه‌ خه‌ڵكی به‌چاوێكی تر ته‌ماشم ده‌كه‌ن، ئه‌و كه‌سه‌ی كه‌ خۆشم ئه‌ویست له‌دوای خێزانه‌كه‌م نه‌یان داپێم، پێم ده‌ڵین ژن بۆ بده‌ین به‌ كه‌سێكی بێ‌ كه‌س، ده‌ قسه‌ بكه‌، بۆ وه‌ڵام ناده‌یته‌وه‌، بۆ، بۆ.
( كوڕ ورته‌ ورتێك له‌گه‌ڵ خۆی ده‌كات) چه‌ند خۆش ده‌بوو، ئه‌و كه‌سه‌ی كه‌ خۆشم ده‌ویست پێكه‌وه‌ ژیانمان له‌م كه‌لاوه‌یه‌ پێكبهێنایه‌.
( له‌وكاته‌دا ده‌نگی ده‌هۆڵ و زوڕنا دێت و قه‌ره‌باڵغی دێت و كوڕ به‌ ده‌م خۆشیه‌وه‌ داده‌به‌زێته‌ ناو بینه‌ران و پێیان ئه‌زێتان كێشا كه‌ ئاماده‌ی ئاهه‌نگه‌كه‌م بوون، ده‌ست خۆشییان لێده‌كات كه‌ ئاماده‌ی ئاهه‌نگه‌كه‌ی بوونه‌)
(كوڕ كاتێك كه‌ ده‌چێته‌ سه‌ر ته‌خته‌ی شانۆ دیسانه‌وه‌ ده‌ست بۆ دادوه‌ر دڕێژ ده‌كات) تۆش هیچت له‌ده‌ست نه‌هات، بۆ ناتوانی ئه‌و تاوانبارانه‌ سزا بده‌یت، به‌ڵام ئه‌گه‌ر دوو داماوی وه‌كو من سه‌رپێچیه‌ك بكه‌ن به‌زووترین كات له‌سێداره‌مان ده‌ده‌ن، زیندانمان ده‌كه‌ن، ئه‌ی به‌نه‌عله‌تی بیت جه‌نابی دادوه‌ر، من هه‌رگیز ئه‌و ڕۆژانه‌م بیر ناچێته‌وه‌ كاتێك هه‌موو ئه‌ندامانی خێزانه‌كه‌م له‌ده‌ستدا، سه‌ردانی هه‌ر جێگایه‌كی ئه‌م وڵاته‌م ده‌كرد داوای ناسنامه‌یان لێمده‌كرد، منیش ناچار بووم خۆم كرد به‌ كوڕی خێزانێكی دیكه‌ و ته‌سكه‌ره‌م ده‌رهێنا، كه‌چی دوای ماوه‌یه‌ك ده‌ستگیركرام و تۆش زیندانیت به‌سه‌رم جێ‌ به‌جێكرد، به‌بیانووی ئه‌وه‌ی ته‌سكه‌ره‌كه‌م ساخته‌یه‌، به‌ڵام من ناچاربووم ئه‌و كاره‌ بكه‌م، تۆ ناتوانی تاوانباران ده‌ستگیر بكه‌یت، زیندانیان بكه‌یت.
( ئاخێك هه‌ڵده‌كیشێت )  كوڕی تاوانباران له‌وڵاتانی دنیا خوێندن ته‌واو ده‌كه‌ن و منیش خوێنده‌واریم نیه‌، كوڕی ئێوه‌ له‌ دیسكۆكانی ئه‌وروپا مه‌ی ده‌خۆنه‌ و له‌یانه‌ شه‌وانه‌كان ڕۆژ ده‌كه‌نه‌وه‌، منیش به‌شانازیه‌وه‌ له‌م خانووه‌ كه‌لاوه‌یه‌ ڕۆژ ده‌كه‌مه‌وه‌، ئێوه‌ ڕوره‌شی هه‌ردوو دنیا، منیش دڵ گه‌ش.
كوڕ: من، تۆ، ئێوه‌، هه‌ر هه‌موومان له‌ وڵاتێك ده‌ژین، گووی گه‌وره‌ وڵاتی بۆگه‌نكردووه‌، پێ خستنه‌ سه‌ر گووه‌كان و شوێن كه‌وته‌ی ئێوه‌ش له‌لایه‌ن ده‌یانی وه‌كو تۆ وه‌یكردووه‌  من له‌چوارچێوه‌ی ئه‌م خانووه‌ كلاوه‌یه‌ نه‌چمه‌ ده‌ره‌وه‌، من ده‌مه‌وێت چیتر چاره‌ی ڕه‌شت نه‌بینم جه‌نابی دادوه‌ر، من ده‌مه‌وێت خه‌مه‌كانم به‌نوسینمه‌وه‌ی یاده‌وه‌ریه‌كانم به‌تاڵ بكه‌م، من ئه‌مه‌وێت چیتر وه‌كو ئاژه‌ڵ بژێم و له‌ چوار چێوه‌ی ئه‌م خانووه‌یه‌ نه‌یه‌مه‌ ده‌ره‌وه‌.
( ڕوناكی ورده‌ ورده‌ كز ده‌بێت له‌گه‌ڵ موسیقا كوڕ له‌سه‌ر جۆلانیه‌كه‌ دانیشتووه‌  و خه‌یاڵ بردویه‌تی له‌وكاته‌ ده‌نگی بانگ ده‌بیسترێت و كوڕ ده‌كه‌وێـته‌ خواره‌وه‌)
كوڕ" مه‌لا كیان هه‌ر كاتێك ویستت بانگ بفه‌مریت، تكات لێده‌كه‌م ئاگادارم بكه‌ره‌وه‌، چونكه‌  دایك و باوكم له‌ كاتی پرۆسه‌ی ئه‌نفال وه‌ك مام ئه‌حمه‌د گێرایه‌وه‌ بۆم زۆر پاڕابوونه‌وه‌ له‌و سه‌ربازه‌ی كه‌ سواری ئیفاكه‌یانیان كردبوو، پێیان گوتبوو بكه‌ بۆ خاتر ئه‌و بانگه‌ ئازادمان بكه‌، به‌ڵام ئه‌و زۆر دڕندانه‌ یه‌ك قۆناغه‌ تفه‌نگێكی كێشابوو به‌ناو ده‌می دایكم، ئێستا تێگه‌یشتی مه‌لا گیان، من به‌و ده‌نگه‌ ڕاده‌چه‌ڵه‌كێم و دایكم به‌بیر دێته‌وه‌.
( ورده‌ ورده‌ روناكی كز ده‌بێت، له‌گه‌ڵ كزبوونه‌وه‌ی روناكی چه‌ندین وێنه‌ی سه‌دام حوسێن ورده‌ ورده‌ شۆڕده‌بنه‌وه‌ بۆ سه‌ر ته‌خته‌ی شانۆ)، نزیك وێنه‌كان ده‌بێته‌وه‌ و پێیان ده‌لێت: ناشرینی چاو زلی، بۆچی بڕیارتدا كه‌سوكاره‌كه‌م زینده‌ به‌چاڵ بكه‌ن( ته‌ماشایه‌كی جێگای دادوه‌ره‌كه‌ ده‌كات و ده‌ڵێت) نه‌توانی تۆمه‌تبارانی دیكه‌ سزا بده‌یت ئه‌بێت ئه‌م تاوانباره‌ به‌سزا بگه‌یه‌نێت،ده‌ی خێراكه‌( له‌وكاته‌ ورده‌ ورده‌ وێنه‌كانی سه‌دام ده‌كه‌ونه‌ خواره‌وه‌ )
كوڕ: سه‌دام ڕۆشت، به‌ڵام ده‌یان سه‌دامی تر ڕۆژانه‌ دێن و ده‌رۆن، سه‌دامه‌كانی ئێستا زینده‌ به‌چاڵمان ناكه‌ن، سه‌دامه‌كانی ئێستا منداڵه‌ ساواكانمان له‌ناو بێ شكه‌ ناكوژن،  به‌ڵام سه‌دامه‌كانی ئێستا پاڕه‌وی گه‌وره‌ قووت ئه‌ده‌ن و سزا نادرێن، سه‌دامه‌كانی ئێستا ئاگایان له‌منی كه‌سوكاری ئه‌نفالكراو نیه‌، ئاگایان له‌باری ده‌رونیم نیه‌، ته‌نها بۆ به‌رژه‌وه‌ندی خۆیان كار ده‌كه‌ن، جه‌نابی دادوه‌ر من بێتاقه‌تم بوو له‌ده‌ستی ئه‌م سه‌دامانه‌، من ئێره‌ت بۆ به‌جێ‌ ده‌هێڵم، تۆ ناتوانی تاوانباران به‌سزا بگه‌یه‌نێ‌، ترست له‌سه‌دامی ئێستا هه‌یه‌، من ده‌ڕۆم، ئێڕه‌ به‌جێده‌هێڵم و ده‌ڕۆم بۆ وڵاتێك كه‌ده‌م و چاوی سه‌دامه‌كان نه‌بینم.
