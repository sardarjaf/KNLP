وه‌ك چۆن له‌ سه‌ر كوشتنی مه‌لایان بێده‌نگ نابین ، له‌سه‌ر ئه‌وه‌ش ناكرێ‌ بێده‌نگ بین مه‌لا فتوای كوشتنی ئه‌وانی تر بدات ، فتوای كوشتنی زه‌رده‌شتیان كارێكی ناجۆره‌و پێچه‌وانه‌ی ده‌قه‌كانی قورئانیشه‌ ” دینی خۆتان بۆ خۆتان و دینی خۆم بۆ خۆم ” ، ” یان كێ‌ ویستی با بڕوا بێنێ‌ و كێش ویستی با بڕوا نه‌هێنێ‌ ” نامه‌وێ‌ لێره‌دا ده‌قی قورئانی بكه‌مه‌ پاڵپشت ، به‌ڵام ناكرێ‌ له‌سه‌ر بیرو باوه‌ڕ فتوای كوشتن بدرێ‌ ، لێره‌دا ئه‌ركی وه‌زاره‌تی ئه‌وقاف و حزبه‌ ئیسلامیه‌كان و یه‌كێتی زانایانی ئیسلامیه‌ وه‌ڵام بده‌نه‌وه‌ نه‌ك نوسه‌رێك .
به‌ڵام یه‌كێتی زانایان له‌ بری وه‌ڵام به‌ فتوای كوشتن له‌ خه‌می نمایش نه‌كردنی فیلمێكان ، نه‌ك ئه‌و هه‌موو سوكایه‌تیه‌ی به‌ مرۆڤ ده‌كرێت ، یه‌كێتی زانایان له‌و ماوه‌دورو درێژه‌دا نامه‌یه‌كیان بۆ حزبه‌كان نه‌نارد ، كه‌ كێشه‌كان چاره‌سه‌ر بكه‌ن، له‌سه‌ر به‌ سه‌گ چواندنی كۆمه‌ڵی كورده‌واری قسه‌یان نه‌بوو ، به‌ بارزانیان نه‌گوت كه‌ كورسیه‌كه‌ به‌شێكی گه‌وره‌ی كێشه‌كانی هه‌رێمه‌ چاره‌سه‌ری بكه‌ ، بۆ یه‌ چاوه‌ڕوانی ئه‌وه‌ش نیم لێیان داكۆكی له‌ ئازادی و جیاوازی بكه‌ن .
چه‌ند ساڵ له‌مه‌وبه‌ر له‌ كۆڕو به‌رنامه‌یه‌كی تایبه‌تا به‌ مامۆستا عه‌لی قه‌ره‌داخیم گوت ،ئێوه‌ وه‌ك زانایانی ئیسلام ،وه‌ك خوێنه‌وارو روناكبیرانی ئیسلام، پێویسته‌ له‌مپه‌رێ‌ بۆ ئه‌و هه‌موو فتواو بڕیاری مه‌رگه‌ دابنه‌ن كه‌ هه‌ر جار ناجارێ‌ كه‌سێك به‌ ناوی ئیسلامه‌وه‌ ده‌یدات ، نمونه‌ی شیعری ئه‌و زاناو شاعیرومه‌لایانه‌م بۆ هێنایه‌وه‌ كه‌ بۆ هۆشیاری خه‌ڵك قوربانیان داوه‌و كه‌سیش به‌ كافری له‌ قه‌ڵه‌م نه‌داون ، نمونه‌ی شیعری خه‌یام ، مه‌لای گه‌وره‌، قانع ، هه‌روه‌ها شیعره‌كانی دڵدارو بێكه‌س ، نامه‌وێ‌ لێره‌دا نمونه‌كان دوو باره‌بكه‌مه‌وه‌ ” ئه‌و ده‌م ته‌واوی شیعره‌كانم باسكردو تۆماریش كران “، به‌ڵام ده‌پرسم فه‌ره‌ج فوده‌ شایانی ئه‌وه‌ بوو بكوژرێ‌ ، عه‌بدولخالق مه‌عروف مابا باشتر نه‌بوو ، له‌ فتوای كوشتنی ئه‌و شاعیرو ئه‌و نووسه‌ر ئیسلام و ئیسلامی سیاسی چی ده‌سكه‌وت ؟.
ئێستا كه‌ هه‌ندێ‌ مه‌لای ئه‌مڕۆ به‌راورد ده‌كه‌م به‌ مه‌لای گه‌وره‌ی كۆیه‌ یان مامۆستا قانع كه‌ ئه‌ویش مه‌لابووه‌ ، توشی سه‌رسامی ده‌بم ، گه‌لۆ كۆمه‌ڵی ئێمه‌ له‌ بری گه‌شه‌ به‌ دین بدات و تازه‌ی بكاته‌وه‌و ئازادی زیاتری تیا فه‌راهه‌م بكات ، له‌ هه‌وڵی كۆت و به‌ندایه‌ ، فتوادژی زه‌رده‌شتیه‌كان ، سه‌ره‌تایه‌كه‌ بۆ به‌ داعشستان بوون ، بۆیه‌ ناكرێ‌ هه‌ڵوه‌سته‌ی له‌ سه‌ر نه‌كرێت ، ئه‌وه‌ی له‌م چه‌ند ساڵه‌ی رابردودا هه‌ندێ‌ ره‌وتی ئاینی و مه‌لا ی نێو ئیسلام له‌ كوردستان بینیوییانه‌ بۆ دواخستنی كۆمه‌ڵ و كرانه‌وه‌ ، به‌ درێژای سه‌دان ساڵ دوژمنان نه‌یانبینیوه‌ ، بڕوانه‌ ئازادی ژن و به‌شداری كارایانه‌ی له‌ كۆمه‌ڵ و بڕیاردا له‌ رۆژگارێكی دورترو ئێستادا ، چه‌ند نمونه‌ی دیاری وه‌ك مه‌ستوره‌ی ئه‌رده‌ڵانی و عادیله‌ خانممان هه‌یه‌ ؟ئه‌مه‌ پرسیاره‌و هیچی تر .
