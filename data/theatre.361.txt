ره‌گوڕیشه‌ی فاشیزم له‌گوتاری شانۆی ئێمه‌دا مێژوویه‌كی نوێی هه‌یه‌، ئه‌وه‌ بونیادێكی ناو په‌یوه‌ندیی شانۆكاره‌، له‌ڕابردوودا ئه‌وه‌ فاشیزم نه‌بووه‌ له‌و په‌یوه‌ندیه‌ بوونی هه‌بووبێت، به‌ڵكو نه‌وه‌كان پۆلین بوونه‌ته‌ سه‌ر دوو جۆر، گوتاره‌ شانۆییه‌كه‌ش له‌نێوان شانۆیپه‌كی سوونه‌تی و شانۆیه‌كی ئه‌زمونكاریدا ده‌ركه‌وتووه‌، به‌مانایه‌كی تر ئه‌توانین بلێین ده‌ركه‌وته‌كانی شانۆی ئێمه‌ خۆی یه‌كلا كردۆته‌وه‌ له‌نێوان ستانسلافسیكی و گرۆتۆفسكی جا تاچه‌ند له‌ئاستی تیۆریه‌وه‌ كار له‌سه‌ر ئه‌وه‌ كراوه‌ ئه‌وه‌یان باسێكیتره‌، ئه‌وه‌ی تر كه‌ په‌یوه‌سته‌ به‌ده‌ركه‌وتنی فاشیزم، ده‌ره‌نجامی سه‌فه‌ری خۆرئاوای هه‌ندێك شانۆكاری ئێمه‌یه‌ هه‌ر له‌و شوێنه‌ی پێیان وایه‌ شتێك نه‌كراوه‌، ئه‌وه‌ش نه‌ك له‌ئه‌نجامی خوێندنه‌وه‌ی ڕه‌خنه‌یی بۆ ئاستی شانۆكه‌مان، به‌ڵكو له‌گه‌ڕانه‌وه‌ بۆ خۆیان، ئه‌و حاله‌ته‌ خودگه‌راییه‌ له‌ كوشتنی (ئێمه‌) و بانگه‌شه‌كردن بۆ (خود) وه‌ك پێشتر وتمان گرێیه‌كی سایكۆلۆژی خه‌ته‌رناكه‌.. ئه‌وه‌ ئه‌و مه‌ترسیه‌ بوو كه‌ ئێمه‌ هه‌ستمان پێی ده‌كرد، ئه‌وه‌ته‌ له‌سه‌ره‌نجامدا ئه‌و گرێیه‌ به‌ره‌و جۆرێك له‌ فاشیزم بوون هه‌نگاو ده‌نێت، زمانی ئه‌و جۆره‌ بیركردنه‌وه‌یه‌ بریتیه‌ له‌ هێنانه‌ ئارای فاشیزمی ڕه‌مزیی.. فاشیزمێك ده‌یه‌وێ شوناسی داهێنه‌ر به‌خۆی ببه‌خشێت تاوه‌كو هه‌موان ڕاوێژ به‌و بكه‌ن و له‌ده‌وری ئه‌و كۆ ببنه‌وه‌.. گرفتی ئه‌و جۆره‌ فاشیزمه‌ ئه‌وه‌یه‌ له‌به‌رگی تیۆره‌وه‌ دێته‌ بوون.. ئه‌وه‌ فاشیزمێكه‌ له‌بنه‌ره‌تدا به‌ته‌نیا په‌یوه‌ست نیه‌ به‌تاكه‌ كه‌سێكه‌وه‌ به‌ڵكو له‌ئاستی فه‌ردیه‌وه‌ ده‌په‌ڕێته‌وه‌ بۆ جۆره‌ ئه‌خلاقێكی فاشیانه‌.. كه‌ له‌ئێستاوه‌ مناڵه‌كان ئه‌و جۆره‌ توندوتیژیه‌ ره‌مزیه‌ی فاشیزم قبوول ده‌كه‌ن به‌سه‌ر خۆیانه‌وه‌ به‌ خه‌ونی ئه‌وه‌ی فاشیزم ببێته‌ ئه‌خلاقێك له‌داهاتوو ئه‌وانیش رۆلی هه‌مان كاره‌كته‌ر وه‌ربگرن.. ئه‌و خۆ بچووكردنه‌وه‌یه‌ له‌به‌رامبه‌ر فاشیزم بمانه‌وێ و نه‌مانه‌وێ زاده‌ی گه‌مژه‌یی و نه‌زانینێكی ترسناكه‌ درۆكردنی فاشیزمه‌ له‌گه‌ڵ تیۆر.. ناكرێت له‌و سه‌ره‌تایه‌ی فاشیزم بێده‌نگ بین.. له‌هه‌ر كوێیه‌ك بینیمان فاشیزم دێته‌ ده‌نگ ده‌بێت بیوه‌ستێنین.. ناكرێ له‌و دۆخه‌ی فاشیزمی ڕه‌مزی نه‌یه‌نیه‌ ده‌نگ.. ئه‌گه‌ر ده‌سه‌ڵاتی كوردیی له‌رێگه‌ی فاشیزمی سه‌ربازیه‌وه‌ ده‌نگه‌ جیاوازه‌كانی كۆمه‌ڵگای كوردیی په‌كده‌خات ئه‌وا فاشیزمی ڕه‌مزی هاتووه‌ بۆ ئه‌وه‌ی ته‌واوی ده‌نگه‌ شانۆییه‌كان بخه‌سێنێت و هه‌مووشی له‌ناو ده‌نگی خۆی چڕ بكاته‌وه‌.. ئیدی له‌وه‌ ده‌رچوو ئه‌و جۆره‌ كاره‌كته‌رانه‌ به‌ره‌نجامی گرێیه‌كی ده‌روونی بن ئیدی ئه‌وان زاده‌ی له‌دایكبوونی سه‌ره‌تایه‌كی ترسناكن بۆ فاشیزمێكی شانۆیی.
په‌یوه‌ندیی نێوان فاشیزم وتێكست.. په‌یوه‌ندیه‌كی ده‌مامكداره‌.. په‌یوه‌ندیه‌كه‌ په‌یوه‌سته‌ به‌دۆخێكی میكاڤیلیانه‌وه‌, ئه‌و په‌یوه‌ندیه‌ ڕه‌وایه‌تی خۆی له‌پشت تێكسته‌كانه‌وه‌ وه‌رده‌گرێته‌وه‌, مێژووی په‌یوه‌ندیه‌كه‌ ده‌گه‌ڕێته‌وه‌ بۆ سه‌ره‌تای له‌دایكبوونی فاشیزم, فاشیزم بۆ ئه‌و په‌یوه‌ندیه‌ ده‌گه‌ڕایه‌وه‌ سه‌ر تێكسته‌كانی نیتچه‌ وكانت, ئه‌و ڕه‌وایه‌تیدانه‌ به‌تێكست كوشتنی فره‌ره‌هه‌ندی تێكست و وه‌رگرتنی دیدێكی ڕووكه‌شانه‌ بوو, وه‌ك چۆن فاشیزم به‌م شێوه‌یه‌ ڕه‌فتاری له‌ته‌ك فه‌لسه‌فه‌ی هێزی نیتچه‌دا ده‌كرد.
فاشیزمه‌ شانۆییه‌كانی ئێمه‌ش هه‌مان دۆخ دووباره‌ ده‌كه‌نه‌وه‌, نایه‌ن قسه‌ له‌و دیده‌ تاركیبینه‌ بكه‌ن كه‌ خه‌ونی كورتكردنه‌وه‌ی شانۆیان له‌خۆیاندا هه‌یه‌, دێن قسه‌ له‌تیۆریست وده‌رهێنه‌رێك ده‌كه‌ن, تا بڕوای ئه‌وه‌مان پێبێنن, كه‌ ئه‌وان سه‌رچاوه‌كه‌یان له‌وێوه‌ هه‌ڵده‌قووڵێ‌, ئه‌و گه‌مه‌ میكاڤیلیه‌ له‌فاشیزمه‌وه‌ سه‌رچاوه‌ی گرت, ئه‌و ده‌مه‌ی میكاڤیلیان ناو ده‌برد به‌ هاوتاترین ومه‌زنترین بیریار.
فاشیزم له‌دنیای ئێمه‌دا دێت ده‌مامك ده‌خاته‌ سه‌ر ناوه‌كان, تا دنیا له‌و دیده‌ فاشیانه‌ی كورتبكاته‌وه‌, خۆی به‌میراتگری مایه‌ر هۆڵد ناو ده‌بات, نه‌ك له‌دنیابینی مایه‌رهۆڵده‌وه‌, به‌ڵكو هێنده‌ی ئینتیمای بۆ فاختانكوف و ستانسلافسكی هه‌یه‌, هێنده‌ ئینتیمای بۆ مایه‌ر هۆڵد خۆی نیه‌, چونكه‌ مایرهۆڵد ته‌نیا ده‌مامكێكه‌و نه‌یتوانیوه‌ له‌دنیابینی ئه‌ودا نزیك بێته‌وه‌, ئه‌و پڕۆسه‌ی ناونانه‌ بنیاتنانی ئه‌خلاقێكی فاشیانه‌یه‌, كه‌ مناڵه‌ نابیناكان سه‌رسامن, سه‌رسامیشیان به‌ته‌نیا په‌یوه‌ست نیه‌ به‌و نائاگاییه‌ی ئه‌وان له‌تیۆر, هێنده‌ی په‌یوه‌سته‌ به‌دۆخی گه‌ڕان به‌دوای شوناس و ئه‌و ڕه‌وانبێژیه‌ ئه‌خلاقیه‌ی به‌ناوی میتۆده‌وه‌ ده‌سخه‌ڕۆیان ده‌دات, كه‌ له‌بنه‌ڕه‌تدا میتۆد ئیشكردنه‌وه‌ نیه‌ له‌ته‌ك تیۆر, به‌قه‌د ئه‌وه‌ی میتۆد به‌ته‌نیا هه‌وڵێكی بانگه‌شه‌كاریه‌, نه‌ له‌ڕووبه‌ری نمایشدا ده‌یبینین, نه‌ له‌ئاستی ئه‌و مه‌عریفه‌یه‌وه‌ ده‌رده‌كه‌وێت.
تا ئه‌و شوێنه‌ی به‌دوای ئارگۆمێنت ده‌گه‌ڕێین بۆ خوێندنه‌وه‌ی ئه‌ركیۆلۆژی مه‌به‌ست له‌و كاره‌ ڕووی ئه‌و دیوی مانایه‌ ئه‌وه‌ گه‌ڕانه‌ له‌پێناو زه‌مینه‌یه‌ك بۆ ده‌یالۆگ.. ئه‌و پڕۆسه‌یه‌ بۆ خۆی وادوه‌كات كاره‌كانمان به‌ ئاكادیمی بكه‌ونه‌وه‌، ئه‌كادیمی بوون ڕیزكردنی داتاو بییلۆگرافیا و ریزكردنی ژماره‌ی كتێب وسه‌رچاوه‌كان نیه‌، ئه‌كادیمی بوون به‌ره‌نجامی ئیشكردنی میتۆدیانه‌یه‌، بۆیه‌ ده‌شێت به‌درلاێژایی نووسینێك قسه‌ له‌ئه‌كادیمی بوون بكرێت له‌كاتێكدا نووسین له‌ده‌ره‌وه‌ی میتۆد بێت، وه‌ك په‌تای ئه‌و جه‌هله‌ كوشنده‌یه‌ی كاریزمای ناوێكی شانۆیی به‌بیانوی ئه‌كادیمی بوون ده‌خوڵقێنێت به‌بێ ئه‌وه‌ی توانای ئه‌وه‌ی هه‌بێت له‌میتۆد نزیك بكه‌وێته‌وه‌، لێره‌وه‌ عه‌فه‌ویه‌ت ناوی ئه‌كادیمی بوونی لێ ده‌نرێت، ئه‌وه‌ش درێژ ده‌بێته‌وه‌ بۆ قسه‌كردن له‌سه‌ر ئارگۆمێنت ئه‌و ده‌مه‌ی ده‌ووترێ ده‌بێت ئارگۆمێنت هه‌بێت به‌بێ ئه‌وه‌ی خۆیان بزانن ئارگۆمێنت چیه‌، كه‌س به‌قه‌د ئه‌و فاشیزمه‌ نوێیه‌ به‌رپرسیار نیه‌ له‌نه‌بوونی ئارگۆمێنت و نائه‌كادیمی بوونی كاره‌كان.. بۆیه‌ ئه‌و فاشیزمه‌ له‌ناو ئاراسته‌یه‌كی سۆفستایی ده‌ژیێت ته‌نیا قسه‌ ده‌كات باس له‌ئه‌كادیمی و ئارگۆمێنت ده‌كات به‌بێ ئه‌وه‌ی ئه‌كادیمی بێت ناو ریز ده‌كات به‌بێ ئینتیماكردن بۆ میتۆدی ناوه‌كان باس له‌ئارگۆمێنت ده‌كات له‌كاتێكدا ئارگۆمێنت بوونی ته‌نیا سۆفستاییه‌ته‌و هیچیتر 
په‌یوه‌ندیی نێوان فاشیزم وزهان په‌یوه‌ندیه‌كی ئۆرگانی نه‌ستیه‌، ئه‌وه‌ بونیادی عه‌قڵی فاشیزمی نوێ ده‌خاته‌ ڕوو، فاشیزمی نوێ كه‌ له‌رۆحى ستالینیزمه‌وه‌ هاتۆته‌ بوون، به‌ڵام ده‌یه‌وێ له‌فۆڕمی شانۆیی خۆی بخاته‌ ڕوو، ئه‌و زمانه‌ی فاشیزم دیدێكی نه‌ستی زهانیه، كه‌ به‌لای ژۆلیا كریستیفا وه‌ زهانیه‌ت دیدی نه‌بینراوی خۆی له‌ناو زمان ده‌شارێته‌وه‌، هاملێت به‌و دۆخه‌ زهانیه‌ ناو ده‌بات، زهانی له‌هه‌ندێك ئاراسته‌دات په‌یوه‌سته‌ به‌ میتا زمانه‌وه‌، فاشیزمی نوێ كه‌ ده‌یه‌وێ له‌پانتاییه‌كانی شانۆی ئێمه‌ ده‌ربكه‌وێت به‌رهه‌می ئه‌و دۆخه‌ زهانییه‌، پێمان ده‌ڵێت ئێمه‌ ده‌رهێنه‌ری شانۆییمان نیه‌، تیۆری شانۆیی كاری پێ نه‌كراوه‌، ئه‌و زمانهی قسه‌كه‌ر له‌بنه‌ره‌تدا دیدێكی تاكره‌هه‌ندانه‌ی فاشیزمه‌ كه‌ ده‌یه‌وێ خۆی وه‌ك فریادڕه‌س و دار عه‌ساكه‌ی موسا بناسێت، به‌وه‌ی بۆ ئه‌وه‌ی شانۆی ئێمه‌ تیۆریانه‌ كار بكات و ببینه‌ به‌ ده‌رهێنه‌ر ده‌بێت دان به‌ فاشیزم بێنین و وه‌ك وێنه‌ی باڵا سه‌یری بكه‌ین، ئه‌و‌ ده‌مامكه‌ی فاشیزم ئاستێكی زهانی زمانه‌، كه‌ توانای ئه‌وه‌ی هه‌یه‌ مناڵه‌ نه‌خوێنه‌واره‌كان فریوه‌ بدات، له‌پاڵ ئه‌و به‌رنامه‌ڕێژیه‌ ئایدیۆلۆژیه‌ ستالینیه‌ی كه‌ دامه‌زاراوه‌ دۆگماكانی حزبی كوردیی پاڵپشتی ده‌كه‌ن.
مه‌ترسی ئه‌و فاشیزمه‌.. به‌ته‌نیا له‌و ئاسته‌ زمانه‌وانیه‌ ‌نیه، ترسه‌كانی ئه‌و فاشیزمه‌ به‌ره‌نجامی دۆخێكی ده‌روونیه‌، بۆیه‌ كاتێ ژاك لاكان نوسی (زمان و نه‌ست وه‌ك یه‌ك بونیاد نراون) ئه‌وه‌ ڕه‌وایه‌تی دانی زیاد له‌پێویست بوو، بۆ زهانیه‌ت، كه‌ كریستیفا زوو هه‌ستی به‌مه‌ترسیه‌كانی كرد، به‌وه‌ی لاكان نه‌توانێ جیاوازی له‌نێوان شیزۆفرینیا و زهان بكات.
هه‌ڵبه‌ت ئه‌وه‌ هه‌مان مه‌ترسیه‌ له‌سه‌ره‌تای ئه‌و گوتاره‌ شانۆییه‌ ئێمه‌ هه‌ست به‌مه‌ترسیه‌كه‌ی ده‌كه‌ین، نه‌ك له‌به‌ر ئه‌وه‌ی شیزۆفرینیاو زهانیه‌ت وه‌ك یه‌ك ده‌ربكه‌ون، به‌ڵكو ترسی ئه‌وه‌مان هه‌یه‌ ئه‌و دووانه‌ ئاوێته‌ی نێو گوتاری عه‌قڵانیه‌تی بزووتنه‌وه‌ی شانۆییمان بێت.
چونكه‌ زهانیه‌ت كه‌ له‌بنه‌ره‌تدا زمانی شیزۆفرینیه‌كه‌ ده‌بێت له‌دۆخی ده‌روونی و دۆخی زمانه‌وانی له‌یه‌كتریان جیا بكه‌ینه‌وه‌، به‌وه‌ی ئه‌و زمانه‌ بۆ خۆی خاوه‌نی دامه‌زراوێكی فاشیانه‌یه‌، فاشیزمی زمان فاشیزمێكی ڕه‌مزیه‌ له‌به‌ره‌نجامی هۆڵی موحازه‌راتی خوێندن و به‌ناوی دكتۆراوه‌ موماسه‌ره‌ ده‌كرێت، ئه‌وه‌یه‌ تراژیدیای مه‌عریفه‌ی شانۆییمان..كه‌ ئه‌و تراژیدیایه‌ ببێته‌ گوتاری باڵاو عه‌قڵانیه‌تی شانۆی كوردیی.
