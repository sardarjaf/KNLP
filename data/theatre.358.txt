زۆرجار شانۆكارانی كه‌ركوك گله‌یی ئه‌وه‌م لێده‌كه‌ن گوایه‌ تێڕوانینم بۆ بزاڤی شانۆیی له‌ شاره‌كه‌دا تێڕوانینێكی ره‌ش بینانه‌یه‌ ، چونكه‌ هه‌میشه‌ پێموایه‌ له‌كه‌ركوكدا شانۆكار نییه‌ ، دیاره‌ ئه‌و تێگه‌یشتنه‌م وه‌ك تێگه‌یشتنی زۆربه‌ی ئه‌و براده‌رانه‌ نییه‌ كه‌ ده‌رئه‌نجامی سڕ بوون و مانه‌وه‌یان له‌نێو زه‌مه‌نی رابردوودا  پێیانوایه‌ ئه‌مرۆیش دوێنێیه‌ ، له‌ راستیدا تێنه‌په‌ڕاندنی ئه‌و رابردووه‌ كه‌  به‌ رۆژه‌ دره‌وشاوه‌كانی شانۆی كه‌ركوكی وه‌سفده‌كه‌ن ، گه‌واهی راستییه‌كانه‌ و ده‌یسه‌لمێنێت كه‌  شانۆكاران جگه‌ له‌ ووتنه‌وه‌ی وێرده‌كانی پیرۆزكردنی شانۆی ساڵانی رابردوو، بۆ ئه‌مڕۆ هیچ باس و خواسێكی دیكه‌یان نییه‌، نامه‌وێ‌ بێمه‌ سه‌ر باسێكی مێژوویی سه‌باره‌ت به‌ سه‌رهه‌ڵدان بزاوتی شانۆیی له‌ كه‌ركوكدا به‌ڵام راستییه‌كی حاشاهه‌ڵنه‌گرهه‌یه‌ و پێویسته‌ هه‌ڵوه‌سته‌ی له‌ سه‌ربكرێت ،پاش كۆتایی ساڵانی حه‌فتاكان ده‌توانین بڵێین جموجۆڵی شانۆیی له‌كه‌ركوكدا نه‌مابوو، تا سه‌ره‌تای ساڵانی نه‌وه‌ده‌كان چه‌ند تیپێكی شانۆیی وه‌ك تیپی شانۆی ئاشتی و تێكۆشان درووست بوون، دیاره‌ كه‌ فاكته‌ری سه‌ره‌كیی سه‌رهه‌ڵدانی ئه‌و تیپانه‌  پتر فاكته‌ری سیاسی بووه‌، چونكه‌ له‌و ده‌مه‌دا كه‌ركوك له‌ رووی كولتوریی و سیاسیی و ته‌واوی بواره‌كانی تره‌وه‌ له‌ شاره‌كانی دیكه‌ی كوردستان دابڕابوو ، رژێمی به‌عس له‌ هه‌وڵی له‌ ناوبردنی زمانی كوردیدا بوو, بۆیه‌ شانۆ تاكه‌ كه‌ناڵێك بوو بۆ كۆكردنه‌وه‌ی دانیشتوانی شاره‌كه‌ و ئاراسته‌كردنی گووتارێكی سیاسی نا راسته‌وخۆ دژ به‌ به‌عس،  هاوكات جێی ئومێد و دڵخۆشی كورده‌كانی ئه‌و شاره‌بوو ، جێی خۆیه‌تی ئه‌و كه‌سانه‌ی كه‌ له‌و ماوه‌یه‌دا كاریان له‌ شانۆده‌كرد رۆڵیان له‌و قۆناغه‌دا به‌هه‌ند وه‌ربگیرێت ،چونكه‌ چه‌ندین جار به‌عس  هه‌ڕه‌شه‌ی گرتن و كوشتنی لێده‌كردن، به‌ڵام ئه‌گه‌ر بڵێین له‌و قۆناغه‌دا به‌ ئاگاییه‌وه‌ له‌ بواری  شانۆدا كاركراوه‌ ئه‌وا پێموایه‌ راستییه‌كان ده‌شارینه‌وه‌ ، و ده‌رئه‌نجام خودی خۆمان فریوده‌ده‌ین ، نه‌خێر له‌و سه‌رده‌مه‌دا هیچ كاتێك گه‌ر به‌ رێككه‌وتیش بوبێت شانۆكار نه‌یتوانیوه‌ به‌ ووشیارییه‌كی شانۆیی تۆكمه‌وه‌ مامه‌ڵه‌ له‌گه‌ڵ شانۆدا بكات ، پرۆسه‌ی ئازادی و كرانه‌وه‌ی كه‌ركوك به‌ رووی شاره‌كانی دیكه‌ی كوردستاندا  ئومێدی كرانه‌وه‌ی رۆژنه‌یه‌كی هێنایه‌ كایه‌وه‌ بۆ ئه‌وه‌ی له‌و ده‌لاقه‌یه‌وه‌ هاوشان له‌گه‌ڵ ته‌واوی گۆڕانه‌كانی دوای (به‌عس)دا  بزاوتی شانۆیش  گۆڕانی ریشه‌یی به‌سه‌ردابێت ، له‌و سه‌ر و به‌نده‌دا په‌یمانگه‌ی هونه‌ره‌جوانه‌كانی كه‌ركوك  دامه‌زرا و چه‌ندین تیپی دیكه‌ی شانۆیی هاتنه‌كایه‌وه‌ كه‌ پێموابێت ژماره‌یان تا ئه‌مڕۆ گه‌یشتووه‌ته‌ زیاتر له‌  15 تیپ  كه‌ هه‌ندێكیان مۆڵه‌تی ره‌سمیان پێبه‌خشراوه‌ و مانگانه‌ بودجه‌ی تایبه‌تیان بۆ ته‌رخانكراوه‌ ، (تیپی شانۆی ئه‌زموونگه‌ری) كه‌ركوكیش یه‌ك له‌و تیپانه‌ بوو كه‌ مۆڵه‌تی ره‌سمی پێبه‌خشرا به‌ڵام بودجه‌ی بۆ ته‌رخاننه‌كراو ده‌رئه‌نجام دوای چه‌ندین پرۆژه‌ی شانۆیی تیپه‌كه‌ به‌یاننامه‌ی هه‌ڵوه‌شانه‌وه‌ی خۆی راگه‌یاند،نكولی له‌وه‌ ناكرێت تیپی ناوبراو له‌ چوارچێوه‌ی كاره‌كانیدا چ له‌ ئاستی تیۆری و چ له‌ ئاستی پراكتیكی رۆلی به‌رچاوی هه‌بوو له‌ هه‌وڵدان بۆ گۆڕینی بزاڤی شانۆیی به‌ره‌و ئاراسته‌یه‌كی جیاواز ، قسه‌م له‌سه‌ر ئه‌وه‌نییه‌ ئاخۆ تا چ راده‌یه‌ك له‌و هه‌وڵه‌ سه‌ركه‌وتنی به‌دستهێناوه‌،ئه‌كرێ‌ ئه‌وه‌یان بابه‌تێكی سه‌ربه‌خۆبێت به‌ڵام ئه‌وه‌ی  مایه‌ی سه‌رنجه‌  زۆربه‌ی ئه‌و لایه‌نانه‌ كه‌ به‌ناو وابه‌سته‌ی شانۆن له‌ دوور و نزیك پرسیاری ئه‌وه‌یان نه‌ورووژاند ئاخۆ هه‌ڵوه‌شاندنه‌وه‌ی  وه‌ها تیپێكی ئه‌كتیڤ چه‌ند زیان به‌ بزووتنه‌وه‌كه‌ ده‌گه‌یه‌نێت ؟
له‌هه‌موی سه‌یرتر وێڕای بێهه‌ڵوێستی شانۆكارانی كه‌ركوك به‌رامبه‌ر به‌و مه‌سه‌له‌یه‌ چه‌ند شانۆكارێكی شاری سلێمانی به‌ده‌نگ هاتن و له‌ چه‌ند شوێنێكدا باڵپشتیان له‌ تیپی ئه‌زموونگه‌ری كرد ، بۆ كه‌سێك كه‌ به‌ڕاستی شانۆكاربێت ئاساییه‌ ئه‌و جۆره‌ هه‌ڵوێسته‌ی هه‌بێت به‌ڵام ده‌بوایه‌ پێشوه‌خت شانۆكارانی كه‌ركوك ئه‌و هه‌ڵوێسته‌یان هه‌بوایه‌،
 خۆ ئه‌گه‌ر ئه‌و تیپه‌ له‌كه‌ركوكدا هه‌ر هیچی نه‌كردبێت ،ده‌ركردنی گۆڤارێكی وه‌رزی به‌ناوی (شانۆكار)ه‌وه‌ به‌سه‌ بۆ ئه‌و شاره‌ كه‌ هه‌ر ژماره‌و له‌چه‌ند دۆسیه‌یه‌كی تایبه‌تدا چه‌ندین بابه‌تی پڕگرنگی خستووه‌ته‌ روو، له‌میژووی ئه‌و شاره‌دا قه‌ت رووی نه‌داوه‌ بڵاوكراوه‌یه‌كی تایبه‌ت به‌ شانۆ ده‌ربكرێت،به‌ بڕوای من له‌كه‌ركوكدا ئه‌گه‌ر بزاڤێكی شانۆیی هه‌بێت ئه‌وا قه‌رزارباری تیپی شانۆی ئه‌زموونگه‌رییه‌ ،كه‌ پێموایه‌ له‌ ئێستادا شتێك نییه‌ به‌ناوی بزاوتی شانۆییه‌وه‌ چونكه‌ جموجۆڵ ده‌رئه‌نجامی مه‌عریفه‌، گۆڕان و ئاراسته‌ی نوێ‌ به‌رهه‌مده‌هێنێت كه‌چی ئه‌وه‌ی به‌دی ده‌كرێت  خولانه‌وه‌یه‌ له‌ نێو بازنه‌یه‌كی داخراودا بۆیه‌ ئه‌وه‌ی به‌رهه‌مدێت ناووشیارییه‌كی چه‌ندباره‌یه‌  كه‌وایه‌ له‌ بنه‌ڕه‌تدا هیچ جوڵه‌یه‌ك له‌ ئارادانییه‌ ،
گرفتی زۆرێك له‌و به‌ناو شانۆكارانه‌ ئه‌وه‌یه‌ ناتوانن دان به‌راستییه‌كاندا بنێن و حوكمی گۆڕان به‌سه‌ر خۆیاندا بسه‌پێنن، كیشه‌ی (تیپی شانۆی ئه‌زموونگه‌ریی)یش له‌گه‌ڵ ئه‌واندا ئه‌وه‌بوو كه‌ به‌رده‌وام داوای گفتوگۆكردنی ده‌كرد، هه‌میشه‌ ده‌یویست له‌گه‌ڵ شانۆكارانی شاره‌كه‌دا كه‌ناڵێكی جدی درووستبكات بۆ قسه‌كردن له‌سه‌ر ئێستا و داهاتووی بزاوتی شانۆیی له‌ شاره‌كه‌دا، چه‌ندین جار له‌ رێی گۆڤاره‌كه‌وه‌   بانگهێشتی شانۆكارانی كردووه‌ بۆ خستنه‌ رووی راو پێشنیاره‌كانیان، به‌ڵام هه‌میشه‌ به‌ بێده‌نگی وه‌ڵامیانداوه‌ته‌وه‌ و خۆیان لێدزیوه‌ته‌وه‌، به‌ڕاستی جێی سه‌رسوڕمانه‌ ، ئه‌گه‌ر شانۆكار سه‌باره‌ت به‌ شانۆ نه‌یاته‌ گۆ  ئه‌ی سه‌باره‌ت به‌ چی دێته‌ گۆ  ، به‌و په‌ڕی نیگه‌رانییه‌وه‌ ده‌ڵێم ، بێده‌نگییه‌كی مه‌ترسیدار  باڵی به‌سه‌ر  نێوه‌نده‌كه‌دا  كێشاوه‌  به‌جۆرێك  كه‌ ده‌رفه‌ت له‌ به‌رده‌م گه‌شه‌سه‌ندنی ترووسكاییه‌كان ناهێڵێت و بانگه‌شه‌ بۆ تاریكاییه‌كی كوشنده‌ده‌كات.
