یه‌ک جنۆکه‌ به‌ده‌نگمه‌وه‌ نه‌هات ، بۆیه‌  له‌ڕقا جنۆکه‌کانم تف و نه‌عله‌ت کرد .
له‌ پڕێکا لێم په‌یدا بوون هه‌ریه‌ک له‌ لای خۆیه‌وه‌ ده‌یکێشا به‌ بنا گوێچکمه‌ ، به‌ چ حالێ توانیم هه‌لبێم و ده‌ربازبم له‌ ده‌ستیان .
کاتێ به‌ گریان و نووزه‌ نووزگه‌ڕامه‌وه‌ ماڵ .
دایکم وتی : کوڕم ئه‌وه‌ چیبووه‌ ، بۆ وا بنا گوێت سوور بووه‌ته‌وه‌؟
دایکم ته‌شتێ پر له‌ ئاوی دانا و منی له‌ باوه‌ش کرد ، چنگێ خوێ به‌ده‌سته‌وه‌ بوو سێ چوار جار له‌ بان ته‌شته‌که‌وه‌ بازی ده‌دا و خوێیه‌که‌ی ده‌رژاند.
خوشکه‌ گه‌وره‌که‌م لێی ده‌پرسی ئه‌وه‌ چی ده‌که‌یت؟ .
دایکم ده‌یگوت : ترسی ئازاد ده‌رده‌که‌م.
هه‌رچه‌ند ئه‌وان ترسی منیان ده‌رده‌کرد به‌لام من دلنیا بووم که‌ جنۆکه‌ تێیان هه‌لداوم .
لام سه‌یر بوو ئه‌وه‌نده‌ هاوارم کرد بۆ جنۆکه‌کان نه‌هاتن ، به‌لام کاتێ جوێنم پییاندا هاتن و تێریان تێهه‌لدام .
له‌ترسا نه‌مده‌وێرا  دووباره‌ جوێنبده‌مه‌وه‌ ، به‌لام وتم : ده‌بوایه‌ له‌ منیان بپرسیایه‌ بۆ جنێوده‌ده‌یت؟
جنۆکه‌کان ڕوخسارێکی ترسناکیان هه‌بوو ، شاخدارو گوێچك پان وچاو زه‌ق و بالا کورت ، لووتێکی زڵ و لچێکی شۆڕ .
سنگییان لووس و ته‌خت بوو، پرچیان گڤن و  درێژ.
کاتێ قسه‌یان ده‌کرد ددانه‌کانیان له‌ پیسیا زه‌رد هه‌ڵگه‌ڕا بوو، زمانیان بچووک بوو له‌گه‌ل قسه‌کردن ناو هه‌ناویانت ده‌بینی، ده‌ستیان درێژ بوو به‌لام پێیان کورت له‌ پێی بزن و مه‌ڕ ده‌چوو، کلکێکی درێژو باریکیان پێوه‌ بوو له‌ چه‌شنی قامچی که‌ به‌ هه‌موو لایه‌ک ده‌یان جولاند .
به‌ گشتی جنۆکه‌کان به‌ تارمایی ده‌چوون و ڕووخساریان نابینرا به‌لام من له‌به‌ر ئه‌وه‌ی ڵێم ئالابوون نێرو مێیانم جیاده‌کرده‌وه‌!
شه‌وی هاوین بوو له‌سه‌ر بان نووستبووم دوو سێ جنۆکه‌ په‌نجه‌کانی پامیان ده‌مژی ، ویستم هاوارکه‌م تووند ده‌میان گرت و وتیان له‌گه‌لمان وه‌ره‌ خواره‌وه‌ .
منیش وتم : نه‌خێر.. نایه‌م.
چوار په‌لمیان گرت و خه‌ریک بوو له‌بان خانوو فڕێم ده‌نه‌ خواره‌وه‌ ، وتیان دێت یان فڕێت ده‌ینه‌ خواره‌وه‌؟  ناچارله‌گه‌لیان چوومه‌ خواره‌وه‌ له‌ ناو حه‌وشه‌ مه‌جلیسێکان به‌ستبوو بۆ دادگایی کردنم .
منیش بیستبووم ئه‌وه‌ی ببێته‌ شه‌ریکی جنۆکه‌ و شه‌یتان خوا قه‌د له‌ گوناحی خۆش نابێت ، وتم : من به‌نده‌ی خودام ..دایک و باوکم نوێژ ده‌که‌ن ده‌مه‌وێ له‌ به‌هه‌شت  له‌گه‌ڵ ئه‌واندابم.
هه‌رئه‌وه‌نده‌ ناوی خودام هێنا ، جنۆکه‌کان کردیان به‌ یه‌ک‌ فیزاح و هاوارێک  کۆلانه‌که‌یان هه‌ژاند .پاشان به‌ هه‌موویان که‌وتنه‌ سه‌رزگم به‌ پێ له‌قه‌ هه‌ناومیان هێنایه‌ خوارێ .
له‌ پڕێکا گوێم له‌ ده‌نگی باوک و دایم بوو : بسم لله‌ بسم لله‌ ..هاتنه‌ خوارێ.
منیش وتم : نا دایه‌ به‌ر نه‌بوومه‌ته‌وه‌ ،‌ جنۆکه‌کان بانگیان کردم منیش ناوی خوام هێنا کردیان به‌ فیزاح  و ڕایان کرد.
دایکم  په‌رێشانتر بوو، هاواری کرد : یا ئه‌بدولقادر گه‌یلانی هاواره‌ ، چاره‌یه‌کمان که‌.
کوڕه‌که‌م شێت بووه‌
هه‌ندێ ئاوی ساردیان به‌ ده‌م و چاوما کرد ، هاوسێکان به‌ بسم لله‌ بسم لله‌ گه‌ڕانه‌وه‌ ماله‌که‌یان.
منیش به‌ باوکم وت : باوکه‌ له‌گه‌ڵ تۆ ده‌نووم ...ئه‌و شه‌وه‌ به‌ ئارامی له‌ باوه‌شی باوکم نوستم .
به‌یانی زوو له‌خه‌و هه‌لسام چووم بۆ میز کردن ، ده‌رگای ئاوده‌سته‌که‌مان که‌ به‌ گونیه‌ گیرابوو ، له‌ کونی گونیه‌که‌وه‌  سه‌یری ناو حه‌وشه‌که‌م ده‌کرد ، بینیم  دایکم و خوشکه‌ گه‌وره‌که‌م فسکه‌ فسکیان بوو .
به‌ته‌نیشتیانه‌وه‌ چه‌ند سێبه‌رێکم بینی ، ڕووانیمه‌ سه‌ربان ، بینیم جنۆکه‌کان هه‌موو کۆ بوو بوونه‌وه‌ به‌ چاوه‌ زه‌قه‌کانیان مۆڕه‌یه‌ن له‌من ده‌کرد .
دایکم جامێ ئاوی ڕشاند و بسم لله‌ی کرد یه‌ک جنۆکه‌ له‌و ناوه‌ نه‌ما .
منیش ده‌ست وچاوم شۆری و چوومه‌ به‌رده‌رگای ماڵ  ڕه‌فیقه‌کانم بینی چووم بۆ لایان که‌چی پێده‌که‌نین وگالته‌یان پێده‌کردم و بانگیان ده‌کردم : ئازه‌ شێت.. ئازه‌ شێت.
به‌ قوڕگی پڕ له‌گریان گه‌ڕامه‌وه‌ ماڵ .
دایکم خێرا له‌ باوه‌شی کردم ، نوشته‌یک که‌ په‌ڕۆکێکی سه‌وزی لێ ئالابوو له‌یه‌خه‌ی به‌ستم ، یه‌ک دوو دوعای به‌ ئاسپایی له‌ بن گوێم خوێند ، پاشان تفێکی کرده‌ سه‌ر زه‌وی وتی: گوێی شه‌یتان که‌ڕ بێت.
له‌وساته‌وه‌ نازناوم بوو به‌ ئازه‌ شێت .
سه‌ره‌تا له‌ گه‌ڕه‌ک پاشان  لای هه‌موو خه‌ڵکی شار، ئێستاش تا نه‌لێین  ئازه‌ شێت که‌س نام ناسێته‌وه‌ .
من له‌ مندالێ شێت نه‌بووم  ، به‌س که‌س نه‌بوو گوێم لێبگرێ یاخود باوه‌ڕم پێبکات .
جنۆکه‌کان به‌دوامه‌وه‌ بوون ، کاتێ بۆ خوشکه‌که‌م باس ده‌کرد ده‌یگوت : جنۆکه‌ بوونی نییه‌ ئه‌ وه‌ تۆ له‌خه‌یالی خۆت دروستی ده‌که‌ی.
جنۆکه‌کان مێرده‌زمه‌ی ژیانن و هیچ کات له‌ کۆلمان نا‌بنه‌‌وه‌.. نیگه‌رانم له‌وه‌ی له‌ نێوان شه‌ڕوخێردا وونبم  ئه‌گه‌ر که‌سێکی مهره‌بان و پاکیت باوه‌ڕت به‌یه‌زدان هه‌یه‌  ئاگات له‌خۆتبێت ، چونکه‌ جنۆکه‌کان حه‌ز به‌ که‌سێ ناکه‌ن له‌ خوا نزیک بێت و کرده‌وه‌ی چاکه‌ بکات .
ئێستا گه‌وره‌ بوومه‌ و‌ باوکی چوار مندالم هه‌ست ده‌که‌م  جنۆکه‌کان له‌ ده‌وروپشتم هه‌ن به‌لام شێوه‌یان گۆڕاوه‌ ، له‌سه‌ر ‌شێوه‌ی مندالی نایانبینم ، به‌لکو وه‌ک من و تۆ وان ، چاویان ده‌ستیان شێوه‌یان له‌ مرۆڤ جیا ناکرێته‌وه‌ ، جار هه‌یه‌ ده‌توانن ببن به‌ مهره‌باترین که‌س تا ئه‌وه‌نده‌ ده‌زانی شلپه‌ له‌ بنا گوێت ده‌هێنن..یاخود خۆیان ده‌که‌ن به‌ ئافره‌ت و فریووت ده‌ده‌ن ..تۆش کاتێ ئاگادار ده‌بیته‌وه‌ ، ئه و‌سا ده‌زانی ئه‌مه‌ ئافره‌ت نه‌بوو عه‌فریت بوو .
جنۆکه‌کان ئه‌مڕۆ له‌ هه‌موو کون و قژبنی  سه‌ر ئه‌م زه‌وییه‌ هه‌ن ،   بۆیه‌ منیش هه‌میشه‌ وه‌ک دایکم چنگێ خوێم پێیه‌  له‌گه‌ل ته‌شتێ ئاو هه‌رکات مندالێکم بترسێ له‌ باوه‌شی ده‌گرم سێ چوار جار له‌سه‌ر ته‌شته‌که‌ باز ده‌ده‌م و ترسی ده‌شکێنم .
له‌ کۆتاییدا هه‌رچه‌ند هه‌وڵ ده‌ده‌م  له‌ شه‌ڕی جنۆکه‌کان خۆم بپارێزم ، که‌چی خه‌لکی به‌شێتم ده‌زانێ .
منیش گه‌ر شێت نه‌بوومایه‌ ئه‌م چیرۆکه‌م نه‌ ده‌نووسی و تۆشم له‌گه‌ل خۆم شێت نه‌ده‌کرد.
