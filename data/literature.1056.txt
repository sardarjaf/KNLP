 باسی سپڵه‌یی و وه‌فاداریم نه‌كرد، پێشتر پشیله‌یه‌ك له‌ زاری سه‌رۆكه‌وه‌ ده‌رپه‌ریِ، له‌به‌ر هه‌ندێ نه‌مویست سپله‌یی و وه‌فاداری بێنمه‌ گۆڕێ، ئه‌وه‌شم گوت:
 حه‌په‌حه‌پ له‌ گوێی من له‌ میاوه‌میاو مۆزیكیتره‌، ئه‌گه‌ر باوه‌ڕیش ناكه‌ن، فه‌رموون ئاهه‌نگێك بۆ سه‌گ و پشیله‌ بگێڕن و له‌ كۆمه‌ڵێك كه‌سی ئاهه‌نگی و مۆزیكناس  بگێڕنه‌وه‌، باكه‌سه‌كانیش رۆژهه‌ڵاتی بن نه‌ك رۆژاوایی، با نه‌ڵێن حه‌په‌حه‌پی سه‌گ به‌ لای ئاهه‌نگی رۆژاوایی ده‌شكێته‌وه‌ و میواوه‌میاوی پشیله‌ش له‌ ئاهه‌نگی رۆژهه‌ڵاتی نزیكده‌بێته‌وه‌..
هه‌ستمكرد زمانم گۆناكات، گوێم له‌ ده‌نگێكه‌ له‌ قووڵاییه‌وه‌ دێ، نه‌مزانی قووڵایی خۆمه‌، یان قووڵایی دنیا، ده‌ڵێ:
مێشكم وه‌ك لێمۆ ده‌گوشی، ده‌مویست له‌ باره‌ی سه‌گ و پشیله‌وه‌ فه‌رمووده‌یه‌كم بیركه‌وێته‌وه‌، فریای بیركردنه‌وه‌ نه‌كه‌وتم، بردمیان، له‌ ته‌نیشت سه‌گێكی سپی به‌ستمیانه‌وه‌، پشیله‌یه‌كی ره‌ش ره‌شیان له‌ پێشم راگرت، زاری لێكده‌كرده‌وه‌، له‌ ددانه‌كانی ترسام، سه‌گه‌كه‌ش چاوی داخست، ره‌نگه‌ ئه‌ویش ترسابێت، نه‌موێرا چیدی ته‌ماشایبكه‌م، سه‌رم بۆ ئاسمان هه‌ڵبڕی، له‌ پشت پشیله‌كه‌وه‌ ده‌نگێ گوتی:
 چه‌ندی كردم شادومان به‌ زارمدا نه‌ده‌هات، له‌ بێ باوه‌ڕیه‌وه‌ نه‌بووكه‌ نه‌ده‌هاته‌ سه‌ر زارم، له‌و ساته‌ باوه‌ڕم به‌ شادومانهێنان هه‌بوو، له‌ ترسی مه‌رگ نه‌هات، ئاخر دوورایی نێوان شادومان و مه‌رگ ره‌نگه‌ چركه‌یه‌ك بێت، شادومان له‌ جێیه‌كی عاسێ خۆی شارده‌وه‌ و نه‌هاته‌ سه‌ر لێو، ئیدی بێ شادومان ته‌واو، مه‌رگ به‌ گولله‌یێ هات، ئه‌وه‌ی ته‌قه‌ی كرد، ئێستاش ده‌ڵێم كوڕه‌كه‌م بوو، ده‌مامكی دابوو، به‌ڵام كه‌س هه‌یه‌ كوڕی خۆی نه‌ناسێـته‌وه‌، تاریكایی دنیا له‌ خۆی لوول بدات ده‌ناسرێته‌وه‌..
نێوان مه‌رگی شه‌شه‌م و یه‌كه‌م نێوان ته‌مه‌نی باپیر و نه‌وه‌كه‌یه‌تی، مه‌رگی یه‌كه‌مم له‌ هی حه‌فته‌م جوانتر له‌ یاده‌، بیستبووم مردنی یه‌كه‌م وه‌ك ماچی یه‌كه‌مه‌، وا ده‌رچوو، بابم هه‌میشه‌ پۆشته‌ و رێكپۆشه‌، به‌ڵام ئه‌و رۆژه‌ له‌ هه‌موو رۆژان، پۆشته‌تر و قۆزتر خوی نواند، ئه‌و رۆژه‌ هێنده‌ی كیژۆله‌ی تازه‌ هه‌ڵچوو له‌ پێش ئاوێنه‌ مایه‌وه‌، پرچی هه‌ر جاره‌ی به‌ لایه‌كدا هه‌ڵده‌دایه‌وه‌، چه‌پ، راست، سه‌رێ، حه‌فت جار تا سه‌ر ده‌رگای حه‌وشه‌ رۆیشت و گه‌ڕایه‌وه‌ به‌ر ئاوێنه‌،  هه‌موو جارێكیش فلچه‌یه‌كی به‌ پێڵاوه‌كه‌یدا داده‌هێنا، دایكم گوتی:
دایكم ده‌ستی  له‌ سه‌ر ده‌می خۆی دانا، به‌ ئاماژه‌ داوای لێبووردنی كرد، شه‌و دره‌نگ بابمم له‌ ته‌له‌فزیۆنه‌وه‌ ده‌ركه‌وت، له‌ سووچێك له‌ سه‌ر كورسی دانیشتووه‌، چه‌ند جارێك ده‌هێنێته‌ پێشێ، دانیشتنه‌كه‌ی له‌ هی بابم ناچێ، ئاخر له‌ دانیشتنی ماڵێ قه‌ت نایبینم پێیه‌كانی له‌ ته‌نیشت یه‌كه‌وه‌ بن، هه‌میشه‌ پێ له‌ سه‌ر پێ داده‌نێ و ده‌ستێكیشی له‌ سمێڵ و برۆی نابێته‌وه‌، بۆ ساتێكیش زاری راناوه‌ستێ، یان گوڵه‌به‌ڕۆژه‌، یان بنێشت، كه‌چی ئێستا، پێیه‌كانی به‌ ته‌نیشت یه‌كه‌وه‌، هه‌ر ده‌ستێك و له‌ سه‌ر رانێ، وه‌ك بڵێی پاڵه‌په‌ستۆی خستبێته‌ سه‌ر رانه‌كانی تا نه‌جوولێن، نه‌ك هه‌ر زار چاویشی جووله‌ی تێدا نیه‌، كتومت بته‌ و له‌وێ دانراوه‌، ده‌ڵێمه‌ دایكم:
زارم لێكده‌نێم و به‌ دیار بابی به‌ بتبوومه‌وه‌ دانیشتین، ئه‌وه‌تا سه‌رۆك قسان ده‌كات، سه‌رۆك باس له‌ ته‌رازوویه‌ك ده‌كات، هه‌ر خۆی هه‌یه‌تی و كه‌س نایبینێ، ده‌ڵێ:
هه‌ر كه‌سێكم بوێت، ده‌یخه‌مه‌ ناو ته‌رازووه‌كه‌ و ده‌زانم بایی چه‌نده‌، بێ ئه‌وه‌ی ئه‌و كه‌سه‌ بزانێت هاتۆته‌ ناو ته‌رازووه‌وه‌، وایه‌؟
سه‌رۆك زۆر هاوكێشه‌ ده‌هێنێته‌گۆڕێ، هاوكێشه‌ی درۆ و راستی، چاكه‌خوازی و شه‌رانگێزی، ئازادی و بنده‌ستی، گاوری و موسلمانێتی..
چاومكرده‌وه‌، بابم له‌ پێش سه‌رۆك راوه‌ستاوه‌، چاوی داخستووه‌، ده‌سته‌كانی شۆڕكردۆته‌وه‌، هه‌ناسه‌ی له‌ خۆی بڕیوه‌، ویستم بڵێم:
گێڕامه‌وه‌، زانیم  له‌ به‌رانبه‌ر كێ راوه‌ستاوه‌، دیتم سه‌رۆك ده‌مانچه‌ی دابه‌شده‌كرد، بابیشم به‌ریكه‌وت، سه‌رۆك دوای به‌خشینی ده‌سته‌ و دیاری گوتی:
له‌ بیرتانه‌، پێش شۆڕش، میلله‌ت نه‌یده‌زانی كه‌پووی خۆی بسڕێت، به‌ زگێك تێربوو به‌ سێ برسی، له‌ ناو كونه‌ شاخ و زۆنگاو و لمدا ده‌ژیا، نه‌یده‌زانی به‌ رێدا بڕوات، وه‌ك پشیله‌ ده‌موچاوی خۆی به‌ توف ده‌شووشت، نه‌یده‌زانی ناوی خۆی بنووسێت، ئێوه‌ بڵێن ئه‌و هه‌موو خۆشگوزه‌رانی و ناز و نیعمه‌ته‌ی به‌ سه‌ریدا باریوه‌، له‌ سایه‌ی كێوه‌یه‌؟!
كه‌واته‌ سه‌گ و پشیله‌تان له‌ بیر بێت، تیكه‌ نانێ بده‌ سه‌گێك كلكه‌له‌قێت بۆ ده‌كات و بۆت وه‌فادار ده‌مێنێته‌وه‌، به‌ڵام پشیله‌ به‌ مه‌نجه‌ڵ ماستی له‌ پێش دانێ، چۆله‌كه‌ی ئاسمانی بۆ بێنه‌ به‌رده‌ستی، هه‌ر سپله‌یه‌،  ئێوه‌ بڵێن، میلله‌ت به‌رانبه‌ر شۆڕشه‌كه‌ی، به‌رانبه‌ر سه‌رۆكه‌كه‌ی سه‌گ بێت باشتره‌، یان پشیله‌..
سه‌رۆك باسی ئه‌وه‌شی كرد ئه‌گه‌ر كه‌سێك پێی خوار دانا، یان زمانی خراپ گه‌ڕا، ئه‌و كه‌سه‌ ئه‌گه‌ر نزیكترین كه‌سیش بوو، برا و خوشك بێت، باب و دایك بێت،  كچ وكوڕ بێت، پێویسته‌ ته‌مبێی یه‌كجاره‌كی بكرێت، ده‌زانن ئه‌و ده‌مانچه‌یه‌ بۆ چیه‌؟
سوێندت بۆ ده‌خۆم ئه‌گه‌ر كوڕی خۆشم ناوی تۆ به‌ خراپه‌ بهێنێت، به‌و ده‌مانچه‌یه‌ ره‌وانه‌ی دۆزه‌خی ده‌كه‌م..
 من هه‌ڵچووم، به‌ر چاوم تاریك بوو، جله‌وی زاری خۆم بۆ نه‌گیرا، دیاره‌ رۆژم هاتبوو، بۆیه‌ گوتم:
 نه‌مدیت بابم رووی ده‌مانچه‌كه‌ له‌ من بكات، به‌ڵام كه‌وتم، هه‌ر ده‌بێت ئه‌و بووبێت، ببینن له‌ پێشه‌وه‌ پێكراوم، یان له‌ پشته‌وه‌؟ من رووم له‌ بابم بوو..
  جارێكیان به‌ هۆی فسی ئیبلیسیشه‌وه‌ مردم، ئه‌و پیاوه‌ی له‌ ئه‌فغانستانه‌وه‌ گه‌ڕِابۆوه‌، ناوی له‌ گێژه‌لووكه‌ نابوو فسی ئیبلیس، ئه‌و كه‌ ده‌بیستێ  ئیبلیس له‌ شاری تبلیس ده‌ژی، بۆ كوشتنی رێی ئه‌وێ ده‌گرێته‌به‌ر، به‌ قسه‌ی خۆی زۆر لێینزیك ده‌بێته‌وه‌، له‌ سه‌ر ئه‌وه‌یه‌ ده‌مانچه‌كه‌ی هه‌ڵكێشێ و ئیبلیس له‌ خوێن وه‌ردات و مرۆڤایه‌تی له‌ داو و چنگی رزگاربكات، ده‌بینێت ئیبلیس قوونی تێده‌كات و فسێك داوێت، فسه‌ ده‌بێته‌ گێژه‌لووكه‌یێ خه‌ریكه‌ جلكی به‌ به‌ره‌وه‌ نه‌هێلێت، رێك وه‌ك  ئه‌وگێژه‌لووكانه‌ی له‌ بیابانی سعوودیه‌وه‌ هه‌ڵده‌كه‌ن، كابرا له‌وه‌وه‌ ئاشكرای ده‌كات كه‌ گێژه‌لووكه‌ فسی ئیبلیسه‌، به‌ سه‌ر منیشیدا چه‌سپاندی كه‌ به‌ فسی ئیبلیس ناوی بێنم..
ئێ، فسی ئیبلیس چۆن منی به‌ كوشت دا؟ له‌ ته‌مه‌نی عاشقایه‌تی، عاشقی كچێكم هه‌ر ده‌ڵێی كچه‌ جۆرجیه‌،  خودا بۆ منی هێناوه‌ته‌ بن مناره‌ی چۆلی، كچه‌ رۆژانه‌ به‌ ده‌وری مناره‌ی چۆلی سووڕێكی ده‌دا و له‌ پڕ به‌ شێوه‌یه‌ك بزرده‌بێت، هه‌ر ده‌ڵێی له‌مه‌چێترانه‌ و نازانم له‌ كوێوه‌ و دێت و بۆ كوێ ده‌چێت، زۆر جار ده‌مویست لێی نزیكبكه‌ومه‌وه‌ و داوای ناونیشانه‌كه‌ی بكه‌م، شه‌رم پێی ده‌به‌ستم و زارمی گرێده‌دا، له‌پڕ به‌ یه‌كجاری بزربوو، نه‌هاته‌وه‌، من بوومه‌ ئێشكگری مناره‌، سوێندم خوارد تا نه‌یبینمه‌وه‌، لێره‌ به‌ ئێشكگری بمێنمه‌وه‌..
دوو وه‌رزی برد، هه‌ر ئێشكگری مناره‌ی چۆلیم، ئۆخه‌یش، رۆژێكی هاوینی وه‌ك خۆی به‌ هه‌مان جلكه‌وه‌ په‌یدا بۆوه‌، كچه‌ زه‌ردی ده‌پۆشی، هێنده‌ تامه‌زرۆی بووم، شه‌رم نه‌ما، نا ره‌نگه‌ به‌ هۆی سه‌رگه‌رمیمه‌وه‌ بووبێت، ئه‌و رۆژه‌ چه‌ند پێكێكم هه‌ڵدابوو، رۆژباشم لێكرد، گه‌رم وه‌ریگرته‌وه‌، گوتم:
ته‌واوێك دووركه‌وتمه‌وه‌، دیتم شتێكی نووسی، له‌ دووره‌وه‌، ئاماژه‌ی دا كه‌ له‌و بن داره‌ی داده‌نێم، وه‌ره‌ هه‌ڵیگره‌، داینا و بزر بوو، كاتێ بۆ بن داره‌كه‌ بۆ ناونیشانی كچه‌كه‌ هه‌نگاوم هه‌ڵده‌هێنا، ده‌بێت دڵم له‌ چه‌ندی لێدابێت!
ئای فسی ئیبلیس له‌ كوێوه‌ هه‌ڵتكرد!
پێش من گه‌یشتێ، په‌ڕه‌ كاخه‌زی برد، دیتم تێكه‌ڵی په‌ڕه‌ كاخه‌ز و پووش و په‌ڵاش و په‌ڕۆو پاتاڵ بوو، زۆر فسی ئیبلیسم دیتووه‌، به‌ڵام هی به‌و شێوه‌یه‌ یه‌كه‌م جارمه‌، سه‌ری له‌ بن تاقی ئاسمان بوو، كه‌وتمه‌ دوای، كۆڵان كۆڵان جاده‌ جاده‌، من به‌ دوایه‌وه‌، به‌ دوای ناونیشانی كچه‌كه‌وه‌، ئه‌و هه‌موو شته‌ له‌ ئاسمانێ تێكه‌ڵ به‌ ناونیشانی كچه‌كه‌ بووه‌، به‌ڵام به‌س بنیشنه‌وه‌ سه‌ر زه‌وی ده‌یبینمه‌وه‌، هه‌ر ده‌بێت بیبینمه‌وه‌، په‌ر په‌ڕ كاخه‌زی به‌ دنیا وه‌ربوو هه‌ڵده‌گرمه‌وه‌، به‌س بێنه‌وه‌ زه‌وی، من چۆن بگه‌مه‌ ئاسمان، چه‌ند ته‌ماشاده‌كه‌م چ پێناچێت ئه‌و فسه‌ی ئیبلیس شته‌كان بداته‌وه‌ زه‌وی، هه‌ر وا شت لوول ده‌دا و به‌ ئاسمانیان دێخێ، سوێندم خوارد، له‌ دوای فسی ئیبلیس نه‌بمه‌وه‌، ئه‌گه‌ر  بچێته‌وه‌ شاری تبلیسیش، نازانم چه‌ند غارده‌ده‌م و ده‌پارێمه‌وه‌، شته‌كانی ئاسمانێ بێنه‌وه‌ سه‌ر زه‌وی، له‌ پڕ قۆڵم كه‌وته‌ ناو ده‌ستێكه‌وه‌، چ ده‌ست، ده‌ستی هه‌مزه‌ پاڵه‌وانیش وا به‌هێز نه‌بووه‌:
مه‌گه‌ر نازانی كه‌وتنه‌ دوای كچێك، یان ناونیشانی كچێك گوناهه‌، دیاره‌ تۆ له‌ ئاگری دۆزه‌خ ناترسیت؟ بۆنی عه‌ره‌قیشت لێدێ؟
هیچم نه‌گوت، ده‌ستی به‌ردام، فسی ئیبلیس زۆر دووركه‌وتبۆوه‌، به‌ڵام دڵنیام ده‌یگه‌مێ، دیتم ده‌یویست به‌ لای زێی گه‌وره‌دا پێچ بكاته‌وه‌، یه‌كپێ هه‌ڵاتم، ته‌قه‌یه‌ك و كه‌وتن و ران گه‌رمداهاتن پێكه‌وه‌ن..
ئه‌و ساڵه‌ی  گه‌رما له‌ سه‌ره‌وه‌ی پله‌ی په‌نجا گه‌رمایه‌تی خۆی ده‌كرد، كاره‌باش هێنده‌ی وه‌زیری كاره‌باش خۆی نیشانی خه‌ڵك نه‌ده‌دا، ئاو ده‌تگوت دێو تووڕه‌ بووه‌ و به‌ مه‌مكه‌كانی به‌ری گرتووه‌، كاتژمێرێك به‌ دیار به‌لووعه‌وه‌ داده‌نیشتم دۆڵكه‌ نیوه‌ نه‌ده‌بوو، نیوه‌ی مووچه‌كه‌م كێله‌ بووزێكی پێنه‌ده‌هات،  پیاوانی نوێژكه‌ر له‌ بێ ئاویدا بوونه‌ دوو ده‌سته‌، هه‌ردوو ده‌سته‌ له‌ كۆڵی خۆ بسیملكردن بوونه‌وه‌، ده‌سته‌ی یه‌كه‌م وازیان له‌ نوێژكردن و ده‌سته‌ی دووه‌م وازیان له‌ چوونه‌ ناو نوێنی ژنه‌كانیان هێنا، له‌و رۆژانه‌ من هه‌میشه‌ خه‌ونم به‌ رووباره‌وه‌ ده‌دیت، ده‌چوومه‌ ده‌و رووبار خۆم رووتده‌كرده‌وه‌، له‌ سه‌رئه‌وه‌بوو خۆم پێدابده‌م، به‌ ئاگاده‌هاتمه‌وه‌، به‌ ده‌نگی ته‌قه‌یه‌ك، ده‌نگی مه‌لا یان ده‌نگی شه‌ڕێك له‌ كۆڵان به‌ئاگاده‌هاتمه‌وه‌...
رۆژێك هاته‌ سه‌رم بچمه‌ سه‌ر زێی گه‌وره‌، بچم و خه‌ونه‌ رووباریه‌كانمی تێدا بهێنمه‌دی، به‌رێكه‌وتم، رێگا هه‌مووی دوو سێ جغاره‌ خۆره‌، به‌ڵام له‌ سێ بازگه‌ راتده‌گرن و له‌ هه‌ر راگرتنێك پاكه‌تێك ده‌كێشیت، گه‌یشتمێ، ئێستا له‌ كه‌ناری زێ نزیكبوومه‌ته‌وه‌، رووبار زۆر داكشایه‌، ده‌بێت به‌ سه‌ر به‌ردی له‌ دۆزه‌خه‌وه‌ هاتوو، هه‌نگاو هه‌ڵێنم، به‌رد پێڵاو ده‌سووتێنێ، نه‌بادا بێ پێڵاو بمێنمه‌وه‌، گه‌ڕامه‌وه‌، ته‌ماشمكرد له‌و به‌ره‌وه‌ كه‌نار نزیكتره‌، ئاو هێنده‌ دوور نیه‌، هاتم به‌ سه‌ر پرده‌كه‌دا بپڕمه‌وه‌، تا بازگه‌كه‌م رازیكرد زۆری برد، په‌ڕیمه‌وه‌، ته‌واوێك هه‌ڵكشام، جێیه‌كی باشم دیته‌وه‌ بۆ مه‌له‌..
هێنده‌ تامه‌زرۆی ئاوم ده‌مویست به‌ جلكه‌وه‌ خۆم هه‌ڵده‌مه‌ ناوی، خه‌ریكی جلك داكه‌ندنم، ئێستا ده‌بێت چتۆ مه‌له‌ بكه‌م، چڵك و گه‌رمایی هاوینێكی دۆزه‌خی له‌ خۆم داماڵم، خۆم رووتكرده‌وه‌، ده‌ی رووبار وا هاتم، هێشتا پێم هه‌ر له‌ وشكایی بوو، له‌ پشته‌وه‌ ده‌نگێك هه‌ر له‌ ده‌نگی نێریه‌ كێوی ده‌چوو، لێیخوڕیم، له‌ زمانی نه‌گه‌یشتم نه‌مزانی چ ده‌ڵێ، ویستم پانتۆڵه‌كه‌م له‌ پێبكه‌م، كلاشینكۆفی راكێشا، پانتۆڵه‌كه‌م دانایه‌وه‌، ناسیمنه‌وه‌ سه‌ربازی ئه‌مریكین، تا ئێستا ته‌نها له‌ ته‌له‌فزیۆن دیتبوومن، به‌ شڕه‌ ئینگلیزیه‌ك گوتم:
پاڵێكی توندیان پێوه‌نام، خراپ كه‌وتم، خراپتر هه‌ڵیانستانده‌مه‌وه‌، به‌ شه‌قێك هه‌ستامه‌وه‌، مردنه‌كه‌ زۆر خۆشتره‌، هه‌ستمكرد له‌ ریز راوه‌ستاوم، ئه‌وانه‌ی ریزێش له‌ بابه‌تی منن، به‌ هێواشی قسه‌یان ده‌كرد، یه‌ك به‌ عه‌ره‌بی قسه‌ی ده‌كرد، ئه‌ویدی به‌كوردی وه‌ڵامی ده‌دایه‌وه‌، گوێم له‌ قسه‌ی توركمانیش بوو، من خۆم كڕكرد، له‌و ساته‌  بووه‌ ته‌قوو تۆق و هه‌رایه‌ك، مه‌گه‌ر هه‌ر چاو نه‌به‌ستراو بزانێت چ روویداوه‌، من ده‌ستم له‌ ده‌ستم شووشتبوو، ده‌مزانی به‌ كه‌ڵه‌پچه‌یه‌ك كه‌ڵه‌پچه‌كراوم نابێته‌وه‌، ده‌مویست شتێك بكه‌م، تووره‌گه‌كه‌ له‌ سه‌رم داماڵم، له‌و ساته‌ بۆ ساتێك وێنه‌ی هێستر و جۆڕكم هاته‌وه‌ پێش چاو، دیتبووم كه‌ جۆڕكی ئالیك له‌ سه‌ر هێستر ده‌كرێ ،كاتێ تێر ده‌خوات و ده‌یه‌وێت له‌ سه‌ری خۆی داماڵێت، چی ده‌كات، له‌ لا سینگی خۆی، له‌ عه‌ردی ده‌خشێنێت، به‌ دیواری دادێنێت، ده‌یه‌وێت سمی لێگیر بكات و بیدڕێنێت، یان له‌ ناوه‌وه‌ به‌ ددانی بیكرۆژێت، من ده‌مویست چاو له‌ هێستر بكه‌م و رێگاكانی وی به‌كار بێنم، به‌ڵام مه‌رگ كه‌ی رێده‌دات فریای زۆر شت بكه‌ویت..
له‌و ساته‌ گولله‌یه‌ك په‌له‌واژی كردمه‌وه‌، ره‌نگه‌ ماوه‌یه‌ك به‌ گیانه‌وه‌ بووبم، نا، ده‌ستبه‌جێ نه‌مردم، دێته‌بیرم به‌ سه‌ر به‌ردێكدا كه‌وتم ده‌تگوت سێری سه‌ر ئاگره‌، زۆر له‌ گولله‌كه‌ گه‌رمتره‌، دڕِكێك له‌ په‌راسووم راده‌چێت، زۆر له‌ گولله‌كه‌ تیژتره‌، مردم دوور له‌ جلكه‌كانم مردم، دوور له‌ ئاو مردم، له‌ سه‌ر مه‌له‌وانیه‌ك مردم، پێشه‌وه‌ی گیانبده‌م ده‌تگوت له‌ ناو به‌فرێم و شه‌ڕه‌به‌فر ده‌كه‌م..
مه‌رگێكیدی نازانم مه‌رگی چه‌نده‌ممه‌، لێ ده‌زانم له‌ ناو به‌فرێ، به‌ هۆی تۆپه‌ڵه‌ به‌فرێكه‌وه‌ گه‌یشت، پێده‌چێ ئاسمان زگی به‌ زه‌وی بسووتێ، ئه‌و قرچه‌ قرچی سوتانی هاوینانه‌ی زه‌ویی بیستووه‌، ده‌یه‌وێ ئه‌گه‌ر به‌ فرمێسكی سپیش بێت فێنكی بكاته‌وه‌، ئه‌و زستانه‌ به‌فرێك كه‌وتووه‌، مناره‌ی چۆلی سه‌د ساڵیش ده‌بێت قه‌ڵاتی به‌و سپیپۆشیه‌ حه‌فته‌ییه‌وه‌ نه‌دیتووه‌...
شار ورد و درشت رژاونه‌ته‌ به‌فره‌وه‌، له‌و رۆژانه‌ چۆله‌كه‌ و منداڵ جێی خۆیان ده‌گۆڕنه‌وه‌، چۆله‌كه‌  به‌ جریوه‌جریو به‌پێی رووته‌وه‌ هه‌ڵده‌فڕنه‌ بن گوێسوانه‌ و ژووه‌ره‌كان و خۆیان گه‌رمده‌كرده‌وه‌، منداڵانیش به‌ پێڵاوی توكنه‌وه‌ دێنه‌ ده‌رێ و خۆ هه‌ڵده‌ده‌نه‌ ناو به‌فره‌وه‌، ده‌یانه‌وی بۆ هاوینیش فێنكایی له‌ له‌شیان داكه‌ن، له‌ ناو جاده‌ و كۆڵانان شه‌ڕه‌ به‌فره‌، هه‌موو بۆ هه‌موو به‌ پێكه‌نینه‌وه‌ به‌فر داوێن و به‌ پێكه‌نینه‌وه‌ به‌فر ده‌گرنه‌وه‌..
  له‌ سه‌ری كۆڵانێ ئه‌و ترومبێله‌ جام ره‌شه‌ بێ ژماره‌یه‌ ده‌بینم، ده‌مێكه‌ راوه‌ستاوه‌، له‌وانه‌یه‌ چاوه‌ڕێی كه‌سێك بكات، ،نه‌مدیتووه‌ پیاوی ئه‌ودیوی جامره‌شانم نه‌دیتووه‌، حه‌زملێیه‌ به‌ چاوی خۆم ببینم و بزانم ئه‌و پیاوانه‌ی خۆیان له‌ پشت جامی ره‌ش شاردۆته‌وه‌، چ جۆره‌ پیاوێكن، ئه‌و پیاوانه‌ی هه‌موو خه‌ڵك ده‌بینن و كه‌س نایانبینێ، شه‌ڕه‌ به‌فری پێكه‌نیناوی درێژه‌ی هه‌یه‌ و ناپسێته‌وه‌، ترومبێله‌ جامره‌شه‌كه‌ش هه‌ر له‌وێ راوه‌ستایه‌، دوو سێ تۆپه‌له‌ به‌فر به‌ ترومبێل ده‌كه‌وێت، ده‌بینم ده‌رگا كرایه‌وه‌، پیاوێك هاته‌ده‌رێ، تووڕه‌و تۆسن، ده‌تگوت له‌ غه‌زه‌بستانه‌وه‌ هاتووه‌ رووی له‌ دوو سێ منداڵ كرد، به‌ڵام هه‌موو كۆڵانی گرته‌وه‌:
له‌و ساته‌ دیتم تۆپه‌ڵه‌ به‌فرێك به‌ شانی پیاوی له‌ ناو جامی ره‌شه‌وه‌ هاتوو، كه‌وت، دیتم به‌ ده‌ستێك شانی خۆی پاككرده‌وه‌ و ده‌سته‌كه‌یدیشی بۆ پشتێنی برد و ده‌مانچه‌ی ده‌رهێنا، سواری كرد، هاواری كرد:
پیاوه‌كه‌ ده‌یویست به‌ جوانی سێره‌ی لێبگرێ و هه‌ڵه‌نه‌كات، من غارم دایه‌ منداڵه‌كه‌، ده‌مویست له‌ لووله‌ی ده‌ممانچه‌ی بپارێزم، ته‌قاندی، كه‌وتم، گوێم له‌ هه‌ستانی تروبێله‌كه‌یه‌، هه‌ستانێك ئاگری كرده‌وه‌، هه‌ستانێك من له‌ ناو خوێنه‌وه‌ بۆنی قیڕ و دووكه‌ڵم كرد، من له‌و جاره‌ی مردن چێژم له‌ خوێنی خۆم وه‌رگرت، له‌ خوێنی خۆمه‌وه‌ گه‌رمیه‌ك ده‌ڕژایه‌ گیانمه‌وه‌،گه‌رمایی خودایی بوو، من له‌ مرۆڤه‌وه‌ گه‌رمایه‌تی وام هه‌ستپێنه‌كردبوو، مه‌رگی ناو به‌فر چ گه‌رم بوو، له‌ ئاویلكه‌دا گوێم له‌ ده‌نگی منداڵه‌كه‌ بوو، ده‌یگوت:
مه‌رگێك له‌و مه‌رگانه‌ دیاریی ژنه‌كه‌مه‌، پێش من چه‌ند كه‌سێك ئه‌و مه‌رگه‌یان تامكردووه‌، زۆر كه‌س ناویان له‌و شێوه‌ مه‌رگه‌ ناوه‌ دیاریی ژنی جوان، رۆژانێكه‌ ژنی جوان به‌ڵای خودایه‌، له‌ به‌دبه‌ختی خۆم منیش ئه‌و به‌ڵایه‌م هه‌یه‌، ژنه‌كه‌م  له‌ جوانه‌كانی شاره‌، له‌ شارئاشووپه‌كانه‌، به‌ر له‌وه‌ی جوانی ببێته‌ به‌ڵا، وامده‌زانی هه‌موو دنیا هی منه‌،كاتێ دوو قۆڵی به‌ شه‌قامدا پیاسه‌مانده‌كرد، پاشای شار بووم، به‌ڵام دوایوه‌ی بووه‌ به‌ڵا، نه‌ پیاسه‌ ما، نه‌ پادشایه‌تی، ئه‌گه‌ر بۆ پێویستیه‌ك، خه‌سته‌خانه‌یه‌ك، تازیمانه‌یه‌ك، لێگێڕانه‌وه‌یه‌ك، شان به‌ شانی به‌ڵا به‌ جاده‌یه‌كدا بڕۆین، وا هه‌ستده‌كه‌م پێ به‌ سه‌ر پشكۆدا ده‌نێم، خواخوامه‌  زوو بگه‌مه‌وه‌ ماڵ و له‌ چاوان بزربم، جگه‌ له‌وه‌ی چاوه‌كان ده‌ڵێی زه‌رگن رووه‌و ژنه‌كه‌م دێن، شوفێر هه‌یه‌ سه‌ری له‌ جام ده‌رده‌هێنێ، ئاگای له‌ سووكان ده‌بڕێ، تامپۆن ده‌كات، كێشه‌ هه‌ر چاوه‌كان نیه‌..
پیاوێك له‌و پیاوانه‌ی له‌ چیاوه‌ هاتبوونه‌وه‌، كه‌ له‌ بازاڕ داده‌به‌زی، گه‌ڕه‌كێك چه‌كدار به‌ دوایه‌وه‌ بوو، نازانم ژنه‌كه‌می له‌ كوێ دیتبوو، دڵی چووبووێ، هه‌رچی شێوازی راوكردنی ژن هه‌یه‌ په‌نای بۆ برد، نه‌گه‌یشته‌ دڵی ژنه‌كه‌م، دوا جار هه‌ڕه‌شه‌ی لێكردبوو، ئه‌گه‌ر نه‌چێته‌ ژێر بار، من بكوژێت، خۆ زۆر بێ شه‌رمانه‌ش پیاوێكی نارده‌ لام:
نا، ئه‌گه‌ر بێ ره‌وشت با، نه‌یده‌ناردمه‌ لای تۆ، تۆ ئه‌گه‌ر ده‌ته‌وێت بژیت، باشتره‌ واز له‌ ژنه‌كه‌ت بێنیت..
ویستمان بچینه‌ دادگاو، داوا له‌ سه‌ر پیاوی له‌ چیاوه‌ هاتوو، تۆماربكه‌م، پارێزه‌رێك ئامۆژگاری كردم، ئه‌و كاره‌ نه‌كه‌م، بۆی روونكردمه‌وه‌ ئه‌و كاره‌ له‌ دوو سه‌ره‌وه‌ خراپه‌، یه‌كه‌م ناوتان به‌د ده‌بێـت، دووه‌م ده‌یدۆڕێنن، چونكێ تۆ به‌لگه‌یه‌كی یاساییت له‌ ده‌ست نیه‌، دوایی تۆ ده‌چیته‌ كام دادگا؟!
دادگا هی خۆیانه‌..
چاره‌؟ خۆ شاردنه‌وه‌،له‌ كوێ؟ بچمه‌ ناو كونی مه‌زراكانیش ده‌مبینێته‌وه‌، ئه‌و هه‌موو سه‌وسواره‌ی به‌ ده‌وره‌وه‌یه‌، به‌ بۆن ده‌مبیننه‌وه‌، تاكه‌ چاره‌ هه‌ڵاتنه‌، شار جێهێشتنه‌، به‌ڵام بۆ كوێ، په‌نایه‌ك شك نابه‌م، بمگرێته‌ خۆی، به‌ دزی خۆمان كۆكرده‌وه‌، نه‌شمده‌زانی روو له‌ كوێ ده‌كه‌م، ئیدی سه‌ر له‌ ئێواره‌یه‌ك، رووه‌و گه‌راج ده‌رۆِیشتین، له‌ فلكه‌ی (به‌ ته‌مای خوای) كه‌ كه‌س نازانێت تروبێل چۆن چۆنی دێت و ده‌ڕوات، به‌ ژاوه‌ژاوترین فلكه‌ی دنیایه‌، هێمنترین و به‌ ره‌وشتترین شوفێر كه‌ ده‌گاته‌ فلكه‌كه‌، ده‌ست ده‌خاته‌ سه‌ر هۆڕن و هه‌ڵیناگرێ و ئاماده‌ نیه‌ سێ چركه‌ بوه‌ستێ و رێ به‌ ترومبێله‌كه‌ی لای ده‌سته‌ راستی یان پیاده‌یه‌ك بدات تێپه‌ڕێ، له‌ به‌ ته‌مای خوایدا په‌ڕینه‌وه‌، به‌ ته‌مای خوایه‌، ئێمه‌ له‌ شپرزه‌یی رێمان برده‌وه‌ به‌ ته‌مای خوای، ده‌نا دوو رێیدی هه‌بوو بۆ گه‌راجی گوندان..
 ده‌مێكه‌ مردووم بۆیه‌ باش نایه‌ته‌وه‌ یادم گه‌یشتبووینه‌ كوێی به‌ ته‌مای خوای، هێنده‌م له‌ یاده‌  به‌ به‌رزایی ترومبێلی به‌رز به‌رزبوومه‌وه‌ و پشتاوپشت كه‌وتم، ترومبێله‌كه‌م نه‌دیت، كه‌سم نه‌دیت،  ژنه‌كه‌م قیژاندی،گوێم لێیه‌  ده‌ڵێ:
دوا مه‌رگ زۆر كتوپڕبوو، نه‌پرژامه‌ سه‌ر هیچ، ته‌واو فریای گواستنه‌وه‌ی گۆڕه‌كانم نه‌كه‌وتم، له‌ گردێك ده‌گه‌ڕام، مناری چۆلی لێوه‌ دیار بێت، به‌ یادی ئه‌و كچه‌ی فسی ئیبلیس ناونیشانه‌كه‌ی بردی حه‌زمده‌كرد مناره‌ی چۆلیم لێوه‌ دیاربێت، گردم دیته‌وه‌، زۆر دووره‌ به‌ڵام جوان لێیه‌وه‌ دیاره‌، وا گۆڕ گۆڕ گۆڕه‌كانم دێنمه‌وه‌، هه‌ر شه‌ش گۆڕم هێنایه‌وه‌، له‌ سه‌ر كێله‌كان ناوی خۆم نووسی، وا وێنه‌ی  بكوژه‌كانم ده‌كێشم، وێنه‌ی بابم، كوڕه‌كه‌م، پیاوه‌كه‌ی له‌ ئه‌فغانستان گه‌ڕابۆوه‌، وێنه‌ی پیاوه‌كه‌ی پشت جامی ره‌ش، وێنه‌ی سه‌ربازێكی ئه‌مریكی، له‌ پڕ پیاوێكی پشتێن قه‌فقه‌فی په‌یداده‌بێت:
له‌ پڕ مێوانخانه‌ترین شار، له‌ پڕ مزگه‌فتترین شار، به‌ دیار شاره‌وه‌  حه‌فت شه‌و له‌ گۆڕستان خه‌وتم، پیاوه‌ حه‌فت گیانه‌كه‌  هه‌ر شه‌وه‌ی مه‌رگێكی خۆی بۆم گێڕایه‌وه‌..
