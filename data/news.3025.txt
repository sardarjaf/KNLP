دروستكردنی ده‌وڵه‌تی نوێی عیراقی له‌دوای‌ هه‌ڵوه‌شاندنه‌وه‌ی‌ ده‌وڵه‌تی عوسمانی نه‌ به‌رهه‌می په‌ره‌سه‌ندنی كۆمه‌ڵایه‌تی كۆمه‌ڵگه‌ی‌ عیراقی بووه‌، نه ‌به‌بڕیاری‌ عیراقیه‌كانیش دروستبوو، به‌ڵكو به‌رژه‌وه‌ندی‌ ده‌وڵه‌تانی بڕیاربه‌ده‌ست له‌ دوای‌ جه‌نگی یه‌كه‌م چۆن چه‌ندین ده‌وڵه‌تی تری‌ له‌ پاشماوه‌ی‌ پیاوه‌ نه‌خۆشه‌كه‌ دروستكرد، به‌هه‌مان شێوه‌ ده‌وڵه‌تێكی ئیفلیجیشی به‌ ناوی‌ عیراقه‌وه‌ له‌ چه‌ند پێكهاته‌یه‌كی ناڕێك و پێك ‌و‌ نه‌گونجاو پێكهێنا كه‌ یه‌كێك له‌و پێکهاتانه‌ كوردی‌ باشوور یان ولایه‌تی موسڵ بوو كرا به‌ پارسه‌نگی سوننه‌ بۆ دروستكردنی هاوسه‌نگی له‌گه‌ڵ شیعه‌دا له‌و ده‌وڵه‌ته‌ نوێیه‌دا.
ئینگلیزه‌كان ته‌نانه‌ت له‌ جوگرافیای ده‌وڵه‌ته‌ نوێیه‌كه‌یاندا كه‌سێكی ئه‌وتۆیان نه‌دۆزییه‌وه‌ بیكه‌ن به‌ سه‌رۆكی ئه‌و ده‌وڵه‌ته‌، به‌ڵكو مه‌لیكێكیان له‌ حیجازه‌وه‌ بۆ قه‌رز كرد.
مه‌لیك فه‌یسه‌ڵی یه‌كه‌م كه‌ باز به‌سه‌رییه‌وه‌ نیشته‌وه‌و كرا به‌ مه‌لیكی ئه‌و ده‌وڵه‌ته‌، له‌ ساڵی 1931 دا یاداشتێكی پڕ بایه‌خی نووسیووه‌ له‌ باره‌ی‌ پێكهاته‌ی‌ گه‌لی ئه‌و وڵاته‌وه‌ كه‌ ئه‌و مه‌لیكی بووه‌و تیایدا ده‌ڵێت: وڵاتی عیراق بریتیه‌ له‌ چه‌ند وڵاتێك كه‌ كه‌موكورتی هه‌یه‌ له‌ گرنگترین بنه‌مای ژیانی كۆمه‌ڵایه‌تیدا، واته‌ له‌ رووی فیكری و نه‌ته‌وه‌یی و ئایینیه‌وه‌، ئه‌م باره‌ش هێزه‌ دابه‌شبووه‌كانی له‌ یه‌كتر په‌رت كردووه‌.
به‌ كورتی ده‌ڵێم له‌ كاتێكدا دڵم ژان ده‌كات، چونكه‌ به‌ بڕوای من له‌ عیراقدا گه‌لی عیراقی بوونی نییه‌، به‌ڵكو چه‌ند پێكهاته‌یه‌كی‌ مرۆیی هه‌ن كه‌ داماڵراون له‌ هه‌ر جۆره‌ بیرێكی نیشتیمانی(بڕوانه‌: عه‌بدلكریم الازری.
مشكله‌ الحكم فی العراق، لندن، 1991).
ئه‌م قسه‌یه‌ نووسراوی پیاوێكه‌ كه‌ ئه‌ركی فه‌رمانڕه‌وایی عیراقی پێسپێردراوه‌.
كه‌واته‌ لێره‌وه‌ ئاشكرایه‌ كه‌ نه‌ له‌و كاته‌داو نه‌ دواترو نه‌ ئێستاش شتێك دروست نه‌بووه‌ ناوی گه‌لی عیراق بێت.
ره‌نگه‌ له‌سه‌رده‌می زه‌بروزه‌نگ و سه‌ركوتكردنی به‌عسدا دیواری نێوان پێكهاته‌كانی ئه‌م وڵاته‌ی‌ كه‌ ناوی لێنراوه‌ عیراق ته‌نكتر بووبێت، به‌ڵام هه‌رگیز شتێك پێكنه‌هاتووه‌ كه‌ ناوی گه‌لی عیراق بووبێت.
دیاره‌ له‌ دوای رووخاندنی رژێمی به‌عسیش ئه‌و دیواره‌ زیاتر ئه‌ستوور بووه‌.
له‌لایه‌ك مێژووی‌ خوێنینی نێوان شیعه‌و سوننه‌ گه‌ڕاوه‌ته‌وه‌ بۆ ئه‌و ساته‌ی كه‌ حسێن به‌ ده‌ستی سوننه‌كان شه‌هید كرا.
له‌و رووه‌شه‌وه‌ شیعه‌كان نه‌یانتوانیووه‌ له‌وه‌ تێبگه‌ن ئه‌وانه‌ی‌ كه‌ حسێنیان شه‌هید كرد نه‌ك هه‌ر ئه‌م سوونانه‌ی‌ ئێستای عیراق نه‌بوون، به‌ڵكو به‌ ئه‌سته‌م په‌یوه‌ندی له‌نێوان ئه‌مان و ئه‌واندا ده‌دۆزرێته‌وه‌.
ره‌نگه‌ تاكه‌ شتێك پێكیانه‌وه‌ ببه‌ستێت ته‌نیا هاومه‌زهه‌بی بێت.
هه‌روه‌ك سوننه‌كانیش ناتوانن به‌وچاوه‌ سه‌یر بكه‌ن كه‌ سوننه‌و شیعه‌ هه‌ردووكیان دوو ئایینزای ئیسلامن و ئه‌وه‌ی‌ له‌ رابردوودا روویداوه‌ بووه‌ته‌ مێژوو، ئێستا سه‌رده‌م و قۆناغێكی تره‌.
دیاره‌ ئه‌م دوو پێكهاته‌یه‌ش هه‌ردووكیان به‌ چاوی نه‌یار سه‌یری كورد ده‌كه‌ن.
شیعه‌ له‌ روانگه‌ی‌ جیاوازی مه‌زهه‌ب و نه‌ته‌وه‌ییه‌وه‌ سه‌یری كورد ده‌كات.
سوننه‌ش له‌ روانگه‌ی‌ جیاوازی نه‌ته‌وه‌ییه‌وه‌ سه‌یری كورد ده‌كات.
كوردیش ئه‌گه‌ر خۆی فریو نه‌دات ئه‌وا له‌ ئاستی یه‌كه‌مدا ده‌بێت بزانێت ئه‌و دووانه‌ هه‌ردووكیان به‌شێكن له‌و نه‌ته‌وه‌ سه‌رده‌سته‌ی‌ كه‌ ده‌سه‌ڵاته‌ جۆراوجۆره‌كانی ئه‌وان به‌رده‌وام ده‌ستی به‌ خوێنی كورد سووربووه‌و هه‌ردووكیشیان وه‌كو ده‌سه‌ڵاتی نه‌ته‌وه‌ی عه‌ره‌ب داگیركه‌ری كوردستانن.
هه‌رچه‌نده‌ قسه‌یه‌كی بۆش و پووچ هه‌یه‌ سیاسیه‌ ساویلكه‌كان به‌كاریده‌هێنن كه‌ گوایه‌ گه‌لان هه‌موو برای یه‌كن و ئه‌وه‌ ده‌سه‌ڵاته‌كانن دوژمنایه‌تی دروست ده‌كه‌ن، ئه‌م قسه‌یه‌ بۆ ئه‌و وڵاتانه‌ راسته‌ كه‌ ئیتنیك و نه‌ته‌وه‌ جیاوازه‌كان له‌سه‌ر شێوه‌ی‌ رێككه‌وتن و پاراستنی به‌رژه‌وه‌ندی‌ هه‌موویان و هاوشانی له‌ شووناسیاندا یه‌كتریان قبوڵكردووه‌و ئینتیمای‌ هاوشێوه‌یان هه‌یه‌ بۆ نیشتمان.
ئه‌گه‌ر ئه‌م قسه‌یه‌ بۆ هه‌ر شوێن و جێگایه‌كی تر راست بێـت ئه‌وا هه‌رگیز بۆ عیراق راست نییه،‌ چونكه‌ عه‌ره‌بی عیراق نه‌ به‌ ده‌سه‌ڵات و نه‌به‌گه‌لیشه‌وه‌ وه‌ها ره‌فتاریان نه‌كردووه‌ كه‌ برای نیشتمانی یان برای ئایینی كورد بن.
بێگومان هه‌ر له‌سه‌تای‌ دروستبوونب عیراقه‌وه‌و دواتریش له‌ قسه‌كه‌ی‌ مه‌لیك فه‌یسه‌ڵه‌وه‌ ده‌رده‌كه‌وێت كه‌ هیچ بنه‌مایه‌كی گه‌ل له‌ عیراقدا دروست نه‌بووه‌.
هه‌ر بۆیه‌ تاكه‌ رێگه‌چاره‌ ئه‌وه‌یه‌ كه‌ ئه‌م سێ پێكهاته‌یه‌ به‌ ته‌واوی لێكتر دوور بخرێنه‌وه‌و ئه‌م وڵاته‌ بگێڕدرێته‌وه‌ بۆ سێ ولایه‌ته‌كه‌ی سه‌رده‌می عوسمانی كه‌ ولایه‌تی موسڵ و به‌غداو به‌سره‌ بوون، یان له‌سه‌ر بنه‌مای‌ جیاوازی‌ نه‌ته‌وه‌یی بگێڕدرێته‌وه‌ بۆ باری‌ پێشتری‌ خۆی‌ كه‌ ئه‌وه‌ی‌ ناونراوه‌ عیراق له‌ عیراقی عه‌ره‌بی و باشووری‌ كوردستان پێكهاتووه‌.
به‌ڵام له‌ دوای هه‌ڵوه‌شاندنه‌وه‌ی‌ ده‌وڵه‌تی عوسمانی هه‌ردوو ولایه‌تی به‌غداو به‌سره‌ پێكه‌وه‌ لكێندران و پاشان ئینگلیزه‌كان تێبینی ئه‌وه‌یان كرد كه‌ هاوسه‌نگی مه‌زهه‌بی نیه‌ له‌و عیراقه‌ نوێیه‌ی‌ كه‌ دروستیان كردووه‌،
بۆیه‌ بیریان له‌ شتێكی تر كرده‌وه‌ كه‌ ئه‌ویش لكاندنی باشووری كوردستان یان ولایه‌تی موسڵ بوو به‌و عیراقه‌ نوێیه‌وه‌، كه‌ ئه‌ویش زیاتر بۆ هاوسنگی دروستكردن بوو له‌ نێوان مه‌زهه‌بی شیعه‌و سوننه‌دا چونكه‌ ژماره‌ی‌ شیعه‌ی عه‌ره‌ب زۆرتر بوون له‌ سوننه‌ی‌ عه‌ره‌ب.. ئینگلیزه‌كان بیریان له‌وه‌ كرده‌وه‌ كه‌ كورده‌كانیش سوننه‌ن مه‌زهه‌بن و به‌ لكاندنی ولایه‌تی موسڵ به‌ عیراقه‌ نوێیه‌كه‌وه‌ ده‌شێ ئه‌و هاوسه‌نگیه‌ دروست بكه‌ن، دیاره‌ هۆكاری‌ ئابووریش بۆ ئه‌و پێكه‌وه‌لكاندنه‌ رۆڵی دیاری‌ هه‌بووه‌.
به‌ڵام ئه‌وان شتێكی زۆر گرنگیان له‌ بیركرد ئه‌ویش جیاوازی نه‌ته‌وه‌یی كوردو عه‌ره‌ب بوو.
به‌ مانایه‌كی تر ده‌توانین بڵێین به‌ریتانیه‌كان جیاوازی مه‌زهه‌بییان له‌پێش جیاوازی نه‌ته‌وه‌ییه‌وه‌ دانا، ئه‌گه‌ر چاو له‌ كاریگه‌ری به‌رژه‌وه‌ندییه‌ ئابووری و ستراتیژییه‌كانی تری‌ ئه‌و ده‌وڵه‌ته‌ داگیركاره‌ی‌ ئه‌و سه‌رده‌مه‌ بپۆشین.
ئه‌وه‌ ئاشكرایه‌ كه‌ به‌ پێی به‌نده‌كانی 61، 63، 63ی رێكه‌وتننامه‌ی‌ سیڤه‌ر ده‌بوو كورد ببوایه‌ته‌ خاوه‌نی ده‌وڵه‌تی خۆی ، به‌ڵام له‌و كۆنگره‌یه‌ی كه‌ له‌ 12ی ئازاری 1921 دا له‌ قاهیره‌ به‌سترا، كێشه‌ی‌ كورد رووبه‌رێكی فراوانی داگیر كردو سیاسیه‌ به‌ریتانییه‌كان بوونه‌ دوو ئاراسته‌وه‌ سه‌باره‌ت به‌و بابه‌ته‌.
ئاراسته‌ی یه‌كه‌میان له‌گه‌ڵ ئه‌وه‌ بوون كه‌ باشووری كوردستان بلكێندرێت به‌و ده‌وڵه‌ته‌ نوێیه‌ی‌ كه‌ تازه‌ له‌ عیراق له‌ هه‌ردوو ولایه‌تی به‌غداو به‌سره‌ دروستیان كردبوو كه‌ بێرسی كۆكس و بێڵ و مێجه‌ر سۆن له‌ رابه‌رانی ئه‌م ئاراسته‌یه‌ بوون.
ئاراسته‌ی‌ دووه‌م لایه‌نگری دروستكردنی ده‌وڵه‌تێكی سه‌ربه‌خۆ بوو له‌ باشووری كوردستان كه‌ هێربێرت یۆنگی جێگری سكرتێر له‌ به‌شی رۆژهه‌ڵاتی ناوه‌ڕاستی وه‌زاره‌تی كۆلۆنیه‌كان و نوئێل و لۆرانسی عه‌ره‌ب و چه‌رچڵ له‌ رابه‌رانی ئه‌م ئاراستیه‌یان بوون.
سه‌رئه‌نجام ئاراسته‌ی‌ یه‌كه‌م سه‌ركه‌وتنیان به‌ده‌ست هێناو حكومه‌تی یه‌كه‌می عیراق به‌ سه‌رۆكایه‌تی عه‌بدوڕه‌حان نه‌قیب بڕیاری لكاندنی باشووری كوردستانی به‌ ده‌وڵه‌تی عیراقیه‌وه‌ داو خستیه‌ بواری جێبه‌جێكردنه‌وه‌ به‌بێ ئه‌وه‌ی‌ راوبۆچوونی خه‌ڵكی باشووری كوردستان وه‌ربگیرێت.
هه‌رچه‌نده‌ چه‌رچڵ خۆی وه‌كو وه‌زیری كۆلۆنیه‌كان دوو دڵ بوو به‌رامبه‌ر به‌و بڕیاره‌و ده‌ترسا له‌وه‌ی‌ كه‌ ئه‌تاتورك بتوانێت تووڕه‌یی كوردی باشوور به‌كاربهێنێت و هه‌وڵی گێڕانه‌وه‌ی ولایه‌تی موسڵ بۆ سه‌ر كۆماری نوێی توركیا بدات.
به‌ڵام له‌گه‌ڵ ئه‌وه‌شدا كرداری پێكه‌وه‌ لكاندنه‌كه‌ سه‌ری گرت و به‌وه‌ش كوردی باشووری كوردستان كرایه‌ پارسه‌نگی سوننه‌ی عه‌ره‌ب بۆ ئه‌وه‌ی‌ جۆرێك له‌ هاوسه‌نگی مه‌زهه‌بی له‌ چوارچێوه‌ی‌ ئه‌و ده‌وڵه‌ته‌ نوێیه‌دا دروست بكات.
به‌ڵام وه‌كو له‌ قسه‌كانی مه‌لیك فه‌یسه‌ڵی یه‌كه‌مدا ده‌رده‌كه‌وێت كه‌ پاش تێپه‌ڕبوونی ده‌ ساڵ به‌سه‌ر دروستكردنی ده‌وڵه‌تی عیراق و لكاندنی باشووری كوردستان پێیه‌وه‌ هێشتا شتێك دروست نه‌بووه‌ به‌ ناوی گه‌لی عیراق، ئه‌وه‌ش نه‌ك له‌ نێوان كوردو عه‌ره‌بدا به‌ڵكو له‌ نێوان عه‌ره‌بی سوننه‌و شیعه‌شدا.
له‌ راستیدا ئه‌و ده‌وڵه‌ته‌ نوێیه‌ی‌ كه‌ به‌ریتانییه‌كان به‌ ناوی ده‌وڵه‌تی عیراقیه‌وه‌ دروستیان كرد، سه‌رباری ئه‌وه‌ی كه‌ باشووری‌ كوردستانیشی پێوه‌ لكێندرا، به‌ڵام هێشتا رێژه‌ی‌ شیعه‌ زیاتر بوو له‌ سوننه‌.
سه‌رباری ئه‌وه‌ش له‌ دانانی یه‌كه‌م مه‌لیكه‌وه‌ كه‌ له‌ بنه‌ماڵه‌ی شه‌ریف خواسترا تاوه‌كو 14ی ته‌موزی 1958 كه‌ عه‌بدولكه‌ریم قاسم كوده‌تای كردو سیستێمی پاشایه‌تی گۆڕی بۆ سیستێكی كۆماری یان سه‌ربازی، سه‌رۆكایه‌تی ده‌وڵه‌تی عیراقی له‌ ده‌ستی تایه‌فه‌ی‌ سوننه‌دا بوو.
دوای ئه‌وه‌ش تا رووخاندنی سیستێمی به‌عس له‌ سه‌رده‌ستی ئه‌مه‌ریكیه‌كان له‌ ساڵی 2003 دا دیسان ده‌سه‌ڵاتی عیراق  هه‌ر له‌ ده‌ستی سوننه‌ی‌ عه‌ره‌بیدا بوو كه‌ سوننه‌ی‌ عه‌ره‌ب به‌ بێ كورد له‌ چاو شیعه‌دا كه‌مینه‌ بوون.
ئه‌مه‌ش نیشانی ده‌دات كه‌ ساده‌ترین بنه‌مای دیموكراسی له‌ عیراقدا بوونی نه‌بووه‌، ئه‌گینا له‌سه‌ر بنه‌مای ده‌نگدان كه‌ له‌م وڵاته‌دا تاكو ئێستاش ده‌نگدانه‌ له‌سه‌ر بنه‌مای ئینتیمای مه‌زهه‌بی و تایه‌فی شیعه‌ به‌ كاریگه‌ری ئه‌وه‌ی‌ زۆرینه‌ن، ئه‌وان ده‌بوونه‌ فه‌رمانڕه‌وای عیراق.، وه‌كو ئه‌وه‌ی‌ له‌ ماوه‌ی‌ سیازده‌دا ساڵی دوای روخاندنی رژێمی سه‌دامدا ده‌یبینین كه‌ له‌و هه‌ڵبژدنانه‌ی‌ كه‌ له‌م ماوه‌یه‌دا كراون ده‌نگی تایه‌فی شیعه‌ زۆرینه‌یه‌و به‌و پێیه‌ش له‌سه‌ر بنه‌مای ئه‌وه‌ی‌ له‌ ده‌ستووری عیراقیدا هاتووه‌ پۆستی سه‌رۆكایه‌تی ئه‌نجوومه‌نی وه‌زیران به‌ مافی شیعه‌ دانراوه‌و له‌راستیشدا له‌و ماوه‌یه‌دا ئه‌م بڕگه‌یه‌ جێبه‌جێكراوه‌و له‌ ئه‌نجامی هه‌ڵبژاردنه‌كاندا تاكو ئێستا چوار حكومه‌تی شیعه‌ پێكهاتووه‌ كه‌ جه‌عفه‌ری و مالیكی عه‌بادی‌ سه‌رۆكایه‌تیان كردووه‌ كه‌ هه‌رسێكیان شیعه‌ن.
دیاره‌ ئه‌و چوار حكومه‌ته‌ی‌  كه‌ له‌ ئه‌نجامی هه‌ڵبژاردنه‌كاندا له‌ عیراقدا پێكهاتوون له‌ژێر ئه‌و ناونیشانه‌دا بوون كه‌ پێی ده‌ڵێن حكومه‌تی یه‌كێتی نیشتیمانی، ئه‌گه‌رچی له‌گه‌وهه‌ردا حكومه‌وتی تاكه‌ كه‌سی شیعه‌ بوون.
به‌ڵام له‌گه‌ڵ ئه‌وه‌شدا سه‌رۆكی ئه‌و حكومه‌تانه‌ی شیعه‌ باسی ئه‌وه‌ ده‌كه‌ن كه‌ كۆكردنه‌وه‌ی‌ زۆرینه‌ی‌ لایه‌نه‌كانی عیراق له‌و حكومه‌تانه‌دا رۆڵی نێگه‌تیفی بینیووه‌ له‌سه‌ر ئیش و كاره‌كانی حكومه‌ت چونكه‌ ئه‌وه‌ كۆكردنه‌وه‌ی‌ كۆمه‌ڵێ هێزی ناكۆك و دژ به‌یه‌كه‌ كه‌ ناتوانن وه‌كو چه‌ند گروپێك به‌ڵام پێكه‌وه‌و به‌یه‌كگرتوویی‌ بۆ یه‌ك ئامانج كاربكه‌ن.
پێشترو له‌سه‌رده‌می مالیكیدا گوێمان له‌وه‌ ده‌بوو كه‌ مالیكی و لیست و هاوپه‌یمانه‌كانی باسی پێكهێنانی حكومه‌تی زۆرینه‌ی‌ سیاسییان ده‌كرد.ئێستا راسته‌خۆو ناڕاسته‌وخۆ ماڵی شیعه‌ ده‌یانه‌وێت ئه‌وه‌ی‌ پێشتر باسیان ده‌كرد بكه‌ن به‌ كردار خۆیان وه‌كو زۆرینه‌ ده‌ست به‌سه‌ر كۆی‌ جومگه‌كانی ده‌سه‌ڵاتدا بگرن.
ئه‌مه‌ش به‌ پێی بنه‌ماكانی دیموكراسی شتێكی ره‌وایه‌، به‌ تایبه‌تی دوای‌ ئه‌وه‌ی‌ كه‌ چ له‌ عیراق و چ له‌ كوردستانیش به‌ ئاشكرا بینیمان حكومه‌تی هه‌مووان به‌ ناوی‌ حكومه‌تی ته‌وافوقه‌وه‌ بێجگه‌ له‌ دامه‌زراوه‌یه‌ك بۆ دزی‌ و گه‌نده‌ڵی هیچی تر نیه‌.
واته‌ ته‌وافوق له‌سه‌ر رێكه‌وتن له‌ نێوان لایه‌نه‌كاندا بۆ ئه‌وه‌ی‌ چاوپۆشی له‌ دزی‌ و گه‌نده‌ڵییه‌كانی یه‌كتر بكه‌ن.
پێشتر دروستكردنی حكومه‌تی زۆرینه‌ بۆ مالیكی كارێكی ئاسان نه‌بوو، له‌ كاتێكدا سوننه‌و كوردو به‌شێك له‌ شیعه‌كانی‌ به‌ ئاسانی بۆ دوور نه‌ده‌خرایه‌وه‌، به‌ڵام عه‌بادی‌ به‌ شێوه‌یه‌كی به‌رنامه‌ بۆ داڕێژراو ئه‌و به‌رنامه‌یه‌ی‌ خسته‌ بواری‌ جێبه‌جێكردنه‌وه‌.
له‌ سه‌رتاوه‌ به‌لادانی جێگرانی سه‌رۆك كۆمارو پاشان لادانی جێگره‌كانی خۆیی و سه‌ندنه‌وه‌ی‌ زۆرێك له‌ پۆسته‌كان له‌ كوردو سونه‌و دواجاریش كه‌ سه‌در هاته‌ سه‌ر هیڵ و داوای‌ پێكهێنانی حكومه‌تی ته‌كنۆكراتی كرد، هه‌ر بۆ زه‌مینه‌سازی‌ زیاتر بوو بۆ پێكهێنانی حكومه‌تی زۆرینه‌ی‌ شیعه‌و په‌راوێزخستنی زیاتری‌ كوردو سوننه‌ بوو.
هه‌ر لێره‌وه‌ به‌پشتبه‌ستن به‌و هه‌نگاوانه‌ی‌ له‌ ماوه‌ی‌ رابردوودا مالیكی دژی كورد ناویه‌تی و ئێستاش عه‌بادی‌ له‌سه‌ری‌ ده‌ڕوات، ئامانجه‌كه‌ ئه‌وه‌ بێت كورد ئاراسته‌ی‌ جیابوونه‌وه‌ له‌ عیراق بگرێته‌به‌ر بۆ ئه‌وه‌ی‌ له‌ لایه‌ك به‌ربه‌ستی كوردی له‌به‌رده‌م شیعه‌دا نه‌مێنێت و به‌وه‌ش ده‌سه‌ڵاتی ره‌های خۆی به‌سه‌ر عیراقدا بسه‌پێنێت.
ئێستا له‌ ئاگایی و به‌به‌رنامه‌ بێت یان به‌ گه‌مژه‌یی و له‌خۆوه‌و بێ بیركردنه‌وه‌ بێت، ره‌فتارو هه‌ڵس و كه‌وته‌كانی مالیكی پیشترو عه‌بادی‌ ئێستا كورد به‌ ئاراسته‌ی‌ دووركه‌وتنه‌وه‌ی‌ زیاتر له‌ عیراق پاڵ ده‌نێنن.
بۆیه‌ ده‌شێ بگووترێت پێ ده‌چێت مالیكی و عه‌بادی‌ به‌تایبه‌تی و شیعه‌ به‌ گشتی ئه‌و عیراقیانه‌ بن كه‌ به‌رنامه‌كه‌ی‌ به‌ریتانیاو سایكس بیكۆ بۆ لكاندنی باشووری كوردستان به‌ عیراقه‌وه‌ هه‌ڵبووه‌شێننـه‌وه‌.
بێگومان به‌وه‌ش شیعه‌ خزمه‌تێكی باشی كورد ده‌كات.
به‌ڵام ئایا كورد خۆی‌ چی ده‌كات؟ ئایا ئاماده‌باشی هه‌یه‌ بۆ ئه‌وه‌ی‌ جیابێته‌وه‌؟ ئایا بارودۆخی ناوخۆی‌ كوردستان به‌م شێواویه‌ له‌باره‌ بۆ جیابوونه‌وه‌؟ ئایا كورد ده‌توانێت ماڵی خۆی‌ بۆ ئه‌و ئامانجه‌ رێكبخات یان ده‌بێت جیابوونه‌وه‌و سه‌ربه‌خۆیی ببێته‌ هۆكار بۆ رێكخستنی ناو ماڵی كورد؟
به‌هه‌رحاڵ ئێستا چ یارودۆخی نێوده‌وڵه‌تی و چ یارودۆخی ناو عیراق هه‌ردووكیان له‌بارن بۆ جیابوونه‌وه‌ی‌ كورد به‌ هێمنی له‌ عیراق.
ئه‌وه‌ی‌ جێگای‌ پرسیار له‌باره‌وه‌ كردنه‌ باری‌ ناوخۆو بكه‌ری‌ كوردی‌ خۆیه‌تی، واته‌ ئایا كه‌ی‌ ئه‌م بكه‌ره‌ ده‌گاته‌ ئه‌وه‌ی‌ بڕیار بدات و به‌و بڕیاره‌ش ده‌شێ هێزی‌ جه‌ماوه‌ریی بۆ یه‌كبوون به‌سه‌ر هێزی‌ پارچه‌ پارچه‌بووی‌ حیزبی و هه‌ڵوێستی لاوازی‌ نوخبه‌ی‌ نه‌خۆش و بێگانه‌په‌رستی سیاسیدا زاڵ ببێت؟
