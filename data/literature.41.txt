ئه‌مشه‌و نایه‌یته‌ ژوره‌وه،به‌قه‌بری دایکی سوێندی لێخواردم.بارانێکی به‌ خوڕه‌م له‌ ده‌ره‌وه‌ ئه‌بارێ.چوونه‌ ده‌ره‌وه‌م پێباش نیه‌،له‌سه‌ر تاکیک له‌ سۆفاکان پاڵکه‌وتم.ئه‌م سوێنده‌ی بۆ لێخواردم؟تۆ بڵیی به‌ڕاستی بێت؟جارێکی تر پاڵم به‌ ده‌رگاکه‌وه‌ نا،بێگومان داخراوه‌.وه‌ک پارچه‌یه‌ک چه‌وری توامه‌وه‌و له‌مدیو ده‌رگاکه‌وه‌ بوم به‌ تۆپه‌ڵێک.بیرم له‌وڕۆژانه‌ ئه‌کرده‌وه‌‌ که‌ بێمن هه‌ناسه‌ی نه‌ئه‌دا.کوان ئه‌وڕۆژانه‌؟ (تۆ ئه‌ڵێی ئه‌مشه‌و نایه‌مه‌وه‌ من ناتوانم ساتێک بێتۆ هه‌ڵ بکه‌م)!!
!که‌یه‌که‌مجار بۆ شه‌وێک به‌جێمهێشت،چارم نه‌بوو هه‌ر ئه‌بوایه‌ بڕۆم،تاگه‌ڕامه‌وه‌ خه‌و له‌ چاوانی نه‌که‌وت.ئه‌ی من ئه‌متوانی ساتێک بێئه‌و بژیم؟ کاتێک گه‌رم بووین،گه‌رم گه‌رم گه‌رم له‌ هه‌موو شتێکدا گه‌رم،گه‌رم له‌ باوه‌ش،گه‌رم له‌ماچ،گه‌رم له‌ هه‌ناسه‌،گه‌رم له‌ ده‌ستو په‌نجه‌ گوشین و ‌ئاخو ناله‌ی پڕ خرۆش،گه‌رم له‌ قاقاو پێکه‌نین،گه‌رم له‌ قوڵپه‌ی گریان،گه‌رم له‌ تێروانین و چاو له‌ چاو بڕین،گه‌رمتر له‌ کوره‌ی پر جۆش،گه‌رمتر گه‌رمتر،،،،،،به‌ڵام ئێستا هه‌موو ئه‌مانه‌ به‌سته‌ڵه‌ک وشه‌خته‌و چلووره‌و سه‌هۆڵبه‌ندانه‌، هه‌مووی نووسراوی سه‌ر سه‌هۆڵی چله‌ی زستانه،ئه‌و سه‌هۆڵه‌ی گه‌رمای هاوین،بۆ ئاسمانی ده‌باو،له‌ کاتێکیتر له‌ وه‌رزێکیتر له‌ شارێکیتر‌ له‌ کێڵگه‌یه‌کیتر به‌سه‌ر خۆڵی تینوو دائه‌بارێ و هیچی لێسه‌وز نابێت.که‌ یه‌که‌م جار له‌دوکانه بچکۆله‌که‌ی باوکم،که‌ بۆ یارمه‌تی دانی،نیوه‌ڕوان ئه‌چووم بینیم.
(نقووڵ هه‌یه‌؟ به‌ڵی هه‌یه‌،،،،به‌چه‌نده‌؟ده‌نکی به‌ماچێک،،،،به‌ڕاست؟ په‌نجا دانه‌م بده‌رێ و دووایی نه‌نکم ئه‌نێرم حسابی له‌ گه‌ڵ بکه‌*.
ئه‌مه‌نده‌ پێکه‌نیم خه‌ریک بوو له‌ هۆش خۆم بچم،،،،ئه‌م کچه‌ چه‌ند زۆر زانه‌،،،،حساب له‌گه‌ڵ نه‌نکی بکه‌م،،،،جاوه‌ره‌ من په‌نجا ده‌نک نووقڵی بده‌‌مێ و دوایی نه‌نکی بێت،،،نازانم، ڕومه‌ته‌کانیم بۆ ڕائه‌گرێ تا ماچی بکه‌م یان حه‌زی له‌ ماچی ده‌مه‌،،،،،یه‌ک دوو سێ،،،،چلوو نۆ،،،لێت خه‌ڵه‌ت بوو کوڕه‌،دووباره‌ی بکه‌وه).ئه‌وه‌ یه‌که‌م بینین بوو دوای ئه‌وه‌ هه‌ر به‌ڕاست هه‌موو ڕۆژێک ماچم ئه‌کردوو ئه‌ویش تێر نقووڵ ئه‌بوو.باوکم وتی ئه‌م نوقڵه‌ دوابڕاوه‌ هیچ قازانج ناکات ئیتر نایهێنم،ماوه‌یه‌ک منی بێماچ و ئه‌وێشی بێنقوڵ کرد.به‌ڵام تازه‌ کار له‌کار ترازابوو،من به‌بێ نقوڵیش هه‌ر ماچی خۆم هه‌بوو.له‌پاڵده‌رگاکه‌ به‌یادی جاران خه‌وم لێکه‌وت.
جه‌سته‌م بوو به‌ دووبه‌ش،به‌شێکی تێشکیکی سه‌وز به‌ماڵه‌که‌دا ئه‌گه‌ڕا،به‌شه‌که‌ی ترم بێگیان له‌پاڵ ده‌رگاکه‌ مایه‌وه.‌( ئه‌وه ‌دڵمه‌ ئه‌و تیشکه‌ سه‌وزه‌) که‌ ده‌ستم له‌سه‌ر دڵم دانا،هیچ هه‌ستم به‌ لێدان نه‌کرد.ئه‌م تیشکه به‌شێوه‌ی گه‌نجێکی جوان،که‌ هه‌رگیز من خۆم ئاوا نه‌بینیوه‌ لێم نزیک بۆوه‌.
منیک ئێستا پیرو که‌نه‌فت،چرجه‌کانی ژێر چاوم و سپی سه‌رو ڕیشم له‌کوێ؟ئه‌م لاوه‌ تازه‌ پێگه‌شتوه‌ ده‌مووچاو سافو بێگه‌رده‌ له‌کوێ؟
من لێره‌م ئه‌وه‌ی تۆ ده‌ستی لێئه‌ده‌یت،له‌وێ نه‌ماوه‌ ئه‌وه‌تا لێره‌،ئه‌وه‌ دڵته‌ ئێستا هێند سه‌ربه‌سته‌ ده‌رگا داخراوه‌کان ئه‌بڕێ.
ئه‌و بێباکانه‌ له‌ کراسێکی ته‌نکی نیوه‌ڕۆتی شینی ئاسمانی،له‌سه‌ر بێده‌که‌ی خه‌وتبوو.جووتێک که‌روێشکی چاو ڕه‌شی خرپن له‌ باخه‌ڵیدا که‌روێشکه‌ خه‌ویان کردبوو.به‌چاوێک خه‌وتو به‌ چاوێک به‌ ئاگا.ئێمه‌یان بینی.ئه‌بێ بڵێم منیان بینی،ئێستا ئێمه‌ بوینه‌وه‌ به‌یه‌ککه‌س.که‌روێشکه‌کان بانگیان کردم،لیان نزیک بوومه‌وه‌.لوته‌ نه‌رمو شله‌که‌یان ئه‌ خشان له‌ به‌ری ده‌ستم.حه‌زیان له‌ یاری بوو،،،،گرتمن ویستم یاریان له‌گه‌ڵ بکه‌م،هه‌ندێک ئه‌م ده‌ست وئه‌وده‌ستم پێکردن.
ئه‌و خه‌به‌ری بۆوه‌،توڕه‌بوو،،،
وه‌ی چه‌ن کورێکی جوانه‌،،،سه‌وزه‌!!
!،،،ئه‌مه‌ یه‌که‌مین جارمه‌ کوڕی وا جوان ببینم،،،ناوت چییه‌؟ که‌روێشکه‌کان وه‌ڵامیان دایه‌وه‌ ‌سه‌وزه‌،،ناوی سه‌وزه‌یه‌،،،
که‌وتینه‌ یاری کردن،،،زۆر یاریمان کرد،،،من هیلاک بووم،،،هه‌ناسه‌ برکێم بوو،،،ئه‌وانیش ئه‌تریقانه‌وه‌،،،من له‌سه‌ر پشت لێی که‌وتم،،،نه‌متوانی زیاتر یاری بکه‌م،،،بانگیان کردم،،،،خه‌وم لێکه‌وتبوو،،،،باڵنده‌کانی کۆڵان گۆرانیان ئه‌وت که‌ به‌خه‌به‌ر هاتم،،، ئه‌وه‌ نه‌موت ئه‌مشه‌و له‌ودیو بخه‌وه‌ زۆر هیلاکم،،،بۆ هاتویته‌وه‌ پاڵم؟،،،،،،ئه‌مزانی خۆت ناگریت هه‌ی ته‌ڕپیر.
تابڵێی ته‌ڕوپاراو نه‌رم وناسك ختووکه‌ده‌ر تاڕاده‌ی ئه‌و قاقا گه‌رمانه‌ی نـیوانتان، ئارام به‌خش له‌شیوه‌ی ئه‌وساتانه‌ی ته‌نیام وژانی شیعر ئه‌م‌گرێ.قه‌ڵه‌م یش له‌وه‌ زیاتر ڕێ نادات نه‌وه‌ك بجـێته‌ خانه‌ی پیاهه‌ڵدان، یانیش بچێته‌ خانه‌ی ڕه‌خنه‌وه‌ چونکه‌ له‌وه‌زیاتر هه‌ڵئه‌گرێت وڕه‌نگه‌ له‌م فرسه‌ته‌ ووردیله‌یه‌دا جێ ی نه‌بێته‌وه‌.
