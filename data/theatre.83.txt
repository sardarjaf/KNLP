له‌م ماوه‌یه‌دا له شاره‌کانی ‌هه‌ولێر و سلێمانی و که‌رکوك چه‌ند وه‌رزێکی شانۆییان پێشکه‌ش کران  له لایه‌ن به‌رێوه‌به‌رایه‌تی هونه‌ری شانۆی ‌سلێمانی وه‌رزه‌ شانۆییه‌که‌ له‌ژێر ناوی(ره‌شابای سلێمانی) سازکرا، له‌ که‌رکوكیش به‌ناوی(فیستیفاڵی شانۆی که‌رکوك) له‌لایه‌ن فه‌رمانگه‌ی رۆشنبیری که‌رکوك به‌رێوه‌‌چوو،له‌ هه‌ولێریش له‌ ژێر ناوی( وه‌رزی هه‌ولێری شانۆیی) که‌ تاوه‌کوئێستا له‌لایه‌ن به‌رێوه‌به‌رایه‌تی هونه‌ری شانۆ به‌رده‌وامی هه‌یه‌  به‌رێوه‌ ده‌چێ... ئه‌وه‌ی جێگای دڵخۆشییه‌ نمایشه‌کان تایبه‌ت نه‌بوو ته‌نها به‌ تیپه‌ شانۆییه‌کانی ناو شاره‌کان به‌ڵکو چه‌ندین تیپ له‌ ده‌ره‌وه‌ی شار(قه‌زایه‌کان) و ده‌ره‌وه‌ی هه‌رێمیش به‌شداری ئه‌م  وه‌رزه‌شانۆیه‌یان کرد... به‌ڵام ئایا ئه‌م هه‌موو چالاکییه‌ شانۆیانه‌ توانی شانۆی کوردی له‌‌م سستییه‌ رزگار بکات؟ یان له‌م قه‌یرانه‌ شانۆییه‌ ده‌رباز بکات؟
سه‌ره‌تا ده‌ڵێم: پاڵپشتی دارایی زۆر پێویسته‌ بۆ هه‌ر کارێك بێت ئه‌گه‌ر به‌ ویژدانیش قسه ‌بکه‌ین ئه‌وه‌ی ئاگادارم له‌م لایانه‌ به‌رێوبه‌رایه‌تی شانۆ له‌رێگه‌ی وه‌زاره‌تی رۆشنبیری و لاوان به‌پێی درخسته‌کان بۆ هه‌ر نمایشێك تاراده‌یه‌کی باش هاوکاری ده‌رهێنه‌ره‌کانی کردوه‌.. ئامانج و په‌یام له‌هه‌موو کارێك زۆر پێویسته‌ به‌تایبه‌ته‌تی له‌ کاری هونه‌ری بۆیه‌ ده‌بێ هه‌ر فیستیفاڵێك که‌ساز ده‌کرێت ده‌بێ ئامانج و په‌یامی دیارکراوی  هه‌بێت.. بۆ نموونه‌ ئایا فستیفاڵه‌که‌ چ جۆرێکه‌: منداڵان، گه‌نجان، ئه‌زمونگه‌ری ، به‌زمه‌ساته ‌، کلاسییکه‌...  ‌له‌م وه‌رزه‌ی هه‌ولێر شانۆیی نمایشه‌کان گشتگیری بوو  له‌به‌ر ئه‌وه‌ی له‌هه‌ولێر زۆربه‌ی نمایشه‌کانی( وه‌رزی هه‌ولێری شانۆیی) بینیوه‌ بۆیه‌ ده‌توانم به‌گشتی هه‌ندێ ئاماژه‌ به‌ نمایشه‌کان بده‌م.
سه‌ره‌تایی وه‌رزه‌که‌ به‌ شانۆگه‌ری"هه‌موومان چاوڕوانین" له‌ ده‌رهێنانی "عوسمان ئیحسان"  ده‌ستی پێکرد که‌ ‌چه‌ندین ئه‌کته‌ری عێراقی و کوردستانی هه‌ریه‌که‌و به‌ زمانی خۆی رۆڵی خۆی ده‌گێرا(کوردی، عه‌ره‌بی، سریانی، تورکمانی) ئه‌گه‌رچی به‌چه‌ندین زمانی جیاواز شانۆگه‌رییه‌که‌ نمایش کرا، به‌ڵام  زۆر سه‌رکه‌وتوانانه‌ چه‌مکی شانۆگه‌ریه‌که‌یان له‌رێگه‌ی ئه‌کته‌ره‌کان گه‌یانده‌ بینه‌ران به‌تایبه‌تی هه‌ردوو ئه‌کته‌ری عێراقی (د.هه‌یسم و د.
عه‌واتف نه‌عیم) که‌به‌راستی به‌ نواندنه‌که‌یان سه‌رسامیان کردین ئه‌م شانۆگه‌رییه‌ سه‌ره‌تاییه‌کی زۆر به‌هێزی وه‌رزه‌که‌ بوو دوایی چه‌ندین شانۆگه‌رییه‌ك پێشکه‌ش کران که‌ به‌بۆچوونی خۆم هه‌ندێکیان ته‌نها  بۆ نوخبه‌(هه‌ڵبژراده‌) ‌بوو وه‌کو (گرێیه‌کانی شکسپیر، شالیر)  هه‌رچه‌نده‌ ‌له‌و‌ بڕواییه‌دام که‌ئه‌گه‌ر شکسپیرله‌ ژیان بماباییه‌ هه‌رگیز ئه‌وه‌ی قبوڵ نه‌ده‌کرد چه‌ند ده‌قێکی خۆی له‌ چوار چێوه‌ی  شانۆگه‌رییه‌ك نمایش بکرێت یان ده‌قێکی وه‌کو (شالیر) به‌و شێوه‌یه‌ که‌سایه‌تیه‌کانی چڕ بکرێنه‌وه‌ چونکه‌ هه‌رده‌قه‌و خاوه‌ن فه‌لسه‌فه‌و چه‌مکێکی تایبه‌تی خۆیه‌تی ئاوا به‌ئاسانی بینه‌ر لێیان حاڵی نابێت ئه‌گه‌ر شانۆگه‌رییه‌که‌ی نه‌خوێندبێته‌وه‌ هه‌رچه‌نده‌ له‌ رووی ده‌رهێنان و نواندن و فۆرمو سینۆگرافیا زۆر جوان و سه‌ر‌کوتوو.
هه‌ر له‌ وه‌رزه‌شانۆییه‌دا هه‌ندێ نمایش هه‌بوو پێکهاته‌ی درامی ته‌واو نه‌بوو یاخود ده‌قه‌که‌ زۆر لاوازبوو، به‌راده‌ییه‌ك کاریگه‌ری سلبی هه‌بوو له‌ ئاماده‌بوونی بینه‌ر له‌هۆڵه‌کانی شانۆدا به‌راده‌یه‌ك نمایش هه‌بوو بینه‌ره‌که‌ی له‌ هه‌رسێ رۆژی نمایش نه‌ده‌گیشته‌(100) بینه‌ر!!
چونکه نمایش هه‌بوو چ وه‌کو ده‌ق،ده‌رهێنان، نواندن،دیکۆر، روناکی... شایانی نمایشکردن نه‌بوون  ده‌بوایه‌ پێش نمایش کردنی هه‌ر شانۆگه‌رییه‌ك لیژنه‌یه‌کی پسپۆر نمایشه‌که‌ ببینی ئه‌و‌ بڕیار له‌سه‌ر شانۆگه‌رییه‌که‌ بدات هه‌رچه‌ند لیژنه‌کی ده‌ق هه‌یه‌ به‌ڵام به‌بۆچونی خۆم ئه‌م لیژنه‌یه‌ وه‌کو پێویست کاری خۆی نه‌کردوه‌.
له‌گه‌ڵ رێزم بۆ هه‌موو نمایشه‌کان وه‌ك پێوست نه‌یانتوانی بینه‌ر بگه‌رێنێته‌وه‌ ناو هۆڵه‌کان فاکته‌ره‌کانیش زۆرن وه‌کو: 1-  بنیه‌ر خۆی له‌ ناونمایشه‌کان نه‌ده‌بینیه‌وه‌ به‌جۆرێك گوزارشت له‌ ناهه‌مووره‌کانی نه‌ده‌کرد.2- هه‌ندێك له‌ ده‌ر‌هێنه‌ره‌کان ستایلی کارکردنیان دیار نه‌بوو واته ‌نه‌بوونی پسپۆری له‌ شێوازی کارکردن بۆ نموونه:‌ ده‌رهێنه‌ر هه‌یه‌ له‌ هه‌موو شێوازه‌کان وه‌کو کلاسیکی ، ئه‌زموونگه‌ری ، واقعی ، منداڵان ، به‌زمه‌سات... کار ده‌کات.
3- به‌رزی ونزمی ئاستی نمایشه‌کان.
4- نه‌بوونی کۆڕو سمینار بۆ نمایشه‌کان .
ئه‌و شانۆگه‌ریانه‌ی نمایش کراون بریتین له‌: ( هه‌موومان  چاوه‌روانین له‌ ده‌رهێنانی "ئیحسان عوسمان"، گرێیه‌کانی شکسپیر له‌ ده‌رهێنانی "دیار عومه‌ر" ، ماندووکراوه‌ له‌ ده‌رهێنانی " ماجد بکر"، شالیر له‌ ده‌ر‌هێنانی "مه‌هدی حه‌سه‌ن"، عه‌با له‌ ده‌رهێنانی "کرمانج مسته‌فا"، ده‌رچوانی گه‌نج له‌ ده‌رهێنانی "سلێمان گه‌ڵالی" ، 2010 له‌ ده‌رهێنانی" له‌یس بنیامین" ، شه‌ل و کوێر له‌ده‌رهێنانی "موئه‌ید فه‌قێ" ، دۆڕاوه‌کان له‌ ده‌رهێنانی"رۆژگار که‌ریم" ، وێستگه‌ی چاوه‌روانی له‌ ده‌رهێنانی" دلڤین موحسین" ، به‌ده‌رخستی شێتێکی پشت گوێ خراو له‌ ده‌رهێنانی "فه‌ره‌یدون محه‌مه‌د" ، دیمه‌نێك له‌ ژیان له‌ ده‌رهێنانی "ئازاد دڵپاك" ، گۆرانی چایکا له‌ ده‌رهێنانی "فریاد که‌رکوکی" ، دادگا له‌ ده‌رهێنانی "له‌تیف نوعمان"بریار وایه‌ ئه‌م شانۆگه‌ریانه‌ش نمایش بکرێت: له‌ چاو تروکانێکدا له‌ ده‌رهێنانی "محه‌مه‌د جه‌میل" ، عه‌بدی لات له‌ ده‌ر‌هێنانی"سه‌باح عه‌بدورحمان" ، فرمێسکی پڕنزاکان له‌ ده‌رهێنانی "یوسف عوسمان" نیوه‌ی مانگه‌ شه‌و له‌ ده‌رهێنانی "نه‌شئه‌ت به‌شیر".
ئه‌گه‌رچی ئه‌م وه‌رزه‌ شانۆیانه له‌هه‌ر سێ شاره‌که‌ ‌ نه‌یانتوانیوه‌ بینه‌ر وه‌کو جاران بگه‌رێنێته‌وه‌ ناو هۆڵه‌کانی شانۆ  پێویسته‌ هونه‌رمه‌ندانی شانۆ زۆر به‌ وورده‌ی به‌کاره‌کانیان دابچنه‌وه‌، بۆ ئه‌وه‌ی بینه‌ری شانۆ ئاشت بکه‌نه‌وه‌.
‌ له‌گه‌ڵ ئه‌مه‌شدا ئێمه‌ی شانۆکاران له‌هه‌ولێر له‌ 15/9/2010 هه‌فته‌ی هه‌موو سێشه‌مه‌یه‌ك ژوانمان هه‌بوو له‌گه‌ڵ نمایشه‌کاندا خۆزگه‌ ئه‌مه‌ ده‌بووه‌ نه‌ریتێك هه‌رچه‌نده‌ هه‌ندێ له‌و ژوانانه‌ ناخۆش بوو له‌کۆتاییدا ده‌ست خۆشی له‌ هه‌موولایه‌ك، به‌هیوای کاری باشتر.
