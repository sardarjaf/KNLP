دوای ئه‌وه‌ی بزوتنه‌وه‌ی تاڵیبان كوژرانی مه‌لا ئه‌خته‌ر مه‌نسور-ی رابه‌ری بزوتنه‌وه‌كه‌ی پشتڕاستكرده‌وه‌، له‌ئێستادا پرسیار له‌چاره‌نووس‌و داهاتووی ئه‌و بزوتنه‌وه‌یه‌ هاتۆته‌ ئاراوه‌، به‌تایبه‌ت كه‌بزوتنه‌وه‌ی تاڵیبان له‌لایه‌ك سه‌رقاڵی رێكخستنه‌وه‌ی ریزه‌كانی‌و له‌لایه‌كی تر گفتوگۆكردن بوو له‌گه‌ڵ حكومه‌تی ئه‌فغانستان.
سه‌ره‌تای ئه‌م هه‌فته‌یه‌ به‌رپرسێكی ئه‌مریكی رایگه‌یاند كه‌ فڕۆكه‌یه‌كی‌ بێفڕۆكه‌وان له‌سنووری‌ نێوان ئه‌فغانستان‌و پاكستان مه‌لا ئه‌خته‌ری كردۆته‌ ئامانج، دواتریش تاڵیبان كوژرانی رابه‌ره‌كه‌ی‌ پشتڕاستكرده‌وه‌.
مه‌لا ئه‌خته‌ر مه‌نسور كه‌ له‌خێڵی (ئیشاكزی)ی سه‌ر به‌هۆزی دۆرانیی شوێنكه‌وتووی پشتۆنه‌كانه‌، له‌نێوان ساڵانی 1963-1965 له‌گوندێكی سه‌ر به‌ناوچه‌ی مایواند له‌ویلایه‌تی قه‌نده‌هار كه‌ده‌كه‌وێته‌ باشوری ئه‌فغانستان‌و هاوسنووره‌ له‌گه‌ڵ هه‌رێمی بلوجستان له‌باشوری خۆرئاوای پاكستان له‌دایكبووه‌.
مه‌لا ئه‌خته‌ر كه‌یه‌كێك بووه‌ له‌دامه‌زرێنه‌رانی بزوتنه‌وه‌ی تاڵیبان له‌قه‌نده‌هار له‌ساڵی 1994دا، له‌نێوان ساڵانی 1996 تاڕووخانی رژێمی تاڵیبان له‌ئه‌فغانستان له‌ساڵی 2001دا، چه‌ندین به‌رپرسیارێتی وه‌رگرتووه‌ وه‌ك وه‌زیری فڕۆكه‌وانی مه‌ده‌نی‌و گواستنه‌وه‌ – حاكمی قه‌نده‌هار تاساڵی 2007 – جێگری مه‌لا محه‌مه‌د عومه‌ر - رابه‌ری پێشووتری بزوتنه‌وه‌ی تاڵیبان-  له‌نێوان ساڵانی 2010- 2015.
رۆژی 29ی ته‌مووزی 2015 ئه‌نجومه‌نی شورای تاڵیبان به‌تێكڕای ده‌نگ، مه‌لا ئه‌خته‌ر مه‌نسور- یان وه‌ك جێگره‌وه‌ی مه‌لا محه‌مه‌د عومه‌ر ده‌ستنیشانكرد، دوای ئه‌وه‌ی به‌فه‌رمی مردنی مه‌لا عومه‌ر راگه‌یه‌ندرا، له‌كاتێكدا بۆ ماوه‌ی دووساڵ مردنی مه‌لا عومه‌ر شاردرایه‌وه‌.
فه‌یز محه‌مه‌د زلاند، مامۆستای زانكۆی ئه‌فغانستان به‌كه‌ناڵی جه‌زیره‌ی راگه‌یاندووه‌ كه‌ كوژرانی مه‌نسور مه‌ترسییه‌ له‌سه‌ر یه‌كڕیزی بزوتنه‌وه‌ی تاڵیبان‌و كاریگه‌ری نه‌رێنی ده‌بێت له‌سه‌ر گفتوگۆكانی ئاشتی نێوان حكومه‌تی كابول‌و ئه‌و بزوتنه‌وه‌یه‌، له‌كاتێكدا هه‌ردوو وه‌زاره‌تی ده‌ره‌وه‌و به‌رگری ئه‌مریكا جه‌ختیان له‌وه‌كرد كه‌ مه‌لا ئه‌خته‌ر گرفت‌و ئاسته‌نگ بووه‌ له‌به‌رده‌م پرۆسه‌ی ئاشتی‌و ئاشته‌وایی له‌نێوان تاڵیبان‌و كابول.
مانگی ته‌موزی ساڵی 2015 به‌فه‌رمی گفتوگۆكانی ئاشتی له‌نێوان كابول‌و نوێنه‌رانی بزوتنه‌وه‌ی تاڵیبان له‌پاكستان  به‌چاودێری ئه‌مریكاو چین ده‌ستیانپێكرد به‌ئامانجی كۆتاییهێنان به‌و جه‌نگه‌ی كه‌ماوه‌ی 15 ساڵه‌ له‌و وڵاته‌ به‌رده‌وامه‌.
به‌ڵام دوای بڵاوبوونه‌وه‌ی هه‌واڵی مردنی مه‌لا عومه‌ر – رابه‌ری پێشووتری تاڵیبان، گفتوگۆكانی ئاشتی له‌نێون ئه‌و بزوتنه‌وه‌یه‌و كابول-دا هه‌ڵپه‌سێردران.
كانونی دووه‌می ئه‌مساڵیش هه‌ریه‌كه‌ له‌ئه‌فغانستان – پاكستان – چین‌و ئه‌مریكا گروپێكی نێوده‌وڵه‌تیان پێكهێنا به‌ئامانجی ده‌ستپێكردنه‌وه‌ی گفتوگۆكانی ئاشتی له‌گه‌ڵ تاڵیباندا.
مانگی شوباتیش گروپه‌كه‌ یه‌كه‌م كۆبونه‌وه‌ی خۆی له‌پاكستان ئه‌نجامدا به‌ئامانجی دانانی نه‌خشه‌رێگایه‌ك بۆ گفتوگۆكانی ئاشتی، بڕیاریش بوو له‌كۆتایی هه‌مان مانگدا گفتوگۆكان ده‌ستپێ بكه‌نه‌وه‌و هه‌ردوولا بگه‌نه‌ رێكه‌وتنێك كه‌خۆی له‌ 25  خاڵدا ده‌بینییه‌وه‌، به‌ڵام به‌هۆی چڕكردنه‌وه‌ی په‌لاماره‌كان له‌لایه‌ن تاڵیبان بۆ سه‌ر دامه‌زراوه‌كانی حكومه‌تی كابول، ئه‌و هه‌وڵه‌ به‌ئه‌نجام نه‌گه‌یشت.
رۆژی 26ی نیسانی ئه‌مساڵیش شاندێكی بزوتنه‌وه‌ی تاڵیبان گه‌یشتنه‌ ئیسلام ئاباد-ی پایته‌ختی پاكستان بۆ گفتوگۆكردن له‌سه‌ر چۆنییه‌تی ده‌ستپێكردنه‌وه‌ی دانوستانه‌كان، هاوكات رۆژی 18ی ئایار جارێكی تر گروپی چوار قۆڵی له‌پاكستان كۆبونه‌وه‌ بۆ تاوتوێكردنی چۆنییه‌تی ده‌ستپێكردنه‌وه‌ی گفتوگۆكان له‌گه‌ڵ تاڵیباندا.
به‌ر له‌ده‌ستپێكردنه‌وه‌ی دانوستانه‌كانیش بزوتنه‌وه‌ی تاڵیبان چه‌ند مه‌رجێكی بۆ به‌شداریكردن خسته‌ڕوو وه‌ك سڕینه‌وه‌ی ناوی بزوتنه‌وه‌كه‌ له‌لیستی ره‌شی تایبه‌ت به‌ڕێكخراوه‌ تیرۆرستییه‌كانی رێكخراوی نه‌ته‌وه‌یه‌كگرتووه‌كان، له‌گه‌ڵ داننان به‌مه‌كته‌بی سیاسی بزوتنه‌وه‌كه‌‌و ئازادكردنی زیندانییه‌كانیان.
ئه‌شره‌ف غه‌نی، سه‌رۆكی ئه‌فغانستان له‌میانی كۆبونه‌وه‌ی به‌ئه‌ندامانی په‌رله‌مان – ئه‌نجومه‌نی باڵای ئاشتی ئه‌فغانی‌و ئه‌نجومه‌نی موجاهدین، ئاماژه‌ی به‌وه‌كردبوو كابول "تاڕاده‌یه‌ك" گه‌شبینه‌ به‌و هه‌وڵانه‌ی ده‌درێن له‌باره‌ی پرۆسه‌ی ئاشته‌وایی له‌ئه‌فغانستان، له‌هه‌مانكاتدا جه‌ختی له‌وه‌كرد له‌میانی كۆبونه‌وه‌كانیان له‌گه‌ڵ پاكستان – چین‌و ئه‌مریكا ره‌زامه‌ندی نیشاندراوه‌ له‌سه‌ر مه‌رجه‌كانی كابول بۆ به‌شداریكردن له‌و گفتوگۆیانه‌دا.
دوای كوژرانی مه‌لا ئه‌خته‌ر مه‌نسور رابه‌ری بزوتنه‌وه‌ی تاڵیبان، له‌ئێستادا بزوتنه‌وه‌كه‌ داوای كۆبونه‌وه‌ی ئه‌نجومه‌نی شورای خۆی كردووه‌ بۆ هه‌ڵبژاردنی رابه‌رێكی نوێ كه‌جێگه‌ی مه‌لا ئه‌خته‌ر بگرێته‌وه‌.
بزوتنه‌وه‌كه‌ جه‌ختی له‌وه‌كردووه‌ كه‌ له‌ئێستادا كه‌سانێكی زۆر پاڵێوراون بۆ جێگرتنه‌وه‌ی مه‌لا ئه‌خته‌ر مه‌نسورو ده‌ستنیشانكردنیان وه‌ك رابه‌ری تاڵیبان وه‌ك كه‌ناڵی (دنیا نیوز)ی پاكستانی بڵاویكردۆته‌وه‌.
خالد په‌شتۆن، ئه‌ندامی پێشووی په‌رله‌مانی ئه‌فغانستان رایگه‌یاند كه‌كوژرانی مه‌نسور پاڵنه‌رو هانده‌رێكی باش ده‌بێت بۆ حكومه‌تی كابول‌و هێزه‌كانی هاوپه‌یمانان بۆ تێكشكاندنی بزوتنه‌وه‌ی تاڵیبان، له‌كاتێكدا پێشبینی ده‌كرێت ئه‌و بزوتنه‌وه‌یه‌ ناكۆكی توند به‌خۆوه‌ ببینێت به‌هۆی ده‌ستنیشانكردنی رابه‌ره‌ نوێیه‌كه‌یه‌وه‌ ئه‌وه‌ش بێگومان ده‌بێته‌هۆی لاوازی توانا سه‌ربازی‌و مه‌یدانییه‌كانی.
لێره‌دا پرسیار ئه‌وه‌یه‌ كێ ده‌بێته‌ جێگره‌وه‌ی مه‌لا ئه‌خته‌ر مه‌نسور؟ بۆ وه‌ڵامی ئه‌و پرسیاره‌ زۆربه‌ی بۆچوونه‌كان له‌سه‌ر ئه‌و ئاڕاسته‌یه‌ن كه‌ سیراجه‌دین حه‌قانی زۆرترین چانسی هه‌یه‌ بۆ رابه‌رایه‌تی كردنی تاڵیبان.
وه‌ك زۆربه‌ی سه‌ركرده‌كانی تاڵیبان زانیاری ورد له‌سه‌ر سیراجه‌دین حه‌قانی تائێستا له‌به‌رده‌ستدا نییه‌، به‌پێی ئه‌و زانیاریانه‌ش كه‌ بڵاوكراونه‌ته‌وه‌ سیراجه‌دین له‌نێوان ساڵانی 1973- 1978 له‌گوندی په‌شتۆنی سه‌ر به‌ویلایه‌تی به‌كتیكا له‌دایكبووه‌.
ساڵی 2008 سیراجه‌دین به‌رپرسیارێتی (بزوتنه‌وه‌ی حه‌قانی) له‌جیاتی جه‌لاله‌دین سیراجی باوكی وه‌رگرت كه‌ له‌هه‌شتاكانی سه‌ده‌ی رابردوودا ئه‌و بزوتنه‌وه‌یه‌ی‌ دامه‌زراندو رابه‌رایه‌تی ئه‌و موجاهیدانه‌ی ده‌كرد كه‌ڕووبه‌ڕووی سوپای سوڤێتی جاران بوونه‌وه‌ دوای ئه‌وه‌ی له‌ساڵی 1979دا ئه‌فغانستانیان داگیركرد.
له‌لایه‌ن ئیداره‌ی ئه‌وكاتی ئه‌مریكاش مامه‌ڵه‌یه‌كی تایبه‌ت له‌گه‌ڵ جه‌لاله‌دین ده‌كراو له‌سه‌رده‌می ده‌سه‌ڵاتی رۆناڵد ریگن- دا سه‌ردانی كۆشكی سپی كرد، به‌ڵام له‌ئێستادا سیراجه‌دین- ی كوڕی به‌كه‌سێكی توند له‌لایه‌ن ئه‌و ئیداره‌یه‌وه‌ ناسراوه‌.
له‌مانگی شوباتی ساڵی 2010دا هێزی ئاسمانی ئه‌مریكا دووجار هه‌وڵی كوشتنی سیراجه‌دین حه‌قانی دا، به‌ڵام له‌هه‌ردوو جاره‌كه‌دا رزگاری بوو، له‌ئاكامی یه‌كێك له‌و هێرشانه‌ش محه‌مه‌د حه‌قانی برا بچووكی سیراجه‌دین كوژرا.
به‌رپرسانی ئه‌فغانستانیش سیراجه‌دین حه‌قانی كه‌ یه‌كێكه‌ له‌جێگرانی مه‌لا مه‌نسورو ئه‌ندامی ئه‌نجومه‌نی شورای ئه‌ماره‌تی ئیسلامی بزوتنه‌وه‌ی تاڵیبان به‌مه‌ترسیدارترین ئه‌میری جه‌نگی تاڵیبان ده‌ناسێنن‌و به‌رپرسیارێتی گه‌لێك له‌و په‌لاماره‌ خوێناویانه‌ی ده‌خه‌نه‌ ئه‌ستۆ كه‌ له‌ئه‌فغانستان روویانداوه‌ له‌نێویشیاندا ئه‌و په‌لاماره‌ی مانگی رابردوو له‌كابول-ی پایته‌خت روویداو بووه‌هۆی كوژرانی 64 كه‌س.
هاوكات وه‌زاره‌تی ده‌ره‌وه‌ی ئه‌مریكا (تۆڕی حه‌قانی) كه‌ له‌لایه‌ن سیراجه‌دین –ه‌وه‌ سه‌رپه‌رشتی ده‌كرێت‌و هاوپه‌یمانی بزوتنه‌وه‌ی تاڵیبانه‌ - كه‌ ده‌وترێت له‌ئێستادا خاوه‌نی 10 هه‌زار چه‌كداره‌-، به‌توندترین گروپی توندڕه‌و ئه‌ژمار ده‌كات كه‌هێزه‌كانی هاوپه‌یمانان‌و سوپاو هێزه‌كانی ئه‌فغانستانیان كردۆته‌ ئامانج، هه‌ر ئه‌وه‌ش وایكردووه‌ بڕی پێنج ملیۆن دۆلار ته‌رخان بكات بۆ هه‌ر كه‌سێك كه‌ زانیاری بدات له‌سه‌ر شوێنی سیراجه‌دین كه‌ ببێته‌هۆی ده‌ستگیركردنی.
له‌ساڵیادی هێرشه‌كانی 11ی سێبتێمبه‌رو له‌ساڵی 2011دا سیراجه‌دین گه‌وره‌ترین په‌لاماری ئه‌نجامدا بۆ سه‌ر بنكه‌یه‌كی هێزه‌كانی ئه‌مه‌ریكا له‌ویلایه‌تی وه‌رداك له‌ئه‌فغانستان‌و به‌هۆیه‌وه‌ 77 سه‌ربازی ئه‌مریكی كوژران.
له‌ 13ی ئه‌یلولی 2007 ئه‌نجومه‌نی ئاسایشی نێوده‌وڵه‌تی سیراجه‌دین حه‌قانی وه‌ك كه‌سێكی په‌یوه‌ندیدار به‌ڕێكخراوی قاعیده‌و بزوتنه‌وه‌ی تاڵیبان خسته‌ لیستی ناوی  تیرۆریستانی داواكراو له‌سه‌ر ئاستی جیهان.
