من حه‌مه‌ غه‌ریبم، سه‌باره‌ت به‌و كێشه‌ی كه‌ دوێنێ زۆربه‌تان له‌ڕێگه‌ی فه‌یسبوكه‌وه‌ ئاگاتان لێ بوو، گوایه‌ كابرایه‌كی ته‌مه‌ن حه‌فتا و دوو ساڵه‌ له‌گه‌ڵ برا ته‌مه‌ن شه‌ست و حه‌فت ساڵه‌كه‌ی خۆیا له‌ ناوه‌ڕاستی بازاڕی شار، ڕێك له‌ چایخانه‌كه‌ی لای وه‌ستا مه‌حمووی كه‌بابچی، بووه‌ته‌ ده‌مه‌قاڵه‌یان و ده‌ستیشیان له‌یه‌كتـر به‌رز كردووه‌ته‌وه‌، ئه‌مه‌وێ هه‌ندێ ڕاستیتان پێ بڵێم، كه‌ تائێستا نه‌زانراو و نه‌بیستراون، وا بۆ یه‌كه‌مین جاره‌ دێـم ئاوا بێ ترس و بێ سڵه‌مینه‌وه‌ باسی ده‌كه‌م، یان ڕاستتر بڵێم ده‌یاننووسمه‌وه‌، چونكه‌ دوای ئه‌مه‌ ئه‌مه‌وێ بچمه‌ دادگه‌و سكاڵا له‌و چیرۆكنووسه‌ حیزه‌ بكه‌م، كه‌ ئه‌م مه‌سه‌له‌ی به‌ناوی چیرۆكێكی واقیعییه‌وه‌ ئاوا ورووژاند، خۆ من ده‌ستم ناگاته‌ ئه‌و خوێڕییه‌ی ساحێب ئه‌م فه‌یسبوكه‌ (مارك زوكه‌ربێرگ - mark zuckerberg)، كه‌ به‌و داهێنانه‌ خراپه‌ی خۆی ماڵی خه‌ڵكی وێران كرد، شه‌ره‌ف و نامووسی دنیای برد.
هه‌رچه‌ن من تاوانی یه‌كه‌م ده‌خه‌مه‌ ئه‌ستۆی براكه‌ی خۆم، به‌ڵام نه‌ده‌بوو ئه‌و چیرۆكنووسه‌ بچێ مامه‌ڵه‌ به‌ كێشه‌یه‌كی كۆمه‌ڵایه‌تی و خێزانیی نێوان دوو كه‌سه‌وه‌ بكات و به‌ناوی گوایه‌ ئه‌ده‌بیات و مه‌ده‌بیاته‌وه‌ بڵاوی بكاته‌وه‌، له‌ كوێیش: له‌ فه‌یسبوك... ئاخر فه‌یسبوك كه‌ی جێی چیرۆكی تیا ئه‌بێته‌ بۆ خاتر قورعان...!
ئه‌و شه‌ڕه‌ی دوێنێی من و (حه‌مه‌كه‌ریم)ی برام، ده‌بوو چل ساڵ له‌مه‌وبه‌ر ڕووی دابا، ئه‌وكاته‌ی كه‌ چوو دڵبه‌ره‌كه‌ی منی خواست.
به‌ڵێ با ئه‌وه‌ش بزانن، قه‌یدی چییه‌، چونكه‌ ئێستا براژنه‌كه‌م یان ڕاستتـر بڵێم دڵبه‌ره‌كه‌م، له‌ژیاندا نه‌ماوه‌ (خوا عه‌فووی كا).
به‌ڵـێ دڵبه‌ره‌كه‌م، نا... ده‌ى تازه‌ شه‌رعیتـره‌ گه‌ر بڵێم براژنه‌كه‌ی من، كه‌ له‌ هه‌مان كاتدا كچی مامیشمان بوو، پێشتـر منی خۆش ده‌ویست نه‌ك حه‌مه‌كه‌ریمی برام، ئه‌ویش ئه‌مه‌ی ده‌زانی و ئه‌سڵه‌ن ڕۆژێك له‌ ڕۆژان چییه‌، هیچ هه‌ستێكی سۆزداریی یان خۆشه‌ویستیی به‌رانبه‌ر ئه‌و نه‌بوو، كه‌چی دوای ئه‌وه‌ی به‌ زۆر ژنیان بۆ هێنام، بێ ئه‌وه‌ی بوێرم ده‌نگ هه‌ڵبڕم و باسی خۆم و دڵبه‌ره‌كه‌م بكه‌م، پاش چه‌ندین ساڵ له‌و ڕووداوانه‌ى كه‌ نامه‌وێ سه‌رتانی پێوه‌ بیه‌شێنم، هه‌موو شتێك كۆتایی هات و حه‌مه‌كه‌ریم چووه‌ خوازبێنی دڵبه‌رى من بۆ خۆی، كه‌ باش باش ده‌یزانی من هێشتا به‌ ته‌مای ئه‌وم و چاوه‌ڕێی مردنی مامم ده‌كرد، یان هیچ نه‌بێ نه‌مده‌ویست دڵبه‌ری ژیانـم بۆ هیچ كه‌سی تر بێ، چونكه‌ دڵنیام ئه‌ویش لای خۆیه‌وه‌ ئه‌وه‌ى باش ده‌زانی، كه‌ من‌ نابه‌دڵ ژنیان بۆ هێناوم و ناچاریشم ئه‌و ژیانه‌ زۆره‌ملێیه‌ به‌رمه‌ سه‌ر له‌گه‌ڵ هاوسه‌ره‌كه‌م، كه‌ هیچ په‌یوه‌ندییه‌كی خۆشه‌ویستی تا ئه‌و ڕۆژه‌ی كۆچی دواییشی كرد له‌نێوانماندا نه‌بوو، بۆیه‌ دڵبه‌ری من لای خۆیه‌وه‌ بڕیاری دابوو، كه‌ بۆ گڵ بێ نه‌ بۆ كه‌سی تر، به‌ڵام تاوانی حه‌مه‌كه‌ریم ئا له‌وێدا بوو، كه‌ توانی نهێنی ئه‌و په‌یوه‌ندییه‌ی نێوان من و خازێ بقۆزێته‌وه‌و وه‌ك كارتی هه‌ڕه‌شه‌ به‌ كاری بێنێ و خازێی دڵبه‌رم ناچار بكات له‌به‌رده‌م ترس له‌ باوك و برادا چۆكی پێ دا بـدات، هه‌ر بۆیه‌ له‌پێناو خۆشه‌ویستیی من بوو، ئه‌ویش ئه‌و ژیانه‌ زۆره‌ملێ و ناچارانه‌یه‌ی بۆ خۆی هه‌ڵبژاردو شووی به‌ حه‌مه‌كه‌ریمی برام كرد، ئیتـر له‌و ڕۆژه‌وه‌ ئێمه‌ له‌ناو ئاگری سه‌ختی عه‌شقێكدا ده‌سووتاین، كه‌ به‌ هه‌موو پێوه‌ره‌كان مه‌ترسیدار ده‌بوو... من نه‌مده‌زانی پێی بڵێم دڵبه‌ری من یان براژنی من!
ده‌ركه‌وتن و سه‌رهه‌ڵدانه‌وه‌ی بچوكترین ئاماژه‌ی عه‌شق و خۆشه‌ویستی له‌نێوان من و خازێدا ده‌بووه‌ مایه‌ی شێواندن و ئاگربه‌ردان له‌ ژیانی چه‌نده‌ها كه‌سی بنه‌ماڵه‌كه‌مان، بۆیه‌ باشترین ڕێگه‌چاره‌م ئه‌وه‌ بوو، كه‌ هه‌تا ده‌توانم له‌م شاره‌ دوور بكه‌ومه‌وه‌و سه‌ری خۆم هه‌ڵگرم، هه‌ر وام كرد، به‌ڵام هه‌رگیز نه‌متوانی عه‌شقی خازێ له‌ دڵم ده‌ركه‌م، هه‌ربۆیه‌ ئه‌و ڕۆژه‌ی كه‌ دوای ئه‌وه‌نده‌ ساڵ ئه‌و چیرۆكه‌م به‌رچاوكه‌وت و خوێندمه‌وه‌، هه‌ستم كرد ده‌ستێك له‌پشته‌وه‌ی ئه‌م مه‌سه‌له‌یه‌وه‌ هه‌یه‌، كه‌ بێگومان ده‌ستی حه‌مه‌كه‌ریمى برامه‌، له‌گه‌ڵ ئه‌وه‌شدا من هه‌قی ئه‌وه‌م دایێ، چونكه‌ ئاگاى له‌ وه‌زعی منه‌ چه‌ن ده‌وڵه‌مه‌ندو پاره‌دارم، كوڕو كچه‌كانم هه‌موو چوونه‌ته‌ سه‌رماڵ و حاڵی خۆیان و... هتد، بۆیه‌ دووباره‌ى ده‌كه‌مه‌وه‌، كه‌ من حه‌ق به‌ حه‌مه‌كه‌ریمی برام ده‌ده‌م، چونكه‌ ئه‌و برامه‌، به‌ڵام كه‌سێكی ئاوا وه‌ك ئه‌و چیرۆكنووسه‌ بۆی نییه‌ بێت له‌ناو چیرۆكی خۆیدا ناوی من و براكه‌م و دڵبه‌ره‌كه‌مان یان ببورن براژنه‌كه‌م بزڕێنی، ئه‌وه‌ ته‌شهیره‌ ته‌شهیر، بۆیه‌ من به‌رله‌وه‌ى بچمه‌ به‌رده‌م جه‌نابی حاكم، پێم باش بوو بچم قسه‌یه‌ك له‌گه‌ڵ حه‌مه‌كه‌ریمی برای خۆم بكه‌م، كه‌چی تومه‌ز ئه‌ویش لای خۆیه‌وه‌ پێی وایه‌ ئه‌و ده‌سته‌ی پشته‌وه‌ی چیرۆكی چیرۆكنووسه‌ به‌ده‌كه‌ هی منه‌، ده‌ستی حه‌مه‌غه‌ریبی برای خۆی....
 هه‌ربۆیه‌ كاتێ چاوم به‌ حه‌مه‌غه‌ریبی برام كه‌وت، دڵم لێیدا، گوتی: حه‌مه‌كه‌ریم ئه‌وه‌تا ئه‌وه‌ حه‌مه‌غه‌ریبی براته‌، كه‌ دڵی به‌ خازێی خێزانی ڕه‌حمه‌تیته‌وه‌ بوو... به‌ڵێ، حه‌مه‌غه‌ریب برای گه‌وره‌م، له‌بری ئه‌وه‌ی ئاسانكارییه‌كی بكردایه‌و قسه‌یه‌كی خێری لای مامم بكردایه‌ بۆم، كه‌چی ده‌چوو خۆی له‌ خازێ نزیك ده‌كرده‌وه‌، ته‌نانه‌ت دوای ئه‌وه‌ی خازێ بۆ من بوو، ئه‌و هه‌ر وازی نه‌ده‌هێناو دایمه‌ له‌ملاو له‌ولا ده‌یگوت: خازێی دڵبه‌رم... ئیتر بیری له‌وه‌ نه‌ده‌كرده‌وه‌، كه‌ قه‌ینا كچه‌مامی هه‌ردووكمانه‌، به‌ڵام تازه‌ به‌پێی شه‌رع و یاسا، بووه‌ته‌ ژنی من، خێزانی شه‌رعی و دایكی هه‌شت منداڵم... بۆیه‌ ڕاستییه‌كه‌ی ئه‌و ڕۆژه‌ی كه‌ له‌ چاخانه‌ى خه‌ڵووزه‌كه‌ چاوم پێ كه‌وت، نه‌متوانی خۆم ڕاگرم و یه‌كسه‌ر لێی چوومه‌ پێشه‌وه‌:
+ حه‌مه‌كه‌ریمی برای پێی گوت: ئه‌وه‌ عه‌یب له ‌خۆت ناكه‌ی حه‌مه‌غه‌ریب، به‌و ڕیشه‌ سپییه‌وه‌ت ده‌چی نهێنی خانه‌واده‌كه‌مان لای ئه‌م و ئه‌و باس ئه‌كه‌ی؟ 
ئیتر حه‌مه‌غه‌ریب و حه‌مه‌كه‌ریم ده‌ستیان له‌ئێكدی گیـربوو، تا خه‌ڵك و خوای ئه‌و ناوه‌ ناوبژییان كردن و ئێستاو ئێستاش حه‌مه‌كه‌ریم هه‌ر واده‌زانێ تاوانی حه‌مه‌غه‌ریبه‌و حه‌مه‌غه‌ریبیش دڵنیایه‌، كه‌ ئه‌و گوناحێكی له‌م مه‌سه‌له‌یدا نییه‌، چونكه‌ حه‌مه‌غه‌ریب ده‌ڵێ: تاوانی سه‌ره‌كی بۆ ئه‌و چیرۆكنووسه‌ خوێڕییه‌ ده‌گه‌ڕێته‌وه‌، كه‌ له‌گه‌ڵ هه‌نێ براده‌ری خۆیا له‌ سووچی چاخانه‌كه‌وه‌ دانیشتبوون و ته‌ماشای ده‌مه‌قاڵێ و شه‌ڕی من و حه‌مه‌كه‌ریمی برامیان ده‌كرد، بۆیه‌ تكایه‌ هه‌ركێ ئه‌و چیرۆكنووسه‌ ده‌ناسێ با پێم بڵێ، به‌یانی ده‌چم شكایه‌تی لێ ده‌كه‌م، هه‌رچه‌نده‌ زۆر دڵم به‌ حاكم و مه‌حكه‌مه‌ى ئه‌م وڵاته‌ی خۆمان خۆش نییه‌، به‌ڵام هه‌ر پێم باش بوو، پێش ئه‌وه‌ی سبه‌ی بێت و بچمه‌ به‌رده‌م حاكم، خۆم بێم ئاوا حه‌قیقه‌تی هه‌موو شتێكتان بۆ بگێڕمه‌وه‌و حه‌مه‌كه‌ریمی برایشم كه‌یفی خۆیه‌تی، ئه‌گه‌ر ئه‌ویش وه‌ك من تۆزێ خوێنه‌واری هه‌بایه‌، هه‌ڵبه‌ت ئێستا ده‌یتوانی عه‌رز و حاڵی خۆی لای ئێوه‌ی خوێنه‌ری به‌ڕێز بكات و داوای كۆمه‌كیشتان لێ بكات.
هه‌ر لێره‌شه‌وه‌ داوای لێبوردن ده‌كه‌م، چونكه‌ من نه‌ چیرۆكنووسم نه‌ ده‌شمه‌وێ ببمه‌ چیرۆكنووس، به‌ڵكو هه‌ر ویستم ئێوه‌ پێش واده‌ی دادگاییكردنه‌كه‌ هه‌نێ شت له‌ حه‌قیقه‌تی شه‌ڕه‌كه‌ی دوێنێی من و حه‌مه‌كه‌ریمی برام بزانن و سبه‌ی له‌ دادگه‌ چاوه‌ڕێتانـم.
