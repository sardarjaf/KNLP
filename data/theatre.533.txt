له‌ ناوه‌ڕاستی حه‌فتاكاندا هێشتا هه‌رزه‌ بووین، له‌ سه‌نته‌ره‌كانی لاوانی كه‌ركووك خۆمان خه‌ریكی جینگه‌جینگی موزیك و ورته‌ورتی شانۆ ده‌كرد، له‌ بیرمه‌ ڕۆژێك خوالێخۆشبوو مامۆستا جه‌میل جێماو خۆی كرد به‌ژووردا و بۆ هه‌ندێك ئامۆژگاری هه‌موومانی له‌ ئامێزه‌ گه‌رمه‌كه‌ی كۆ كرده‌وه‌، ئه‌و كۆمه‌ڵێك شتی له‌باره‌ی شانۆوه‌ بۆ باس كردین كه‌ به‌لای ئێمه‌ی هه‌رزه‌كارانی ناو دنیای هونه‌ر نامۆ بوون، ئاخر جارێ له‌و بوارانه‌دا فه‌ریك بووین و هه‌شته‌بامان لێ نه‌ده‌زانی، به‌ڵام ئه‌وه‌نده‌مان لێ هه‌ڵكڕاند كه‌ ئه‌م مرۆیه‌ له‌ ئامۆژگارییه‌كانیدا مه‌به‌ستیه‌تی دواڕۆژێكی گه‌شه‌داری هونه‌ریمان پێشان بدات، ئێمه‌یش هه‌ر ده‌بوو به‌گوێی بكه‌ین، چونكه‌ بۆ ئه‌و قۆناغه‌ له‌ كه‌ركووك و له‌ هونه‌ردا هه‌ر ئه‌و زاتانه‌مان شك ده‌برد، واته‌ مامۆستا جێماو و هاوڕێكانی.. بۆیه‌ زێده‌بێژیم نه‌كردووه‌ ئه‌گه‌ر بڵێم وه‌ك خۆم تا ئێستایش ڕۆحی ئه‌و مرۆیه‌م به‌دیاره‌وه‌یه‌.
مه‌به‌ستم نییه‌ له‌سه‌ر جه‌میل جێماو بوه‌ستم، چونكه‌ له‌وه‌ به‌دوا لێی دابڕاین و تا ساڵانێك به‌ر له‌ كۆچی دوایی ئیتر له‌ ناوه‌نده‌كه‌دا نه‌ماندیته‌وه‌، بگره‌ ئه‌وه‌ی مه‌به‌ستم بوو له‌و نموونه‌ باڵایه‌، كاریگه‌ریی ئه‌وانی پێش خۆمان له‌سه‌ر نه‌وه‌كانی دواتر ڕه‌نگدانه‌وه‌یه‌كی ئاشكرای هه‌بووه‌.. له‌نێو ئه‌و هونه‌رمه‌ندانه‌ی ئه‌وكات له‌ كه‌ركووكدا ده‌كه‌وتنه‌ به‌رچاو و كاری باڵایان ده‌كرد (سه‌لمان فایه‌ق كاكه‌یی - جیهاد دڵپاك - جه‌لیل زه‌گنه‌ - عه‌لی كه‌ریم - عه‌بدولڕه‌حمان مه‌حموود - مه‌حموود گلیی جوعه‌ كه‌چه‌ڵ) و چه‌ندانی دیكه‌ تا ئه‌م ده‌مه‌ش له‌ دیدی كه‌ركووكییاندا به‌باڵایی ماونه‌ته‌وه‌، بێگومان هه‌ندێكیان دوور كه‌وتنه‌وه‌ و هه‌ندێكی دیكه‌یشیان له‌ ئاستی مامناوه‌نددا مانه‌وه‌، به‌ڵام له‌وه‌شدا زێده‌بێژیم نه‌كردووه‌ ئه‌گه‌ر بڵێم (جیهاد دڵپاك) له‌ هه‌ره‌ دیارترینیان بووه‌، ڕه‌نگه‌ هۆكاره‌كه‌شی ئه‌وه‌ بێت كه‌ (دڵپاك) هه‌ر له‌ "به‌هاری دزراو" و "مه‌ره‌زه‌" و پاشان "ژاڵه‌"، له‌ پاڵه‌وانیه‌تی و ڕۆڵی سه‌ره‌كیدا گه‌مه‌ی هونه‌ریی ده‌كرد، هاوكات له‌ هه‌موو ئه‌وانه‌شدا ڕۆڵی كه‌سی چه‌وساوه‌ و هه‌روه‌ها شۆڕشگێڕی به‌رجه‌سته‌ ده‌كرد، بۆیه‌ نه‌ك ته‌نیا له‌ كه‌ركووك، بگره‌ له‌ ته‌واوی كوردستاندا بووه‌ هونه‌رمه‌نده‌ خۆشه‌ویسته‌كه‌ی خه‌ڵك.. لێره‌شدا مه‌به‌ستمه‌ بڵێم ته‌واوی هونه‌رمه‌ندانی ئه‌و قۆناغه‌ تا ئه‌م ده‌مه‌یش كاریگه‌رییان به‌سه‌ر هونه‌رمه‌ندانی بواری شانۆ له‌ كه‌ركووكدا ماوه‌، به‌ڵام له‌ نێویاندا (جیهاد دڵپاك) زۆرترین پشكی له‌و كاریگه‌رییه‌ به‌رده‌كه‌وێت.. ڕه‌نگه‌ ئه‌سته‌م بێت ئاماژه‌ به‌چه‌ند ناوێك بده‌م و جیقڵدانه‌ ته‌نگیی هونه‌رمه‌ندانی ئێمه‌ له‌ كه‌ركووكدا ئه‌و ته‌رزه‌ هه‌ڵوێستانه‌یان به‌ئاسان پێ هه‌رس ناكرێت، به‌ڵام ڕاستیتان ده‌وێ چه‌ند ناوێك له‌ بواری شانۆ له‌ كه‌ركووكدا له‌ سه‌ره‌تایانه‌وه‌ تا ئه‌م ده‌مه‌یش نه‌یانتوانیوه‌ ئه‌و كاریگه‌رییه‌ له‌ كاره‌كانیاندا بشارنه‌وه‌ یاخود له‌ لاساییكردنه‌وه‌ی (دڵپاك) خۆ به‌دوور بگرن كه‌ پێم وانییه‌ ته‌نانه‌ت (دڵپاك) خۆیشی به‌و جۆره‌ هونه‌ره‌ ئاسووده‌ بێت.
ئه‌مه‌ به‌شانوباڵ هه‌ڵدانی هونه‌رمه‌ند (جیهاد دڵپاك) نییه‌ و ئه‌وه‌نده‌یش لێیه‌وه‌ نزیك نیم تا پێویستیم پێبێت، ئه‌دی خۆ ئه‌و گه‌یشتووه‌ته‌ ئه‌و پێگه‌ و ئاسته‌ هونه‌رییه‌ی له‌ ناسیندا پێویستیی به‌ئێمه‌مانان نه‌بێت، به‌ڵام بۆ ئێمه‌یش ویژدان مردووییه‌ ئه‌گه‌ر ڕاستییه‌كان وه‌ك ئه‌وه‌ی هه‌ن نه‌یانگێڕینه‌وه‌.. له‌وه‌ش واوه‌تر ئه‌و دۆخه‌ شلۆقه‌یه‌ كه‌ له‌ كه‌ركووكدا شانۆی كوردی تێی كه‌وتووه‌، ئه‌مه‌ نه‌ك به‌و واتایه‌ی له‌و شاره‌دا هونه‌رمه‌ندی باش و تیپی باش و كاری باش بوونیان نه‌بێت، بگره‌ ئه‌گه‌ر بكه‌وینه‌ به‌راورده‌وه‌، ده‌بوو له‌ناو ئه‌وه‌نده‌ ساڵه‌ی كاركردن به‌ئازادی، هه‌نگاوه‌كان ڕووه‌ و داهێنان و كاره‌ نوێخوازییه‌كان له‌و شاره‌دا خێراتر بوونایه‌ نه‌ك وه‌ك ئه‌وه‌ی ده‌یبینین هه‌رچه‌ندی بڵێی خاووخلیچكن.
ئه‌و پۆله‌ هونه‌رمه‌نده‌ی ئه‌و قۆناغه‌ كه‌ نه‌ك ته‌نیا له‌ كه‌ركووك، بگره‌ له‌ ته‌واوی كوردستاندا ڕۆڵی به‌رچاویان هه‌بووه‌ و تا ئێستایش هه‌مان كاریگه‌رییان هه‌یه‌ و به‌ستوونه‌كانی هونه‌ری شانۆ هه‌ژمار ده‌كرێن، دوای به‌جێهێشتنی شاره‌كه‌ له‌ ناوه‌ڕاستی حه‌فتاكاندا، ئه‌و بواره‌ هونه‌رییه‌ له‌ كه‌ركووكدا ڕووی له‌ كزی كرد، وێڕای ئه‌وه‌ی له‌ دوای ساڵانی نه‌وه‌ته‌كانیشه‌وه‌ دیسان هه‌ڵپه‌یه‌كی تری شانۆمان له‌ كه‌ركووكدا بینییه‌وه‌، به‌ڵام تا ئه‌م ده‌مه‌یش نه‌گه‌یشته‌ ئه‌و ئاسته‌ی ته‌واوی بۆشاییه‌كانی پێشتر پڕ بكرێنه‌وه‌، واته‌ جێهێشتنی كه‌ركووك له‌لایه‌ن هونه‌رمه‌ندانی نه‌وه‌ی پێشتره‌وه‌ و جێگیربوونیان له‌ شاره‌كانی دیكه‌ی كوردستان، جیا له‌ دروستكردنی بۆشایی له‌ شاره‌كه‌، هاوكات ڕۆڵیان له‌ پڕكرنه‌وه‌ی بۆشایی دیكه‌ بینی له‌و شارانه‌ی بۆیان چوون.. ئه‌وان تا ئه‌م ده‌مه‌ش به‌هه‌مان هه‌ناسه‌ی پێشتر و واوه‌تریش كاری هونه‌ری ده‌كه‌ن، بۆیه‌ له‌ ئێستادا له‌ پێویستییه‌ گرنگه‌كانن بۆ كه‌ركووك، مه‌به‌ستم موزایه‌ده‌ی گه‌ڕانه‌وه‌ی ئه‌وان بۆ شاره‌كه‌ نییه‌، ئه‌وه‌نده‌ی مه‌به‌ستمه‌ بڵێم بۆ گه‌شه‌دان به‌شانۆی كوردی له‌ كه‌ركووكدا جیهادێكی دیكه‌مان پێویسته‌، واته‌ هه‌وڵدان بۆ دروستكردنی جیهاده‌كان ده‌كرێ له‌ ڕێگه‌ی كه‌سانی وه‌ك (جیهاد دڵپاك) و هاوڕێكانیه‌وه‌ بێت.. ئه‌وان ئه‌گه‌ر ده‌رفه‌تی گه‌ڕانه‌وه‌یشیان بۆ كه‌ركووك لاواز بێت و نه‌گونجێت، بێگومان ئه‌ویش به‌هۆی تایبه‌تمه‌ندیی ژیانی كه‌سی و خێزانیی هه‌ر یه‌كێكیان، به‌ڵام ده‌كرێ له‌ ڕێگه‌ی گواستنه‌وه‌ی ئه‌زموونه‌كه‌یانه‌وه‌ ئه‌و ڕۆڵه‌ جیهادییه‌ ببینن.
