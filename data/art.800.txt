دایک و باوکی هه‌ر له‌ منداڵییه‌وه ، له‌ ماڵه‌وه‌ به‌ زمانی شیرینی کوردیمان له‌گه‌ڵ شۆخانی کچیاندا ده‌دوان‌ .. بۆیه‌ شۆخان به‌ کوردییه‌کی په‌تی و 
شۆخان هه‌موو ساڵێک بۆ ماوه‌ی دوو هه‌فته‌ له‌ مانگی ئادارد دا ، له‌گه‌ڵ دایکی یا باوکی ، یا هه‌رسێکیان پێکه‌وه‌ ده‌هاتنه‌وه‌ بۆ شاری سلێمانی ، بۆ ئه‌وه‌ی له‌گه‌ڵ خزمه‌کانیاندا ، به‌شداری له‌ ئاهه‌نگ و سه‌یرانی و خۆشیی جه‌ژی نه‌ته‌وه‌ییمان ( نه‌ورۆز ) بکه‌ن .
له‌ نێو هه‌موو گۆڵه‌ سروشتییه‌ جوانه‌کانی به‌هاری قه‌شه‌نگ و ره‌نگینی کوردستاندا .. ‌شۆخان زۆر هۆگری‌ گوڵه‌ نیرگزه .. جوانترین گۆڵه‌ به‌لایه‌وه‌ و به‌ شاگوڵی ده‌زانێ .. ئه‌و جۆره‌ نێرگزه‌ی که‌ له‌ نه‌رویجدا هه‌یه‌ نه‌ جوانی و نه‌بۆنی ، ناگاته‌ جوانی و بۆنی نێرگزی کوردستان .
له‌ مانگی ئاداری پاردا (2011 ) ، شۆخان و دایکی هاتنه‌وه‌ بۆ سلێمانی .. دوو رۆژ به‌ر له‌ جه‌ژنی نه‌ورۆز ، شۆخان و دایکی له‌ شه‌قامی (مه‌وله‌وی) دا .. کچێکی خوێن شیرینی ناسکۆڵه‌یان بینی که‌ سه‌به‌ته‌یه‌ک پڕ له‌ چه‌پکه‌ گوڵه‌ نێرگزی گه‌شی له‌ به‌رده‌م  خۆێدا  بۆ فرۆشتن دانابوو .. شۆخان و دایکی چوونه‌ لای كچه‌ نێرگز فرۆشه‌که‌ و دوو چه‌پکیان لێ کڕی .. شۆخان یه‌کسه‌ر تێر و پڕ بۆنی نێرگزه‌که‌ی هه‌ڵمژی.
‌  
دایکی شۆخان ( بیست وپێنج هه‌زارییه‌کی دایه‌ ده‌ست کچه‌ نێرگز فرۆشه‌که‌) وه‌ و .. پێی وت : هه‌مووی هه‌ڵگره‌ بۆ خۆت .
نێرگزفرۆشه‌که‌ : ( به‌ رامانێکه‌وه‌ ) ئه‌م پاره‌یه‌ زۆره‌ .. ئه‌و دوو چه‌پکه‌ نێرگزه‌ بایی دوو هه‌زار دینارن .. له‌ دوو هه‌زاره‌که‌ زیاترتان لێ وه‌رناگرم .
شۆخان : خوشکه‌ جوانه‌که‌م ، ده‌توانیت بۆ سبه‌ینێ ، له‌م کاته‌ و هه‌ر بۆ ئه‌م شوێنه‌ .. پانزه‌ چووزه‌ره‌ نێرگزم به‌ سه‌لکه‌کانیانه‌وه‌ بۆ بهێنیت .. ؟ 
شۆخان : دایکه‌ گیان ده‌یان به‌مه‌وه‌ بۆ وڵاتی نه‌رویج و له‌وێدا له‌ مه‌به‌سته‌که‌م تێ ده‌گه‌یت که‌ بۆ چیمه‌.
بۆ سبه‌ینی نێرگز فرۆشه‌که‌ ، بیست و پێنج چووزه‌ره‌ نێرگزی به‌ سه‌لک و ره‌گ و ریشاڵه‌وه‌ ، که‌ پارچه‌ که‌تانێکی ته‌ڕی پێوه‌ ئاڵاندبوون بۆ شۆخانی هێنابوون.. له‌ هه‌مان کات و شوێندا دانشتبوو .. چاوه‌ڕێی شۆخان و دایکی ده‌کرد .. شۆخان و دایکی له‌م کاته‌دا گه‌یشتنه‌ لای کچه‌ نێرگز فرۆشه‌که‌ و.. بۆخچه‌یه‌کی گه‌وره‌ی پڕیشیان پێ بوو.
نێرگزفرۆشه‌که‌ ( به‌ شۆخان و دایکی به‌ زه‌رده‌ خه‌نه‌یه‌کی ناسکه‌وه‌) : به‌خێر بێن .. شۆخان خان ئه‌وه‌ی ‌ داوات لێ کردم بۆتم هێناوه .. ئه‌مه‌ بیست و پێنج دانه‌ چووزه‌ره‌ نێرگزی هه‌راشن ، به‌ سه‌لک و ره‌گ و ریشاڵه‌وه‌ بۆم هێناوی  پێشکه‌شتبن .. هیوادارم که‌ به‌ دڵت بێ .
شۆخان باوه‌شی له‌ كچه‌ نێرگزفرۆشه‌که‌ دا و ، ئه‌م لاو ئه‌ولای ماچ کرد و ، زۆر سۆپاسی کرد .. وتیشی : هه‌رگیز خۆت و ئه‌م دیارییه‌ت له‌ بیر ناکه‌م .
دایکی شۆخان سه‌د دۆلاری ده‌رهێنا و له‌ گه‌ڵ پاکه‌تێک گه‌زۆی نایاب و به‌ کێژۆڵه‌ نێرگز فرۆشه‌که‌ی وت :   وابزانه‌ که‌ من پوورتم ، ئه‌م بوخچه‌یه‌ش ده‌ستێک جلی تازه‌ی جوانی کوردی و جووتێ پێڵاوی جوانیشی تێدایه‌ .. که‌ ئه‌مڕۆ به‌ تایبه‌تی بۆ تۆمان کڕیون و ، دیاریین له‌ شۆخانی خوشکته‌وه‌.. بۆ تۆی ئازیز .. بۆ ئه‌وه‌ی به‌یانی که‌ جه‌ژنی نه‌ورۆزه‌ له‌ به‌ریان بکه‌یت و ده‌می خوشک و براکانیشت شیرین بکه‌یت .. ئه‌گه‌ر ئه‌م دیارییه‌مان لێ وه‌رنه‌گری ،‌ ئێمه‌ش دیارییه‌که‌ت لێ وه‌رناگرین .
نێرگزفرۆشه‌که‌ : من شۆخان چووه‌ دڵمه‌وه‌ و.. ئه‌م سه‌لکه‌ نێرگزانه‌شم به‌ دیاری بۆ هێناوه‌ .. چۆن ده‌بێ له‌ بریتییان پاره‌ و جلوبه‌رگتان لێ وه‌ربگرم.
شۆخان و دایکی زۆریان هه‌وڵدا .. تا ‏نێرگز فرۆشه‌که‌یان رازی کرد و .. دیارییه‌کانیانی لێ وه‌رگرتن .
به‌ ئه‌وپه‌ڕی رێزه‌وه‌ دیارییه‌کانی لێ وه‌رگرتن و .. باوه‌شی له‌ شۆخان و له‌ دایکی داو .. ماچیانی کرد و .. زۆریش سوپاسیانی کرد.. هه‌ر بۆ یادگاریش هه‌رسێکیان پێکه‌وه‌ وێنه‌یه‌کیان گرت .
یه‌که‌م رۆژی دوای جه‌ژنی نه‌ورۆز ، شۆخان و دایکی گه‌ڕانه‌وه‌ بۆ نه‌رویج .. هه‌ر ئه‌و رۆژه‌ی که‌ گه‌یشتنه‌وه‌ ماڵ .. شۆخان له‌ ناو سێ ئینجانه‌ی گه‌وره‌ و جواندا ، سه‌لکه‌‌ نێرگزه‌کانی چاند .. له‌ ماوه‌ی دوو هه‌فته‌دا .. نێرگزه‌کانی گوڵیان کرد و .. بۆن و به‌رامه‌ی خۆشیان هه‌موو لایه‌کی حه‌وشه‌که‌یانی پڕ کرد .. هه‌ر میوانێکیشیان بهاتایه‌ له‌ جوانی و بۆنخۆشی گوڵه‌ نێرگزه‌کان سه‌رسام ده‌بوون .
له‌ رۆژی چوار شه‌ممه‌ی هه‌فته‌ی یه‌که‌می مانگی گوڵاندا .. شوخان داوای له‌ باوکی کرد ، که‌ ئه‌مڕۆ به‌ ئۆتۆمبیله‌که‌ی بیگه‌یه‌نێته‌ قوتابخانه‌که‌یان چونکه‌ یه‌کێک له‌و دوو ئینجانه‌ نێرگزانه‌ ده‌بات بۆ قوتابخانه‌که‌یان .. له‌وێوه‌ ده‌ینێرن بۆ شوێنێکی تر .. تا ئه‌و کاته‌ش نه‌ دایکی و نه‌ باوکی له‌ مه‌به‌ستی شۆخان له‌ بردنی ئه‌و گوڵه‌ نێرگزانه‌ بۆ قوتابخانه‌که‌ی تێ نه‌گه‌یشتبوون .. شۆخانیش ده‌یویست پێش وه‌خت قسه‌ نه‌کات تا ئه‌و کاته‌ی که‌ هیوا و مه‌به‌ستی دێته‌دی .
‌   
شۆخان ئینجانه‌ نێرگزه‌که‌ی برد بۆ قوتابخانه‌که‌یان .. له‌گه‌ڵ چه‌ند ئینجانه‌ پڕ له‌ گوڵێکی تری چه‌ند قوتابییه‌کی تری قوتابخانه‌که‌یان .. که‌ هه‌ر یه‌که‌یان جۆره‌ گۆڵێکیان هێنابوو .. هه‌ر ئه‌و رۆژه‌ش هه‌موو گوڵه‌کانیان برد بۆ هۆڵی پێشبڕکێی ( پێشانگای شاگوڵی ساڵ ـ له‌ نه‌رویج ) دا .. كه هه‌موو جۆره‌ گوڵه‌ هه‌ڵبژارده‌کانی شاره‌کانی نه‌رویجیان له‌و پێشانگایه‌دا پێشان ده‌دان .
له‌ رۆژی هه‌ینی هه‌مان هه‌فته‌دا پێشانگاکه‌ کرایه‌وه‌ ..  بینه‌رانی گوڵه‌کان زۆر به‌ جوانی و ناسکی و بۆن خۆشی گوڵه‌ نێرگزه‌کانی شۆخان سه‌رسام بوون .. سه‌ره‌نجی ( لیژنه‌ی بڕیاده‌ر ) راکێشا .. له‌ ئه‌نجامدا به‌ تێکڕای ده‌نگه‌کانی ئه‌ندامانی ( لیژنه‌ی هه‌ڵسه‌نگاندن و بڕیارده‌ر) .. گۆڵه‌ نێرگزه‌کانی شۆخانیان .. به‌ جوانترین و به‌ بۆن خۆشترین و به‌ یه‌که‌مین گوڵی ساڵ هه‌ڵیان بژارد و .. نازناوی .. ( شاگوڵی ساڵی 2011 ) یان پێ به‌خشی .
بانگی شۆخان و دایک و باوکیشیان کرد .. که‌ بۆ ئاهه‌نگی پێشکه‌شکردنی خه‌ڵاته‌کان ئاماده‌بن .. بۆ ئه‌و رۆژه‌ش شۆخان خۆی رازانده‌وه‌ و ده‌ستێ جلی کوردی جوانی له‌ شێوه‌ی هه‌ر چوار ره‌نگه‌کانی ئاڵای کوردستان له‌به‌ر کرد .. دوو هێنده‌ی تر شۆخان شۆختر و جوانتر بوو .. که‌ گه‌یشتنه‌ هۆڵی ئاهه‌نگه‌که‌ .. سه‌رۆکی شاره‌وانی ی ئۆسڵۆ زۆر به‌گه‌رمی به‌خێرهاتن و پێشوازی له‌ شۆخان و دایک و باوکی کرد .
له‌گه‌ڵ چه‌پڵه‌ رێزانی ئاماده‌بوواندا.. سه‌رۆکی شاره‌وانی ی ئۆسڵۆ.. میدالیای یه‌که‌می زێڕینی ( شاگوڵی ساڵی 2011 ) ی له‌ ملی شۆخان کرد و .. زۆر به‌ گه‌رمییش ته‌وقه‌ی له‌ گه‌ڵیدا کرد و روومه‌تی ماچ کرد و .. ده‌ستخۆشی و پیرۆزبایی لێ کرد .. به‌ زه‌رده‌خه‌نه‌یه‌کیشه‌وه‌ پێی وت : " خۆتیش و جله‌ جوانه‌کانیشت شایانی ئه‌وه‌ن که‌ به‌ ( شا خونچه‌ و شاجلوبه‌رگی ساڵ ) هه‌ڵبژێردرێن .. هیوای سه‌رکه‌وتن و دواڕۆژێکی گه‌شت بۆ ده‌خوام . "
لێشی پرسی : " ئه‌م گۆڵه‌ ناسک و جوانه‌ بۆن خۆشه‌ت له‌ کوێوه‌ هێناوه‌‌ ؟ "
شوخان به‌ زمانی نه‌رویجی ، بۆ سه‌رۆکی شاره‌وانی و ئاماده‌بووانی گێڕایه‌وه‌ .. که‌ گوڵی نێرگز یه‌کێکه‌ له‌ گوڵه‌ جوانه‌کانی کوردستان و .. سیمبولی جه‌ژنی نه‌ورۆز و به‌هاری ره‌نگینی کوردستانه‌ .. هه‌ر به‌ نه‌رویجیش کورته‌ باسێکی ده‌رباره‌ی کورد و کوردستان بۆ ئاماده‌بووان کرد .. ئاماده‌بووانیش به‌ قایمی  چه‌پڵه‌یان بۆ لێدا و .. پیرۆزباییان لێ کرد و .. وێنه‌یان له‌گه‌ڵ خۆی و له‌گه‌ڵ گوڵه‌ نێرگزه‌کانی کوردستانیشدا گرت .
