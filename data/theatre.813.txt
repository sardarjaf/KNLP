به‌مه‌به‌ستی قسه‌كردن له‌سه‌ر ره‌وتی ئه‌مڕۆی شانۆی كوردی و دۆزینه‌وه‌ی میكانیزمێكی گونجاو بۆ چاره‌سه‌ركردنی قه‌یران و كێشه‌كانی شانۆ، له‌ بیرۆكه‌ی فازیل جاف و به‌ پاڵپشتی و سه‌رپه‌رشتی وه‌زاره‌تی رۆشنبیری و سه‌ندیكای هونه‌رمه‌ندان به‌ ئاماده‌بوونی دیارترین شانۆكاره‌كانی كوردستان و شاره‌زایانی بواره‌كه‌ یه‌كه‌مین (دیداری شانۆی كوردی)  بۆ ماوه‌ی 3 رۆژ به‌ڕێوه‌چوو.
به‌ڵام ده‌ركه‌وت كه‌ به‌شداربووه‌كان ئه‌وه‌نده‌ی ره‌خنه‌ و قسه‌یان له‌سه‌ر مه‌راسیمی  دیداره‌كه‌ هه‌بوو،ئه‌وه‌نده‌ قسه‌یان له‌سه‌ر ئامانجی سه‌ره‌كی
بۆیه‌ زۆربه‌ی شانۆكارانی ئاماده‌بووی دیداره‌كه‌ پێیان وابوو كه‌ ئه‌و مه‌راسیمه‌ بۆئه‌وه‌ سازكراوه‌ كه‌ ناوی چه‌ند كه‌سێكی پێ‌ به‌رزبكرێته‌وه‌ و به‌سه‌ر شاره‌زایانی بواره‌كه‌ی فه‌رزی بكه‌ن.
ره‌خنه‌یه‌كی دیكه‌ كه‌ له‌سه‌ر دیداره‌كه‌ هه‌بوو، په‌یوه‌ندی پته‌وی هه‌ردوو وه‌زاره‌تی په‌روه‌رده‌ و خوێندنی باڵابوو به‌ بواری شانۆ ، كه‌ زۆربه‌ی سه‌رنج و ته‌وه‌ره‌ سه‌ره‌كیه‌كان له‌سه‌ر ئه‌كادیمیه‌تی كاره‌كه‌بوو كه‌ له‌ رێگه‌ی په‌روه‌رده‌ و خوێندنی باڵاوه‌ ده‌گه‌یشتنه‌ ئامانج، كه‌چی ئه‌و وه‌زاره‌تانه‌ نوێنه‌ریان له‌و دانیشتنه‌ نه‌بوو ،چونكه‌ به‌ شێوه‌یه‌كی فه‌رمی داوای به‌شداربوون و ئاماده‌بوونیان لێ‌ نه‌َكرابوو تا نوێنه‌ریان له‌وێ‌ سه‌رنجی شانۆكاره‌كانیان هه‌ڵسه‌نگاندبا ، بۆیه‌ هه‌موو ئه‌و قسانه‌ی له‌و باره‌یه‌وه‌ كران هیچ بایه‌خێكی ئه‌وتۆی بۆ شانۆی كوردی نه‌بوو ، به‌ڵكو به‌ بۆچوونی ئاماده‌بووان كاتی زۆری  دیداره‌كه‌ی به‌هه‌ده‌ر برد.
ئه‌و سیمیناره‌ی كه‌ له‌ مه‌راسیمی دیداره‌كه‌دا زۆرترین قسه‌ی له‌سه‌ر كرا و تێیدا ره‌خنه‌یه‌كی زۆر له‌ وه‌زاره‌تی رۆشنبیریی و شانۆكاران گیرا، له‌ گفتوگۆی هونه‌رمه‌ند سه‌باح عه‌بدوڵڕه‌حمان دا بوو، كه‌ پێیان وابوو سیسته‌می هاندان و كاركردنی وه‌زاره‌تی رۆشنبیری بۆ هونه‌ری شانۆ وێرانه‌ و له‌وپه‌ڕی بێ‌ سه‌روبه‌ری دایه‌ ،  له‌و ئه‌نجامه‌شدا زۆربه‌ی شانۆكاران گه‌نده‌ڵ بوونه‌ و ته‌نها بۆ پاره‌ كارده‌كه‌ن و چاویان له‌ داهێنان و كاری جوان نیه‌.
چونكه‌ تا ئیستا وه‌زاره‌تی رۆشنبیری له‌ رێگه‌ی به‌رێوبه‌رایه‌تی شانۆوه‌ ته‌نها زانیوویه‌تی پاره‌ خه‌رج بكات و نه‌یتووانیووه‌ هه‌ڵسه‌نگاندن و پێوه‌ری هه‌بێت بۆ چاودێریكردنی كاره‌كان .
له‌ به‌رامبه‌ریشدا قسه‌كانی ئه‌و هونه‌رمه‌نده‌ ره‌خنه‌ی لێگیراو ره‌تكرایه‌وه‌، چونكه‌ به‌ بۆچوونی هیوا سوعادی به‌ڕێوبه‌ری هونه‌ری شانۆ و چه‌ند شانۆكارێكی دیكه‌، هونه‌رمه‌ند سه‌باح عه‌بده‌لڕه‌حمان كاتێك ده‌یتووانی كه‌ ئه‌وه‌نده‌ به‌ زه‌قی ره‌خنه‌ له‌ وه‌زاره‌ت بگرێت كه‌ به‌ده‌ربێت له‌ گه‌نده‌ڵیه‌كان و كێشه‌ هه‌ڵپه‌سێردراوه‌كان و له‌سه‌ر ئه‌و سیسته‌مه‌ی كه‌ به‌ گه‌نده‌ڵ ناوی ده‌بات كاری نه‌كردبا.
له‌به‌ر ئه‌وه‌ی ئه‌و هونه‌رمه‌نده‌ به‌ر له‌ ئێستا خۆی به‌رێوبه‌ری به‌رێوه‌به‌رایه‌تی شانۆ بووه‌ و ده‌سه‌ڵاتی ئه‌وه‌یان پێداوه‌ كا چاكسازی بكات و دۆخه‌كه‌ به‌ره‌و باشترببات كه‌ نه‌وه‌ك نه‌یكردووه‌ به‌ڵكو له‌ شوێنی ئه‌وه‌ی رێزلێنان بۆ داهێنه‌ره‌كانی شانۆ رێكبخات، مه‌راسیمی رێزلێنانی بۆ وه‌زیرێك كردووه‌ كه‌ هه‌موومان ده‌زانین بۆ چ مه‌به‌ستێك بووه‌.
به‌ هه‌مان شێوه‌ هونه‌رمه‌ند سه‌عدون یونس پێی وابوو كه‌ ئه‌و شانۆگه‌ریانه‌ی له‌ ده‌ره‌وه‌ی ووڵات  نمایش ده‌كرێت ته‌نها بۆ گه‌شت و پاره‌یه‌ و هیچ بینه‌رێكیان نیه‌، به‌ڵام بۆ وه‌ڵامدانه‌وه‌ی ره‌خنه‌كانی سه‌عدون یونس چه‌ند شانۆكارێك پێیان وابوو كه‌ ئه‌و هونه‌رمه‌نده‌ ئاگاداری هیچ كارێكی شانۆیی نه‌بووه‌ له‌ ده‌ره‌وه‌ و ناوه‌وه‌ی ووڵات، بۆیه‌ بۆی نیه‌ قسه‌ له‌سه‌ر مه‌سه‌له‌یه‌ك بكات به‌بێ‌ زانیاری دروست، ته‌نانه‌ت سه‌عدونیان به‌ هونه‌رمه‌ندێكی غروور و  خه‌مسارد ناوزه‌ند كرد.
