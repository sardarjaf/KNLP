ته‌ماشایه‌كی كۆڵانی كرد ، هیچ كام له‌ هاوڕێكانی نه‌هاتبوونه‌ ده‌ره‌وه‌ بۆیه‌ ئه‌میش چووه‌ ژووره‌وه‌ و له‌سه‌ر جۆلانه‌ی حه‌وشه‌كه‌یان دانیشت.هێنده‌ی زانی دایكی نامه‌یه‌كی بچكۆلانه‌ی بۆ هێنا و وتی: گوڵێ گیان ئه‌م نامه‌یه‌ شه‌نی هاوڕێت بۆی هێناوی .ئه‌ویش له‌ دایكی وه‌رگرت و خوێندییه‌وه‌...نه‌سرابوو((سڵاو گوڵێ گیان....ئێمه‌ ده‌چین بۆ لای بریسكه‌،تۆش وه‌ره‌.....هاوڕێكانت.))
دوای خوێندنه‌وه‌ی نامه‌كه‌ ،كه‌مێك دڵگران بووله‌ هاوڕێكانی ،چونكه‌ ده‌یوت ده‌بوایه‌ ئه‌وان پێشتر ئاگاداریان بكردمایه‌ و به‌ ده‌ستی به‌تاڵ نه‌ڕۆیشتنایه‌ بۆ لای بریسكه‌ چونكه‌ له‌ نه‌خۆشخانه‌ ده‌رچووه‌....گوڵێ گوێی نه‌دایه‌ و وتی نا ...با لێبوردن لای منه‌وه‌ بێت ...ئه‌ی ئه‌گه‌ر پێیشیان نه‌وتمایه‌ چیم بكردایه‌..بۆیه‌ خێۆا له‌گه‌ڵ دایكی چوو بۆ سۆپه‌ر ماركێته‌كه‌ی گه‌ڕه‌ك چه‌پكێك گوڵ و قوتویه‌ك شیرینی كڕی و له‌سه‌ر پارچه‌یه‌ك كاغه‌ز به‌ كۆمپیوته‌ره‌كه‌ی نووسی (دیاری گوڵێ و هاوڕێكانی بۆ بریسكه‌) ..به‌ پرینته‌ره‌كه‌ی رایكێشا...له‌ ناو چه‌پكه‌ گوڵه‌كه‌دا چه‌سپی كرد...ئیتر به‌ره‌ و ماڵی بریسكه‌ كه‌وته‌ ڕێ..كه‌ زه‌نگی لێدا بانوی خوشكی بریسكه‌ ده‌رگاكه‌ی لێكرده‌وه‌ و وتی فه‌رموو فه‌رموو گوڵێ گیان..هاوڕێكانی كه‌ گوڵێ یان به‌و شێوه‌یه‌ بینی ،رویان نه‌ده‌هات ته‌ماشای بكه‌ن، هه‌ر سه‌یری یه‌كتریان ده‌كرد...چونكه‌ ئه‌وان هه‌م گوڵێ یان به‌جێهێشتبوو ، هه‌م به‌ ده‌ستی به‌تاڵیش هاتبوون بۆ لای بریسكه‌.
گوڵێ به‌ رویه‌كی خۆشه‌وه‌ چووه‌ ژووره‌وه‌ و رۆژباشی لێكردن و چه‌پكه‌ گوڵ و شیرینییه‌كه‌شی لای سه‌ری بریسكه‌وه‌ دانا و ته‌وقه‌ی له‌گه‌ڵدا كردو هه‌واڵی پرسی..ئه‌وسا یه‌كه‌ یه‌كه‌ ته‌وقه‌شی له‌گه‌ڵ هاوڕێكانیدا كرد و وتی ها هاوڕێیان ببورن ، من به‌ به‌هۆی ئاماده‌كردنی ئه‌و دیارییانه‌وه‌ دواكه‌وتم كه‌ ئێوه‌ له‌ نامه‌كه‌تاندا نووسیبووتان...ئه‌وانیش ته‌ماشایه‌كی یه‌كتریان كرد ..بانوویش وتی گوڵێ گیان ئه‌وان ده‌زانن كه‌ تۆ له‌و كارانه‌دا شاره‌زا و لێهاتووی بۆیه‌ ئه‌و ئه‌ركه‌یان به‌تۆ سپاردووه‌...ئیتر پێكه‌وه‌ دانیشتن و كه‌وتنه‌ قسه‌ و هه‌ریه‌كه‌یان باسی ئه‌و وانانه‌ی بۆ ده‌كرد كه‌ له‌وێ نه‌بووه‌ ...دایكی بریسكه‌ش به‌خۆشییه‌وه‌ ده‌هات و ده‌چوو ..میوه‌ و شیرینیشی بۆ هێنان ...پاش ماوه‌یه‌ك هه‌ستان و هه‌ریه‌كه‌یان رۆیشته‌وه‌ بۆ ماڵی خۆیان......
