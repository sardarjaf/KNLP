هه‌رچه‌نده‌ من چه‌ند جارێك رامگه‌یاندووه‌ كه‌ له‌باره‌ی یه‌كێتی نیشتیمانیه‌وه‌ هیچ نه‌ڵێم و هیچ نه‌نووسم، به‌ڵام هه‌ندێجار ناتوانم ئه‌و به‌ڵێنانه‌ی خۆم ببه‌مه‌سه‌ر، هۆكاره‌كه‌ش ئه‌وه‌یه‌ كه‌ مرۆڤـ هه‌ست ده‌كات هه‌ندێ كار ده‌كرێت كه‌ سه‌رئه‌نجام به‌ زیانی گشتی میلله‌ته‌كه‌مان ده‌شكێته‌وه‌، بۆیه‌ له‌و كاتانه‌دا ناتوانێت بێ ده‌نگ بێت.
ره‌نگه‌ ئه‌گه‌ر یه‌كێتی به‌ ته‌واوه‌تی بگه‌یشتیایه‌ته‌ ئه‌و قۆناغه‌ی كه‌ شكست و هه‌ره‌سهێنانی هیچ جۆره‌ كاریگه‌رییه‌كی له‌سه‌ر نه‌ته‌وه‌كه‌مان نه‌بوایه‌، ئه‌وا بێگومان نه‌ قسه‌مان له‌باره‌وه‌ ده‌كردو نه‌ هیچ هه‌نگاوێكی نادرووستی ئه‌ویش نیگه‌رانی ده‌كردین، وه‌كو چه‌ندین ده‌سته‌و گروپی بچووك كه‌ هیچ رۆڵ و كاریگه‌رییه‌كیان له‌سه‌ر چاره‌نووس و ته‌نانه‌ت بارودۆخی ئێستای كوردستانیش نییه‌ چی بكه‌ن كه‌س به‌لایه‌وه‌ گرنگ نییه‌، به‌ڵام ئه‌وه‌ی ئاشكرایه‌ تاكو ئێستاش یه‌كێتی سه‌رباری كێشه‌كانی ده‌توانێت به‌ هه‌ردووباری ئه‌رێنی و نه‌رێنیدا رۆڵی دیار له‌ باشووری كوردستاندا ببینێت، بۆیه‌ به‌و ئه‌ندازه‌ی كه‌ چاره‌نووسی كوردستان په‌یوه‌ندی به‌ هه‌موومانه‌وه‌ هه‌یه‌، به‌و ئه‌ندازه‌یه‌ش نه‌ك هه‌ر مافی ئه‌وه‌مان هه‌یه‌ قسه‌ بكه‌ین به‌ڵكو ئه‌ركی سه‌رشانیشمانه‌ ئه‌وه‌ بیری برایانی سه‌ركردایه‌تی ئه‌م حیزبه‌ بخه‌ینه‌وه‌ كه‌ ئه‌وان به‌رپرسیارێتیه‌كی گه‌وره‌یان له‌ ئه‌ستۆیه‌ به‌ تایبه‌تی دوای ته‌واوبوونی قۆناغی سه‌ركردایه‌تی مام جه‌لال، كه‌ ناكرێت به‌وشێوه‌یه‌ی ئێستاو به‌شێك له‌به‌رپرسان هه‌رزه‌كارانه‌ بێ گوێدانه‌ ئه‌و ئاسه‌وارو كاریگه‌رییه‌ نێگه‌تیفیانه‌ی قسه‌كانیان به‌جێی ده‌هێڵن قسه‌بكه‌ن و ته‌سریح بده‌ن.
نامه‌كه‌ی من به‌ پرسیارێك ده‌ست پێده‌كات: ئایا كێ یه‌كێتی به‌ڕێوه‌ده‌بات؟ ئایا كێ مافی ئه‌وه‌ی هه‌یه‌ به‌ بێ گه‌ڕانه‌وه‌ بۆ مه‌كته‌بی سیاسی ته‌سریح بدات به‌ تایبه‌تی ئه‌و ته‌سریحانه‌ی كه‌ په‌یوه‌ندییان به‌ كۆی بارودۆخی كوردستانه‌وه‌ هه‌یه‌؟ هه‌ر لێره‌شه‌وه‌ ئه‌م نامه‌ برایانه‌ به‌ڵام تفت و تاڵه‌یه‌ ئاراسته‌ی كاك مه‌لا به‌ختیار ده‌كه‌م چونكه‌ ئه‌وه‌ی كه‌ رۆژانه‌ له‌م حیزبه‌دا قسه‌ده‌كات و له‌سه‌ر هه‌موو جۆره‌ بابه‌تێك له‌ سیاسه‌تی حیزبه‌كه‌ی و سیاسه‌تی نێوده‌وڵه‌تیه‌وه‌ بۆ دیموكراسی و ئازادی و ته‌نویرو بواره‌كانی ئه‌ده‌ب و تا ده‌گاته‌ مافی ژنان و. .
قسه‌ی هه‌یه‌.
من له‌ ماوه‌ی ئه‌م ساڵدا ئه‌مه‌ دووه‌م نامه‌یه‌ كه‌ بۆ به‌ڕێزی بنووسم، دڵنیاشم كه‌ ئه‌و هیچ گومانی له‌وه‌ نییه‌ كه‌ من حه‌زده‌كه‌م هه‌م خۆی و هه‌م حیزبه‌كه‌شی سه‌ركه‌وتوو بن بۆ خزمه‌تی میلله‌ته‌كه‌مان، به‌ڵام گومانم هه‌یه‌ ئه‌م شێوه‌ كاركردن و ته‌سریحدانه‌ی به‌ڕێزیان خزمه‌تی كوردو ته‌نانه‌ت یه‌كێتیش بكات.
پێشتریش من زیاترین ره‌خنه‌م له‌ ئه‌دای حكومه‌تداری به‌ڕێزان كاك كۆسره‌ت و كاك به‌رهه‌م ساڵح گرت و به‌داخێكی زۆریشه‌وه‌ ئه‌نجامه‌كان وه‌كو ئه‌وه‌ ده‌رچوون كه‌ من پێشبینیم كردبوون، چ یۆ یه‌كێتی و چ بۆ رۆڵی سه‌ركردایه‌تی ئه‌م حیزبه‌و سیاسه‌ت و ستراتیژه‌كه‌ی كه‌ ئه‌وه‌ش رۆژ له‌ دوای رۆژ بچوكی ده‌كاته‌وه‌، به‌ڵام مه‌ترسی كاره‌كانی ئه‌وكاته‌ی ئه‌وان وه‌كو ئه‌وه‌ی ئێستا ترسناك نه‌بوو چونكه‌ ئه‌وكاته‌ پیاوێك هه‌بوو كه‌ به‌ توانای خۆی ده‌یتوانی به‌ هه‌موو كه‌مورتیه‌كانه‌وه‌ هه‌م یه‌كێتی له‌ روخاندن بپارێزێت، هه‌م ئه‌و كێشانه‌ی له‌گه‌ڵ لایه‌نه‌كانی تر درووست ده‌بوون به‌ تواناو لێهاتوویی خۆی چاره‌سه‌ریان بكات كه‌ ئه‌ویش مام جه‌لال بوو.
له‌و بڕوایه‌دام هه‌مووان كۆكین له‌سه‌ر ئه‌وه‌ كه‌ ئێستا قۆناغی مام جه‌لال ته‌واو بووه‌و هه‌ڵه‌ی ترسناك یه‌كێتی به‌ره‌و كه‌ناری هه‌ره‌س ده‌بات و لایه‌نه‌كانی تریش به‌و ئه‌ندازه‌ی كه‌ گوێیان له‌ مام جه‌لال ده‌گرت گوێ له‌و برا به‌ڕێزانه‌ی ئێستای مه‌كته‌بی سیاسی ناگرن به‌ تایبه‌تی كه‌ زۆرێك له‌و ته‌سریحاتانه‌ هه‌ست ده‌كرێت ئامانجی نانه‌ته‌وه‌یی و نانیشتیمانی له‌ پشتیانه‌وه‌یه‌و ئه‌گه‌ر راڕه‌وی رووداوه‌كان به‌وجۆره‌ بڕۆن ئه‌وا جارێكی تر كوردستان به‌ره‌و كاره‌ساته‌ تراژیدیه‌كانی رابردوو ده‌به‌ن.
له‌ راستیدا ئه‌وه‌ی له‌ رووی ده‌ره‌وه‌و له‌ ناو راگه‌یاندنه‌كان و له‌ ته‌سریحدانه‌كاندا ده‌بینرێت و هه‌ستی پێده‌كرێت، ئه‌وه‌یه‌ كه‌ ئێستا كاك مه‌لا به‌ختیار واره‌فتار ده‌كات كه‌ ئه‌و سكرتێری یه‌كێتی نیشتیمانی كوردستان بێـت و پێش ده‌چێت كه‌ بڕیاری ره‌های حیزب له‌ ده‌ستی به‌ڕێزیدا بێت و هه‌ر ئه‌ویش حیزب به‌ڕێوه‌ببات، به‌ تایبه‌تی له‌و هه‌موو ته‌سریاتانه‌ی كه‌ ده‌یدات، به‌ ئه‌سه‌فیشه‌وه‌ كه‌ به‌ڕێزیان ده‌چێـتـه‌ دوای مایكرۆفۆنه‌وه‌، ئیتر بیری ده‌چێـه‌وه‌ كه‌ ئێستا كوردستان له‌ چ بارودۆخێكدایه‌و، ته‌نانه‌ت بیری ئه‌وه‌شی نییه‌ ئه‌و سه‌رو ریشی سپی بووه‌و به‌ پێی هه‌موو لۆجیكێك قۆناغی گوتاردانی حه‌ماسی به‌سه‌رچووه‌و ئێستا بۆ ئه‌و ته‌مه‌نه‌ی به‌ڕێزیان گوتاری پڕ له‌ حه‌ماسه‌ت گونجاو نییه‌ چونكه‌ سه‌رچاوه‌ی زۆرینه‌ی هه‌ڵه‌كانی مرۆڤـ سیاسی بێت یان ناسیاسی هه‌ڵچوون و حه‌ماسه‌ته‌.
من لێره‌دا نامه‌كه‌م له‌ چه‌ند خاڵێكدا كۆده‌كه‌مه‌وه‌، كه‌ سه‌رباری راوبۆچوونی خۆم ئه‌م خاڵانه‌ راوبۆچوونی ناو خه‌ڵكیشی تێدایه‌ به‌ تایبه‌تی خه‌ڵكی نوخبه‌و خه‌ڵكی ئاسایی ده‌ره‌وه‌ی یه‌كێتیش.
یه‌كه‌م: من پێموایه‌ كاك به‌ختیار به‌ حوكمی ئه‌وه‌ی كه‌ یه‌كێكه‌ له‌ كۆنترین كادیره‌كانی یه‌كێتی نیشتیمانی كه‌ تاكو ئێستا له‌سه‌ركردایه‌تی ئه‌و حیزبه‌دا ماوه‌، ده‌بوو باش سه‌یری به‌رژه‌وه‌ندییه‌كانی حیزبه‌كه‌ی و كوردیشی بكردایه‌و له‌ پێناوی تموحی بوون به‌ سكرتێری یه‌كێتیدا كه‌ ئه‌سته‌مه‌ ئه‌و ئاواته‌ی بێـته‌دی، نه‌بوایه‌ته‌ لایه‌نگری لایه‌ك به‌ دژی لایه‌كی تر.
كاتی خۆی كاك به‌ختیار زۆر ره‌خنه‌ی له‌ رۆڵی ره‌حمه‌تی فه‌ره‌نسۆ هه‌ریری ده‌گرت له‌وه‌ی كه‌ ده‌روێشانه‌ له‌ خزمه‌تی بنه‌ماڵه‌ی بارزانیدا بوو، ئێستا به‌شێك له‌ خه‌ڵك پێی وایه‌ كاك به‌ختیار خۆی ئه‌و رۆڵه‌ ده‌بینێت، هه‌روه‌ك كه‌سانێكی تریش هه‌ن پێیان وایه‌ كاك به‌ختیار له‌ گرێی ئه‌و سوكایه‌تیه‌ رزگاری نه‌بووه‌ كه‌ سه‌ركردایه‌تی یه‌كێتی گرتوویانه‌و زیندانیان كردووه‌و پاكانه‌یان پێنووسیووه‌، بۆیه‌ ئێستا ده‌شێ پاڵنه‌ری تۆڵه‌ یه‌كێك له‌ نه‌ستیدا له‌ جوڵێنه‌ره‌كانی ئه‌و بێت بۆ ئه‌م شێوازه‌ كاركردنه‌ به‌ تایبه‌تی كه‌ كێوی مام جه‌لال و كێوی نه‌وشیروان مسته‌فای له‌به‌رده‌مدا نه‌ماوه‌.
دووه‌م: تا ئێستا كاك به‌ختیار باسی رۆڵی یه‌كێتی ده‌كات له‌ دیموكراسی و ئازادی و داواكاری مافه‌كانی كورددا، به‌ڵام به‌داخه‌وه‌ ئه‌و شێوه‌كاركردنه‌ی به‌ڕێزیان و به‌شێك له‌سه‌ركردایه‌تی یه‌كێتی، یه‌كێتی له‌و خاسێتانه‌ رووت كردووه‌ته‌وه‌.
ساده‌ترین شت مه‌سه‌له‌ی حكومه‌تی خۆجێیه‌تی سلێمانیه‌ كه‌ جارێكی تر سیاسه‌تی شكستخواردووی په‌نجا به‌ په‌نجا بۆ به‌ڕیوه‌بردنی ئه‌م شاره‌یه‌ زیندوو كرایه‌وه‌، هه‌روه‌ك له‌و كاته‌وه‌ كاك به‌ختیارو كه‌سانێكی تری وه‌كو به‌ڕێزیان به‌ ناوی عه‌رشی بابان و خانه‌قای مه‌ولاناو بیلاتی شێخ مه‌حموده‌و ئه‌زموونی شیعری نالی و گۆران و ته‌نویری پیره‌مێردو ئاینده‌بینی شه‌ریف پاشاو ره‌فیق حیلمیه‌وه‌ قسه‌ده‌كه‌ن، سلێمانی له‌ پاشه‌كشه‌دایه‌.
هه‌روه‌ك شێوازی به‌ڕێوه‌بردنی یه‌كێتی له‌لایه‌ن ئه‌م به‌ڕێزانه‌وه‌، یه‌كێتی له‌ رابه‌رایه‌تی پرۆژه‌ی نه‌ته‌وایه‌تی رووتكردووه‌ته‌وه‌و ئێستا له‌سه‌ر ئاستی ناوه‌وه‌و ده‌ره‌وه‌ش له‌و رووه‌وه‌ كه‌وتووه‌ته‌ په‌راوێزه‌وه‌.
كاتی خۆی برایانی یه‌كێتی باسی ئه‌وه‌یان ده‌كرد كه‌ گوایه‌ سه‌ركردایه‌تی پارتی گوتوویانه‌ ئه‌گه‌ر شۆڕمان پێنه‌كرێت خۆ پێمان تێكده‌درێت، به‌ داخه‌وه‌ ئێستا زۆرینه‌ی ته‌سریحدانه‌كانی كاك به‌ختیار له‌م بارودۆخه‌ی ئێستای كوردستاندا هه‌مان مانا و ده‌لاله‌تی هه‌یه‌.
سێیه‌م: به‌هه‌رحاڵ من ده‌زانم یه‌كێتی نیشتیمانی هه‌رچۆن له‌ درووستبوونیه‌وه‌ له‌ خه‌باتی پێشمه‌رگایه‌تیدا پێشه‌نگ بووه‌، ئێستاش زۆرینه‌ی فه‌رمانده‌كانی پێشمه‌رگه‌ی له‌وانه‌ی ناو مه‌كته‌بی سیاسیه‌وه‌ بۆ خواره‌وه‌ تا ده‌گاته‌ ئاستی پێشمه‌رگه‌ی ئاسایی خه‌باتگێڕو تێكۆشه‌رن و بۆ به‌رگری له‌ گه‌ل و نیشتیمان جیگای بچوكترین گومان نین، به‌ڵكو به‌ پێچه‌وانه‌وه‌ ئه‌وان تاكه‌ به‌شن له‌ ناو یه‌كێتیدا كه‌ رۆڵی پێشه‌نگیی خۆیانیان ون نه‌كردووه‌، واته‌ ئه‌وه‌ی له‌ ناو یه‌كێتیدا كه‌ تاكو ئێستا به‌ زیندوویی ماوه‌ته‌وه‌ پێشمه‌رگه‌یه‌، به‌ڵام زۆرجاریش خراپی به‌ڕێوه‌بردنی یه‌كێتی له‌سه‌ره‌وه‌ ره‌نجی ئه‌وانیش به‌ باده‌دات، بۆ نموونه‌ به‌ خوێنی پێشمه‌رگه‌ جه‌له‌ولاو سه‌عدیه‌ رزگار ده‌كرێت، كه‌چی وه‌ك باس ده‌كرێت به‌شێك له‌سه‌ركردایه‌تی یه‌كێتی ئه‌و ناوچانه‌ ته‌سلیمی میلیشیا شیعه‌كان ده‌كات.
بێگومان ئه‌وه‌ی پێشتر ده‌بینرا ئه‌وه‌بوو ململانێكانی ناو یه‌كێتی زیاتر زیانه‌كه‌ی بۆ حیزبه‌كه‌ خۆی بێت، به‌ڵام ئێستا ئاسه‌واری نێگه‌تیفی ئه‌و ململانێیانه‌ ده‌گوازرێته‌وه‌ بۆ سه‌ر ئاستی نه‌ته‌وه‌یی و نیشتیمانی به‌ تایبه‌تی له‌و رووه‌وه‌ كه‌ ده‌شێ ته‌سریحاتی له‌وجۆره‌ی ئه‌م دواییه‌ی كاك به‌ختیار له‌ ساڵیادی شه‌هید كاكه‌ عه‌به‌دا، به‌ ئاشكرا دیاره‌ هه‌ڕه‌شه‌یه‌ بۆ درووستكردنی هێزی یه‌كگرتووی پێشمه‌رگه‌ی كوردستان، ببێته‌ هۆی ئاژاوه‌و تێكچوونی زیاتر له‌بارودۆخی ناوخۆی كوردستاندا.
هه‌روه‌ك لایه‌نگری یه‌كێتی له‌ هه‌ندێ له‌ بۆچوونه‌كانی پێكه‌كه‌ سه‌باره‌ت به‌ كردنی شه‌نگاڵ به‌ كانتۆن ئه‌ویش نه‌ك له‌گه‌ڵ یه‌كێتی و یه‌كبوونی نه‌ته‌وه‌یدا ناكۆكه‌ به‌ڵكو دژی ساده‌ترین بنه‌مای یه‌كێتی نه‌ته‌وه‌ییه‌، كه‌ بێگومان ئه‌م جۆره‌ هه‌ڵوێستانه‌ هه‌م زیانی بۆ یه‌كێتی كورد له‌ باشووری كوردستان ده‌بێت، هه‌م به‌ نه‌رێنی به‌سه‌ر یه‌كێتی نیشتیمانیدا ده‌شكێتـه‌وه‌.
كه‌ هیوادارم سه‌رئه‌نجام لۆجیكی عه‌قڵ لای كاك به‌ختیارو كۆی سه‌ركردایه‌تی یه‌كێتی باڵاده‌ست بێت و عه‌قڵی ته‌سكی حیزبایه‌تی و به‌رژه‌وه‌ندی ته‌سكی تاكه‌كه‌سی و رقی رابردوو، ململانێی نادرووست به‌لاوه‌ بنێن و هه‌م كێشه‌كانی خۆیان چاره‌سه‌ر بكه‌ن هه‌م به‌ زوترین كات مه‌رجه‌عێك بۆ حیزبه‌كه‌یان درووست بكه‌ن چونكه‌ ئه‌م شێ,ه‌ پاشا گه‌ردانیه‌ی ئێستا پێش هه‌موو كه‌سێ زیان له‌ یه‌كێتی خۆی ده‌دات و ئه‌گه‌ر ئه‌وان چاره‌سه‌ری كێشه‌كانیان نه‌كه‌ن بێگومان مێژوو ره‌حمیان پێناكات و وه‌كو چۆن چه‌ندین حیزب و رێكخراوو بزووتنه‌وه‌و ته‌نانه‌ت شارستانێتیشی فڕێداوه‌ته‌ په‌راوێزه‌وه‌، ئه‌وانیش فڕێ ده‌دات په‌راوێزه‌وه‌.
