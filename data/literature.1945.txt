جارێكیان هه‌روا به‌ڕێكه‌وت له‌ مه‌مله‌كه‌تی وێرانستان رێم كه‌وته‌ شارێك ناوی ( باغی داد ) بوو,  بینیم خه‌ڵكی ئه‌م شاره‌ سه‌رجه‌م ئۆقره‌یان لێبڕاوه‌ , كۆڵان به‌ كۆڵان به‌دووای ون بوونی خۆیاندا ده‌گه‌ڕێن , كه‌به‌نێو شاریشدا كه‌وتمه‌ گه‌ڕان , دیتم حه‌وشه‌و ماڵه‌كان,  له‌ترسی وه‌ڕه‌ وه‌ڕی كۆمه‌ڵێ‌ سه‌گی به‌ره‌ڵادا ده‌رگاو په‌نجه‌ره‌كانیان خڕ داخراوه‌ , هه‌روه‌ها له‌م شاره‌ سه‌یرو سه‌مه‌ره‌یه‌دا دیتم دایكه‌كان له‌جیاتی شیر ئاوی تۆڵه‌كه‌ ده‌ده‌ن به‌مناڵه‌كانیان , ئه‌وانیش له‌جیاتی گڕوگاڵ جنێو به‌باوكانیان ده‌ده‌ن , باوكه‌ داماوه‌كانیش كڵاوی چه‌رمه‌سه‌ری و نه‌گبه‌تیان له‌سه‌ر نابوو, له‌ سێبه‌ری بێكاریدا جگاره‌ی مات ومه‌لولیان ده‌كێشا , سه‌رسامیش له‌لای خۆیه‌وه‌ ناوی ئه‌م شاره‌ی نابوو شاری( عاجه‌واتی) كاتێكیش زیاتر به‌ گه‌ڕه‌كه‌ ده‌وڵه‌مه‌نه‌كانی  شاردا كه‌وتمه‌ خولانه‌وه‌ , دیتم كه‌روێشكه‌كان په‌ڕوباڵیان لێ‌ په‌یدا بووه‌ و داڵه‌ كه‌ر خۆره‌كانیش بوونه‌ته‌ مریشك و قه‌لی ماڵان , هه‌ر له‌و گه‌ڕه‌كانه‌دا به‌سه‌ربانی ماڵێكه‌وه‌ مه‌ڕێكم بینی ببووه‌ مه‌یمون,  كه‌متیارێك ببووه‌ پڵنگ, ڕێویه‌ك ببووه‌ شێر و قه‌له‌ڕه‌شێك ببووه‌ تاوس, كاتێكیش چوومه‌ ده‌روازه‌ی شار بینیم ترۆمبێله‌كان به‌كۆمه‌ڵ ته‌ویله‌ دراون , ئه‌سپه‌كانیش ورگن  ورگن  وه‌ك قورباقه‌یان لێ‌ هاتووه‌ , كۆمه‌ڵێ‌ بزنه‌ لاتیش خۆیان كردبوو به‌ئاسك , ژاژه‌ڵانی و زه‌لكاو گه‌نداوه‌كانیش ببوونه‌ سه‌یرانگاو هاوینه‌ هه‌وار , چه‌قه‌ڵ و به‌رازه‌كانیش چایخانه‌و مه‌یخانه‌یان تێدا دانابوو , له‌و هاوینه‌ هه‌واره‌ ده‌گمه‌نه‌دا , یه‌كانه‌یه‌كی ڕیش درێژ ده‌ستی به‌ڕیشه‌كه‌یدا ده‌هێناو باسی هه‌وارگه‌ی دڵانی بۆ ماڵۆسێك ده‌كرد, كه‌ ئه‌م هه‌موو دیمه‌نه‌ ناقۆڵاو قه‌ڵپ و هه‌ڵه‌گه‌ره‌م بینی له‌به‌ر خۆمه‌وه‌ گوتم: ئه‌ی باشه‌ به‌رپرسانی شار له‌كوێن ؟ هه‌رئه‌وه‌نده‌م خۆش بوو وام گوت,   شامیلۆ وته‌نی گه‌رده‌لوولی پێكه‌نین هه‌ستا , هه‌ر مانكا بوو قوفیلكه‌ی ده‌می ده‌ته‌قان و ده‌یفیشكان و ده‌یگوت : كابرا ده‌ڵێی كوێری ئه‌ی ئێمه‌ نابینی؟!
به‌درێژی ڕێگاكه‌ , به‌م به‌رو به‌و به‌ردا, ئاپوره‌ی جه‌ماوه‌ری خه‌ڵك له‌ پێشوازی شاندی نوێنه‌راندا ڕاوه‌ستابوون , گه‌وره‌و بچوك،  یه‌كی چه‌پكه‌ گوڵێكیان هه‌ڵگرتبوو به‌ ته‌مابوون شانده‌كه‌ گوڵباران كه‌ن ,  قرچه‌ قرچی گه‌رمای هاوین بوو , خه‌ڵك له‌ژێر مه‌شریقانی ئه‌و گه‌رمایه‌دا وه‌ستابوون چاوه‌ڕوانی نوێنه‌ركانیان بوون تا هه‌واڵی خۆشیان بۆ بێننه‌وه‌و مژده‌ی به‌ده‌ست هێنان و مسۆگه‌ر كردنی مافه‌كانیان بده‌نێ‌, پایته‌خت له‌و جێگایه‌وه‌ به‌ده‌یان كیلۆمه‌تر دوور بوو , چه‌ند رۆژێك بوو جه‌ماوه‌ر له‌چاوه‌ڕوانی  مزگێنی خۆشدا  بوو,  ئه‌ورۆژه‌ هه‌واڵیان به‌ خه‌ڵك دابوو كه‌وا  نوێنه‌ره‌كانیان له‌ پایته‌خت گه‌ڕاونه‌ته‌وه‌و له‌ ده‌روازه‌ی شارن , بۆیه‌ ئه‌وانیش چووبوونه‌ پێشوازیان , خه‌ڵك چاوه‌ڕوانی بینینی ڕومه‌تی گه‌شاوه‌ی شانده‌كه‌ بوون  كه‌ قافڵه‌ی نوێنه‌ران قه‌ڵه‌وڵی كردی به‌شارداو گه‌شتنه‌ نێوجه‌ماوه‌ر , خه‌ڵك له‌ جۆش وخرۆشی هاواركردن وفیكه‌ كێشاندا بوون , له‌پڕ دیتیان زۆربه‌ی نوێنه‌ران خه‌وه‌نوچكه‌یانه‌ و باوێشك ده‌ده‌ن , ئاپوره‌ی خه‌ڵك  كه‌ ئه‌م دیمه‌نه‌ سه‌یره‌یان بینی سه‌ریان سوڕماو هاواریان كرد: ئه‌مه‌ چیه‌ ئه‌ی نوێنه‌رانی گه‌ل؟ یه‌كێك له‌ نوێنه‌ران له‌ جامی سه‌یاره‌كه‌وه‌ سه‌ری ده‌رهێنا و به‌ ده‌م و چاوێكی خواڵوه‌وه‌ به‌ خه‌ڵكه‌كه‌ی وت: هه‌موو شتێكمان ته‌واو كرد, هه‌موو مافه‌كانتانمان مسۆگه‌ر كرد ,  كاری ئێمه‌ ته‌واو ده‌چینه‌وه‌ ده‌خه‌وین,  چه‌ند رۆژه‌ نه‌خه‌وتوین  خه‌ومان دێت ؟!
به‌بیستنی ئه‌م قسه‌ سه‌یره‌ خه‌ڵك له‌ جیاتی گوڵ باران كه‌وتنه‌ تانه‌باران كردنی نوێنه‌ره‌كانیان , له‌و ده‌مه‌دا نوێنه‌رێكی خه‌واڵوی تر  سه‌ری ده‌رهێناو  به‌ خه‌ڵكه‌كه‌ی گوت: ئه‌ڕێ‌ كاكه‌ ئێوه‌ بۆ بڕواناكه‌ن كه‌ ئێمه‌ هه‌موو شتێكمان بۆ ته‌واو كردوون !
باِوه‌ڕكه‌ن هه‌ر هه‌مووی یه‌ك دوو مانگێكی تره‌ هه‌موو هه‌قوو هقوقه‌كانتان به‌ باری سه‌ت وچل  وشتر ده‌گاته‌ ده‌ستتان !
