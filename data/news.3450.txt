پێنج لایه‌نه‌ سیاسیه‌كانی‌ هه‌رێمی‌ كوردستان، جه‌خت له‌سه‌ر پێویستی‌ یه‌كخستنی‌ ناوماڵی‌ كورد‌و چاره‌سه‌ری‌ ناكۆكییه‌كانی‌ نێوانیان ده‌كه‌نه‌وه‌، به‌بێ‌ ئه‌وه‌ی‌ هیچ پرۆژه‌یه‌كی‌ نوێیان بۆ ئه‌و هه‌نگاوه‌ هه‌بێت‌و ئاماده‌كارییان بۆ ئه‌نجامدانی‌ كۆبونه‌وه‌یه‌كی‌ هاوبه‌ش كردبێت.
وروژاندنه‌وه‌ی‌ پرسی‌ كێشه‌ ناوخۆییه‌كانی‌ هه‌رێم‌و پێداگری‌ حزبه‌كان له‌سه‌ر چاره‌سه‌ركردنیان، هاوكاته‌ له‌گه‌ڵ ئه‌و دۆخه‌ سیاسیه‌ ئاڵۆزه‌ی‌ عێراقی‌ به‌گشتی‌ گرتووه‌ته‌وه‌‌و كورد ده‌یه‌وێت، به‌یه‌كده‌نگی‌ مامه‌ڵه‌ی‌ له‌گه‌ڵدا بكات.
بڕیاره‌ كۆبوونه‌وه‌ی‌ لایه‌نه‌كان‌و فراكسیۆنه‌ كوردستانیه‌كان، بۆ تاوتوێكردنی‌ پرسی‌ گه‌رانه‌وه‌ بۆ به‌غدا یان نه‌گه‌رانه‌وه‌ ئه‌نجامبدرێت، به‌ڵام به‌بێ‌ باسی‌ ململانێ‌ ناوخۆییه‌كان له‌كاتێكدا هه‌رێمی‌ كوردستان نزیكه‌ی‌ حه‌وت مانگه‌ له‌‌قه‌یرانێكی‌ ئاڵۆزی‌ سیاسیدایه‌، دوای‌ ئه‌وه‌ی‌ سه‌رۆكی‌ په‌رله‌مان‌و وه‌زیره‌كانی‌ گۆڕان له‌‌حكومه‌ت دوورخرانه‌وه‌ به‌بڕیاری‌ پارتی‌ له‌ 12/10ی‌ ساڵی‌ رابردوودا.
شۆڕش حاجی به‌خه‌دان-ی‌ وت "وه‌ك چۆن پارتی له‌سه‌ر هه‌ڵوێسته‌كانی خۆی به‌رده‌وامه‌ له‌سه‌ر پرسه‌كانی ناوخۆی هه‌رێم، كه‌ یه‌كێك له‌ سه‌ركرده‌كانی پارتی له‌م رۆژانه‌دا رایگه‌یاندوه‌ زه‌ره‌یه‌ك هه‌ڵوێستمان به‌رانبه‌ر بزوتنه‌وه‌ی گۆران نه‌گۆراوه‌، ئێمه‌ ناڵێین هه‌ڵوێستمان به‌رانبه‌ر به‌پارتی، ده‌ڵێین هه‌ڵوێستمان به‌رانبه‌ر بارودۆخی هه‌رێمی كوردستان نه‌گۆڕاوه‌".
ئه‌وه‌شی‌ خسته‌ڕو كه‌ داوایان كردوه‌ "شه‌رعیه‌ت بگه‌رێته‌وه‌" بۆ په‌رله‌مان‌و حكومه‌تی هه‌رێمی كوردستان، گفتوگۆش له‌پێناوی گفتوگۆ نه‌بێت.
وته‌بێژی گۆڕان ده‌ڵێت "ده‌بێت گفتوگۆ له‌پێناو هاوسه‌نگی هێز بێت، هه‌مو لایه‌ك درك به‌وه‌ بكه‌ن كه‌ هه‌مو لایه‌ك شه‌ریكی راسته‌قینه‌بێت بۆ چاره‌سه‌ری كێشه‌كان، ئێمه‌ له‌سه‌ر بنه‌مای گه‌ڕانه‌وه‌ی شه‌رعیه‌ت ئاماده‌یمان تێدایه‌ بۆ گفتوگۆكردن".
له‌دوای‌ گه‌ڕانه‌وه‌ی‌ نه‌وشیروان مسته‌فا بۆ هه‌رێم، چه‌ندین كۆبوونه‌وه‌ی‌ لایه‌نه‌كان ئه‌نجامدراوه‌ بۆ تاووتوێكردنی‌ چاره‌سه‌ری‌ كێشه‌كانی‌ هه‌رێم، به‌ڵام دیارنیه‌ كۆبونه‌وه‌ی‌ هاوبه‌ش بۆ ئه‌و مه‌به‌سته‌ ئه‌نجامده‌درێت یان نا.
نه‌وشیروان مسته‌فا، رێكخه‌ری‌ گشتی‌ بزووتنه‌وه‌ی‌ گۆڕان له‌‌مانگی ‌نۆی‌ 2015دا چووه‌ ئه‌وروپا‌و دوای‌ زیاتر له‌ حه‌وت مانگ، كۆتایی‌ مانگی‌ رابردوو گه‌ڕایه‌وه‌ هه‌رێمی‌ كوردستان.
عه‌لی حوسێن، ئه‌ندامی سه‌ركردایه‌تی پارتی به‌خه‌ندانی‌ وت، له‌گه‌ڵ یه‌كێتی هه‌مو ئه‌و خاڵانه‌ گفتوگۆی له‌سه‌ر ده‌كرێت بۆ كاراكردنه‌وه‌ی دۆخی سیاسی هه‌رێم هه‌یه‌، پێشیوابوو گۆڕان به‌شێكه‌ له‌پرۆسه‌ی سیاسی‌و ده‌بێت مامه‌ڵه‌ له‌گه‌ڵدا بكرێت.
ئه‌وه‌شی‌ رونكرده‌وه‌ كه‌ گفتوگۆكانیان بۆ چاره‌سه‌ری‌ كێشه‌كان له‌گه‌ڵ یه‌كێتیدا به‌رده‌وامی‌ هه‌یه‌، "كۆمه‌ڵێك شتمان  له‌گه‌ڵ یه‌كێتی باس كردوه‌ له‌سه‌ر چۆنێتی كاراكردنه‌وه‌ یان پێشه‌وه‌ چونی پرۆسه‌ی سیاسی له‌كوردستان، له‌گه‌ڵ ئه‌وه‌ین هه‌موو هێزه‌كان كۆببینه‌وه‌ دابنیشین بۆچونی هه‌مو لایه‌ك وه‌ربگرین".
ئاشكراشیكرد به‌مزوانه‌ له‌گه‌ڵ یه‌كێتی داده‌نیشنه‌وه‌ "چی مه‌سڵه‌حه‌ت بێت بۆ پێكه‌وه‌ كاركردن له‌گه‌ڵ هه‌مو لایه‌نه‌كان وا ده‌كه‌ین".
پارتی‌ له‌دوای‌ روداوه‌كانی‌ ئۆكتۆبه‌ری‌ ساڵی‌ رابردووه‌وه‌، داوایده‌كرد گۆڕان یوسف محه‌مه‌د، سه‌رۆكی‌ په‌رله‌مانی‌ كوردستان له‌پۆسته‌كه‌ی‌ لابدات‌و ئه‌وكات گفتوگۆ بكه‌ن بۆ دوباره‌ رێككه‌وتن له‌سه‌ر چۆنیه‌تی‌ به‌شدارییان له‌حكومه‌ت.
له‌به‌رانبه‌ریشدا، گۆڕان پێداگری‌ ده‌كرد له‌سه‌ر گه‌ڕانه‌وه‌ وه‌زیره‌كان‌و سه‌رۆكی‌ په‌رله‌مان بۆ سه‌ر كاره‌كانیان، پێش ئه‌نجامدانی‌ هه‌ر گفتوگۆیه‌ك.
سه‌ركردایه‌تییه‌كه‌ی‌ پارتی‌ ده‌ڵێت دۆخی ئێستای كوردستان به‌رپرسیارێتی‌ زیاتر ده‌خاته‌ سه‌ر شانی سه‌ركرده‌و لایه‌نه‌ سیاسیه‌كان، جه‌ختیكرده‌وه‌ "پارتی ئاماده‌یی تێدایه‌ بۆ گفتوگۆ له‌گه‌ڵ هه‌مو لایه‌نه‌كان‌و رێككه‌وتن ئه‌نجامبدرێت، به‌ڵام رێكه‌وتنه‌كان له‌سه‌ر بنه‌مای نیشتمانی‌و دروست بێت‌و هه‌وڵنه‌ده‌ین رێكه‌وتنه‌كان تێكبده‌ین".
پارتی‌ له‌گه‌ڵ یه‌كێتیدا تائێستا چه‌ند جارێك كۆبوونه‌ته‌وه‌، ده‌رباره‌ی‌ كێشه‌ ناوخۆییه‌كانی‌ هه‌رێمی‌ كوردستان‌و په‌یوه‌ندی‌ نێوانیان، پێشتریش هه‌ردولا رایانگه‌یاندبوو كه‌ له‌گفتوگۆكانیاندا هه‌وڵده‌ده‌ن رێگاچاره‌ی‌ گونجاو بۆ ناكۆكیه‌كانی‌ نێوان لایه‌نه‌كان گه‌ڵاڵه‌ بكه‌ن.
سه‌عدی ئه‌حمه‌د پیره‌، ئه‌ندامی مه‌كته‌بی سیاسی یه‌كێتی نیشتمانی كوردستان له‌لێدوانێكدا بۆ خه‌ندان وتی‌، كۆبونه‌وه‌ بڕیاره‌ بكرێت بۆ به‌رچاوڕونی فراكسیۆنه‌ كوردیه‌كان له‌باره‌ی‌ ئه‌وه‌ی‌ بچنه‌وه‌ یان نه‌چنه‌وه‌ بۆ به‌غدا، هه‌روه‌ها هه‌رێمی كوردستان چی له‌عێراق داوا ده‌كات.
نوێنه‌رانی‌ كورد له‌به‌غدا، هه‌فته‌ی‌ رابردوو بڕیاریاندا نه‌چنه‌وه‌ به‌غدا تاوه‌كو دۆخی‌ عێراق رونده‌بێته‌وه‌‌و سه‌ركردایه‌تی‌ سیاسی‌ كورد داوایان لێده‌كات، دوای‌ ئه‌وه‌ی‌ له‌لایه‌ن خۆپیشانده‌رانه‌وه‌ روبه‌روی‌ هێرش بونه‌وه‌ كاتێك كه‌ چونه‌ ناو باڵه‌خانه‌ی‌ په‌رله‌مانه‌وه‌ له‌ناوچه‌ی‌ سه‌وز.
ته‌قینه‌وه‌ی‌ ململانێ سیاسییه‌كانی‌ ‌ساڵی‌ رابردوو، هاوكاتبوو له‌گه‌ڵ كۆبوونه‌وه‌ پێنج قۆڵییه‌كان بۆ چاره‌سه‌ری‌ پرسی‌ سه‌رۆكایه‌تی‌ هه‌رێمی‌ كوردستان، به‌ڵام دوای‌ نۆ كۆبوونه‌وه‌ لایه‌نه‌كان نه‌گه‌یشتن به‌ڕێككه‌وتن‌و تائێستا پرسه‌كه‌ به‌هه‌ڵپه‌سێردراوی‌ ماوه‌ته‌وه‌.
یاریده‌ده‌ری ئه‌مینداری گشتی یه‌كگرتووی‌ ئیسلامی‌ ئاشكرایده‌كات، تائێستا پارتی‌و گۆڕان له‌سه‌ر هه‌ڵوێسته‌كانی خۆیان سورن، بۆیه‌ بۆنی چاره‌سه‌ركردنی كێشه‌كانی بارودۆخی هه‌رێم ناكرێت تا پرۆژه‌یه‌ك پێشكه‌شی لایه‌نه‌كان بكه‌ن.
محه‌مه‌د ئه‌حمه‌د له‌لێدوانێكدا بۆ خه‌ندان ئاماژه‌ی به‌وه‌كرد، له‌سه‌ر بارودۆخی ناوخۆ هیچ شتێك نیه‌‌و "پارتی‌و گۆران ئاماده‌ییان تێدا نیه‌‌و بۆنی ئه‌وه‌ش ناكه‌ین، چونكه‌ سورن له‌سه‌ر مه‌واقیفه‌كانی خۆیان، ته‌نانه‌ت دوای گه‌رانه‌وه‌ی نه‌وشیروان مسته‌فا ده‌رگایه‌ك شك نابه‌ین كه‌ پرۆژه‌یه‌ك پێشكه‌ش به‌لایه‌نه‌كان بكه‌ین بۆ ئه‌وه‌ی جارێكی تر كۆبونه‌وه‌ی پێنج لایه‌نه‌كه‌ ده‌ستپێبكاته‌وه‌".
یه‌كگرتوو، له‌دوای‌ گه‌ڕانه‌وه‌ی‌ نه‌وشیروان مسته‌فا داوایه‌كی‌ بۆ كۆبونه‌وه‌ی‌ لوتكه‌ی‌ لایه‌نه‌كان خسته‌ڕو، ئه‌وه‌ش هاوكات بوو له‌گه‌ڵ ده‌ستپێشخه‌رییه‌كی‌ بزووتنه‌وه‌ی‌ ئیسلامی‌، بۆ ده‌ستپێكردنه‌وه‌ی‌ گفتوگۆكانی‌ چاره‌سه‌ری‌ كێشه‌ ناوخۆییه‌كان، به‌ڵام سه‌رباری‌ چه‌ندین كۆبونه‌وه‌ له‌وباره‌یه‌وه‌ تائێستا هیچ ئه‌نجامێكی‌ دیارنییه‌.
یاریده‌ده‌ری ئه‌مینداری گشتی یه‌كگرتوو جه‌ختیكرده‌وه‌ كه‌ به‌چركه‌ كۆڵیان له‌فشار دروستكردن نه‌داوه‌ بۆ یه‌كڕیزی‌و یه‌كخستی ناوماڵی كورد، "ئه‌گه‌ر بزانین ده‌رگایه‌ك هه‌یه‌ بۆ ئه‌وه‌ی هه‌مو لایه‌نه‌كان له‌سه‌ر مێزی گفتوگۆ كۆبكه‌ینه‌وه‌‌و ئه‌و بارودۆخه‌ سیاسیه‌ ئاسۆی چاره‌سه‌ركردنی هه‌بێت پرۆژه‌ پێشكه‌ش ده‌كه‌ین".
لایه‌نه‌ سیاسیه‌كاتی‌ تر (یه‌كێتی‌، كۆمه‌ڵ‌و یه‌كگرتوو)، هاوشێوه‌ی‌ گۆڕان له‌ناوه‌ڕاستی‌ ساڵی‌ رابردوودا پرۆژه‌‌و بۆچوونی‌ هاوبه‌شیان هه‌بوو بۆ هه‌مواركردنه‌وه‌ی‌ یاسای سه‌رۆكایه‌تی‌ هه‌رێم، كه‌ كاردانه‌وه‌ی‌ توندی‌ پارتی‌ لێكه‌وته‌وه‌.
له‌دوای‌ دورخستنه‌وه‌ی‌ گۆڕان له‌ده‌سه‌ڵات، ئه‌و لایه‌نانه‌ وه‌زیره‌كانیان له‌كابینه‌ی‌ هه‌شته‌می‌ حكومه‌تدا مانه‌وه‌‌و جه‌ختیان له‌وه‌كرده‌وه‌، ده‌یانه‌وێت ببنه‌ به‌شێك له‌چاره‌سه‌ری‌ ناكۆكییه‌ سیاسیه‌كان به‌تایبه‌تی‌ له‌نێوان گۆڕان‌و پارتیدا.
محه‌مه‌د حه‌كیم، ئه‌ندامی ده‌سته‌ی كارگێڕی ئه‌نجومه‌نی سیاسی كۆمه‌ڵی ئیسلامی ده‌ڵێت هیچ پرۆژه‌یه‌كی نوێ نیه‌ بۆ كۆبونه‌وه‌ی پێنج لایه‌نه‌كه‌.
هه‌روه‌ها بۆ خه‌ندان رونیكرده‌وه‌ "بۆ خۆی كۆبونه‌وه‌ی نوێنه‌ری پێنج لایه‌نه‌كه‌ له‌گه‌ڵ فراكسیۆنه‌كان –ده‌رباره‌ی‌ كێشه‌كان له‌گه‌ڵ به‌غدا- هه‌وڵێكی باشه‌ بۆ ئه‌وه‌ی ده‌رفه‌ت‌و پرۆژه‌یه‌ك هه‌بێت بۆ ئه‌وه‌ی له‌سه‌ر بارودۆخی ناوخۆی‌ هه‌رێمی‌ كوردستانیش كۆمانبكاته‌وه‌".
له‌ماوه‌ی‌ حه‌وت مانگی‌ رابردوودا، ده‌یان ده‌ستپێشخه‌ری‌ ناوخۆیی‌‌و ده‌ره‌كی‌ كراون، به‌ڵام هیچ كامێكیان نه‌یتوانیووه‌ كۆتایی‌ به‌ئاڵۆزیه‌ سیاسیه‌كان بهێنێت‌و لایه‌نه‌كان ناكۆكه‌كان پێكه‌وه‌ له‌سه‌ر مێزی‌ گفتوگۆ كۆبكاته‌وه‌.
كێشه‌ی‌ نێوان لایه‌نه‌كان، دانیشتنه‌كانی‌ په‌رله‌مانی‌ كوردستانی‌ به‌ته‌واوی‌ په‌كخستووه‌، كاریگه‌ری‌ راستوخۆیشی‌ له‌سه‌ر كاروباره‌كانی‌ حكومه‌ت هه‌بووه‌ وه‌كو به‌رپرسان خۆیان ئاماژه‌ی‌ پێده‌كه‌ن، تائێستاش دیارنیه‌ دۆخی‌ ئێستای‌ هه‌رێمی‌ كوردستان تاكه‌ی‌ به‌هه‌ڵپه‌سێردراوی‌ ده‌مێنێته‌وه‌.
