روانگه‌ی هه‌ردوو وه‌زاره‌تی ده‌ره‌وه‌و به‌رگری ئه‌مریكا له‌سه‌ر به‌ره‌و پێشچوونه‌كانی شه‌ڕ دژی داعش له‌عیراق جیاوازه‌، ئه‌وه‌ش هاوكاته‌ له‌گه‌ڵ زیادبوونی ئه‌و بۆچوونانه‌ی‌ كه‌پێیوایه‌ سوپای‌ عیراق هێزێكی‌ ئاماده‌كراونین بۆ كۆنتڕۆڵكردنه‌وه‌ی‌ شاری‌ موسڵ‌ به‌بێ‌ هاوكاری‌ ئه‌مریكا.
له‌كاتێكدا وه‌زاره‌تی ده‌ره‌وه‌ی ئه‌مریكا پێیوایه‌ شه‌ڕی كۆنترۆڵكردنه‌وه‌ی فه‌لوجه‌ له‌ژێر ده‌سه‌ڵاتی داعش به‌ره‌وپێشچوونی باشی به‌خۆوه‌ بینیوه‌، به‌ڵام وه‌زاره‌تی به‌رگری له‌ئاستی خێرایی شه‌ڕه‌كان دڵخۆش نین.
ئیلیزابێت ترۆدۆ، وته‌بێژ به‌ناوی وه‌زاره‌تی ده‌ره‌وه‌ی ئه‌مریكا له‌كۆنفرانسی رۆژنامه‌وانی رۆژانه‌دا، له‌وه‌ڵامی رۆڵی ئه‌مریكا له‌ئۆپه‌راسیۆنی كۆنترۆڵكردنه‌وه‌ی شاری فه‌لوجه‌دا وتی " ئه‌گه‌رچی ئاماده‌نیم له‌سه‌ر ورده‌كارییه‌كانی ئۆپه‌راسیۆنه‌كه‌ روونكردنه‌وه‌ بده‌م، به‌ڵام رۆژی هه‌ینی ئێمه‌ به‌رده‌وامیماندا به‌پشتیوانییه‌كانمان بۆ هێزه‌ ئه‌منییه‌كانی عیراق ئه‌وه‌ش به‌هێرشی ئاسمانی بۆ سه‌ر بنكه‌و باره‌گاكانی داعش‌و یارمه‌تی ستراتیجی‌و راوێژكاریی‌و پێداویستی سه‌ربازی".
ئه‌وه‌ش له‌كاتێكدایه‌ پیته‌ر كوك، وته‌بێژی وه‌زاره‌تی به‌رگری ئه‌مریكا ناڕازیبوونی ئاشتۆن كارته‌ر له‌به‌ره‌وپێشچوونه‌كانی شه‌ڕ دژی داعش له‌عیراق ئاشكراده‌كات.
ئه‌و جیاوازیی بۆچونانه‌ له‌ئاستی خێرایی شه‌ڕ دژی داعش له‌عیراق له‌كاتێكدایه‌، كه‌ ئه‌مه‌ریكا له‌ڕێی بۆردومانی بنكه‌و باره‌گاكانی داعش له‌شه‌ڕی فه‌لوجه‌‌و پێشكه‌شكردنی راوێژ رۆڵی هه‌یه‌، كه‌ له‌ 23ی (ئایار/5) شه‌ڕی فه‌لوجه‌ ده‌ستپێكردووه‌و زۆربه‌ی‌ ناوچه‌كانی ده‌وروبه‌ری قه‌زاكه‌ كۆنترۆڵكراوه‌، به‌ڵام نه‌یانتوانیوه‌ بچنه‌ ناو سه‌نته‌ری قه‌زاكه‌وه‌و شه‌ڕ به‌رده‌وامی هه‌یه‌.
پیته‌ر كوك، ده‌رباره‌ی دواپه‌ره‌سه‌ندنه‌كانی ئۆپه‌راسیۆنی كۆنترۆڵكردنه‌وه‌ی فه‌لوجه‌ وتی "ئێمه‌ رۆژانه‌ هه‌ماهه‌نگیمان هه‌یه‌، له‌وباره‌یه‌وه‌ جه‌نه‌راڵ ماك فارلاندو تیمه‌كه‌ی به‌ورییایه‌وه‌ به‌رده‌وامه‌ له‌گه‌ڵ عیراقییه‌كان له‌سه‌ر شه‌ڕی كۆنترۆڵكردنه‌وه‌ی فه‌لوجه‌".
هه‌روه‌ها وتی "به‌رده‌وام ده‌بین له‌پشتیوانی عیراقییه‌كان، هێرشه‌ ئاسمانییه‌كانیش بۆ سه‌ر بنكه‌و باره‌گاكانی داعش له‌فه‌لوجه‌ نمونه‌یه‌كی ئه‌و پشتیوانی‌و پاڵپشتییانه‌ی ئه‌مریكاو هاوپه‌یمانانن، جه‌نه‌راڵ ماك فارلاند به‌رده‌وام ده‌بێت له‌په‌یوه‌ندییه‌كانی له‌گه‌ڵ عیراقییه‌كان بۆ پێشكه‌شكردنی راوێژكاری پێویست به‌درێژایی شه‌ڕه‌كان".
ئیلیزابێت ترۆدۆ، وته‌بێژ به‌ناوی وه‌زاره‌تی ده‌ره‌وه‌ی ئه‌مریكاش وتی "ئێمه‌ به‌رده‌وام ده‌بین له‌گفتوگۆكانمان له‌گه‌ڵ حكومه‌تی عیراق له‌پێناو دۆزینه‌وه‌ی ده‌ره‌چه‌یه‌ك بۆ رزگاركردنی هاوڵاتینی سڤیلی فه‌لوجه‌"، ئه‌وه‌ش له‌وه‌ڵامی ئه‌و ده‌نگۆیانه‌ی‌ له‌سه‌ر ترسی هاوڵاتیانی مه‌ده‌نی فه‌لوجه‌ له‌میلیشیا شیعه‌كان‌و هه‌وڵه‌كانی حكومه‌تی ئه‌مریكا بۆ پاراستنی گیانی ئه‌و هاوڵاتییه‌ مه‌ده‌نییانه‌.
له‌ماوه‌ی دوو ساڵی ڕابردوودا، هاوكارییه‌كانی ئه‌مریكا بۆ سوپای عیراق له‌ڕووی لۆجێستیه‌وه‌ به‌رزبوونه‌ته‌وه‌، ئه‌وه‌ش به‌هۆی ئه‌وه‌ی كه‌سوپای عیراق روبه‌ڕوی چه‌ندین كێشه‌ی خواردن، ئاوو ته‌قه‌مه‌نی بووه‌ته‌وه‌، نه‌یتوانیوه‌ وه‌ك پێویست پێداویستی‌ هێزه‌كانیان دابینبكه‌ن.
له‌و چوارچێوه‌یه‌شه‌دا ئه‌و بۆچونانه‌ زیادبوون كه‌ پێیوایه‌ به‌بێ هاوكارییه‌كانی ئه‌مریكا ئۆپه‌راسیۆنی هێزه‌ عیراقییه‌كان بۆ كۆنترۆڵكردنه‌وه‌ی موسڵ سه‌ركه‌وتوو نابێت‌و شكست ده‌هێنێت.
پاش چه‌ندین ساڵ راهێنان به‌سوپای عیراق، جۆرێك له‌نائاماده‌یی به‌سوپای عیراقییه‌وه‌ ده‌بینرێت‌و وادیاره‌ كۆنترۆڵكردنه‌وه‌ی موسڵیش دواخراوه‌، به‌وته‌ی‌ چه‌ند سه‌رچاوه‌یه‌ك بۆ رۆژنامه‌ی نیویۆرك تایمز.
به‌پێی نیویۆرك تایمز سوپای‌ عیراق، هێزێكی‌ ئاماده‌كراو نین كۆنتڕۆڵكردنه‌وه‌ی‌ شاری‌ موسڵ‌و به‌بێ هاوكاری‌ ئه‌مریكاش، ئه‌و هه‌نگاوه‌ به‌مه‌حاڵ ده‌زانرێت.
ترۆدۆ، وته‌بێژی وه‌زاره‌تی ده‌ره‌وه‌ی ئه‌مریكا ده‌ڵێت كه‌ ناچێته‌ ناو ورده‌كارییه‌كان له‌سه‌ر پشتیوانی پێداویستی سه‌ربازی بۆ سوپای عیراق، "به‌ڵام دڵنیایی ده‌ده‌م له‌وه‌ی كه‌ئێمه‌ به‌رده‌وام ده‌بین له‌پشتیوانی هێزه‌ ئه‌منییه‌كانی عیراق، وایده‌بینین له‌سێ ساڵی ڕابردوودا به‌ره‌وپێشچوونی باش هه‌بوو، باڵوێزی ئه‌مریكاش له‌عیراق سه‌باره‌ت به‌نوێترین هاوكارییه‌كانی ئه‌مریكا بۆ عیراق روونكردنه‌وه‌ی داوه‌".
هاوكات هه‌ر له‌و كۆنفرانسه‌ رۆژانه‌ییه‌كه‌یدا پیته‌ر كوك، وتی كه‌ وه‌زیری به‌رگری رازییه‌ له‌هه‌وڵه‌كانیان بۆ په‌ره‌پێدانی زیاتری ئاستی شه‌ڕ دژی داعش له‌عیراق‌و سوریا، "به‌ڵام ڕای وایه‌ كه‌ تازووتر كۆتایی به‌م هه‌ڕه‌شه‌یه‌ بهێنرێت باشتره‌، له‌به‌رئه‌وه‌ ده‌بینین كه‌به‌رده‌وام وه‌زاره‌تی به‌رگری جه‌خت له‌سه‌ر زیادكردنی هێرشه‌ ئاسمانییه‌كان‌و هاوكاری سه‌ربازی ده‌كاته‌وه‌ نه‌ك ته‌نها هی ئه‌مریكا، به‌ڵكو فراوانتركردنی هاوكارییه‌ سه‌ربازییه‌كانی هاوپه‌یمانی نێوده‌وڵه‌تی دژی داعش بۆ به‌ره‌وپێشبردنی زیاتری شه‌ڕی دژی داعش".
"ئه‌مه‌ شه‌ڕه‌و شه‌ڕیش پێویستی به‌هه‌ماهه‌نگی وردو وریایانه‌یه‌"، بۆیه‌ وه‌زاره‌تی به‌رگری ده‌یه‌وێت به‌ره‌وپێشچوونه‌كان له‌عیراقدا خێراتر ببینێت، هه‌ر به‌وته‌ی‌ كوك.
بۆ ئه‌وه‌ش ئاماژه‌ به‌كۆبوونه‌وه‌یه‌كی هه‌فته‌ی‌ رابردووی وه‌زیری به‌رگری ئه‌مریكا ده‌كات له‌گه‌ڵ چه‌ند وڵاتێكی ئاسیای وه‌ك سه‌نگافوره‌ تیایدا گفتوگۆیان كردووه‌ ده‌رباره‌ی شه‌ڕی داعش‌و وه‌زیری به‌رگری "مه‌به‌ستییه‌تی خێراتر ئه‌م شه‌ڕانه‌ ئه‌نجامی دڵخۆشكه‌رییان هه‌بێت".
وتیشی "هه‌موو هه‌وڵه‌كانمان ده‌خه‌ینه‌گه‌ڕو له‌گه‌ڵ هاوپه‌یمانانی شه‌ڕی دژه‌ داعش گفتوگۆی به‌رده‌واممان ده‌بێت تابزانین كامه‌ باشترین‌و خێراترینه‌ كه‌بیكه‌ین".
ده‌رباره‌ی ده‌نگۆكانیش له‌سه‌ر ترسی هاوڵاتیانی مه‌ده‌نی فه‌لوجه‌ له‌ میلیشیا شیعه‌كان، و هه‌وڵه‌كانی حكومه‌تی ئه‌مریكا بۆ پاراستنی گیانی ئه‌و هاوڵاتییه‌ مه‌ده‌نییانه‌ ، وته‌بێژی وه‌زاره‌تی ده‌ره‌وه‌ی ئه‌مریكا رایگه‌یاند: حه‌یده‌ر عه‌بادی سه‌رۆك وه‌زیرانی عیراق كۆتایی هه‌فته‌ سه‌ردانی فه‌لوجه‌ی كردو له‌نزیكه‌وه‌ له‌ بارودۆخه‌كه‌ ده‌كۆڵێته‌وه‌ و قسه‌شی له‌سه‌ر پێویستییه‌ خێراكان بۆ پاراستنی گیانی هاوڵاتی مه‌ده‌نی فه‌لوجه‌ كرد، ئێمه‌ش به‌رده‌وام ده‌بین له‌گفتوگۆكانمان له‌گه‌ڵ حكومه‌تی عیراق له‌پێناو دۆزینه‌وه‌ی ده‌ره‌چه‌یه‌ك بۆ رزگاركردنی هاوڵاتینی سڤیلی فه‌لوجه‌".
