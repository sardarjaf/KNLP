مامۆستا سه‌لاَح یه‌كێكه‌ له‌ وه‌رزشوانه‌ دێرینه‌كانى كوردستان، ده‌رچووى كۆلێژى په‌روه‌رده‌ى وه‌رزشى زانكۆى به‌غدایه‌ له‌ ساڵى 1965-1966،له‌ ساڵى 1970 خاوه‌نى برِوانامه‌ى جیهانی له‌ فۆتبۆڵ بووه‌ له‌ ولاَتى ئه‌ڵمانیا، جگه‌ له‌وه‌ى مامۆستاى وه‌رزشى و به‌رِێوه‌به‌رى چالاكى وه‌رزشى دیده‌وانى هه‌ولێر بووه‌ به‌رِێوه‌به‌رى گشتى بووه‌ له‌ وه‌زاره‌تى رۆشنبیرى له‌ كابینه‌ى یه‌كه‌مى حكومه‌تى هه‌رێم، په‌رتوكێكى له‌ باره‌ى فوتبۆڵ چاپكردووه‌، ئێستاش لێپرسراوى نوێنه‌رایه‌تیه‌كانه‌ له‌ لیژنه‌ى ئۆڵۆمپى كوردستان و جێگرى سه‌رۆكى لیژنه‌ى ئۆڵۆمپى كوردستانه‌.
- لیژنه‌ى ئۆڵۆمپى ئه‌داتێكى ته‌نفیزییه‌و سه‌رپه‌رشتى 30 یه‌كێتى وه‌رزشى كوردستان ده‌كات به‌ یارى ئۆڵۆمپى و جیا له‌ ئۆڵۆمپییه‌كان، ئه‌ندامانیان له‌ نێوان 4 - 6 - 8 كه‌سى هه‌یه‌ چالاكییه‌كانى ئه‌و یارییه‌ به‌رِێوه‌ ده‌بن.
توانیومانه‌ له‌سه‌ر ئاستى زۆربه‌ى ئه‌و یه‌كێتیانه‌ له‌ ئاستى پێشكه‌وتووان و لاوان و تازه‌ پێگه‌شتووان چالاكییان بۆ بكه‌ین، له‌ ئاستى كچانیش جموجوڵمان هه‌یه‌، نووسینگه‌ى ئافره‌تانمان هه‌یه‌ به‌هه‌ماهه‌نگى له‌گه‌ڵ یه‌كێتیه‌ وه‌رزشییه‌كان ئه‌و یارییانه‌ ئه‌نجام ده‌ده‌ین، به‌ شانازییه‌وه‌ زۆربه‌ى یاریزانانى كورِوكچى كوردستان له‌ ریزى هه‌ڵبژارده‌ى عێراقدان، له‌ زۆربه‌ى یارییه‌كاندا سىَ ئه‌نداممان له‌ یه‌كێتیه‌كانى عێراق سه‌رۆك یه‌كێتیه‌كانن، زۆر جێگرو ئه‌ندامانیشمان هه‌یه‌، زیاتر له‌ 25 یه‌كێتى وه‌رزشى ئه‌نداممان هه‌یه‌، پێوه‌ندییه‌كى باشمان هه‌یه‌ له‌گه‌ڵ وه‌رزشى عێراق تاكو بتوانین په‌ره‌ به‌و چالاكیانه‌ بده‌ین.
- ئه‌و جۆره‌ قسانه‌ به‌ راست نازانم.
لیژنه‌ى ئۆڵۆمپى عێراقى هیچ كاتێك برِیارى له‌م جۆره‌یان نه‌داوه‌ كه‌وا ئێمه‌ ناشه‌رعى بین، دواترین چاوپێكه‌وتنیش له‌گه‌ڵیان رۆژى 1ی ئه‌م مانگه‌ بوو له‌گه‌ڵ ره‌عه‌د حه‌مودى و چه‌ند براده‌رێكى ترى لیژنه‌ى ئۆڵۆمپى عێراقى له‌ هه‌ولێر سه‌یرى پاڵه‌وانێتى رۆژئاوای ئاسیامان ده‌كرد له‌ یارى بۆكسێن، ئه‌وان گله‌ییان له‌ ئێمه‌ ده‌كرد.
كه‌وا ئێوه‌ چیتان له‌ ئێمه‌ داواكردووه‌ كه‌وا ئێمه‌ بۆتانمان جێبه‌جىَ نه‌كردووه‌، ئێمه‌ زۆر رێزى ئێوه‌ ده‌گرین به‌ چاوێكى به‌رز سه‌یرتان ده‌كه‌ین هه‌رچیتان بوێت بۆ ئێمه‌ قابیلى تاوتوێكردنه‌و ئه‌و قسانه‌ى له‌سه‌ر ئێمه‌ و ئێوه‌ ده‌كرێت دووره‌ له‌ راستى، ئێمه‌ راپۆرتمان داوه‌ته‌ ئه‌وان و له‌سه‌ر هه‌ماهه‌نگى كردنى یه‌كترى و بۆ به‌رِێوه‌بردنى كاره‌كانمان و چاوپێكه‌وتن بكه‌ین له‌ كوردستان و له‌ به‌غدا .
- ئه‌وان له‌ هیچ دانیشتنێك به‌ ئێمه‌یان نه‌وتووه‌ ئێوه‌ ناشه‌رعین، به‌ پێچه‌وانه‌وه‌ له‌ سلێمانى دانیشتنێكمان كرد.
سه‌رۆكى لیژنه‌ى ئۆڵۆمپى خۆى ئاماده‌ى كۆبوونه‌وه‌كه‌بوو هه‌ردوو یه‌كێتى كۆن و نوىَ هاتبوون، پرۆژه‌یه‌كمان پێشكه‌شیان كردبوو به‌ تێرو ته‌سه‌لى ته‌وافوقى له‌نێوان لیژنه‌ى ئۆڵۆمپى عێراقى و یه‌كێتى تۆپى پێی عێراقى كراو ده‌ست خۆشیان له‌ ئێمه‌ كرد.
به‌لاَم له‌ به‌غدا ده‌ستێوه‌ردانى حكومه‌ت هه‌بوو، نه‌یانتوانى كاره‌كه‌ ئه‌نجام بده‌ن.
ئامۆژگارى ئێمه‌ بۆ ئه‌وان ئه‌وه‌ بوو به‌بىَ گه‌رِانه‌وه‌ بۆ مه‌رجه‌عیه‌تى ده‌ولى، گه‌رِانه‌وه‌ بۆ لیژنه‌ى ئۆڵۆمپى و یه‌كێتى تۆپى پىَ، ئێوه‌ ناتوانن هیچ هه‌نگاوێك بنێن، په‌له‌یان كرد له‌دامه‌زراندنى ئه‌م یه‌كێتیه‌ كاتییه‌، به‌لاَم بۆیان نه‌چووه‌ سه‌ر گه‌رِانه‌وه‌ سه‌ر قسه‌كه‌ى ئێمه‌، ئێمه‌ به‌شانازیه‌وه‌ ئه‌م پرِۆژه‌مان پێشكه‌شى ئه‌وان كرد، بۆیه‌ ته‌كید ئه‌كه‌مه‌وه‌ كه‌وا له‌ هیچ ئه‌ندامێك و هیچ كۆبوونه‌وه‌یه‌ك ئه‌وان به‌ ئێمه‌یان نه‌وتووه‌ كه‌وا ئێمه‌ ناشه‌رعین.
- سىَ ساڵ له‌مه‌وبه‌ر كه‌ یه‌كمان گرته‌وه‌، ئێمه‌ له‌ سلێمانى كه‌ لیژنه‌ى ئۆڵۆمپى هه‌بوو نوێنه‌رایه‌تیمان نه‌بوو، یه‌كێتیه‌ وه‌رزشیه‌كان له‌وىَ بوو هیچ پێویستى به‌ نوێنه‌رایه‌تى نه‌بوو، لێره‌ هه‌مان بیرورِامان پێشكه‌ش كردو وتمان هه‌ولێر پێوسیتى به‌وه‌ نییه‌، وتیان ده‌بىَ هه‌بێت، له‌به‌ر هه‌ندێك بابه‌ت وتمان بابمێنىَ، هه‌موو كه‌سێك ده‌بێت سه‌یرى پاشه‌رِۆژ بكات، ئه‌گه‌ر به‌ قوتابیان بڵێم خوێنكارو به‌ ئافره‌ت بڵێم ژن و به‌ لیژنه‌ى ئۆڵۆمپى بڵێم ئه‌نجومه‌نى بالاَى وه‌رزشى، گرنگ ئه‌ویه‌ كاره‌كه‌مان برِوات به‌رِێوه‌، ئه‌و براده‌رانه‌ى بیر له‌وه‌ ده‌كه‌نه‌وه‌ چاره‌نووسى كورد چ له‌ رووى سیاسى و چ له‌ رووى كۆمه‌لاَیه‌تى بێننه‌وه‌ سه‌ر وه‌رزش، ئێمه‌ كێشه‌ دروست بكه‌ین، شتێكى راست نییه‌، من به‌ كه‌مى نازانم، ده‌ڵێم ده‌بێت پاشه‌رِۆژى كورد و وه‌رزش له‌به‌رچاو بگیرێت، مه‌سه‌له‌ى یانه‌ى هه‌ولێر كه‌وا دوورخرایه‌وه‌، زیانى به‌ركه‌وت، ئه‌گه‌ر ئه‌و براده‌رانه‌ دژى یانه‌ى هه‌ولێر بووانه‌، خۆ نه‌جه‌فیش به‌شدار نه‌بوو، به‌لاَم ئه‌وكاته‌ لیژنه‌ى ئۆڵۆمپى عێراقى له‌ ژێر فشار بوون و ئێستاش له‌ ژێر فشارن، به‌لاَم كاره‌كه‌ له‌ راى گشتى له‌ ناو عێراق ئه‌بێت هه‌ڵوێست پێشان بده‌ین.
- واژۆ نه‌كرا، تا له‌ ئه‌ورِوپا بووم گه‌لاَڵه‌ى بیرۆكه‌كه‌مان كردبوو، من كه‌ گه‌رِامه‌وه‌ چووینه‌ بوارى گفتوگۆ له‌گه‌ڵ براده‌ران، لێت ناشارمه‌وه‌ كۆمیته‌ى ته‌نفیزى به‌كۆى گشتى 11كه‌س له‌گه‌ڵ ئه‌وه‌ بوون كه‌وا نه‌مێنێت، به‌لاَم وه‌كو راى خۆم له‌گه‌ڵ ئه‌وه‌بووم كه‌وا بمێنىَ، به‌لاَم ملكه‌چى برِیارى كۆمیته‌ى ته‌نفیزیم.
* راسته‌ له‌ رۆژى جه‌ژنى نه‌ورۆز ئه‌ندامانى لیژنه‌ى ئۆڵۆمپى كوردستان هه‌ر یه‌كه‌و برِى یه‌ك ملیۆن دیناریان وه‌رگرتووه‌؟
- كه‌ ساڵ ته‌واو بوو چالاكى ماوه‌ى سىَ ساڵی ئه‌و ئه‌ندامانه‌ و ماندوبوونیان و ده‌وامكردنیان به‌ نیو مووچه‌، هه‌روه‌ها پاداشت نه‌كردنیان له‌م ماوه‌یه‌دا، بۆیه‌ ئه‌م برِه‌ پاره‌مان پێدان.
- ئه‌وه‌ كارێكى سروشتیه‌ ئه‌و كه‌سانه‌ى له‌ وه‌رزش نزیكن باش ئه‌زانن كه‌ تۆپى پىَ ده‌كه‌یت ژماره‌ى یاریزانان و ده‌سته‌ى كارگێرِى و یانه‌كان زۆرن، به‌لاَم تێنس و شه‌تره‌نج و یارییه‌كانى تر ژماره‌یان كه‌مه‌ و چالاكیه‌كانیان كه‌مه‌، تۆپى پىَ به‌ پله‌ یه‌ك بودجه‌ى باشه‌ و دواتریش باله‌ و باسكه‌.
- من وه‌كو سه‌لاَح حه‌ز ناكه‌م پێكهاته‌ى ئۆڵۆمپى ببێته‌ پێكهاته‌ى پارت و نه‌ته‌وه‌، حه‌زده‌كه‌م كادیرى وه‌رزشى خاوه‌ن كه‌سایه‌تى بێت سه‌ركرده‌ى وه‌رزش بێت، به‌لاَم بارى كوردستان ئێستا هه‌ر له‌ رووى هه‌موو سه‌ندیكاو حزب و په‌رله‌مانى كوردستان و عێراقیش به‌ ته‌وافوقێكى سیاسى دانراوه‌، لیژنه‌ى ئۆڵۆمپى كوردستانیش وایه‌، هیوادارم له‌ داهاتوودا كه‌سى شایسته‌ ببێته‌ سه‌ركردایه‌تى وه‌رزشى.
- هه‌ست به‌وه‌ ناكه‌م، دوو نموونه‌ ده‌هێَنم، ئه‌میندارى لیژنه‌ى ئۆڵۆمپى كه‌وا له‌ لایه‌نێكى تره‌ به‌رپرسه‌ له‌ هه‌موو یه‌كێتیه‌ وه‌رزشیه‌كانى كوردستان، بگره‌ هه‌ر له‌ خانه‌قین و مه‌نده‌لى تاكو سنوورى توركیا به‌ یه‌ك چاو سه‌یریان ده‌كات و بودجه‌یان بۆ دابین ده‌كات.
من نوێنه‌رایه‌تى چوار پارێزگارم له‌ ژێر ده‌سته‌، هه‌موو كاره‌كانیان له‌ ژێر ده‌ستى منه‌، ئه‌و حوكمه‌ى به‌سه‌ر سلێمانى ده‌كه‌م به‌سه‌ر هه‌ولێر و دهۆكیش ئه‌كه‌م.
- ئاماده‌بوونى سه‌رۆكى لیژنه‌ى ئۆڵۆمپى هه‌موو رۆژىَ بێته‌ ئێره‌ ئه‌رك نییه‌ فه‌رمانگه‌یه‌كى فه‌رمى نییه‌ كاره‌كانى رۆژانه‌ واژۆ بكات، فه‌رمانگه‌ نییه‌ وه‌كو په‌روه‌رده‌ و ته‌ندروستى هه‌موو رۆژێك ده‌وام بكات، ده‌سه‌لاَتى هه‌یه‌ دوایه‌تى به‌جێگره‌كانى و ئه‌میندارى دارایى و ئه‌میندارى لیژنه‌ى ئۆڵۆمپى و ئه‌ندامانى بۆ ئه‌وه‌ى كاره‌كانى خۆیان بكه‌ن، چى پێویست بێت به‌ برِیارى ئه‌و له‌ كۆبوونه‌وه‌ گرنگه‌كان ئاماده‌ ده‌بێت و برِیارى خۆى ده‌دات.
* بۆچى یه‌كێتیه‌ وه‌رزشییه‌كانى عێراق له‌ هه‌ولێر پاڵه‌وانه‌تیه‌كان ئه‌نجام ده‌ده‌ن، ئایا له‌به‌ر هۆكارى ئه‌وه‌یه‌ كه‌وا پاره‌كه‌ ئێوه‌ ده‌یده‌ن؟
