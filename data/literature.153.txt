قه‌ینا من و تۆ ته‌نیا هه‌ژده‌ مه‌تر لێكه‌وه‌ دوورین، به‌ڵام ئه‌و كاته‌ی نامه‌یه‌كتم پێده‌گا، ئیتر تاسه‌یه‌كی شیرینی خوێندنه‌وه‌ به‌رۆكم ئه‌گرێ و چوون ئه‌وه‌ی له‌ دۆستێكی هه‌ره‌ عه‌زیزی هه‌زار ساڵه‌ دووره‌وه‌ هاتبێ، ئاوا به‌ تامه‌زرۆییه‌وه‌ په‌یتاپه‌یتا زه‌رفه‌كه‌ له‌ نووكه‌وه‌ هه‌ڵه‌پچڕم و به‌كاوه‌خۆ كاغه‌زه‌كه‌ ڕائه‌كێشم.
هه‌نێجاریش دڵم خێرا خورپه‌ ئه‌كا و وه‌ك ئه‌وه‌ی هه‌ست به‌ كاره‌ساتێك بكه‌م، ئاوا به‌ په‌له‌پرووزێیه‌كی بێوێنه‌وه‌ زه‌رفه‌كه‌ هه‌ڵه‌پچڕم و نازانم چۆنچۆنی كاغه‌زه‌كه‌ ده‌ره‌كه‌م و به‌ فڕكانێ ده‌ست ئه‌كه‌م به‌ خوێندنه‌وه‌ی ئه‌و قسه‌ قۆڕو بێمه‌عنایانه‌ی كه‌ نووسیوتن.. زۆرجاری تریش سی چل ڕۆژ، یان ڕه‌نگه‌ كه‌متر به‌سه‌ر ناردنی نامه‌كه‌تدا تێپه‌ڕ بێ، به‌ڵام ئه‌وه‌ هێشتا نه‌گه‌شتووه‌ته‌ ده‌ستم، مه‌سه‌له‌كه‌ش پێوه‌ندی به‌ ئاسۆی برامه‌وه‌ هه‌یه‌، كه‌ ئه‌گه‌ر وابڕوا به‌ عه‌مه‌لی ئه‌وه‌ نایه‌، باوكم به‌و زووانه‌ ژنی بۆ بینێ، چونكه‌ خۆت ئه‌زانی ژنگه‌ل به‌ ته‌نها شه‌وان یه‌ك داواكارییان هه‌یه‌، ده‌نا به‌ ڕۆژ پیاوم ئه‌وێ له‌ یه‌ككاتدا بتوانێ ئیسكیف و زه‌رده‌چه‌وه‌و حه‌فازه‌و لاسقی ناو شه‌واڵ و زه‌نگیانه‌ی بیركه‌وێته‌وه‌.
هه‌ڵبه‌ت ئه‌و ته‌بیعه‌تی له‌بیرچوونه‌وه‌یه‌ش به‌لای ئاسۆی برامه‌وه‌ هه‌یه‌، هه‌روه‌ك ته‌بیعه‌تی زۆرخه‌وتنه‌كه‌ی، هۆكاری خۆی هه‌یه‌، كه‌ هه‌ریه‌كیان گومانێكیان له‌ پشته‌وه‌ڕا ئاماده‌یه‌، له‌وانه‌ش: ئه‌گه‌رچی خۆی كه‌مانچه‌ژه‌نه‌، به‌ڵام ڕه‌نگه‌ حه‌ز نه‌كا له‌ دائیره‌ی به‌رید دابمه‌زرێ، چونكه‌ ئه‌زانم كه‌پڵه‌كه‌ی حه‌زی له‌پیاوی پۆسته‌چی نییه‌، یان له‌وانه‌یه‌ به‌ خۆی و كه‌مانچه‌كه‌یه‌وه‌، كه‌ له‌گه‌ڵ هاوڕێكانیدا تیپێكی حیزانه‌ی مۆسیقایان دامه‌زراندووه‌، خه‌ریكی خۆئاماده‌كردن بێت بۆ ئه‌و ئاهه‌نگه‌ی كه‌ هه‌رگیز ساز ناكرێ، كه‌چی شانزه‌ مانگ خۆیانی پێوه‌ قاڵ ئه‌كه‌ن!
یان.. چی؟ نازانم.. گرنگ، نامه‌كانت له‌كاتی خۆیا نایگه‌یه‌نێ و ته‌واو..
چواره‌م.. چواره‌م.. چواره‌م، ئێێی..  ئا باسی كه‌پڵ و مه‌پڵم كرد، به‌ڵام ڕاستییه‌كه‌ی نازانم ئاخۆ ئه‌و برایه‌م كچێك له‌ ژیانیدا هه‌یه‌، یان نا؟ ئاسۆ ئه‌ڵێم، ئه‌و كه‌مانچه‌ژه‌نه‌ گوناحه‌ی كه‌ (مسته‌ر فرانس) له‌ هۆڵه‌نداوه‌ به‌ڕێگه‌ی سوریادا ده‌عوه‌تی كرد، به‌ڵام نه‌چوو، چونكه‌ هه‌موومان له‌ ماڵه‌وه‌، باوكیشم، پێمانوت: نه‌كه‌ی بڕۆی ئه‌وانه‌ به‌رازخۆرن.. ته‌نانه‌ت خۆمیش وتم: دوور نییه‌ هۆمۆسێكسیواڵ بێ، مه‌قه‌یه‌ت به‌، ئاسۆ!!
به‌هه‌رحاڵ له‌به‌ر ئه‌وه‌ی زۆرجار نامه‌كانت دره‌نگ ئه‌گه‌یه‌نێته‌ ده‌ستم، بۆیه‌ پێده‌چێ ئه‌ویش وه‌ك تۆ ده‌لاله‌كچێك، مێشكی ونجڕونجڕ كردبێ، خۆمان گوته‌نی: كچێك مێشكی گابێ!!
ئێستا كێشه‌ ئه‌وه‌ نییه‌، فه‌رمانبه‌ره‌كه‌ی پۆسته‌ نامه‌كان زوو ئه‌گه‌یه‌نێته‌ جێی خۆی یان نا، به‌ڵكو كێشه‌كه‌ ئه‌مه‌یانه‌:
هه‌رجاره‌ی كه‌ نامه‌یه‌كتم پێده‌گاو ده‌یخوێنمه‌وه‌، هه‌ر له‌خۆمه‌وه‌ حه‌زه‌كه‌م هیچ نه‌بێ‌ ئه‌گه‌ر به‌ دوو دێڕیش بێت، شتێكت هه‌ر بۆ بنووسم، كه‌چی نازانم نهێنییه‌كه‌ چییه‌ وا نامپه‌رژێ؟
تێناگه‌م بۆچی هه‌سته‌كه‌م ئه‌و قسانه‌ی كه‌ پێویسته‌ بۆتیان بنووسم هێشتا به‌ ته‌واوی نه‌خه‌مڵیون و چرۆیان نه‌كردووه‌؟
ناتوانم بڕوا به‌خۆم بێنم، له‌وه‌ی ئه‌م لقه‌ لاولاوه‌ی له‌ ده‌ستم ئاڵاوه‌و بوارم نادا بنووسم، هه‌مان مۆته‌كه‌ی ئه‌و عه‌شق و خۆشه‌ویستییه‌یه‌، كه‌ سه‌رده‌مانێك یاسه‌مینه‌كه‌ی لای من چوون ئه‌و ساشینكایه‌ی تۆ تووشی سه‌ره‌ سووڕێ و ماخولانی كردبووم.
سه‌باره‌ت به‌وه‌ی زۆر بێزاری له‌م شاره‌ و كه‌چی ناشتوانی لێی دووركه‌ویته‌وه‌، تكایه‌ لاپه‌ڕه‌ (54)ی (شاره‌كه‌م، پیره‌ قه‌حبه‌یه‌كی ڕموزنه‌) بخوێنه‌وه‌، ئه‌نجا بیرت نه‌چێ تۆ یه‌كه‌م كه‌س نیت، كه‌ ناتوانی ئێره‌ جێبهێڵی، ئه‌وه‌ نییه‌ هه‌موو ڕۆژێ منیش ئه‌ڕشێنێته‌وه‌، كه‌چی هیچ ده‌سه‌ڵاتیشم نییه‌.
هۆكاری سیانزه‌هه‌م، ڕه‌نگه‌ ئه‌وه‌بێ: كه‌ جاری واهه‌یه‌، به‌چه‌ند ڕۆژێك، من و پۆسته‌چییه‌كه‌، یه‌كتر نابینین.. (ئه‌م دێڕه‌یان په‌یوه‌ندی به‌ من و ئاسۆوه‌ هه‌یه‌، بۆیه‌ ئه‌توانن هه‌ر ئێستا بیسڕنه‌وه‌، یان هه‌ر مه‌یخوێننه‌وه‌).
قسه‌م زۆره‌، به‌ نووسینیش نه‌بێ‌ بۆم ناوترێن، نووسینیش كاتی ئه‌وێ كات.
كاتیش وه‌كئه‌وه‌ی گووی سه‌ی هاری خواردبێ، وه‌كئه‌وه‌ی بزانێ ئێمه‌ ئه‌مانه‌وێ بیگرین و بیگێین، ئاوا به‌لاماندا تێئه‌په‌ڕێ و گوزه‌ر ئه‌كا.
